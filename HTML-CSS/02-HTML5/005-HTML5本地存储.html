一、概念讲解

1.1 Web存储方案对比

特性 Cookie localStorage sessionStorage IndexedDB
容量 4KB 5MB 5MB 无上限（通常数百MB+）
过期 可设过期时间 永久 标签页关闭 永久
请求携带 ✅ 每次HTTP请求 ❌ ❌ ❌
API 字符串 同步键值对 同步键值对 异步
作用域 同域+路径 同源 同源+标签页 同源
数据类型 字符串 字符串 字符串 结构化数据
Worker访问 ❌ ✅ ❌ ✅

1.2 localStorage / sessionStorage

// Set item (auto-converts to string)
localStorage.setItem('username', '铁蛋');

// Get item
var name = localStorage.getItem('username'); // '铁蛋'

// Remove item
localStorage.removeItem('username');

// Clear all
localStorage.clear();

// Key by index
var key = localStorage.key(0);

// Length
var count = localStorage.length;

1.3 IndexedDB基础

IndexedDB是浏览器内置的NoSQL数据库，支持事务、索引和游标：

// Open database
var request = indexedDB.open('MyDatabase', 1);

request.onupgradeneeded = function(e) {
    var db = e.target.result;
    var store = db.createObjectStore('users', { keyPath: 'id' });
    store.createIndex('name', 'name', { unique: false });
};

request.onsuccess = function(e) {
    var db = e.target.result;
    // Start transaction, add data, etc.
};

二、知识脑图

HTML5本地存储
├── Cookie（传统）
│   ├── 4KB限制
│   ├── 每次请求携带
│   ├── document.cookie操作
│   └── 适用：会话token、用户偏好
├── Web Storage
│   ├── localStorage
│   │   ├── 5MB（同源）
│   │   ├── 永久存储
│   │   └── 同源共享
│   ├── sessionStorage
│   │   ├── 5MB（同源+标签页）
│   │   ├── 标签页关闭清除
│   │   └── 独立标签页隔离
│   └── API
│       ├── setItem / getItem
│       ├── removeItem / clear
│       ├── key / length
│       └── storage事件
├── IndexedDB
│   ├── 数据库（Database）
│   ├── 对象仓库（Object Store）
│   ├── 事务（Transaction）
│   ├── 索引（Index）
│   └── 游标（Cursor）
└── 最佳实践
    ├── 存储JSON用JSON.stringify/parse
    ├── try-catch处理quota
    ├── 敏感数据不存本地
    └── 定期清理过期数据

    <script>
        // ========== Tab switching ==========
        function switchTab(name) {
            document.querySelectorAll('.tab-btn').forEach(function(btn, i) {
                btn.classList.toggle('active', btn.textContent === name ||
                    (i === 0 && name === 'localStorage') ||
                    (i === 1 && name === 'sessionStorage') ||
                    (i === 2 && name === 'indexedDB') ||
                    (i === 3 && name === 'notes'));
            });

            document.querySelectorAll('.tab-btn').forEach(function(btn) {
                btn.classList.remove('active');
            });
            event.target.classList.add('active');

            document.querySelectorAll('.tab-panel').forEach(function(panel) {
                panel.classList.remove('active');
            });
            document.getElementById('panel-' + name).classList.add('active');

            // Refresh data
            if (name === 'localStorage') refreshLS();
            if (name === 'sessionStorage') refreshSS();
            if (name === 'indexedDB') refreshIDB();
            if (name === 'notes') refreshNotes();
        }

        // ========== localStorage ==========
        function refreshLS() {
            var body = document.getElementById('ls-body');
            body.innerHTML = '';
            var totalSize = 0;

            for (var i = 0; i < localStorage.length; i++) {
                var key = localStorage.key(i);
                var value = localStorage.getItem(key);
                totalSize += key.length + value.length;

                var tr = document.createElement('tr');
                tr.innerHTML = '<td class="key">' + escapeHtml(key) + '</td>' +
                    '<td class="value">' + escapeHtml(value) + '</td>' +
                    '<td class="actions"><button class="btn btn-danger" onclick="removeLS(\'' +
                    escapeHtml(key) + '\')">删除</button></td>';
                body.appendChild(tr);
            }

            document.getElementById('ls-count').textContent = localStorage.length;
            document.getElementById('ls-size').textContent = (totalSize * 2).toLocaleString(); // UTF-16
        }

        function addLSItem() {
            var key = document.getElementById('ls-key').value.trim();
            var value = document.getElementById('ls-value').value.trim();
            if (!key) return;

            try {
                localStorage.setItem(key, value);
                document.getElementById('ls-key').value = '';
                document.getElementById('ls-value').value = '';
                refreshLS();
            } catch (e) {
                alert('存储失败：可能超出容量限制 - ' + e.message);
            }
        }

        function removeLS(key) {
            localStorage.removeItem(key);
            refreshLS();
        }

        function clearLS() {
            if (confirm('确定清空所有localStorage数据？')) {
                localStorage.clear();
                refreshLS();
            }
        }

        // ========== sessionStorage ==========
        function refreshSS() {
            var body = document.getElementById('ss-body');
            body.innerHTML = '';

            for (var i = 0; i < sessionStorage.length; i++) {
                var key = sessionStorage.key(i);
                var value = sessionStorage.getItem(key);

                var tr = document.createElement('tr');
                tr.innerHTML = '<td class="key">' + escapeHtml(key) + '</td>' +
                    '<td class="value">' + escapeHtml(value) + '</td>' +
                    '<td><button class="btn btn-danger" onclick="sessionStorage.removeItem(\'' +
                    escapeHtml(key) + '\');refreshSS();">删除</button></td>';
                body.appendChild(tr);
            }
        }

        function addSSItem() {
            var key = document.getElementById('ss-key').value.trim();
            var value = document.getElementById('ss-value').value.trim();
            if (!key) return;
            sessionStorage.setItem(key, value);
            document.getElementById('ss-key').value = '';
            document.getElementById('ss-value').value = '';
            refreshSS();
        }

        // ========== IndexedDB ==========
        var idb = null;

        function initIDB() {
            var request = indexedDB.open('ContactDB', 1);

            request.onupgradeneeded = function(e) {
                var db = e.target.result;
                if (!db.objectStoreNames.contains('contacts')) {
                    var store = db.createObjectStore('contacts', { keyPath: 'id', autoIncrement: true });
                    store.createIndex('name', 'name', { unique: false });
                    store.createIndex('email', 'email', { unique: true });
                }
            };

            request.onsuccess = function(e) {
                idb = e.target.result;
                refreshIDB();
            };
        }

        function addIDBContact() {
            var name = document.getElementById('idb-name').value.trim();
            var email = document.getElementById('idb-email').value.trim();
            if (!name || !email) return;

            var tx = idb.transaction('contacts', 'readwrite');
            var store = tx.objectStore('contacts');
            var request = store.add({ name: name, email: email });

            request.onsuccess = function() {
                document.getElementById('idb-name').value = '';
                document.getElementById('idb-email').value = '';
                refreshIDB();
            };

            request.onerror = function(e) {
                alert('添加失败（邮箱可能重复）：' + e.target.error);
            };
        }

        function refreshIDB() {
            if (!idb) return;
            var body = document.getElementById('idb-body');
            body.innerHTML = '';

            var tx = idb.transaction('contacts', 'readonly');
            var store = tx.objectStore('contacts');
            var request = store.openCursor();

            request.onsuccess = function(e) {
                var cursor = e.target.result;
                if (cursor) {
                    var contact = cursor.value;
                    var tr = document.createElement('tr');
                    tr.innerHTML = '<td>' + escapeHtml(contact.name) + '</td>' +
                        '<td>' + escapeHtml(contact.email) + '</td>' +
                        '<td><button class="btn btn-danger" onclick="deleteIDBContact(' +
                        contact.id + ')">删除</button></td>';
                    body.appendChild(tr);
                    cursor.continue();
                }
            };
        }

        function deleteIDBContact(id) {
            var tx = idb.transaction('contacts', 'readwrite');
            tx.objectStore('contacts').delete(id);
            tx.oncomplete = function() { refreshIDB(); };
        }

        // ========== Notes App ==========
        function getNotes() {
            try {
                return JSON.parse(localStorage.getItem('my_notes') || '[]');
            } catch (e) {
                return [];
            }
        }

        function saveNotes(notes) {
            localStorage.setItem('my_notes', JSON.stringify(notes));
        }

        function addNote() {
            var text = document.getElementById('noteInput').value.trim();
            if (!text) return;

            var notes = getNotes();
            notes.unshift({
                id: Date.now(),
                text: text,
                time: new Date().toLocaleString('zh-CN')
            });
            saveNotes(notes);
            document.getElementById('noteInput').value = '';
            refreshNotes();
        }

        function deleteNote(id) {
            var notes = getNotes().filter(function(n) { return n.id !== id; });
            saveNotes(notes);
            refreshNotes();
        }

        function refreshNotes() {
            var notes = getNotes();
            var list = document.getElementById('notesList');

            if (notes.length === 0) {
                list.innerHTML = '<div class="empty-state"><div class="icon">📝</div><p>还没有备忘录</p></div>';
                return;
            }

            list.innerHTML = notes.map(function(note) {
                return '<div class="note-item">' +
                    '<div class="note-content">' +
                    '<div>' + escapeHtml(note.text) + '</div>' +
                    '<div class="note-time">' + note.time + '</div>' +
                    '</div>' +
                    '<button onclick="deleteNote(' + note.id + ')">×</button>' +
                    '</div>';
            }).join('');
        }

        // ========== Storage Event Listener ==========
        window.addEventListener('storage', function(e) {
            var log = document.getElementById('eventLog');
            var entry = document.createElement('div');
            entry.className = 'log-entry';
            entry.innerHTML = '<span class="log-time">[' + new Date().toLocaleTimeString() + ']</span> ' +
                e.key + ': "' + (e.oldValue || 'null') + '" → "' + (e.newValue || 'null') + '"';
            log.appendChild(entry);
            log.scrollTop = log.scrollHeight;
        });

        // ========== Utility ==========
        function escapeHtml(str) {
            var div = document.createElement('div');
            div.textContent = str;
            return div.innerHTML;
        }

        // ========== Initialize ==========
        refreshLS();
        initIDB();
        refreshNotes();
    </script>
</body>

执行预览

四标签页管理界面：

  1. localStorage — 统计条目数/已用空间，可增删数据，展示所有键值对
    1. sessionStorage — 同上，标签页关闭后数据消失
      1. IndexedDB — 联系人管理（姓名+邮箱），演示数据库事务操作
        1. 备忘录Demo — 实际应用案例，自动保存到localStorage，带删除和时间戳

          2. 四、注意事项

          注意点 说明 建议
          只存字符串 对象会变成"[object Object]" 用JSON.stringify/parse
          容量限制 localStorage约5MB 大数据用IndexedDB
          同步阻塞 localStorage是同步操作 大量数据可能阻塞主线程
          安全性 用户可在DevTools中查看修改 敏感数据不要存本地
          隐私模式 Safari隐私模式写入会报错 用try-catch包裹
          storage事件 只在同源的其他标签页触发 当前标签页不触发

          五、避坑指南

          ❌ 直接存对象

          localStorage.setItem('user', { name: '铁蛋' });
// Actual stored: "[object Object]"

          ✅ JSON序列化

          localStorage.setItem('user', JSON.stringify({ name: '铁蛋' }));
var user = JSON.parse(localStorage.getItem('user'));

          ❌ 不处理写入失败

          localStorage.setItem('data', hugeString); // May throw QuotaExceededError

          ✅ try-catch保护

          try {
    localStorage.setItem('data', hugeString);
} catch (e) {
    console.error('存储失败:', e);
    // Fallback: clear old data or use IndexedDB
}

          ❌ IndexedDB不用事务

          var store = db.objectStore('contacts'); // Error!

          ✅ 通过事务操作

          var tx = db.transaction('contacts', 'readwrite');
var store = tx.objectStore('contacts');
store.add(data);

          六、练习题

          🟢 入门

              2. 🟡 进阶

                  2. 🔴 挑战

                      2. 七、知识点总结

                      HTML5本地存储
├── Web Storage
│   ├── localStorage（永久，同源共享）
│   ├── sessionStorage（标签页隔离）
│   ├── API：setItem/getItem/removeItem/clear
│   └── storage事件（跨标签页）
├── IndexedDB
│   ├── Database → Object Store
│   ├── Transaction（事务）
│   ├── Index（索引）
│   ├── Cursor（游标遍历）
│   └── 异步API（request/result模式）
├── 对比选择
│   ├── Cookie：会话token（4KB）
│   ├── localStorage：用户偏好（5MB）
│   └── IndexedDB：大量数据（无限制）
└── 最佳实践
    ├── JSON序列化
    ├── try-catch
    ├── 过期清理
    └── 敏感数据不存本地

                      八、举一反三

                      场景 推荐方案 说明
                      用户偏好（主题/语言） localStorage 小数据、持久化
                      表单草稿自动保存 sessionStorage 标签页内临时
                      离线数据缓存 IndexedDB 大量结构化数据
                      购物车（未登录） localStorage 跨会话保留
                      多标签页同步 localStorage + storage事件 跨标签页通知
                      文件/图片缓存 IndexedDB + Blob 二进制数据

                        v2 — 带封装的存储工具

                        var Store = {
    get: function(key) {
        try { return JSON.parse(localStorage.getItem(key)); }
        catch(e) { return null; }
    },
    set: function(key, value) {
        try { localStorage.setItem(key, JSON.stringify(value)); }
        catch(e) { console.error('Storage full:', e); }
    }
};
