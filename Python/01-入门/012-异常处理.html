一、概念讲解

什么是异常（Exception）？

异常就是程序运行时发生的错误事件。Python 用异常对象来表示这些错误，你可以捕获并处理它们，而不是让程序崩溃。

# Without exception handling -> program crashes
result = 10 / 0  # ZeroDivisionError: division by zero

# With exception handling -> graceful recovery
try:
    result = 10 / 0
except ZeroDivisionError:
    print("Cannot divide by zero!")
    result = float('inf')

异常处理的价值

没有 try/except 有 try/except
程序崩溃退出 优雅降级，继续运行
用户看到一堆报错 用户看到友好提示
数据可能丢失 可以安全回滚
日志难以追踪 可以记录详细错误信息

Python 异常层级

BaseException
├── SystemExit          # sys.exit()
├── KeyboardInterrupt   # Ctrl+C
├── GeneratorExit       # generator.close()
└── Exception           # <- 你应该捕获的层级
    ├── ValueError      # 值不正确
    ├── TypeError       # 类型不匹配
    ├── KeyError        # 字典键不存在
    ├── IndexError      # 索引越界
    ├── FileNotFoundError  # 文件不存在
    ├── ZeroDivisionError  # 除以零
    ├── AttributeError     # 属性不存在
    ├── ImportError        # 导入失败
    ├── RuntimeError       # 运行时错误
    └── ... (100+ built-in exceptions)

三、完整代码示例

3.1 try/except 基础

# === Basic exception handling ===

# Catch a specific exception
try:
    number = int("hello")  # ValueError
except ValueError as e:
    print(f"Invalid number: {e}")

# Catch multiple exceptions
try:
    result = 10 / 0
except (ZeroDivisionError, TypeError) as e:
    print(f"Calculation error: {e}")

# try/except/else/finally
try:
    result = 10 / 2
except ZeroDivisionError:
    print("Division by zero!")
else:
    print(f"Result: {result}")  # Runs only if no exception
finally:
    print("Cleanup done")        # Always runs
# Output:
# Result: 5.0
# Cleanup done

3.2 常见内置异常处理

# === Common built-in exceptions ===

# ValueError: wrong value
try:
    age = int("twenty")
except ValueError:
    print("Please enter a valid number")

# TypeError: wrong type
try:
    result = "hello" + 42
except TypeError:
    print("Cannot add string and number")

# KeyError: missing dict key
data = {"name": "Alice"}
try:
    age = data["age"]
except KeyError:
    print("Key 'age' not found")

# IndexError: out of range
numbers = [1, 2, 3]
try:
    print(numbers[10])
except IndexError:
    print("Index out of range")

# FileNotFoundError
try:
    with open("nonexistent.txt") as f:
        content = f.read()
except FileNotFoundError:
    print("File not found")

3.3 raise 与自定义异常

# === Raise exceptions and custom exceptions ===

# Raise built-in exception
def set_age(age):
    if age < 0:
        raise ValueError("Age cannot be negative")
    if age > 150:
        raise ValueError("Age seems unrealistic")
    return age

# Custom exception class
class InsufficientFundsError(Exception):
    """Raised when account has insufficient funds."""

    def __init__(self, balance, amount):
        self.balance = balance
        self.amount = amount
        super().__init__(
            f"Cannot withdraw ${amount:.2f}: "
            f"balance is ${balance:.2f}"
        )

# Use custom exception
class BankAccount:
    def __init__(self, balance=0):
        self.balance = balance

    def deposit(self, amount):
        if amount <= 0:
            raise ValueError("Deposit amount must be positive")
        self.balance += amount
        return self.balance

    def withdraw(self, amount):
        if amount <= 0:
            raise ValueError("Withdrawal amount must be positive")
        if amount > self.balance:
            raise InsufficientFundsError(self.balance, amount)
        self.balance -= amount
        return self.balance

# Test the bank account
account = BankAccount(100)
try:
    account.withdraw(150)  # More than balance
except InsufficientFundsError as e:
    print(e)  # Cannot withdraw $150.00: balance is $100.00

3.4 异常链与上下文

# === Exception chaining ===
import json

class ConfigError(Exception):
    pass

# raise from: explicit chaining
def load_config(filepath):
    try:
        with open(filepath) as f:
            return json.load(f)
    except FileNotFoundError as e:
        raise ConfigError(f"Config file missing: {filepath}") from e

# Example: wrap low-level error with domain error
try:
    config = load_config("missing.json")
except ConfigError as e:
    print(f"Config error: {e}")
    print(f"Caused by: {e.__cause__}")

3.5 实战：健壮的数据处理管道

# === Robust data processing pipeline ===

import json
from datetime import datetime

class DataProcessingError(Exception):
    """Base error for data processing."""
    pass

def process_user_record(record, line_num):
    """Process a single user record with validation."""
    errors = []

    # Validate required fields
    if "name" not in record:
        errors.append("missing 'name'")

    if "email" not in record:
        errors.append("missing 'email'")

    if "age" in record:
        try:
            record["age"] = int(record["age"])
            if record["age"] < 0 or record["age"] > 150:
                errors.append(f"invalid age: {record['age']}")
        except ValueError:
            errors.append(f"age is not a number: {record['age']}")

    if errors:
        raise DataProcessingError(
            f"Line {line_num}: {', '.join(errors)}"
        )

    return record

def process_data_file(filepath):
    """Process a JSON lines file with error handling."""
    results = {"success": [], "errors": []}

    try:
        with open(filepath, "r", encoding="utf-8") as f:
            for line_num, line in enumerate(f, 1):
                line = line.strip()
                if not line:
                    continue

                try:
                    record = json.loads(line)
                    processed = process_user_record(record, line_num)
                    results["success"].append(processed)
                except json.JSONDecodeError as e:
                    results["errors"].append(
                        f"Line {line_num}: Invalid JSON - {e}"
                    )
                except DataProcessingError as e:
                    results["errors"].append(str(e))

    except FileNotFoundError:
        print(f"Error: File '{filepath}' not found")
        return None

    print(f"Processed: {len(results['success'])} successful, "
          f"{len(results['errors'])} errors")
    return results

五、注意事项

事项 说明 建议
裸 except except: 捕获一切，包括 KeyboardInterrupt 禁止使用，至少 except Exception
吞异常 except: pass 静默忽略所有错误 至少记录日志
过宽捕获 except Exception 捕获范围太大 尽量捕获具体异常类型
finally 中的异常 可能覆盖 try 中的原始异常 finally 中避免 raise
性能影响 异常处理比条件判断慢 用条件判断做预期检查
异常 vs 返回值 异常用于异常情况，返回值用于正常流程 不要用异常控制正常逻辑

六、避坑指南

❌ 坑1：裸 except 吞掉所有错误

# ❌ Catches EVERYTHING, including KeyboardInterrupt
try:
    do_something()
except:
    pass  # Silently swallows all errors!

# ✅ Catch specific exceptions
try:
    do_something()
except ValueError as e:
    print(f"Value error: {e}")
except FileNotFoundError as e:
    print(f"File error: {e}")

❌ 坑2：用异常控制正常流程

# ❌ Using exception for normal control flow
try:
    value = data["key"]
except KeyError:
    value = "default"

# ✅ Use dict.get() for expected cases
value = data.get("key", "default")

❌ 坑3：finally 中的 return 覆盖异常

# ❌ finally's return hides the exception
def bad_func():
    try:
        raise ValueError("Important error!")
    finally:
        return 42  # Exception is silently lost!

# ✅ Don't return in finally
def good_func():
    try:
        raise ValueError("Important error!")
    finally:
        cleanup()  # Just cleanup, no return

              v3：企业级错误处理框架

              # Production-grade error handling framework
import logging
import time
from functools import wraps
from typing import TypeVar, Callable, Any, Tuple, Type

T = TypeVar("T")
logger = logging.getLogger(__name__)

def retry(
    max_retries: int = 3,
    delay: float = 1.0,
    backoff: float = 2.0,
    exceptions: Tuple[Type[Exception], ...] = (Exception,),
):
    """Retry decorator with exponential backoff."""
    def decorator(func: Callable[..., T]) -> Callable[..., T]:
        @wraps(func)
        def wrapper(*args, **kwargs) -> T:
            retries = 0
            current_delay = delay
            while True:
                try:
                    return func(*args, **kwargs)
                except exceptions as e:
                    retries += 1
                    if retries > max_retries:
                        logger.error(
                            f"{func.__name__} failed after "
                            f"{max_retries} retries: {e}"
                        )
                        raise
                    logger.warning(
                        f"{func.__name__} retry {retries}/{max_retries}: {e}"
                    )
                    time.sleep(current_delay)
                    current_delay *= backoff
        return wrapper
    return decorator

class AppError(Exception):
    """Base application error with error code."""
    def __init__(self, message: str, code: str = "UNKNOWN"):
        self.message = message
        self.code = code
        super().__init__(message)

class ValidationError(AppError):
    def __init__(self, message: str):
        super().__init__(message, code="VALIDATION_ERROR")

class NotFoundError(AppError):
    def __init__(self, resource: str, identifier: Any):
        super().__init__(
            f"{resource} '{identifier}' not found",
            code="NOT_FOUND",
        )
