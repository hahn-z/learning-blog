031 - 实战：多人聊天室

📂 Swoole从入门到精通 🏷️ Swoole

难度标注

难度：★★★☆☆（中级）
前置知识：PHP 基础、Swoole WebSocket、Redis 基本操作

概念讲解

多人聊天室是实时通信的典型场景。传统 PHP 基于 HTTP 请求-响应模型，无法实现服务器主动推送。Swoole 的 WebSocket Server 提供了持久连接能力，使得实时聊天变得简单高效。

核心概念：

  • WebSocket：全双工通信协议，客户端和服务器建立持久连接后可双向发送消息
  • Fd（File Descriptor）：Swoole 中每个客户端连接的唯一标识
  • 广播：服务器收到消息后转发给所有在线用户
  • 房间（Room）：将用户分组，实现群聊隔离

    • 架构设计：

    Client A ←→ │            │ ←→ Client B
Client C ←→ │  Swoole    │ ←→ Client D
Client E ←→ │  WebSocket │ ←→ Client F
            │   Server   │
            └─────┬──────┘
                  │
            ┌─────┴──────┐
            │   Redis    │
            │ (用户状态)  │
            └────────────┘

    完整代码

    服务端（chat_server.php）

    <?php
// chat_server.php - Swoole WebSocket Chat Room Server

Swoole\Runtime::enableCoroutine();

$server = new Swoole\WebSocket\Server("0.0.0.0", 9501);

// Global state (use Redis in production)
$rooms = [];       // room_id => [fd => nickname]
$userFdMap = [];   // fd => ['nickname' => x, 'room' => y]

$server->on('WorkerStart', function ($server, $workerId) {
    echo "[Worker {$workerId}] Started\n";
});

$server->on('open', function ($server, $request) use (&$userFdMap) {
    echo "[Open] New connection: fd={$request->fd}\n";
    // Send welcome message
    $server->push($request->fd, json_encode([
        'type' => 'system',
        'msg'  => 'Welcome! Please set your nickname.',
        'time' => date('H:i:s'),
    ]));
});

$server->on('message', function ($server, $frame) use (&$rooms, &$userFdMap) {
    $fd   = $frame->fd;
    $data = json_decode($frame->data, true);

    if (!$data || !isset($data['type'])) {
        return;
    }

    switch ($data['type']) {
        case 'login':
            // User sets nickname and joins a room
            $nickname = trim($data['nickname'] ?? 'Anonymous');
            $roomId   = trim($data['room'] ?? 'lobby');

            $userFdMap[$fd] = [
                'nickname' => $nickname,
                'room'     => $roomId,
            ];

            if (!isset($rooms[$roomId])) {
                $rooms[$roomId] = [];
            }
            $rooms[$roomId][$fd] = $nickname;

            // Notify room
            broadcastToRoom($server, $rooms, $roomId, json_encode([
                'type' => 'system',
                'msg'  => "{$nickname} joined the room.",
                'time' => date('H:i:s'),
            ]), $excludeFd = null);

            // Send room user list to the new user
            $server->push($fd, json_encode([
                'type' => 'userlist',
                'users' => array_values($rooms[$roomId]),
                'room'  => $roomId,
            ]));

            echo "[Login] {$nickname} joined room '{$roomId}'\n";
            break;

        case 'chat':
            // Regular chat message
            if (!isset($userFdMap[$fd])) {
                $server->push($fd, json_encode([
                    'type' => 'error',
                    'msg'  => 'Please login first.',
                ]));
                return;
            }

            $userInfo = $userFdMap[$fd];
            $roomId   = $userInfo['room'];
            $msg      = trim($data['msg'] ?? '');

            if ($msg === '') {
                return;
            }

            $payload = json_encode([
                'type'     => 'chat',
                'nickname' => $userInfo['nickname'],
                'msg'      => $msg,
                'time'     => date('H:i:s'),
            ]);

            broadcastToRoom($server, $rooms, $roomId, $payload);
            break;

        case 'ping':
            // Heartbeat response
            $server->push($fd, json_encode(['type' => 'pong']));
            break;
    }
});

$server->on('close', function ($server, $fd) use (&$rooms, &$userFdMap) {
    if (!isset($userFdMap[$fd])) {
        return;
    }

    $userInfo = $userFdMap[$fd];
    $nickname = $userInfo['nickname'];
    $roomId   = $userInfo['room'];

    // Remove from room
    unset($rooms[$roomId][$fd]);
    if (empty($rooms[$roomId])) {
        unset($rooms[$roomId]);
    }

    // Remove user mapping
    unset($userFdMap[$fd]);

    // Notify room
    if (isset($rooms[$roomId])) {
        broadcastToRoom($server, $rooms, $roomId, json_encode([
            'type' => 'system',
            'msg'  => "{$nickname} left the room.",
            'time' => date('H:i:s'),
        ]), $excludeFd = null);
    }

    echo "[Close] {$nickname} left room '{$roomId}'\n";
});

/**
 * Broadcast message to all users in a room
 */
function broadcastToRoom($server, &$rooms, $roomId, $message, $excludeFd = null) {
    if (!isset($rooms[$roomId])) {
        return;
    }

    foreach ($rooms[$roomId] as $fd => $nickname) {
        if ($fd === $excludeFd) {
            continue;
        }
        if ($server->isEstablished($fd)) {
            $server->push($fd, $message);
        }
    }
}

$server->set([
    'heartbeat_check_interval' => 30,
    'heartbeat_idle_time'      => 60,
]);

echo "Chat Server started on ws://0.0.0.0:9501\n";
$server->start();

    <script>
        let ws = null;

        function doLogin() {
            const nick = document.getElementById('nickname').value;
            const room = document.getElementById('room').value;
            if (!nick) return alert('Enter a nickname');

            ws = new WebSocket('ws://localhost:9501');
            ws.onopen = () => {
                ws.send(JSON.stringify({ type: 'login', nickname: nick, room: room }));
                document.getElementById('login-area').style.display = 'none';
                document.getElementById('chat-box').style.display = 'block';
                document.getElementById('input-area').style.display = 'flex';
            };
            ws.onmessage = (e) => {
                const data = JSON.parse(e.data);
                const box = document.getElementById('chat-box');
                if (data.type === 'chat') {
                    box.innerHTML += `<div class="msg-chat"><span class="nick">${data.nickname}:</span> ${data.msg} <small>${data.time}</small></div>`;
                } else if (data.type === 'system') {
                    box.innerHTML += `<div class="msg-system">${data.msg}</div>`;
                }
                box.scrollTop = box.scrollHeight;
            };
            ws.onclose = () => {
                box.innerHTML += '<div class="msg-system">Disconnected.</div>';
            };
        }

        function sendMsg() {
            const input = document.getElementById('msg-input');
            if (!input.value.trim()) return;
            ws.send(JSON.stringify({ type: 'chat', msg: input.value }));
            input.value = '';
        }
    </script>
</body>
</html>

    注意事项

    项目 说明
    并发模型 每个 Worker 进程维护独立的 $rooms 数组，多 Worker 下需用 Table 或 Redis 共享状态
    内存泄漏 onClose 必须清理用户数据，否则数组持续增长
    心跳配置 设置 heartbeat_check_interval 防止僵尸连接
    消息大小 默认 max_frame_size 为 2MB，大文件传输需调整
    安全过滤 生产环境需过滤 XSS、限制消息频率
    集群部署 单机 $rooms 无法跨进程共享，集群需 Redis Pub/Sub

    避坑指南

    ❌ 常见错误 ✅ 正确做法
    用 $_SESSION 存用户状态 Swoole 是常驻内存，用 $userFdMap 或 Redis
    onMessage 中直接 print 输出 用 echo 输出到终端，push 给客户端
    忽略 isEstablished 检查 push 前先检查 $server->isEstablished($fd)
    多 Worker 直接操作全局变量 用 Swoole\Table 或 Redis 实现跨进程共享
    不处理心跳 配置 heartbeat_check_interval + 客户端定时 ping
    JSON 编码忽略 UTF-8 json_encode 加 JSON_UNESCAPED_UNICODE

$redis = new Swoole\Coroutine\Redis();
$redis->connect('127.0.0.1', 6379);

// Publish message to Redis channel
$redis->publish("room:{$roomId}", json_encode($payload));

// Subscribe in each worker
Swoole\Runtime::go(function () use ($redis) {
    $sub = new Swoole\Coroutine\Redis();
    $sub->connect('127.0.0.1', 6379);
    $sub->subscribe(['room:lobby']);
    while ($msg = $sub->recv()) {
        // Push to local clients in this worker
    }
