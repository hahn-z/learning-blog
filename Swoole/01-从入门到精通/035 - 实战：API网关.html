一、概念讲解

API网关是微服务架构中的统一入口，负责请求路由、负载均衡、限流熔断、认证鉴权等横切关注点。用Swoole构建API网关，可以获得极高的吞吐量和毫秒级响应延迟。

API网关核心职责

职责 说明 Swoole实现
路由转发 URL映射到后端服务 路由表 + 正则匹配
负载均衡 请求分发到多个实例 加权轮询 / 一致性哈希
限流熔断 保护后端服务 令牌桶 / 滑动窗口
认证鉴权 JWT/OAuth校验 协程HTTP客户端
日志监控 请求追踪 异步日志 + metrics
协议转换 HTTP ↔ gRPC等 协程管道

三、代码演进

v1: 基础反向代理

<?php
// v1_reverse_proxy.php - Simple reverse proxy gateway

$gateway = new Swoole\Http\Server("0.0.0.0", 9500);

// Route table: path prefix => backend service
$routes = [
    '/api/users' => 'http://user-service:8081',
    '/api/orders' => 'http://order-service:8082',
    '/api/products' => 'http://product-service:8083',
];

$gateway->on('request', function ($request, $response) use ($routes) {
    $path = $request->server['request_uri'] ?? '/';
    
    // Find matching backend
    $backend = null;
    foreach ($routes as $prefix => $service) {
        if (strpos($path, $prefix) === 0) {
            $backend = $service;
            break;
        }
    }
    
    if (!$backend) {
        $response->status(404);
        $response->end(json_encode(['error' => 'Route not found']));
        return;
    }
    
    // Forward request to backend
    go(function () use ($request, $response, $backend, $path) {
        $client = new Swoole\Coroutine\Http\Client(
            parse_url($backend, PHP_URL_HOST),
            parse_url($backend, PHP_URL_PORT) ?: 80
        );
        
        $client->setHeaders($request->header ?: []);
        $client->setMethod($request->server['request_method'] ?? 'GET');
        
        if ($request->rawcontent()) {
            $client->setData($request->rawcontent());
        }
        
        $fullPath = $path . '?' . ($request->server['query_string'] ?? '');
        $client->execute($fullPath);
        
        $response->status($client->statusCode);
        $response->header('content-type', $client->headers['content-type'] ?? 'application/json');
        $response->end($client->body);
        $client->close();
    });
});

echo "[Gateway] Listening on 0.0.0.0:9500\n";
$gateway->start();

v2: 带中间件的网关

<?php
// v2_middleware_gateway.php - Gateway with middleware chain

class ApiGateway
{
    private array $routes = [];
    private array $middlewares = [];
    private array $services = []; // service => [endpoints]
    
    public function addRoute(string $prefix, array $endpoints): void
    {
        $this->routes[$prefix] = [
            'prefix' => $prefix,
            'endpoints' => $endpoints,
            'current' => 0, // Round-robin index
        ];
    }
    
    public function addMiddleware(callable $middleware): void
    {
        $this->middlewares[] = $middleware;
    }
    
    public function handle(Swoole\Http\Request $request, Swoole\Http\Response $response): void
    {
        $path = $request->server['request_uri'] ?? '/';
        $startTime = microtime(true);
        
        // Build context
        $ctx = [
            'request' => $request,
            'response' => $response,
            'path' => $path,
            'method' => $request->server['request_method'] ?? 'GET',
            'headers' => [],
            'status' => 200,
            'body' => '',
            'start_time' => $startTime,
        ];
        
        // Run middleware chain
        $this->runMiddleware($ctx, 0);
    }
    
    private function runMiddleware(array &$ctx, int $index): void
    {
        if ($index >= count($this->middlewares)) {
            // All middleware passed, forward to backend
            $this->forward($ctx);
            return;
        }
        
        $next = function () use (&$ctx, $index) {
            $this->runMiddleware($ctx, $index + 1);
        };
        
        $this->middlewares[$index]($ctx, $next);
    }
    
    private function forward(array &$ctx): void
    {
        $path = $ctx['path'];
        $route = null;
        
        foreach ($this->routes as $prefix => $r) {
            if (strpos($path, $prefix) === 0) {
                $route = $r;
                break;
            }
        }
        
        if (!$route) {
            $this->sendError($ctx, 404, 'Route not found');
            return;
        }
        
        // Round-robin load balancing
        $idx = $route['current'] % count($route['endpoints']);
        $endpoint = $route['endpoints'][$idx];
        $route['current']++;
        $this->routes[$route['prefix']]['current'] = $route['current'];
        
        go(function () use ($ctx, $endpoint) {
            try {
                $host = parse_url($endpoint, PHP_URL_HOST);
                $port = parse_url($endpoint, PHP_URL_PORT) ?: 80;
                
                $client = new Swoole\Coroutine\Http\Client($host, $port);
                $client->setMethod($ctx['method']);
                $client->setHeaders(array_merge(
                    $ctx['request']->header ?? [],
                    ['X-Forwarded-For' => $ctx['request']->server['remote_addr'] ?? ''],
                    ['X-Request-Id' => uniqid('req_')],
                ));
                
                if ($raw = $ctx['request']->rawcontent()) {
                    $client->setData($raw);
                }
                
                $query = $ctx['request']->server['query_string'] ?? '';
                $fullPath = $ctx['path'] . ($query ? "?{$query}" : '');
                
                $client->execute($fullPath);
                
                $ctx['status'] = $client->statusCode;
                $ctx['body'] = $client->body;
                
                $resp = $ctx['response'];
                $resp->status($client->statusCode);
                foreach ($client->headers ?? [] as $k => $v) {
                    $resp->header($k, $v);
                }
                $resp->end($client->body);
                $client->close();
                
            } catch (Throwable $e) {
                $this->sendError($ctx, 502, 'Backend error: ' . $e->getMessage());
            }
        });
    }
    
    private function sendError(array &$ctx, int $code, string $msg): void
    {
        $ctx['response']->status($code);
        $ctx['response']->header('content-type', 'application/json');
        $ctx['response']->end(json_encode(['error' => $msg, 'code' => $code]));
    }
}

// === Build Gateway ===
$gw = new ApiGateway();

// Register backend services with multiple endpoints
$gw->addRoute('/api/users', ['http://user-svc-1:8081', 'http://user-svc-2:8081']);
$gw->addRoute('/api/orders', ['http://order-svc-1:8082']);
$gw->addRoute('/api/products', ['http://product-svc-1:8083', 'http://product-svc-2:8083']);

// Middleware: CORS
$gw->addMiddleware(function ($ctx, $next) {
    $ctx['response']->header('Access-Control-Allow-Origin', '*');
    $ctx['response']->header('Access-Control-Allow-Methods', 'GET, POST, PUT, DELETE');
    if (($ctx['method'] ?? '') === 'OPTIONS') {
        $ctx['response']->status(204);
        $ctx['response']->end();
        return;
    }
    $next();
});

// Middleware: Rate limiting (simple token bucket)
$rateLimitBucket = new Swoole\Table(1024);
$rateLimitBucket->column('tokens', Swoole\Table::TYPE_INT);
$rateLimitBucket->column('last_refill', Swoole\Table::TYPE_INT);
$rateLimitBucket->create();

$gw->addMiddleware(function ($ctx, $next) use ($rateLimitBucket) {
    $ip = $ctx['request']->server['remote_addr'] ?? 'unknown';
    $now = time();
    $maxTokens = 100;
    $refillRate = 10; // tokens per second
    
    $info = $rateLimitBucket->get($ip);
    if (!$info) {
        $rateLimitBucket->set($ip, ['tokens' => $maxTokens - 1, 'last_refill' => $now]);
        $next();
        return;
    }
    
    // Refill tokens
    $elapsed = $now - $info['last_refill'];
    $tokens = min($maxTokens, $info['tokens'] + $elapsed * $refillRate);
    
    if ($tokens < 1) {
        $ctx['response']->status(429);
        $ctx['response']->end(json_encode(['error' => 'Rate limit exceeded']));
        return;
    }
    
    $rateLimitBucket->set($ip, ['tokens' => $tokens - 1, 'last_refill' => $now]);
    $next();
});

// Middleware: JWT Auth (for protected routes)
$gw->addMiddleware(function ($ctx, $next) {
    $path = $ctx['path'] ?? '';
    $publicPaths = ['/api/products', '/api/auth'];
    
    foreach ($publicPaths as $public) {
        if (strpos($path, $public) === 0) {
            $next();
            return;
        }
    }
    
    $auth = $ctx['request']->header['authorization'] ?? '';
    if (!$auth || !preg_match('/Bearer\s+(.+)/', $auth, $m)) {
        $ctx['response']->status(401);
        $ctx['response']->end(json_encode(['error' => 'Missing auth token']));
        return;
    }
    
    // Validate JWT (simplified - use firebase/php-jwt in production)
    $token = $m[1];
    // ... JWT validation logic ...
    $ctx['user_id'] = 'extracted_user_id'; // From token payload
    $next();
});

// Middleware: Access log
$gw->addMiddleware(function ($ctx, $next) {
    $start = microtime(true);
    $next();
    $duration = round((microtime(true) - $start) * 1000, 2);
    $method = $ctx['method'] ?? 'GET';
    $path = $ctx['path'] ?? '/';
    $status = $ctx['status'] ?? 200;
    echo sprintf("[%s] %s %s -> %d (%.2fms)\n", date('H:i:s'), $method, $path, $status, $duration);
});

$server = new Swoole\Http\Server("0.0.0.0", 9500);
$server->on('request', [$gw, 'handle']);
$server->set(['worker_num' => 4]);
echo "[Gateway] Listening on 0.0.0.0:9500 (4 workers)\n";
$server->start();

v3: 完整网关（含服务发现与熔断）

<?php
// v3_full_gateway.php - Production gateway with service discovery and circuit breaker

class CircuitBreaker
{
    private Swoole\Table $states;
    private const CLOSED = 'closed';
    private const OPEN = 'open';
    private const HALF_OPEN = 'half_open';
    
    public function __construct()
    {
        $this->states = new Swoole\Table(256);
        $this->states->column('state', Swoole\Table::TYPE_STRING, 16);
        $this->states->column('failures', Swoole\Table::TYPE_INT);
        $this->states->column('last_failure', Swoole\Table::TYPE_INT);
        $this->states->column('successes', Swoole\Table::TYPE_INT);
        $this->states->create();
    }
    
    public function isAvailable(string $service): bool
    {
        $info = $this->states->get($service);
        if (!$info) return true;
        
        if ($info['state'] === self::OPEN) {
            // Check if cooldown period passed (30s)
            if (time() - $info['last_failure'] > 30) {
                $this->states->set($service, ['state' => self::HALF_OPEN, 'successes' => 0]);
                return true; // Allow one test request
            }
            return false;
        }
        return true;
    }
    
    public function recordSuccess(string $service): void
    {
        $info = $this->states->get($service);
        $successes = ($info['successes'] ?? 0) + 1;
        
        if ($info && $info['state'] === self::HALF_OPEN && $successes >= 3) {
            $this->states->set($service, ['state' => self::CLOSED, 'failures' => 0, 'successes' => $successes]);
        } else {
            $this->states->set($service, ['state' => $info['state'] ?? self::CLOSED, 'failures' => 0, 'successes' => $successes]);
        }
    }
    
    public function recordFailure(string $service): void
    {
        $info = $this->states->get($service);
        $failures = ($info['failures'] ?? 0) + 1;
        
        if ($failures >= 5) {
            $this->states->set($service, [
                'state' => self::OPEN,
                'failures' => $failures,
                'last_failure' => time(),
                'successes' => 0,
            ]);
            echo "[CircuitBreaker] {$service} -> OPEN ({$failures} failures)\n";
        } else {
            $this->states->set($service, [
                'state' => $info['state'] ?? self::CLOSED,
                'failures' => $failures,
                'last_failure' => time(),
                'successes' => 0,
            ]);
        }
    }
}

class ServiceRegistry
{
    private Swoole\Table $services;
    
    public function __construct()
    {
        $this->services = new Swoole\Table(256);
        $this->services->column('endpoints', Swoole\Table::TYPE_STRING, 1024);
        $this->services->column('updated_at', Swoole\Table::TYPE_INT);
        $this->services->create();
    }
    
    public function register(string $name, array $endpoints): void
    {
        $this->services->set($name, [
            'endpoints' => json_encode($endpoints),
            'updated_at' => time(),
        ]);
    }
    
    public function getEndpoints(string $name): array
    {
        $info = $this->services->get($name);
        return $info ? json_decode($info['endpoints'], true) : [];
    }
}

// === Production Gateway Server ===
$registry = new ServiceRegistry();
$breaker = new CircuitBreaker();

// Register services
$registry->register('user-service', ['192.168.1.10:8081', '192.168.1.11:8081']);
$registry->register('order-service', ['192.168.1.20:8082']);

$routeMap = [
    '/api/users' => 'user-service',
    '/api/orders' => 'order-service',
];

$server = new Swoole\Http\Server("0.0.0.0", 9500);
$server->set(['worker_num' => 4, 'max_coroutine' => 10000]);

$server->on('request', function ($req, $resp) use ($routeMap, $registry, $breaker) {
    $path = $req->server['request_uri'] ?? '/';
    
    // Find service
    $serviceName = null;
    foreach ($routeMap as $prefix => $svc) {
        if (strpos($path, $prefix) === 0) {
            $serviceName = $svc;
            break;
        }
    }
    
    if (!$serviceName) {
        $resp->status(404)->end(json_encode(['error' => 'Not found']));
        return;
    }
    
    // Circuit breaker check
    if (!$breaker->isAvailable($serviceName)) {
        $resp->status(503)->end(json_encode(['error' => 'Service unavailable (circuit open)']));
        return;
    }
    
    // Get endpoints and pick one (round-robin with random)
    $endpoints = $registry->getEndpoints($serviceName);
    if (empty($endpoints)) {
        $resp->status(503)->end(json_encode(['error' => 'No available endpoints']));
        return;
    }
    
    $endpoint = $endpoints[array_rand($endpoints)];
    
    go(function () use ($req, $resp, $endpoint, $serviceName, $breaker) {
        try {
            [$host, $port] = explode(':', $endpoint);
            $client = new Swoole\Coroutine\Http\Client($host, (int)$port, false);
            $client->setHeaders($req->header ?? []);
            $client->setMethod($req->server['request_method'] ?? 'GET');
            $client->setData($req->rawcontent() ?: '');
            $client->execute($req->server['request_uri'] ?? '/');
            
            $resp->status($client->statusCode);
            $resp->end($client->body);
            $client->close();
            
            if ($client->statusCode >= 500) {
                $breaker->recordFailure($serviceName);
            } else {
                $breaker->recordSuccess($serviceName);
            }
        } catch (Throwable $e) {
            $breaker->recordFailure($serviceName);
            $resp->status(502)->end(json_encode(['error' => $e->getMessage()]));
        }
    });
});

echo "[Gateway] Production gateway on :9500\n";
$server->start();

四、注意事项

项目 说明
连接池 网关高并发下必须用连接池，避免频繁创建TCP连接
超时设置 后端请求必须设超时，否则协程会一直挂起
Worker数 网关属于IO密集型，worker_num建议设为CPU核心数×2
请求体大小 注意设置package_max_length防止大请求攻击
热重载 路由配置变更不应重启服务，用共享内存或文件watch
健康检查 定时检测后端服务可用性，自动摘除故障节点

五、避坑指南

❌ 错误做法 ✅ 正确做法
每个请求new一个HTTP Client 复用连接或使用连接池
网关内做复杂业务逻辑 网关只做路由、转发、横切关注点
不设后端超时 必须设置timeout，建议3-5秒
同步阻塞IO 全部使用协程API
忽略连接关闭 用defer确保client->close()
不做限流 至少实现IP级别限流
