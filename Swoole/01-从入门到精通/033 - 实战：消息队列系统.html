概念讲解

消息队列（Message Queue）是分布式系统中解耦和异步处理的核心组件。生产者把消息投递到队列，消费者按自己的节奏取出处理，实现解耦、削峰、异步。

为什么需要消息队列？

场景 没有队列 有队列
用户注册发邮件 注册接口要等邮件发完才返回，慢！ 注册即返回，邮件异步发
秒杀抢购 数据库直接扛，容易挂 请求先进队列，慢慢消费
日志收集 每条日志写文件，I/O 阻塞 批量攒一批再写

消息队列模型：

Producer → [Queue/Topic] → Consumer

常见模式：
1. 点对点（PTP）：一条消息只被一个消费者处理
2. 发布/订阅（Pub/Sub）：一条消息被所有订阅者处理
3. 延迟队列：消息在指定时间后才可消费

本文实现：

  • 基于 Swoole Table + 进程间通信实现轻量级消息队列
  • 支持：投递、消费、确认（ACK）、延迟消息、死信队列

    消息队列核心（QueueServer.php）

    <?php
// QueueServer.php - Swoole-based Message Queue Server

class Message
{
    public string $id;
    public string $body;
    public int $createdAt;
    public int $deliverAt;  // 0 = immediate
    public int $retryCount  = 0;
    public int $maxRetries  = 3;
    public string $status   = 'pending'; // pending, processing, done, dead

    public function __construct(string $body, int $delayMs = 0)
    {
        $this->id        = uniqid('msg_', true);
        $this->body      = $body;
        $this->createdAt = time();
        $this->deliverAt = $delayMs > 0 ? (int)(microtime(true) * 1000) + $delayMs : 0;
    }

    public function toArray(): array
    {
        return [
            'id'         => $this->id,
            'body'       => $this->body,
            'created_at' => $this->createdAt,
            'deliver_at' => $this->deliverAt,
            'retry'      => $this->retryCount,
            'max_retry'  => $this->maxRetries,
            'status'     => $this->status,
        ];
    }
}

class QueueManager
{
    private array $queues    = [];  // topic => [Message, ...]
    private array $deadLetters = []; // failed messages
    private array $processing = []; // msgId => Message (in-flight)

    /**
     * Push a message to a topic queue
     */
    public function produce(string $topic, string $body, int $delayMs = 0): string
    {
        $msg = new Message($body, $delayMs);
        if (!isset($this->queues[$topic])) {
            $this->queues[$topic] = [];
        }
        $this->queues[$topic][] = $msg;
        return $msg->id;
    }

    /**
     * Consume one message from a topic (blocking if empty)
     */
    public function consume(string $topic, int $timeoutMs = 3000): ?Message
    {
        if (!isset($this->queues[$topic]) || empty($this->queues[$topic])) {
            return null;
        }

        $now = (int)(microtime(true) * 1000);

        foreach ($this->queues[$topic] as $i => $msg) {
            // Check delay
            if ($msg->deliverAt > 0 && $now < $msg->deliverAt) {
                continue; // Not ready yet
            }

            // Take message out of queue
            unset($this->queues[$topic][$i]);
            $this->queues[$topic] = array_values($this->queues[$topic]);

            $msg->status = 'processing';
            $this->processing[$msg->id] = $msg;

            return $msg;
        }

        return null;
    }

    /**
     * Acknowledge successful processing
     */
    public function ack(string $msgId): bool
    {
        if (isset($this->processing[$msgId])) {
            unset($this->processing[$msgId]);
            return true;
        }
        return false;
    }

    /**
     * Negative acknowledge - requeue or dead-letter
     */
    public function nack(string $msgId): bool
    {
        if (!isset($this->processing[$msgId])) {
            return false;
        }

        $msg = $this->processing[$msgId];
        unset($this->processing[$msgId]);

        $msg->retryCount++;
        if ($msg->retryCount >= $msg->maxRetries) {
            // Move to dead letter queue
            $msg->status = 'dead';
            $this->deadLetters[] = $msg;
            echo "[DeadLetter] {$msg->id} after {$msg->retryCount} retries\n";
            return false;
        }

        // Requeue for retry
        $msg->status = 'pending';
        $topic = $this->findTopicByMessage($msg);
        if ($topic) {
            $this->queues[$topic][] = $msg;
        }

        return true;
    }

    /**
     * Get queue stats
     */
    public function stats(): array
    {
        $stats = [
            'queues'      => [],
            'processing'  => count($this->processing),
            'dead_letters' => count($this->deadLetters),
        ];

        foreach ($this->queues as $topic => $messages) {
            $stats['queues'][$topic] = count($messages);
        }

        return $stats;
    }

    private function findTopicByMessage(Message $target): ?string
    {
        // Simple linear search (use a reverse index in production)
        foreach ($this->queues as $topic => &$messages) {
            foreach ($messages as $msg) {
                if ($msg->id === $target->id) {
                    return $topic;
                }
            }
        }
        return null;
    }
}

    队列服务器（mq_server.php）

    <?php
// mq_server.php - Message Queue TCP Server

$queueManager = new QueueManager();

$server = new Swoole\Server("0.0.0.0", 9503);

$server->on('receive', function ($server, $fd, $reactorId, $data) use ($queueManager) {
    $request = json_decode(trim($data), true);
    if (!$request || !isset($request['action'])) {
        $server->send($fd, json_encode(['error' => 'Invalid request']));
        return;
    }

    $response = match ($request['action']) {
        'produce' => handleProduce($queueManager, $request),
        'consume' => handleConsume($queueManager, $request),
        'ack'     => handleAck($queueManager, $request),
        'nack'    => handleNack($queueManager, $request),
        'stats'   => ['action' => 'stats', 'data' => $queueManager->stats()],
        default   => ['error' => "Unknown action: {$request['action']}"],
    };

    $server->send($fd, json_encode($response) . "\n");
});

function handleProduce(QueueManager $qm, array $request): array
{
    $topic = $request['topic'] ?? 'default';
    $body  = $request['body'] ?? '';
    $delay = $request['delay_ms'] ?? 0;

    $msgId = $qm->produce($topic, $body, $delay);

    echo "[Produce] topic={$topic} msgId={$msgId}" . ($delay ? " delay={$delay}ms" : "") . "\n";

    return [
        'action' => 'produce',
        'msg_id' => $msgId,
        'topic'  => $topic,
    ];
}

function handleConsume(QueueManager $qm, array $request): array
{
    $topic = $request['topic'] ?? 'default';

    $msg = $qm->consume($topic);

    if (!$msg) {
        return ['action' => 'consume', 'status' => 'empty'];
    }

    echo "[Consume] topic={$topic} msgId={$msg->id}\n";

    return [
        'action' => 'consume',
        'msg'    => $msg->toArray(),
    ];
}

function handleAck(QueueManager $qm, array $request): array
{
    $msgId = $request['msg_id'] ?? '';
    $ok = $qm->ack($msgId);
    return ['action' => 'ack', 'msg_id' => $msgId, 'ok' => $ok];
}

function handleNack(QueueManager $qm, array $request): array
{
    $msgId = $request['msg_id'] ?? '';
    $ok = $qm->nack($msgId);
    return ['action' => 'nack', 'msg_id' => $msgId, 'requeued' => $ok];
}

$server->set([
    'worker_num'    => 1, // Single worker for queue consistency
    'dispatch_mode' => 2, // Fixed fd to same worker
]);

echo "MQ Server started on tcp://0.0.0.0:9503\n";
$server->start();

    消费者 Worker（mq_worker.php）

    <?php
// mq_worker.php - Consumer Worker Process

Swoole\Runtime::enableCoroutine();

class QueueConsumer
{
    private Swoole\Coroutine\Client $client;
    private string $topic;
    private int $pollInterval;

    public function __construct(string $topic = 'default', int $pollIntervalMs = 100)
    {
        $this->topic       = $topic;
        $this->pollInterval = $pollIntervalMs;
    }

    private function connect(): Swoole\Coroutine\Client
    {
        $client = new Swoole\Coroutine\Client(SWOOLE_SOCK_TCP);
        $client->connect('127.0.0.1', 9503, 2.0);
        return $client;
    }

    private function sendCommand(array $command): array
    {
        $client = $this->connect();
        $client->send(json_encode($command) . "\n");
        $resp = $client->recv();
        $client->close();
        return json_decode(trim($resp), true) ?: [];
    }

    /**
     * Start consuming messages
     */
    public function run(callable $handler): void
    {
        echo "[Consumer] Started, topic={$this->topic}\n";

        while (true) {
            $resp = $this->sendCommand([
                'action' => 'consume',
                'topic'  => $this->topic,
            ]);

            if (($resp['status'] ?? '') === 'empty') {
                Swoole\Coroutine\System::sleep($this->pollInterval / 1000);
                continue;
            }

            $msg = $resp['msg'] ?? null;
            if (!$msg) {
                continue;
            }

            try {
                // Call user handler
                $result = $handler($msg['body'], $msg);

                // ACK on success
                $this->sendCommand(['action' => 'ack', 'msg_id' => $msg['id']]);
                echo "[ACK] {$msg['id']}\n";

            } catch (Throwable $e) {
                // NACK on failure
                $this->sendCommand(['action' => 'nack', 'msg_id' => $msg['id']]);
                echo "[NACK] {$msg['id']}: {$e->getMessage()}\n";
            }
        }
    }
}

// --- Run consumer ---
Swoole\Coroutine\run(function () {
    $consumer = new QueueConsumer('orders', 100);

    $consumer->run(function ($body, $msg) {
        echo "[Process] {$body}\n";

        // Simulate processing
        $data = json_decode($body, true);
        if (isset($data['fail']) && $data['fail']) {
            throw new RuntimeException("Simulated failure");
        }

        // Simulate work
        Swoole\Coroutine\System::sleep(0.1);
        echo "[Done] Processed order: " . ($data['order_id'] ?? 'unknown') . "\n";
    });
});

    注意事项

    项目 说明
    单 Worker 队列服务器用单 Worker 保证消息顺序和状态一致
    ACK 超时 消费者取走消息后若不 ACK，消息会"丢失"；生产环境需加超时回退机制
    持久化 本文用内存存储，重启丢消息；生产用 Redis 或磁盘持久化
    消费幂等 消费者可能收到重复消息，业务逻辑需幂等
    延迟精度 延迟队列基于轮询检查，精度取决于 poll 间隔
    并发消费 多消费者时需锁机制保证消息不被重复消费

    避坑指南

    ❌ 常见错误 ✅ 正确做法
    消费成功不 ACK 必须在处理完成后 ACK，否则消息可能丢失
    消费者不 catch 异常 处理失败要 NACK，让消息重试或进死信队列
    无限重试 设置 maxRetries，超过进死信队列
    队列无上限 设置队列最大深度，防止内存溢出
    单点故障 队列服务要做 HA（主备或集群）
    同步消费阻塞 用 Swoole 协程实现异步非阻塞消费

                // Redis-based persistent queue
class RedisQueue
{
    private Redis $redis;

    public function produce(string $topic, string $body, int $delayMs = 0): string
    {
        $msgId = uniqid('msg_', true);
        $msg = json_encode(['id' => $msgId, 'body' => $body, 'created' => time()]);

        if ($delayMs > 0) {
            $deliverAt = time() + (int)($delayMs / 1000);
            $this->redis->zAdd("delayed:{$topic}", $deliverAt, $msg);
        } else {
            $this->redis->rPush("queue:{$topic}", $msg);
        }

        return $msgId;
    }

    public function consume(string $topic, int $timeout = 3): ?array
    {
        // Check delayed messages first
        $now = time();
        $ready = $this->redis->zRangeByScore("delayed:{$topic}", 0, $now);
        foreach ($ready as $msg) {
            $this->redis->zRem("delayed:{$topic}", $msg);
            $this->redis->rPush("queue:{$topic}", $msg);
        }

        // Blocking pop
        $result = $this->redis->blPop(["queue:{$topic}"], $timeout);
        return $result ? json_decode($result[1], true) : null;
    }
}
