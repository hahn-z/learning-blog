概念讲解

RPC（Remote Procedure Call，远程过程调用）让调用远程服务就像调用本地函数一样简单。在微服务架构中，RPC 是服务间通信的核心机制。

为什么需要 RPC？

  • 微服务拆分：系统拆成多个服务后，服务间需要高效通信
  • 性能：比 HTTP REST 更轻量，使用二进制协议和长连接
  • 透明调用：客户端不需要知道服务端在哪、用什么语言

    • RPC 框架核心组件：

    ┌──────────┐         ┌──────────────┐         ┌──────────┐
│ Consumer │──Call──→│  RPC Server  │──Call──→│ Provider │
│ (Client) │←Result──│  (Registry)  │←Result──│ (Server) │
└──────────┘         └──────────────┘         └──────────┘

调用流程：
1. Consumer → 获取 Provider 地址 → 发送请求
2. Provider → 接收请求 → 调用本地方法 → 返回结果
3. Consumer → 接收结果 → 返回给调用方

    协议设计：

    +--------+--------+----------+---------+--------+
| Magic  | Version| Length   | Request │ Body   |
| 4bytes | 1byte  | 4bytes   | ID 4B   | N bytes|
+--------+--------+----------+---------+--------+

    完整代码

    协议类（Protocol.php）

    <?php
// Protocol.php - RPC Protocol Definition

class RpcProtocol
{
    const MAGIC    = 0x52504331; // "RPC1"
    const VERSION  = 0x01;
    const HEADER_SIZE = 13; // 4 + 1 + 4 + 4

    /**
     * Encode a request into binary packet
     */
    public static function encode(string $body): string
    {
        $bodyLen = strlen($body);
        // Pack: magic(N) + version(C) + length(N) + requestId(N)
        $header = pack('NCCN', self::MAGIC, self::VERSION, 0, $bodyLen);
        // Fix: use proper pack format
        $header = pack('NnNN', self::MAGIC, self::VERSION, 0, $bodyLen);
        return $header . $body;
    }

    /**
     * Decode header from binary data
     */
    public static function decodeHeader(string $data): ?array
    {
        if (strlen($data) < self::HEADER_SIZE) {
            return null;
        }

        $arr = unpack('Nmagic/nversion/Nrequest_id/Nbody_len', $data);

        if ($arr['magic'] !== self::MAGIC) {
            throw new RuntimeException("Invalid magic number");
        }

        return $arr;
    }

    /**
     * Create a request payload
     */
    public static function makeRequest(string $service, string $method, array $args): string
    {
        return json_encode([
            'service' => $service,
            'method'  => $method,
            'args'    => $args,
            'traceid' => uniqid('rpc_', true),
        ]);
    }
}

    RPC Server（rpc_server.php）

    <?php
// rpc_server.php - RPC Service Provider

$server = new Swoole\Server("0.0.0.0", 9502);

// Service registry: class => object
$services = [];

/**
 * Register a service class
 */
function registerService(string $name, object $instance) use (&$services) {
    $services[$name] = $instance;
}

// --- Define Sample Services ---

class UserService
{
    public function getById(int $id): array
    {
        // Simulate database query
        $users = [
            1 => ['id' => 1, 'name' => 'Alice', 'email' => 'alice@example.com'],
            2 => ['id' => 2, 'name' => 'Bob', 'email' => 'bob@example.com'],
            3 => ['id' => 3, 'name' => 'Charlie', 'email' => 'charlie@example.com'],
        ];
        return $users[$id] ?? ['error' => 'User not found'];
    }

    public function getList(int $page = 1, int $size = 10): array
    {
        return [
            'page' => $page,
            'size' => $size,
            'list' => [
                ['id' => 1, 'name' => 'Alice'],
                ['id' => 2, 'name' => 'Bob'],
            ],
        ];
    }
}

class OrderService
{
    public function create(array $order): array
    {
        return [
            'order_id' => rand(10000, 99999),
            'status'   => 'created',
            'items'    => $order['items'] ?? [],
            'total'    => $order['total'] ?? 0,
        ];
    }
}

// Register services
$services['UserService'] = new UserService();
$services['OrderService'] = new OrderService();

$server->on('receive', function ($server, $fd, $reactorId, $data) use (&$services) {
    try {
        $header = RpcProtocol::decodeHeader($data);
        if (!$header) {
            $server->send($fd, json_encode(['error' => 'Invalid header']));
            return;
        }

        $bodyLen = $header['body_len'];
        $body = substr($data, RpcProtocol::HEADER_SIZE, $bodyLen);
        $request = json_decode($body, true);

        $serviceName = $request['service'] ?? '';
        $method      = $request['method'] ?? '';
        $args        = $request['args'] ?? [];

        echo "[RPC] {$serviceName}::{$method}(" . json_encode($args) . ")\n";

        // Check service exists
        if (!isset($services[$serviceName])) {
            $response = json_encode(['error' => "Service '{$serviceName}' not found"]);
            $server->send($fd, RpcProtocol::encode($response));
            return;
        }

        $service = $services[$serviceName];

        // Check method exists
        if (!method_exists($service, $method)) {
            $response = json_encode(['error' => "Method '{$method}' not found"]);
            $server->send($fd, RpcProtocol::encode($response));
            return;
        }

        // Call the service method
        $result = call_user_func_array([$service, $method], $args);

        $response = json_encode([
            'result'  => $result,
            'traceid' => $request['traceid'] ?? '',
        ]);

        $server->send($fd, RpcProtocol::encode($response));

    } catch (Throwable $e) {
        $error = json_encode([
            'error' => $e->getMessage(),
            'file'  => $e->getFile(),
            'line'  => $e->getLine(),
        ]);
        $server->send($fd, RpcProtocol::encode($error));
    }
});

$server->set([
    'worker_num'            => 2,
    'max_request'           => 10000,
    'dispatch_mode'         => 1, // Round-robin
    'open_length_check'     => true,
    'package_length_type'   => 'N',
    'package_length_offset' => 8,
    'package_body_offset'   => 13,
]);

echo "RPC Server started on tcp://0.0.0.0:9502\n";
$server->start();

    RPC Client（rpc_client.php）

    <?php
// rpc_client.php - RPC Client with connection pool

Swoole\Runtime::enableCoroutine();

class RpcClient
{
    private string $host;
    private int $port;
    private float $timeout;
    private ?Swoole\Coroutine\Client $client = null;

    public function __construct(string $host = '127.0.0.1', int $port = 9502, float $timeout = 3.0)
    {
        $this->host    = $host;
        $this->port    = $port;
        $this->timeout = $timeout;
    }

    /**
     * Lazy connect
     */
    private function connect(): void
    {
        if ($this->client && $this->client->isConnected()) {
            return;
        }

        $this->client = new Swoole\Coroutine\Client(SWOOLE_SOCK_TCP);
        $this->client->set([
            'open_length_check'     => true,
            'package_length_type'   => 'N',
            'package_length_offset' => 8,
            'package_body_offset'   => 13,
        ]);

        if (!$this->client->connect($this->host, $this->port, $this->timeout)) {
            throw new RuntimeException("Connect failed: {$this->client->errMsg}");
        }
    }

    /**
     * Call a remote service method
     */
    public function call(string $service, string $method, array $args = [])
    {
        $this->connect();

        $body    = RpcProtocol::makeRequest($service, $method, $args);
        $packet  = RpcProtocol::encode($body);

        $this->client->send($packet);

        $response = $this->client->recv(-1);
        if (!$response) {
            throw new RuntimeException("Recv failed: {$this->client->errMsg}");
        }

        $header = RpcProtocol::decodeHeader($response);
        $result = substr($response, RpcProtocol::HEADER_SIZE, $header['body_len']);

        return json_decode($result, true);
    }

    public function close(): void
    {
        if ($this->client) {
            $this->client->close();
            $this->client = null;
        }
    }
}

// --- Demo Usage ---

Swoole\Coroutine\run(function () {
    $client = new RpcClient();

    // Call UserService::getById
    $result = $client->call('UserService', 'getById', [1]);
    echo "getById(1): " . json_encode($result, JSON_PRETTY_PRINT) . "\n\n";

    // Call UserService::getList
    $result = $client->call('UserService', 'getList', [1, 10]);
    echo "getList(1, 10): " . json_encode($result, JSON_PRETTY_PRINT) . "\n\n";

    // Call OrderService::create
    $result = $client->call('OrderService', 'create', [[
        'items' => ['Widget A', 'Widget B'],
        'total' => 99.99,
    ]]);
    echo "create(order): " . json_encode($result, JSON_PRETTY_PRINT) . "\n\n";

    // Call non-existent service (error handling)
    $result = $client->call('PaymentService', 'pay', [100]);
    echo "pay(100): " . json_encode($result, JSON_PRETTY_PRINT) . "\n";

    $client->close();
});

    注意事项

    项目 说明
    包协议 必须用长度检查协议（open_length_check），否则 TCP 粘包
    异常处理 服务端异常要捕获并返回，不能让 Worker 崩溃
    超时控制 客户端必须设置超时，防止服务端卡死导致客户端挂起
    服务发现 生产环境用注册中心（Consul/Nacos），不用硬编码地址
    连接池 高并发场景用连接池管理 TCP 连接，避免频繁创建销毁
    序列化 JSON 简单但性能一般，生产推荐 MsgPack 或 Hessian

    避坑指南

    ❌ 常见错误 ✅ 正确做法
    不处理 TCP 粘包 设置 open_length_check + package_length_type
    用 file_get_contents 模拟 RPC 用 Swoole Coroutine Client 做长连接
    服务端异常直接 throw 到 Worker try-catch 包裹，返回错误 JSON
    客户端不设超时 connect 和 recv 都加 timeout
    硬编码服务地址 用配置中心或注册中心动态获取
    serialize() 做序列化 有安全风险，用 JSON 或 MsgPack
