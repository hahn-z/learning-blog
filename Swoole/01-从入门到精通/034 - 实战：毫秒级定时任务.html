实战：毫秒级定时任务

难度：⭐⭐⭐中级 | 预计阅读时间：20分钟

一、概念讲解

传统PHP的定时任务依赖系统crontab，最小粒度只有1分钟。Swoole内置的定时器基于epoll/kqueue实现，支持毫秒级精度，彻底改变了PHP在定时任务领域的能力边界。

核心概念

概念 说明
tick 周期性定时器，每隔N毫秒执行一次
after 一次性定时器，N毫秒后执行一次
Timer Wheel Swoole内部的时间轮算法，O(1)插入/删除
协程定时器 在协程环境中使用的定时器API

Swoole定时器 vs 传统Crontab：

维度 Crontab Swoole Timer
最小粒度 1分钟 1毫秒
进程模型 每次fork新进程 进程内回调
性能开销 高（进程创建） 极低（epoll事件）
精度 ±1分钟 ±1毫秒
适用场景 日常运维 实时监控、心跳检测

三、代码演进

v1: 基础定时器

<?php
// v1_basic_timer.php - Basic timer usage

$server = new Swoole\Timer();

// One-shot timer: execute after 2000ms
Swoole\Timer::after(2000, function () {
    echo "[After] 2 seconds passed!\n";
});

// Periodic timer: execute every 1000ms
$timerId = Swoole\Timer::tick(1000, function ($id) {
    static $count = 0;
    $count++;
    echo "[Tick #{$count}] Timer ID: {$id}, Time: " . date('H:i:s') . "\n";
    
    // Stop after 5 ticks
    if ($count >= 5) {
        Swoole\Timer::clear($id);
        echo "[Tick] Timer cleared.\n";
    }
});

echo "Timers started. Waiting...\n";

v2: Server中的定时任务

<?php
// v2_server_timer.php - Timers inside Swoole Server

$server = new Swoole\WebSocket\Server("0.0.0.0", 9501);

// Track connected clients and their last heartbeat
$clients = [];

$server->on('open', function ($server, $request) use (&$clients) {
    $clients[$request->fd] = [
        'fd' => $request->fd,
        'last_heartbeat' => time(),
        'connect_time' => time(),
    ];
    echo "[Open] Client #{$request->fd} connected. Total: " . count($clients) . "\n";
});

$server->on('message', function ($server, $frame) use (&$clients) {
    if ($frame->data === 'ping') {
        $clients[$frame->fd]['last_heartbeat'] = time();
        $server->push($frame->fd, 'pong');
    }
});

$server->on('close', function ($server, $fd) use (&$clients) {
    unset($clients[$fd]);
    echo "[Close] Client #{$fd} disconnected. Total: " . count($clients) . "\n";
});

$server->on('workerStart', function ($server, $workerId) use (&$clients) {
    // Heartbeat check every 10 seconds
    $server->tick(10000, function () use ($server, &$clients) {
        $now = time();
        $timeout = 30; // 30 seconds timeout
        
        foreach ($clients as $fd => $info) {
            if ($now - $info['last_heartbeat'] > $timeout) {
                echo "[Heartbeat] Client #{$fd} timeout, closing.\n";
                $server->close($fd);
                unset($clients[$fd]);
            }
        }
    });
    
    // Stats reporter every 60 seconds
    $server->tick(60000, function () use ($server, &$clients) {
        $count = count($clients);
        $mem = memory_get_usage(true) / 1024 / 1024;
        echo "[Stats] Connections: {$count}, Memory: " . round($mem, 2) . " MB\n";
    });
    
    echo "[Worker #{$workerId}] Timers started.\n";
});

$server->start();

v3: 企业级定时任务管理器

<?php
// v3_task_manager.php - Enterprise-grade cron task manager

class CronTaskManager
{
    private array $tasks = [];
    private array $timers = [];
    private Swoole\Table $statusTable;
    
    public function __construct()
    {
        // Shared memory table for cross-worker status
        $this->statusTable = new Swoole\Table(1024);
        $this->statusTable->column('task_name', Swoole\Table::TYPE_STRING, 64);
        $this->statusTable->column('last_run', Swoole\Table::TYPE_INT);
        $this->statusTable->column('run_count', Swoole\Table::TYPE_INT);
        $this->statusTable->column('status', Swoole\Table::TYPE_STRING, 16);
        $this->statusTable->column('error', Swoole\Table::TYPE_STRING, 255);
        $this->statusTable->create();
    }
    
    /**
     * Register a cron task
     */
    public function register(string $name, int $intervalMs, callable $callback, array $options = []): void
    {
        $this->tasks[$name] = [
            'name' => $name,
            'interval' => $intervalMs,
            'callback' => $callback,
            'timeout' => $options['timeout'] ?? 5000,
            'retry' => $options['retry'] ?? 0,
            'retry_delay' => $options['retry_delay'] ?? 1000,
        ];
    }
    
    /**
     * Start all registered tasks
     */
    public function start(): void
    {
        foreach ($this->tasks as $name => $task) {
            $this->startTask($task);
        }
        echo "[CronManager] " . count($this->tasks) . " tasks started.\n";
    }
    
    private function startTask(array $task): void
    {
        $timerId = Swoole\Timer::tick($task['interval'], function () use ($task) {
            $this->executeTask($task);
        });
        $this->timers[$task['name']] = $timerId;
        
        // Initialize status
        $this->statusTable->set($task['name'], [
            'task_name' => $task['name'],
            'last_run' => 0,
            'run_count' => 0,
            'status' => 'idle',
            'error' => '',
        ]);
    }
    
    private function executeTask(array $task): void
    {
        $name = $task['name'];
        
        // Check if previous run is still active
        $status = $this->statusTable->get($name);
        if ($status && $status['status'] === 'running') {
            echo "[CronManager] Task '{$name}' skipped (previous run still active).\n";
            return;
        }
        
        $this->statusTable->set($name, [
            'status' => 'running',
            'task_name' => $name,
            'last_run' => time(),
            'run_count' => ($status['run_count'] ?? 0) + 1,
            'error' => '',
        ]);
        
        try {
            // Execute with timeout protection
            $timeout = $task['timeout'];
            $result = Swoole\Coroutine\run(function () use ($task, $timeout) {
                $chan = new Swoole\Coroutine\Channel(1);
                
                go(function () use ($task, $chan) {
                    try {
                        $task['callback']();
                        $chan->push(['success' => true]);
                    } catch (Throwable $e) {
                        $chan->push(['success' => false, 'error' => $e->getMessage()]);
                    }
                });
                
                $result = $chan->pop($timeout / 1000);
                return $result ?: ['success' => false, 'error' => 'Timeout'];
            });
            
            if (!$result['success']) {
                throw new RuntimeException($result['error'] ?? 'Unknown error');
            }
            
            $this->statusTable->set($name, [
                'status' => 'success',
                'task_name' => $name,
                'error' => '',
            ]);
            
            echo "[CronManager] Task '{$name}' completed successfully.\n";
            
        } catch (Throwable $e) {
            $this->statusTable->set($name, [
                'status' => 'error',
                'task_name' => $name,
                'error' => $e->getMessage(),
            ]);
            echo "[CronManager] Task '{$name}' failed: {$e->getMessage()}.\n";
            
            // Retry logic
            if ($task['retry'] > 0) {
                $this->scheduleRetry($task, $task['retry']);
            }
        }
    }
    
    private function scheduleRetry(array $task, int $remaining): void
    {
        Swoole\Timer::after($task['retry_delay'], function () use ($task, $remaining) {
            echo "[CronManager] Retrying task '{$task['name']}' ({$remaining} attempts left).\n";
            $this->executeTask(array_merge($task, ['retry' => $remaining - 1]));
        });
    }
    
    /**
     * Get status of all tasks
     */
    public function getStatus(): array
    {
        $statuses = [];
        foreach ($this->statusTable as $key => $row) {
            $statuses[] = $row;
        }
        return $statuses;
    }
    
    /**
     * Stop a specific task
     */
    public function stop(string $name): bool
    {
        if (isset($this->timers[$name])) {
            Swoole\Timer::clear($this->timers[$name]);
            unset($this->timers[$name]);
            echo "[CronManager] Task '{$name}' stopped.\n";
            return true;
        }
        return false;
    }
}

// === Usage Example ===
$manager = new CronTaskManager();

// Register tasks with different intervals
$manager->register('cache_warmup', 60000, function () {
    // Refresh cache every 60 seconds
    $redis = new Swoole\Coroutine\Redis();
    $redis->connect('127.0.0.1', 6379);
    $redis->set('cache:warmup:time', date('Y-m-d H:i:s'));
    echo "  -> Cache warmed up.\n";
}, ['timeout' => 3000, 'retry' => 2]);

$manager->register('metrics_collect', 5000, function () {
    // Collect metrics every 5 seconds
    $mem = memory_get_usage(true);
    echo "  -> Memory: " . round($mem / 1024 / 1024, 2) . " MB\n";
});

$manager->register('data_sync', 30000, function () {
    // Sync data every 30 seconds
    echo "  -> Data sync completed.\n";
}, ['timeout' => 10000]);

$manager->start();

// Keep the process running
Swoole\Event::wait();

四、注意事项

项目 说明
定时器精度 实际精度受事件循环影响，通常误差在±1ms内
Worker进程 定时器在Worker进程中运行，不要在Task进程中创建
内存泄漏 定时器回调中创建的对象会在回调结束后释放，注意闭包引用
最大数量 默认最多65535个定时器，可通过max_timer调整
信号冲突 Swoole\Timer与PCNTL信号不兼容，不要混用
协程环境 在协程中优先使用Coroutine::sleep()替代Timer::after()

五、避坑指南

❌ 错误做法 ✅ 正确做法
Timer::tick(1, ...) 设置1ms间隔做重IO 根据任务耗时设置合理间隔，≥100ms
在定时器回调中使用sleep() 使用Swoole\Coroutine\sleep()或Timer
定时器内抛出异常不捕获 所有回调必须try-catch包裹
在onReceive中创建定时器不清理 使用workerStart统一管理定时器
用全局变量在多个定时器间共享状态 使用Swoole\Table或Redis共享状态
忘记clear导致定时器泄漏 注册时记录timerId，结束时clear
