一、概念讲解

每年ECMAScript都会发布新特性。ES2020-ES2023带来了多个实用语法糖，显著提升代码可读性和安全性。

ES2020

  • Optional Chaining ?.：安全访问深层属性
  • Nullish Coalescing ??：仅null/undefined时用默认值
  • BigInt：任意精度整数
  • Promise.allSettled：等待全部完成
  • Dynamic import()`：按需加载模块
  • globalThis：跨环境全局对象

    • ES2021

    • Logical Assignment &&=/||=/??=：逻辑运算+赋值
    • Numeric Separators _：数字分隔符
    • Promise.any：第一个成功
    • WeakRef/FinalizationRegistry：弱引用

      • ES2022

      • Top-level await：模块顶层直接await
      • Class Fields：#private、公共字段
      • at()：数组/字符串反向索引
      • Object.hasOwn：安全的hasOwnProperty

        • ES2023

        • Change Array by Copy：toSorted/toReversed/toSpliced/with
        • Hashbang Grammar：#!支持
        • Symbols as WeakMap keys

          • 二、脑图

                  ES2020-2023新特性
        /    |    |    \
     ES2020 ES2021 ES2022 ES2023
      / | \    |    / \     |
   ?. ?? BigInt ??= #  at() toSorted
   allSettled  _  await hasOwn toReversed

          三、完整代码示例

          // ============================================
// 1. ES2020: Optional Chaining & Nullish Coalescing
// ============================================
const user = {
  profile: {
    name: "Alice",
    address: null,
  },
};

// Optional chaining — safe deep access
console.log(user.profile?.name);         // "Alice"
console.log(user.profile?.address?.city); // undefined (not error!)
console.log(user.profile?.phone?.code);   // undefined

// Nullish coalescing — only null/undefined trigger default
console.log(user.profile.address ?? "no address"); // "no address"
console.log(0 ?? "default");    // 0 (not "default"!)
console.log("" ?? "default");   // "" (not "default"!)
console.log(false ?? "default");// false (not "default"!)
console.log(null ?? "default"); // "default"
console.log(undefined ?? "default"); // "default"

// vs || (OR) which treats all falsy as default
console.log(0 || "default");    // "default" (often unexpected!)
console.log("" || "default");   // "default"

// ============================================
// 2. ES2021: Logical Assignment & Numeric Separators
// ============================================
let x = null;
x ??= 42; // x = x ?? 42
console.log(x); // 42

let count = 0;
count ||= 1; // count = count || 1 → 0 is falsy → 1
console.log(count); // 1

let valid = true;
valid &&= false; // valid = valid && false
console.log(valid); // false

// Numeric separators
const million = 1_000_000;
const bytes = 0xFF_FF_FF_FF;
const binary = 0b1010_0101;
console.log(million, bytes, binary); // 1000000 4294967295 165

// ============================================
// 3. ES2022: Class Fields, at(), Object.hasOwn
// ============================================
class Modal {
  #isOpen = false;          // private field
  title = "Untitled";       // public field

  static #count = 0;        // private static

  constructor(title) {
    this.title = title;
    Modal.#count++;
  }

  open() { this.#isOpen = true; }
  close() { this.#isOpen = false; }
  get isOpen() { return this.#isOpen; }
  static get instanceCount() { return Modal.#count; }
}

const m1 = new Modal("Settings");
const m2 = new Modal("Help");
console.log(m1.title, m1.isOpen); // Settings false
m1.open();
console.log(m1.isOpen);           // true
console.log(Modal.instanceCount); // 2
// m1.#isOpen; // SyntaxError: private field

// at() — negative indexing
const arr = [10, 20, 30, 40, 50];
console.log(arr.at(0));    // 10
console.log(arr.at(-1));   // 50
console.log(arr.at(-2));   // 40
console.log("hello".at(-1)); // "o"

// Object.hasOwn
const obj = Object.create({ inheritedProp: 1 });
obj.ownProp = 2;
console.log(Object.hasOwn(obj, "ownProp"));      // true
console.log(Object.hasOwn(obj, "inheritedProp")); // false
console.log(obj.inheritedProp);                   // 1

// ============================================
// 4. ES2023: Change Array by Copy
// ============================================
const original = [3, 1, 4, 1, 5];

// toSorted — sort without mutating
const sorted = original.toSorted();
console.log(sorted);    // [1, 1, 3, 4, 5]
console.log(original);  // [3, 1, 4, 1, 5] — unchanged!

// toReversed
const reversed = original.toReversed();
console.log(reversed); // [5, 1, 4, 1, 3]

// toSpliced
const spliced = original.toSpliced(1, 2, 9, 9);
console.log(spliced); // [3, 9, 9, 1, 5]

// with — replace single element
const replaced = original.with(2, 99);
console.log(replaced); // [3, 1, 99, 1, 5]

// ============================================
// 5. Combined practical example
// ============================================
function processUsers(users) {
  return users
    .filter(user => user?.active ?? false)
    .toSorted((a, b) => (a.name ?? "").localeCompare(b.name ?? ""))
    .map(user => ({
      name: user.name ?? "Anonymous",
      email: user.contacts?.email ?? "no email",
      phone: user.contacts?.phone ?? "no phone",
    }));
}

const rawUsers = [
  { active: true, name: "Charlie", contacts: { email: "c@test.com" } },
  { active: true, name: "Alice", contacts: { email: "a@test.com", phone: "123" } },
  { active: false, name: "Bob" },
  { active: true, name: null, contacts: null },
];

console.log(processUsers(rawUsers));

          五、注意事项

          特性 版本 注意
          ?. ES2020 不能用于赋值左侧
          ?? ES2020 优先级低于`\ \ `，混用时加括号
          ??= ES2021 仅null/undefined时赋值
          at(-1) ES2022 等价于arr[arr.length-1]
          toSorted() ES2023 不修改原数组（immutable）
          #private ES2022 必须在class body中声明

          六、避坑指南

          ❌ ??和||混用不加括号

          const x = null || undefined ?? "default"; // SyntaxError

          ✅ 明确加括号

          const x = (null || undefined) ?? "default"; // "default"

          ❌ toSorted/toReversed不可用（旧环境）

          const sorted = arr.toSorted(); // may not exist in older Node/ browsers

          ✅ polyfill或使用slice+sort

          const sorted = [...arr].sort((a, b) => a - b); // universal

                场景 新特性 旧写法对比

                  const city = user?.address?.city ?? "unknown";

                  // Immutable data processing pipeline
const result = users
  ?.filter(u => u.active)

                  十二、总结

                  核心要点 说明
                  ?. 最常用新特性，安全访问深层属性
                  ?? 比`\ \ `更精确的默认值
                  ??= 优雅的默认值赋值
                  at(-1) 反向索引，Python风格
                  toSorted等 不可变操作，React友好
                  #private 真正的类私有字段
