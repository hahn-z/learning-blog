## 一、概念讲解

### JavaScript错误类型

JS内置7种错误类型，理解它们是高效调试的第一步：

| 错误类型 | 触发场景 | 示例 |
|---------|---------|------|
| `SyntaxError` | 代码语法错误 | `const x =` |
| `ReferenceError` | 引用未声明变量 | `console.log(x)` 未声明x |
| `TypeError` | 类型操作错误 | `null.foo` |
| `RangeError` | 值超出范围 | `new Array(-1)` |
| `URIError` | URI编码错误 | `decodeURI('%')` |
| `EvalError` | eval使用错误（现代少见） | - |
| `AggregateError` | 多错误聚合（ES2021） | `Promise.any` 全部失败 |

### 错误处理三件套

```
try { ... }       // 可能出错的代码
catch (e) { ... } // 捕获错误
finally { ... }   // 无论成功失败都执行
```

## 三、完整代码示例

### 3.1 自定义错误体系

```javascript
// Custom error hierarchy for business logic
class AppError extends Error {
  constructor(message, code = 'UNKNOWN', statusCode = 500) {
    super(message);
    this.name = 'AppError';
    this.code = code;
    this.statusCode = statusCode;
    this.timestamp = new Date().toISOString();
    // Maintain proper stack trace in V8
    Error.captureStackTrace?.(this, this.constructor);
  }

  toJSON() {
    return {
      name: this.name,
      message: this.message,
      code: this.code,
      statusCode: this.statusCode,
      timestamp: this.timestamp,
    };
  }
}

class ValidationError extends AppError {
  constructor(message, fields = []) {
    super(message, 'VALIDATION_ERROR', 400);
    this.name = 'ValidationError';
    this.fields = fields;
  }
}

class NotFoundError extends AppError {
  constructor(resource, id) {
    super(`${resource} not found: ${id}`, 'NOT_FOUND', 404);
    this.name = 'NotFoundError';
    this.resource = resource;
    this.id = id;
  }
}

class AuthError extends AppError {
  constructor(message = 'Authentication required') {
    super(message, 'AUTH_ERROR', 401);
    this.name = 'AuthError';
  }
}
```

### 3.2 异步错误处理

```javascript
// Safe async wrapper - converts rejected promises to [error, data]
function asyncHandler(fn) {
  return (...args) => fn(...args)
    .then(data => [null, data])
    .catch(err => [err, null]);
}

// Usage with Promise
const [err, user] = await asyncHandler(fetchUser(userId));
if (err) {
  if (err instanceof NotFoundError) {
    return res.status(404).json(err.toJSON());
  }
  throw err;
}

// Safe async for Express middleware
function asyncMiddleware(fn) {
  return (req, res, next) => {
    Promise.resolve(fn(req, res, next)).catch(next);
  };
}

// Usage in Express
app.get('/users/:id', asyncMiddleware(async (req, res) => {
  const user = await getUser(req.params.id);
  if (!user) throw new NotFoundError('User', req.params.id);
  res.json(user);
}));
```

### 3.3 全局错误捕获

```javascript
// Global error handlers
// 1. Synchronous errors
window.onerror = (message, source, lineno, colno, error) => {
  console.error('[Global Error]', { message, source, lineno, colno });
  reportError(error);
  return false; // false = still show in console
};

// 2. Unhandled promise rejections
window.addEventListener('unhandledrejection', (event) => {
  console.error('[Unhandled Rejection]', event.reason);
  event.preventDefault(); // prevent console warning
  reportError(event.reason);
});

// 3. Resource loading errors (using capture phase)
window.addEventListener('error', (event) => {
  if (event.target !== window) {
    const tag = event.target.tagName;
    const src = event.target.src || event.target.href;
    console.warn(`[Resource Error] ${tag} failed: ${src}`);
  }
}, true); // capture phase catches resource errors

// Error reporting
function reportError(error) {
  // Send to error tracking service
  navigator.sendBeacon?.('/api/errors', JSON.stringify({
    message: error?.message,
    stack: error?.stack,
    url: location.href,
    timestamp: Date.now(),
  }));
}
```

### 3.4 日志工具

```javascript
// Structured logger with levels
const LogLevel = { DEBUG: 0, INFO: 1, WARN: 2, ERROR: 3 };

class Logger {
  constructor(name, level = LogLevel.INFO) {
    this.name = name;
    this.level = level;
  }

  _log(level, levelName, ...args) {
    if (level < this.level) return;
    const timestamp = new Date().toISOString();
    const prefix = `[${timestamp}] [${levelName}] [${this.name}]`;
    const method = level >= LogLevel.ERROR ? 'error' : level >= LogLevel.WARN ? 'warn' : 'log';
    console[method](prefix, ...args);
  }

  debug(...args) { this._log(LogLevel.DEBUG, 'DEBUG', ...args); }
  info(...args)  { this._log(LogLevel.INFO, 'INFO', ...args); }
  warn(...args)  { this._log(LogLevel.WARN, 'WARN', ...args); }
  error(...args) { this._log(LogLevel.ERROR, 'ERROR', ...args); }
}

const log = new Logger('App', LogLevel.DEBUG);
log.info('Server started on port 3000');
log.error('Database connection failed', new Error('ECONNREFUSED'));
```

## 五、注意事项

| 场景 | 注意点 | 建议 |
|------|--------|------|
| `try/catch` | 只能捕获同步错误 | async函数内用 `try/await` |
| `Promise.catch` | 不捕获语法错误 | 开发阶段用 ESLint 检查 |
| `window.onerror` | 无法捕获资源加载错误 | 用 `window.addEventListener('error', fn, true)` |
| `unhandledrejection` | 部分旧浏览器不支持 | 垫片或polyfill |
| `Error.captureStackTrace` | 仅V8引擎支持 | 用可选链 `?.` 调用 |
| `throw` 字符串 | 丢失堆栈信息 | 始终 `throw new Error()` |

## 六、避坑指南

| ❌ 错误做法 | ✅ 正确做法 | 原因 |
|------------|-----------|------|
| `catch(e) {}` 空捕获 | 至少 `console.error` | 静默吞错最难调试 |
| `throw 'something'` | `throw new Error('something')` | 保留堆栈追踪 |
| 只用 `try/catch` 包Promise | 用 `.catch()` 或 `try/await` | Promise错误不冒泡到外层try |
| 异步回调里 throw | 回调里做错误处理 | throw在回调外无法捕获 |
| `finally` 里 `return` | 避免在finally中返回 | 会覆盖try/catch的返回值 |
| 暴露堆栈给用户 | 生产环境隐藏敏感信息 | 安全隐患 |

## 十二、总结

错误处理不是"写完代码再补"的事后工作，而是架构的一部分。建立**自定义错误体系**，使用**asyncHandler模式**统一异步错误，配置**全局错误捕获**兜底，配合**结构化日志**——让Bug无处遁形。
