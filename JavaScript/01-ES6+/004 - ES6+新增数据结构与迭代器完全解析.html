一、概念讲解

1.1 新数据结构

  • Map： 键值对集合，任意类型作键（vs Object只允许字符串/Symbol）
  • Set： 值的唯一集合，自动去重
  • WeakMap/WeakSet： 键必须是对象，且是弱引用（不阻止GC）

    • 1.2 迭代协议

    • 可迭代协议： 对象实现[Symbol.iterator]()方法，返回迭代器
    • 迭代器协议： 对象实现next()方法，返回{ value, done }

      • 1.3 Generator

      function*声明的函数，可通过yield暂停和恢复执行。天然实现迭代器协议。

// 3. WeakMap — weak-referenced keys
// ============================================
const privateData = new WeakMap();

function createSecret(obj, secret) {
  privateData.set(obj, secret);
}
function getSecret(obj) {
  return privateData.get(obj);
}

const myObj = { id: 1 };
createSecret(myObj, "hidden value");
console.log(getSecret(myObj)); // "hidden value"
// When myObj is GC'd, the WeakMap entry is also cleaned up

// ============================================
// 4. Iterators & Iterable protocol
// ============================================
// Custom iterable
class NumberRange {
  constructor(start, end, step = 1) {
    this.start = start;
    this.end = end;
    this.step = step;
  }

  [Symbol.iterator]() {
    let current = this.start;
    const { end, step } = this;
    return {
      next() {
        if (current < end) {
          const value = current;
          current += step;
          return { value, done: false };
        }
        return { value: undefined, done: true };
      },
    };
  }
}

const range = new NumberRange(0, 5, 2);
console.log([...range]); // [0, 2, 4]

for (const num of range) {
  console.log(num); // 0, 2, 4
}

// ============================================
// 5. Generator functions
// ============================================
function* idGenerator() {
  let id = 1;
  while (true) {
    yield id++;
  }
}

const gen = idGenerator();
console.log(gen.next()); // { value: 1, done: false }
console.log(gen.next()); // { value: 2, done: false }
console.log(gen.next()); // { value: 3, done: false }

// Fibonacci generator
function* fibonacci() {
  let [prev, curr] = [0, 1];
  while (true) {
    yield curr;
    [prev, curr] = [curr, prev + curr];
  }
}

// Take first 8 Fibonacci numbers
const fib = fibonacci();
const first8 = Array.from({ length: 8 }, () => fib.next().value);
console.log(first8); // [1, 1, 2, 3, 5, 8, 13, 21]

// Generator with delegation
function* concatIterables(...iterables) {
  for (const iter of iterables) {
    yield* iter;
  }
}

const combined = concatIterables([1, 2], "hi", new Set([3, 4]));
console.log([...combined]); // [1, 2, 'h', 'i', 3, 4]

// ============================================
// 6. Async generators (ES2018)
// ============================================
async function* asyncCounter(max) {
  for (let i = 1; i <= max; i++) {
    await new Promise(r => setTimeout(r, 10));
    yield i;
  }
}

(async () => {
  const results = [];
  for await (const num of asyncCounter(3)) {
    results.push(num);
  }
  console.log("Async gen:", results); // [1, 2, 3]
})();

      五、注意事项

      结构 特点 适用场景 注意
      Map 任意键类型 频繁增删的键值对 比Object更安全
      Set 自动去重 去重、成员检测 NaN === NaN在Set中
      WeakMap 弱引用键 私有数据、缓存 不可遍历
      WeakSet 弱引用值 标记对象 不可遍历
      Generator 惰性求值 无限序列、异步流 需手动调用next()

      六、避坑指南

      ❌ 用Object做频繁增删的字典

      const dict = {};
dict[key] = value;
delete dict[key]; // slow, triggers V8 deoptimization

      ✅ 用Map

      const dict = new Map();
dict.set(key, value);
dict.delete(key); // optimized for frequent changes

      ❌ Generator只遍历一次

      function* gen() { yield 1; yield 2; }
const g = gen();
console.log([...g]); // [1, 2]
console.log([...g]); // [] — exhausted!

      ✅ 创建新的Generator实例

      const g1 = gen();
const g2 = gen();
console.log([...g1]); // [1, 2]
console.log([...g2]); // [1, 2]

              十二、总结

              核心要点 说明
              Map vs Object Map键类型灵活、有序、大小可查
