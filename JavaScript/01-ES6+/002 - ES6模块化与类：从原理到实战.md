# ES6模块化与类：从原理到实战

> **难度标注：** 🟡 中级 | **前置知识：** 函数、对象、基本模块概念
> **阅读时间：** 约18分钟 | **代码量：** 约150行

---

## 一、概念讲解

### 1.1 ES6模块系统

- **静态结构：** import/export必须在顶层，编译时确定依赖
- **严格模式：** 模块自动启用`"use strict"`
- **单例：** 模块只执行一次，多次import共享同一实例
- **命名导出 vs 默认导出**

### 1.2 ES6类

- **语法糖：** 本质仍是原型继承
- `constructor`：初始化方法
- `extends`/`super`：继承链
- `static`：静态方法
- `#private`：私有字段（ES2022）

## 三、完整代码示例

```javascript
// ============================================
// === FILE: math-utils.js (simulated) ===
// ============================================

// Named exports
const PI = 3.14159;
function square(x) { return x * x; }
function cube(x) { return x * x * x; }

// Simulating export
const mathUtils = { PI, square, cube };

// Default export
const Calculator = {
  add: (a, b) => a + b,
  subtract: (a, b) => a - b,
};

// ============================================
// === FILE: shapes.js (class hierarchy) ===
// ============================================

class Shape {
  constructor(name) {
    this.name = name;
  }

  // Instance method
  describe() {
    return `${this.name}: area = ${this.area().toFixed(2)}`;
  }

  // Static method
  static create(type, ...args) {
    switch (type) {
      case "circle": return new Circle(...args);
      case "rectangle": return new Rectangle(...args);
      default: throw new Error(`Unknown shape: ${type}`);
    }
  }
}

class Circle extends Shape {
  #radius; // private field (ES2022)

  constructor(radius) {
    super("Circle");
    this.#radius = radius;
  }

  get radius() { return this.#radius; }
  set radius(r) {
    if (r <= 0) throw new Error("Radius must be positive");
    this.#radius = r;
  }

  area() {
    return mathUtils.PI * this.#radius ** 2;
  }

  // Static method
  static fromDiameter(d) {
    return new Circle(d / 2);
  }
}

class Rectangle extends Shape {
  #width;
  #height;

  constructor(width, height) {
    super("Rectangle");
    this.#width = width;
    this.#height = height;
  }

  area() {
    return this.#width * this.#height;
  }

  get perimeter() {
    return 2 * (this.#width + this.#height);
  }
}

// ============================================
// === FILE: main.js (usage) ===
// ============================================

// Using the classes
const circle = new Circle(5);
console.log(circle.describe());         // Circle: area = 78.54
console.log(circle.radius);             // 5 (getter)

circle.radius = 10;
console.log(circle.describe());         // Circle: area = 314.16

const rect = new Rectangle(4, 6);
console.log(rect.describe());           // Rectangle: area = 24.00
console.log(rect.perimeter);            // 20

// Static factory method
const shape = Shape.create("circle", 3);
console.log(shape.describe());          // Circle: area = 28.27

// Static method on subclass
const c2 = Circle.fromDiameter(10);
console.log(c2.describe());             // Circle: area = 78.54

// Private field is truly private
// console.log(circle.#radius);        // SyntaxError

// instanceof checks
console.log(circle instanceof Circle);  // true
console.log(circle instanceof Shape);   // true
console.log(rect instanceof Circle);    // false

// ============================================
// === Module patterns (simulated) ===
// ============================================

// Re-export pattern
const mathModule = { ...mathUtils, Calculator };

// Namespace pattern
const Shapes = { Shape, Circle, Rectangle };

console.log("---");
console.log("mathModule keys:", Object.keys(mathModule));
console.log("Shapes keys:", Object.keys(Shapes));
```

## 五、注意事项

| 特性 | 模块 | 类 | 说明 |
|------|------|-----|------|
| 严格模式 | 自动启用 | 不自动 | 模块内总是strict |
| 提升 | 不提升（import） | 不提升（class） | 必须先定义后使用 |
| this | undefined（顶层） | 实例（方法中） | 模块顶层this≠window |
| 重导出 | `export { x } from` | N/A | 模块特有 |
| 私有 | 模块作用域天然私有 | `#field` | 不同机制 |

## 六、避坑指南

### ❌ 循环依赖
```javascript
// a.js
import { b } from './b.js';
export const a = b + 1;

// b.js
import { a } from './a.js';
export const b = a + 1;
// → ReferenceError or undefined values
```
### ✅ 重构避免循环
```javascript
// shared.js - extract shared types/constants
export const BASE = 0;

// a.js
import { BASE } from './shared.js';
export const a = BASE + 1;

// b.js
import { BASE } from './shared.js';
export const b = BASE + 2;
```

### ❌ 类中直接用箭头方法覆盖
```javascript
class Foo {
  bar = () => { this.x = 1; }; // arrow field: on instance, not prototype
}
// Each instance gets its own copy → more memory
```
### ✅ 普通方法 + 构造函数bind（按需）
```javascript
class Foo {
  bar() { this.x = 1; } // on prototype, shared
  constructor() {
    this.bar = this.bar.bind(this); // bind once if needed
  }
}
```

## 十一、代码演进

### v1: ES5原型继承
```javascript
function Shape(name) { this.name = name; }
Shape.prototype.describe = function() {
  return this.name;
};
function Circle(radius) {
  Shape.call(this, "Circle");
  this.radius = radius;
}
Circle.prototype = Object.create(Shape.prototype);
Circle.prototype.constructor = Circle;
Circle.prototype.area = function() {
  return Math.PI * this.radius * this.radius;
};
```

### v2: ES6 class
```javascript
class Shape {
  constructor(name) { this.name = name; }
  describe() { return this.name; }
}
class Circle extends Shape {
  constructor(radius) {
    super("Circle");
    this.radius = radius;
  }
  area() { return Math.PI * this.radius ** 2; }
}
```

### v3: ES2022私有字段
```javascript
class Circle extends Shape {
  #radius;
  constructor(radius) {
    super("Circle");
    this.#radius = radius;
  }
  get radius() { return this.#radius; }
  area() { return Math.PI * this.#radius ** 2; }
}
```

## 十二、总结

| 核心要点 | 说明 |
|----------|------|
| **模块静态化** | 编译时确定依赖，利于tree-shaking |
| **命名导出优先** | 更好的IDE支持和重构 |
| **class是语法糖** | 理解原型链才能用好类 |
| **私有字段#** | ES2022真私有，不再需要闭包 |
| **避免循环依赖** | 模块设计的基本原则 |
