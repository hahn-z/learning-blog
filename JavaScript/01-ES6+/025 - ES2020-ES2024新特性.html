三、完整代码示例

3.1 ES2020: Optional Chaining & Nullish Coalescing

// Optional Chaining (?.) - safe property access
const user = { address: { street: 'Main St' } };

// Before ES2020
const zip1 = user && user.address && user.address.zip;

// After ES2020
const zip2 = user?.address?.zip; // undefined, no error
const name = user?.profile?.name?.length ?? 0; // 0

// Works with methods and arrays
const result = data?.items?.[0]?.getName?.() ?? 'default';

// Nullish Coalescing (??) - only null/undefined trigger fallback
const count = 0 ?? 10;   // 0 (not 10!)
const name = '' ?? 'N/A'; // '' (not 'N/A!')
const val = null ?? 10;   // 10
const val2 = undefined ?? 10; // 10

// Practical: config with defaults
function createConfig(options) {
  return {
    port: options?.port ?? 3000,
    host: options?.host ?? 'localhost',
    debug: options?.debug ?? false,
    timeout: options?.timeout ?? 5000,
  };
}

3.2 ES2020: BigInt & Dynamic Import

// BigInt: arbitrary precision integers
const big = 9007199254740993n; // n suffix
const big2 = BigInt('9007199254740993');
console.log(big === big2); // true
console.log(big + 1n);     // 9007199254740994n

// BigInt cannot mix with Number
// big + 1; // TypeError! Must convert:
Number(big); // loses precision for large values

// Dynamic import: load modules on demand
async function loadEditor() {
  const { Editor } = await import('./editor.js');
  return new Editor(document.getElementById('editor'));
}

// Conditional loading
if (supportsWebGL()) {
  const { render3D } = await import('./webgl-renderer.js');
  render3D(scene);
} else {
  const { render2D } = await import('./canvas-renderer.js');
  render2D(scene);
}

3.3 ES2021: Logical Assignment & Promise.any

// Logical Assignment Operators
let config = { debug: false, retries: 3 };

config.debug ||= true;       // ||= : assign if falsy
config.retries &&= 0;        // &&= : assign if truthy
config.name ??= 'default';   // ??= : assign if nullish

// Promise.any: resolve with first fulfilled promise
async function fetchFastest(mirrors) {
  try {
    const response = await Promise.any(
      mirrors.map(url => fetch(url).then(r => {
        if (!r.ok) throw new Error(`${r.status}`);
        return r;
      }))
    );
    return response;
  } catch (aggregateError) {
    // All promises rejected
    console.error('All mirrors failed:');
    aggregateError.errors.forEach((e, i) => console.error(`  Mirror ${i}:`, e.message));
    throw aggregateError;
  }
}

// Numeric Separators: improve readability
const budget = 1_000_000;     // 1000000
const bytes = 0xFF_FF_FF_FF;  // 4294967295
const binary = 0b1010_0001;   // 161

3.4 ES2022: Top-level await, Object.hasOwn, Error Cause

// Top-level await (in ES modules)
// config.mjs
const response = await fetch('/api/config');
export const config = await response.json();

// Object.hasOwn: safer than hasOwnProperty
const obj = Object.create(null); // no prototype
obj.name = 'test';
// obj.hasOwnProperty('name'); // TypeError! No prototype
Object.hasOwn(obj, 'name'); // true - works everywhere

// Error Cause: chain errors with context
try {
  await connectDatabase();
} catch (originalError) {
  throw new Error('Failed to initialize database', {
    cause: originalError, // Link to original error
  });
}

// Access the chain
// err.message → 'Failed to initialize database'
// err.cause.message → 'ECONNREFUSED'
// err.cause.cause?.message → deeper errors...

// Array.prototype.at(): negative indexing
const arr = [10, 20, 30, 40, 50];
arr.at(0);    // 10 (first)
arr.at(-1);   // 50 (last)
arr.at(-2);   // 40 (second to last)

// Class Fields (public & private)
class Timer {
  #startTime = null;       // Private field
  #running = false;        // Private field
  interval = 1000;         // Public field

  get isRunning() { return this.#running; }

  start() {
    this.#startTime = Date.now();
    this.#running = true;
  }

  elapsed() {
    if (!this.#running) return 0;
    return Date.now() - this.#startTime;
  }
}

3.5 ES2023-2024: Array Grouping & Promise.withResolvers

// ES2023: Array.prototype.group()
const products = [
  { name: 'Apple', category: 'fruit' },
  { name: 'Carrot', category: 'vegetable' },
  { name: 'Banana', category: 'fruit' },
];

const grouped = Object.groupBy(products, p => p.category);
// { fruit: [{name:'Apple',...}, {name:'Banana',...}], vegetable: [{name:'Carrot',...}] }

// ES2023: Change Array by Copy (non-mutating methods)
const original = [3, 1, 4, 1, 5];
const sorted = original.toSorted();    // [1, 1, 3, 4, 5] - original unchanged
const reversed = original.toReversed(); // [5, 1, 4, 1, 3]
const spliced = original.toSpliced(1, 2, 9); // [3, 9, 1, 5]
const withNew = original.with(0, 99);  // [99, 1, 4, 1, 5]

// ES2024: Promise.withResolvers
function createDeferred() {
  const { promise, resolve, reject } = Promise.withResolvers();
  return { promise, resolve, reject };
}

// Usage: resolve/reject from outside the executor
const { promise, resolve } = Promise.withResolvers();
setTimeout(() => resolve('done!'), 1000);
const result = await promise; // 'done!' after 1s

// ES2024: Well-Formed Unicode Strings
'\uD800'.isWellFormed(); // false (lone surrogate)
'hello'.isWellFormed();  // true
'\uD800'.toWellFormed(); // '\uFFFD' (replacement char)

四、执行预览

ES2020-ES2024 Features Demo
============================

ES2020:
  user?.address?.zip → undefined (no crash)
  0 ?? 10 → 0 (nullish coalescing preserves 0)
  1_000_000n + 1n → 1000001n (BigInt)

ES2021:
  config.debug ||= true → true (was false)
  Promise.any([fetch(url1), fetch(url2)]) → first success

ES2022:
  arr.at(-1) → last element
  Object.hasOwn(obj, 'key') → true
  new Error('msg', { cause: original })

ES2023-2024:
  Object.groupBy(items, fn) → grouped object
  arr.toSorted() → sorted copy
  Promise.withResolvers() → { promise, resolve, reject }

五、注意事项

特性 注意点 建议
Optional Chaining 不能用于赋值左侧 obj?.foo = 1 ❌
Nullish Coalescing 运算符优先级低于 ` ` 必要时加括号
BigInt 不能与Number混算 显式转换 BigInt(n) 或 Number(b)
Top-level await 仅在ES模块中可用
