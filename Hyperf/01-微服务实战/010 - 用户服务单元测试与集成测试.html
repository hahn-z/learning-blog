用户服务单元测试与集成测试

难度： ⭐⭐⭐⭐
前置知识： CRUD API（009）、PHPUnit 基础、测试驱动开发（TDD）
预估用时： 80 分钟

一、概念讲解

一句话定义： 单元测试是对代码最小单元（函数、方法）进行隔离测试，集成测试是测试多个组件协同工作的整体行为。

现实类比： 测试就像汽车质检——单元测试是检查每个零件（螺丝、齿轮）是否合格，集成测试是检查整辆车能否正常行驶、刹车是否灵敏。只有两个层面都通过，才能放心交付给用户。

技术场景： 用户服务需要测试：Service 层的业务逻辑（单元测试）、API 接口的端到端行为（集成测试）。测试覆盖率 >80% 才能上线，确保核心功能（注册、登录、JWT）不出现回归 Bug。

3.2 测试基类（集成测试）

<?php
// ✅ tests/TestCase.php

declare(strict_types=1);

namespace Tests;

use Hyperf\Testing\TestCase;
use Hyperf\Context\ApplicationContext;
use Hyperf\DbConnection\Db;
use PHPUnit\Framework\TestCase as BaseTestCase;

abstract class TestCase extends BaseTestCase
{
    protected function setUp(): void
    {
        parent::setUp();
        
        // 🟢 Run migrations for each test
        $this->runMigrations();
        
        // 🟢 Clear database
        $this->refreshDatabase();
    }

    protected function tearDown(): void
    {
        parent::tearDown();
        
        // 🟢 Close database connection
        Db::connection()->disconnect();
    }

    // ✅ Run migrations
    protected function runMigrations(): void
    {
        $this->artisan('migrate:fresh');
    }

    // ✅ Refresh database
    protected function refreshDatabase(): void
    {
        Db::table('users')->truncate();
        Db::table('user_profiles')->truncate();
        Db::table('user_tokens')->truncate();
    }

    // ✅ Create test user
    protected function createUser(array $data = []): array
    {
        $user = \App\Model\User::create(array_merge([
            'email' => 'test@example.com',
            'username' => 'testuser',
            'password' => 'Test1234',
            'status' => 'active',
        ], $data));

        return (array) $user->toArray();
    }

    // ✅ Generate JWT token for testing
    protected function generateToken(int $userId = 1): string
    {
        $jwt = ApplicationContext::getContainer()->get(\App\Service\JwtService::class);
        return $jwt->generateAccessToken($userId);
    }

    // ✅ Create authenticated request
    protected function withAuth(?string $token = null): array
    {
        return [
            'Authorization' => 'Bearer ' . ($token ?? $this->generateToken()),
        ];
    }
}

// ✅ tests/Unit/Service/UserServiceTest.php

declare(strict_types=1);

namespace Tests\Unit\Service;

use PHPUnit\Framework\TestCase;
use App\Service\UserService;
use App\Validator\RegisterValidator;
use PHPUnit\Framework\MockObject\MockObject;
use Psr\Log\LoggerInterface;

class UserServiceTest extends TestCase
{
    private UserService $userService;
    private MockObject $logger;

    protected function setUp(): void
    {
        parent::setUp();

        // 🟢 Mock logger
        $this->logger = $this->createMock(LoggerInterface::class);
        $this->userService = new UserService();
        $this->userService->setLogger($this->logger);
    }

    // ✅ Test registration validation
    public function test_register_validation_fails_with_weak_password(): void
    {
        $data = [
            'email' => 'test@example.com',
            'username' => 'testuser',
            'password' => '123',  // ❌ Weak password
            'password_confirmation' => '123',
        ];

        $errors = RegisterValidator::validate($data);

        $this->assertNotEmpty($errors);
        $this->assertContains('Password must be at least 8 characters', $errors);
    }

    // ✅ Test registration validation passes
    public function test_register_validation_succeeds_with_valid_data(): void
    {
        $data = [
            'email' => 'test@example.com',
            'username' => 'testuser',
            'password' => 'Test1234',
            'password_confirmation' => 'Test1234',
        ];

        $errors = RegisterValidator::validate($data);

        $this->assertEmpty($errors);
    }

    // ✅ Test password strength calculation
    public function test_password_strength_calculation(): void
    {
        $weak = \App\Validator\PasswordValidator::getStrength('123');
        $medium = \App\Validator\PasswordValidator::getStrength('Test123');
        $strong = \App\Validator\PasswordValidator::getStrength('Test@1234');

        $this->assertLessThan(50, $weak);
        $this->assertGreaterThanOrEqual(50, $medium);
        $this->assertGreaterThan(80, $strong);
    }

    // ✅ Test email format validation
    public function test_email_validation(): void
    {
        $validEmail = 'test@example.com';
        $invalidEmail = 'invalid-email';

        $this->assertTrue(filter_var($validEmail, FILTER_VALIDATE_EMAIL) !== false);
        $this->assertFalse(filter_var($invalidEmail, FILTER_VALIDATE_EMAIL));
    }
}

3.4 AuthService 单元测试

<?php
// ✅ tests/Unit/Service/AuthServiceTest.php

declare(strict_types=1);

namespace Tests\Unit\Service;

use PHPUnit\Framework\TestCase;
use App\Service\AuthService;
use App\Service\JwtService;
use App\Model\User;
use PHPUnit\Framework\MockObject\MockObject;

class AuthServiceTest extends TestCase
{
    private AuthService $authService;
    private MockObject $jwtService;

    protected function setUp(): void
    {
        parent::setUp();

        // 🟢 Mock JWT service
        $this->jwtService = $this->createMock(JwtService::class);
        $this->authService = new AuthService();
        $this->authService->setJwtService($this->jwtService);
    }

    // ✅ Test login with valid credentials
    public function test_login_with_valid_credentials(): void
    {
        // Setup mock
        $this->jwtService
            ->expects($this->exactly(2))  // access + refresh
            ->method('generateAccessToken')
            ->willReturn('mock-access-token');

        $this->jwtService
            ->expects($this->once())
            ->method('generateRefreshToken')
            ->willReturn('mock-refresh-token');

        // Create test user
        $user = User::create([
            'email' => 'test@example.com',
            'username' => 'testuser',
            'password' => password_hash('Test1234', PASSWORD_BCRYPT),
            'status' => 'active',
        ]);

        // Test login
        $result = $this->authService->login('test@example.com', 'Test1234');

        $this->assertEquals(0, $result['code']);
        $this->assertArrayHasKey('access_token', $result['data']);
        $this->assertArrayHasKey('refresh_token', $result['data']);
    }

    // ✅ Test login with invalid credentials
    public function test_login_with_invalid_credentials(): void
    {
        $user = User::create([
            'email' => 'test@example.com',
            'username' => 'testuser',
            'password' => password_hash('Test1234', PASSWORD_BCRYPT),
            'status' => 'active',
        ]);

        $result = $this->authService->login('test@example.com', 'wrongpassword');

        $this->assertEquals(401, $result['code']);
        $this->assertEquals('Invalid credentials', $result['message']);
    }

    // ✅ Test login with disabled account
    public function test_login_with_disabled_account(): void
    {
        $user = User::create([
            'email' => 'test@example.com',
            'username' => 'testuser',
            'password' => password_hash('Test1234', PASSWORD_BCRYPT),
            'status' => 'banned',
        ]);

        $result = $this->authService->login('test@example.com', 'Test1234');

        $this->assertEquals(403, $result['code']);
        $this->assertEquals('Account is disabled', $result['message']);
    }
}

3.5 API 集成测试

<?php
// ✅ tests/Feature/UserApiTest.php

declare(strict_types=1);

namespace Tests\Feature;

use Tests\TestCase;
use App\Model\User;

class UserApiTest extends TestCase
{
    // ✅ Test user registration
    public function test_user_registration(): void
    {
        $response = $this->post('/auth/register', [
            'email' => 'new@example.com',
            'username' => 'newuser',
            'password' => 'Test1234',
            'password_confirmation' => 'Test1234',
        ]);

        $response->assertStatus(201);
        $response->assertJson([
            'code' => 0,
            'message' => 'Registration successful',
        ]);

        // Verify user was created
        $this->assertDatabaseHas('users', [
            'email' => 'new@example.com',
            'username' => 'newuser',
        ]);
    }

    // ✅ Test user login
    public function test_user_login(): void
    {
        // Create test user
        $user = $this->createUser([
            'email' => 'test@example.com',
            'password' => password_hash('Test1234', PASSWORD_BCRYPT),
        ]);

        $response = $this->post('/auth/login', [
            'account' => 'test@example.com',
            'password' => 'Test1234',
        ]);

        $response->assertStatus(200);
        $response->assertJson([
            'code' => 0,
            'message' => 'Login successful',
        ]);

        $data = $response->json('data');
        $this->assertArrayHasKey('access_token', $data);
        $this->assertArrayHasKey('refresh_token', $data);
    }

    // ✅ Test get user profile (protected)
    public function test_get_user_profile(): void
    {
        $user = $this->createUser();
        $token = $this->generateToken($user['id']);

        $response = $this->get('/auth/me', $this->withAuth($token));

        $response->assertStatus(200);
        $response->assertJson([
            'code' => 0,
            'data' => [
                'id' => $user['id'],
                'username' => $user['username'],
            ],
        ]);
    }

    // ✅ Test unauthorized access
    public function test_unauthorized_access(): void
    {
        $response = $this->get('/auth/me');

        $response->assertStatus(401);
        $response->assertJson([
            'code' => 401,
        ]);
    }

    // ✅ Test user list (paginated)
    public function test_user_list(): void
    {
        // Create 25 test users
        for ($i = 1; $i <= 25; $i++) {
            $this->createUser([
                'email' => "user{$i}@example.com",
                'username' => "user{$i}",
            ]);
        }

        $token = $this->generateToken(1);

        $response = $this->get('/users?page=1&size=10', $this->withAuth($token));

        $response->assertStatus(200);
        $data = $response->json();
        $this->assertCount(10, $data['data']);
        $this->assertEquals(25, $data['pagination']['total']);
    }

    // ✅ Test update user profile
    public function test_update_user_profile(): void
    {
        $user = $this->createUser();
        $token = $this->generateToken($user['id']);

        $response = $this->put("/users/{$user['id']}", [
            'nickname' => 'Updated Name',
            'bio' => 'Updated bio',
        ], $this->withAuth($token));

        $response->assertStatus(200);

        $this->assertDatabaseHas('user_profiles', [
            'user_id' => $user['id'],
            'nickname' => 'Updated Name',
            'bio' => 'Updated bio',
        ]);
    }

    // ✅ Test delete user (soft delete)
    public function test_delete_user(): void
    {
        $user = $this->createUser();
        $token = $this->generateToken($user['id']);

        $response = $this->delete("/users/{$user['id']}", [], $this->withAuth($token));

        $response->assertStatus(200);
        $response->assertJson([
            'code' => 0,
            'message' => 'User deleted',
        ]);

        // Verify soft delete
        $this->assertSoftDeleted('users', ['id' => $user['id']]);
    }

    // ✅ Test token refresh
    public function test_token_refresh(): void
    {
        $user = $this->createUser();

        // Login to get refresh token
        $loginResponse = $this->post('/auth/login', [
            'account' => $user['email'],
            'password' => 'Test1234',
        ]);

        $refreshToken = $loginResponse->json('data.refresh_token');

        // Refresh token
        $response = $this->post('/auth/refresh', [
            'refresh_token' => $refreshToken,
        ]);

        $response->assertStatus(200);
        $response->assertJson([
            'code' => 0,
            'message' => 'Token refreshed',
        ]);

        $this->assertArrayHasKey('access_token', $response->json('data'));
    }

    // ✅ Test logout
    public function test_logout(): void
    {
        $user = $this->createUser();
        $token = $this->generateToken($user['id']);

        $response = $this->post('/auth/logout', [], $this->withAuth($token));

        $response->assertStatus(200);
        $response->assertJson([
            'code' => 0,
            'message' => 'Logout successful',
        ]);

        // Verify tokens are revoked
        $this->assertDatabaseHas('user_tokens', [
            'user_id' => $user['id'],
            'revoked' => true,
        ]);
    }
}

五、注意事项

约束 说明 违反后果
测试覆盖率 >80% 核心功能必须覆盖 上线后出现回归 Bug
每个测试独立 不依赖其他测试运行顺序 测试结果不稳定
使用内存数据库 SQLite :memory: 测试慢，数据库污染
Mock 外部依赖 不调用真实 API/邮件服务 测试不稳定，慢
CI 自动运行测试 每次提交自动运行测试 代码合并引入 Bug

六、避坑指南

典型错误 现象 正确做法
❌ 测试依赖顺序 单独运行通过，一起运行失败 setUp() 中重置数据库
❌ 不清理测试数据 测试数据积累污染环境 tearDown() 或事务回滚
❌ Mock 不正确 Mock 方法未调用 expects($this->once()) 验证调用
❌ 测试环境用真实数据库 测试慢，可能破坏生产数据 用 SQLite 内存数据库
❌ 测试用例描述模糊 test1()、test2() test_user_login_with_valid_credentials()

<?php
// 🟢 Comprehensive test suite with integration tests
class UserApiTest extends TestCase {
    public function test_user_registration_flow() {
        // Step 1: Register user
        $registerResponse = $this->post('/auth/register', [
            'email' => 'test@example.com',
            'username' => 'testuser',
            'password' => 'Test1234',
            'password_confirmation' => 'Test1234',
        ]);
        $registerResponse->assertStatus(201);
        $userId = $registerResponse->json('data.user_id');

        // Step 2: Login
        $loginResponse = $this->post('/auth/login', [
            'account' => 'test@example.com',
            'password' => 'Test1234',
        ]);
        $loginResponse->assertStatus(200);
        $token = $loginResponse->json('data.access_token');

        // Step 3: Access protected endpoint
        $profileResponse = $this->get('/auth/me', [
            'Authorization' => 'Bearer ' . $token,
        ]);
        $profileResponse->assertStatus(200);
        $profileResponse->assertJson([
            'data' => [
                'id' => $userId,
                'username' => 'testuser',
            ],
        ]);

        // Step 4: Logout
        $logoutResponse = $this->post('/auth/logout', [], [
            'Authorization' => 'Bearer ' . $token,
        ]);
        $logoutResponse->assertStatus(200);

        // Step 5: Verify token is revoked
        $this->assertDatabaseHas('user_tokens', [
            'user_id' => $userId,
            'revoked' => true,
        ]);
    }
}
