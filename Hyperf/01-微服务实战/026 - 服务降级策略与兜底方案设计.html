服务降级策略与兜底方案设计

难度标注

⭐⭐⭐（中级）

前置知识： 服务熔断概念、Hyperf 3.x、缓存基础

预估用时： 45-50 分钟

概念讲解

服务降级（Service Degradation） 是一种有策略地牺牲非核心功能或降低服务质量，以保障核心功能可用的容灾手段。与熔断的"断开"不同，降级更像是"换挡"——从高质量服务切换到低质量但可用的替代方案。

现实类比： 高铁站突然停电，售票系统不可用。车站立即启动降级方案：停止网上订票（非核心），开放人工窗口用纸质票（降级服务），确保旅客仍能上车。等电力恢复，再切回正常模式。降级不是不做，而是用更简陋的方式做。

技术场景： 用户服务中，获取用户详细信息需要调用多个下游服务（用户资料、订单统计、积分信息、优惠券）。当积分服务超时时，不是让整个请求失败，而是返回用户基本信息 + "积分数据暂时不可用"的提示，核心的用户查询仍然正常。

1. 降级策略定义

<?php
// app/Enum/DegradationLevel.php

declare(strict_types=1);

namespace App\Enum;

// ✅ Degradation levels from normal to extreme
enum DegradationLevel: string
{
    case NORMAL = 'normal';           // Full service
    case DEGRADED = 'degraded';       // Non-core features degraded
    case MINIMAL = 'minimal';         // Only core features
    case EMERGENCY = 'emergency';     // Read-only, cached data only
}

2. 降级上下文管理

<?php
// app/Service/DegradationManager.php

declare(strict_types=1);

namespace App\Service;

use App\Enum\DegradationLevel;
use Hyperf\Context\Context;
use Hyperf\Redis\Redis;

class DegradationManager
{
    // ✅ Redis key for global degradation level
    private const LEVEL_KEY = 'degradation:level';
    
    // ✅ Feature availability per degradation level
    private const FEATURE_MAP = [
        'user_profile' => [DegradationLevel::NORMAL, DegradationLevel::DEGRADED, DegradationLevel::MINIMAL, DegradationLevel::EMERGENCY],
        'order_stats' => [DegradationLevel::NORMAL, DegradationLevel::DEGRADED, DegradationLevel::MINIMAL],
        'points_query' => [DegradationLevel::NORMAL, DegradationLevel::DEGRADED],
        'recommendations' => [DegradationLevel::NORMAL],
        'activity_feed' => [DegradationLevel::NORMAL],
    ];

    public function __construct(
        private Redis $redis
    ) {}

    /**
     * ✅ Get current degradation level from Redis (config center)
     */
    public function getCurrentLevel(): DegradationLevel
    {
        $level = Context::get('degradation.level');
        if ($level !== null) {
            return $level;
        }

        // 🟢 Default to NORMAL, can be overridden via Redis
        $stored = $this->redis->get(self::LEVEL_KEY);
        $level = $stored ? DegradationLevel::from($stored) : DegradationLevel::NORMAL;
        Context::set('degradation.level', $level);
        return $level;
    }

    /**
     * ✅ Check if a feature is available at current degradation level
     */
    public function isFeatureAvailable(string $feature): bool
    {
        $level = $this->getCurrentLevel();
        $allowedLevels = self::FEATURE_MAP[$feature] ?? [];
        return in_array($level, $allowedLevels, true);
    }

    /**
     * ✅ Set degradation level (admin operation)
     */
    public function setLevel(DegradationLevel $level): void
    {
        $this->redis->set(self::LEVEL_KEY, $level->value);
        Context::set('degradation.level', $level);
        
        logger()->warning('Degradation level changed', [
            'new_level' => $level->value,
            'operator' => Context::get('auth.user_id', 'system'),
        ]);
    }
}

3. 用户服务聚合查询（带降级）

<?php
// app/Service/UserDetailService.php

declare(strict_types=1);

namespace App\Service;

use App\Enum\DegradationLevel;
use App\Model\User;
use Hyperf\Di\Annotation\Inject;

class UserDetailService
{
    #[Inject]
    private DegradationManager $degradation;

    #[Inject]
    private OrderClientService $orderClient;

    #[Inject]
    private CacheService $cache;

    /**
     * ✅ Get user details with degradation support
     */
    public function getUserDetail(int $userId): array
    {
        $level = $this->degradation->getCurrentLevel();

        // ✅ Core: user profile is always available
        $result = [
            'user' => $this->getUserProfile($userId),
            'degraded' => $level !== DegradationLevel::NORMAL,
            'degradation_level' => $level->value,
        ];

        // 🟡 Conditionally add features based on degradation level
        $result['order_stats'] = $this->safeCall(
            fn() => $this->orderClient->getOrderStats($userId),
            fn() => $this->cache->getCachedOrderStats($userId),
            'order_stats'
        );

        $result['points'] = $this->safeCall(
            fn() => $this->degradation->isFeatureAvailable('points_query')
                ? $this->orderClient->getPoints($userId)
                : null,
            fn() => ['balance' => 0, 'cached' => true],
            'points'
        );

        $result['recommendations'] = $this->safeCall(
            fn() => $this->degradation->isFeatureAvailable('recommendations')
                ? $this->orderClient->getRecommendations($userId)
                : null,
            fn() => [],
            'recommendations'
        );

        return $result;
    }

    private function getUserProfile(int $userId): ?array
    {
        $user = User::find($userId);
        return $user?->toArray();
    }

    /**
     * ✅ Safe call wrapper: try primary → fallback → null
     */
    private function safeCall(callable $primary, callable $fallback, string $feature): mixed
    {
        try {
            $result = $primary();
            if ($result !== null) {
                return $result;
            }
        } catch (\Throwable $e) {
            logger()->warning("Feature call failed, using fallback", [
                'feature' => $feature,
                'error' => $e->getMessage(),
            ]);
        }

        // 🟢 Try fallback (cache / default)
        try {
            return $fallback();
        } catch (\Throwable $e) {
            return null;
        }
    }
}

4. 缓存兜底服务

<?php
// app/Service/CacheService.php

declare(strict_types=1);

namespace App\Service;

use Hyperf\Redis\Redis;

class CacheService
{
    // ✅ Cache TTL: 5 minutes for degraded data
    private const CACHE_TTL = 300;

    public function __construct(
        private Redis $redis
    ) {}

    /**
     * ✅ Get cached order stats as fallback
     */
    public function getCachedOrderStats(int $userId): ?array
    {
        $cached = $this->redis->get("order_stats:{$userId}");
        if ($cached) {
            $data = json_decode($cached, true);
            $data['from_cache'] = true;
            $data['cache_time'] = date('Y-m-d H:i:s');
            return $data;
        }
        
        // 🟢 No cache available, return safe defaults
        return [
            'total_orders' => 0,
            'total_amount' => 0,
            'no_data' => true,
        ];
    }

    /**
     * ✅ Warm cache with fresh data (called when service is healthy)
     */
    public function warmOrderStatsCache(int $userId, array $data): void
    {
        $this->redis->setex(
            "order_stats:{$userId}",
            self::CACHE_TTL,
            json_encode($data)
        );
    }
}

5. 降级管理接口

<?php
// app/Controller/DegradationController.php

declare(strict_types=1);

namespace App\Controller;

use App\Enum\DegradationLevel;
use App\Service\DegradationManager;
use Hyperf\Di\Annotation\Inject;
use Hyperf\HttpServer\Annotation\Controller;
use Hyperf\HttpServer\Annotation\GetMapping;
use Hyperf\HttpServer\Annotation\PutMapping;

#[Controller(prefix: '/api/degradation')]
class DegradationController
{
    #[Inject]
    private DegradationManager $manager;

    // ✅ Get current degradation status
    #[GetMapping(path: '')]
    public function status(): array
    {
        return [
            'level' => $this->manager->getCurrentLevel()->value,
            'features' => [
                'user_profile' => $this->manager->isFeatureAvailable('user_profile'),
                'order_stats' => $this->manager->isFeatureAvailable('order_stats'),
                'points_query' => $this->manager->isFeatureAvailable('points_query'),
                'recommendations' => $this->manager->isFeatureAvailable('recommendations'),
            ],
        ];
    }

    // ✅ Update degradation level
    #[PutMapping(path: '/{level}')]
    public function setLevel(string $level): array
    {
        $degLevel = DegradationLevel::tryFrom($level);
        if ($degLevel === null) {
            return ['error' => 'Invalid level', 'valid' => array_column(DegradationLevel::cases(), 'value')];
        }
        
        $this->manager->setLevel($degLevel);
        return ['success' => true, 'level' => $degLevel->value];
    }
}

注意事项

约束 说明 违反后果
核心功能不能降级 用户登录、支付等必须可用 用户体验严重受损
降级要有开关 支持动态调整降级级别 只能改代码重启，不灵活
兜底数据要安全 返回的默认值不能误导用户 用户看到0余额以为真没钱了
降级要可观测 日志记录每次降级触发 不知道降级是否在工作
降级要及时恢复 服务恢复后自动切回正常 一直降级浪费服务能力

避坑指南

典型错误 现象 正确做法
降级返回硬编码空数据 用户看到"暂无数据"以为真没有 标注 degraded: true 提示用户
所有功能一起降级 一刀切，该可用的也不可用 分级降级，核心不降
降级缓存不过期 恢复后仍返回旧数据 设置合理的缓存 TTL
降级操作无权限控制 任何人都能触发降级 管理接口需要鉴权
降级忘了通知前端 前端展示异常 降级响应中包含状态标识

练习题

🟢 基础题

  1. 实现缓存兜底：当积分服务不可用时，从 Redis 缓存返回上次成功的结果。
    1. 添加降级标注：在返回数据中添加 degraded 字段，标识哪些数据是降级结果。

      2. 🟡 进阶题

      1. 自动降级：基于错误率自动调整降级级别（连续 10 次失败自动从 NORMAL 切到 DEGRADED）。

        2. 🔴 开放题

        1. 设计多级降级配置中心：支持 per-service、per-feature、per-user-group 三种维度的降级策略，支持热更新。

          2. 知识点总结

          服务降级
├── 降级策略
│   ├── 功能降级（关闭非核心功能）
│   ├── 数据降级（缓存/默认值）
│   ├── 体验降级（简化UI/延迟更新）
│   └── 流量降级（拒绝部分请求）
├── 降级级别
│   ├── NORMAL（全量服务）
│   ├── DEGRADED（非核心降级）
│   ├── MINIMAL（仅核心）
│   └── EMERGENCY（只读缓存）
├── Hyperf 实现
│   ├── DegradationManager
│   ├── Feature Map 配置
│   ├── safeCall 包装器
│   └── CacheService 兜底
└── 运维
    ├── 动态开关（Redis/配置中心）
    ├── 监控面板
    └── 自动恢复

          举一反三

          场景 变种说明 关键差异
          降级 → 熔断 熔断是自动断开，降级是有策略的切换 熔断关注保护，降级关注可用
          缓存兜底 → 静态兜底 不同粒度的兜底数据 静态兜底更简单但信息少
          单服务降级 → 全局降级 范围不同 全局需要配置中心协调

          代码演进

          v1 ❌ 无降级，任何依赖失败就整体失败

          public function getUserDetail(int $userId): array
{
    // ❌ All or nothing - any service failure = total failure
    $user = User::find($userId);
    $orders = $this->orderClient->getOrderStats($userId); // fails → exception
    $points = $this->pointsClient->getPoints($userId);    // fails → exception
    return compact('user', 'orders', 'points');
}

          v2 ✅ try-catch + 缓存兜底

          public function getUserDetail(int $userId): array
{
    $result = ['user' => User::find($userId)];
    
    try {
        $result['orders'] = $this->orderClient->getOrderStats($userId);
    } catch (\Throwable $e) {
        $result['orders'] = $this->cache->getCachedOrderStats($userId);
    }
    
    return $result;
}
