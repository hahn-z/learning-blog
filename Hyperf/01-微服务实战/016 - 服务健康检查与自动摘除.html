016 - 服务健康检查与自动摘除

难度： ⭐⭐ | 前置知识： 服务注册（014）、服务发现（015）| 预估用时： 40 分钟

概念讲解

一句话定义： 健康检查是定期向服务实例发送探测请求，验证其是否正常响应，不健康的实例自动从注册中心摘除，避免流量打到故障节点。

现实类比： 就像医生定期体检——医生给你做检查（健康检查），结果正常就标记健康（healthy=true），有问题就建议休息（摘除）。同事发现你"病了"后就不会给你派活了。

技术场景： 用户服务实例进程崩溃、数据库连接失败或响应超时时，Nacos 通过健康检查标记为不健康，其他服务（订单服务）调用时自动跳过该实例，继续使用健康实例。

完整代码

健康检查控制器 — HealthController.php

<?php

declare(strict_types=1);

namespace App\Controller;

use Hyperf\DbConnection\Db;
use Hyperf\Di\Annotation\Inject;
use Hyperf\HttpServer\Annotation\Controller;
use Hyperf\HttpServer\Annotation\GetMapping;
use Hyperf\Redis\Redis;
use Psr\Log\LoggerInterface;

#[Controller(prefix: '/health')]
class HealthController
{
    #[Inject]
    private LoggerInterface $logger;

    #[Inject]
    private Redis $redis;

    /**
     * ✅ Simple liveness check — process is running
     */
    #[GetMapping(path: '/live')]
    public function liveness(): array
    {
        return [
            'status' => 'UP',
            'timestamp' => time(),
            'service' => 'user-service',
        ];
    }

    /**
     * ✅ Readiness check — dependencies are ready
     */
    #[GetMapping(path: '/ready')]
    public function readiness(): array
    {
        $checks = [
            'database' => $this->checkDatabase(),
            'redis' => $this->checkRedis(),
            'nacos' => $this->checkNacos(),
        ];

        $allHealthy = !in_array(false, $checks, true);

        return [
            'status' => $allHealthy ? 'UP' : 'DOWN',
            'checks' => $checks,
            'timestamp' => time(),
        ];
    }

    /**
     * ✅ Detailed health check for monitoring
     */
    #[GetMapping(path: '/detail')]
    public function detail(): array
    {
        return [
            'status' => 'UP',
            'timestamp' => time(),
            'service' => 'user-service',
            'version' => env('APP_VERSION', '1.0.0'),
            'checks' => [
                'database' => $this->checkDatabase(),
                'redis' => $this->checkRedis(),
                'nacos' => $this->checkNacos(),
                'disk' => $this->checkDisk(),
            ],
            'metrics' => [
                'connections' => $this->getConnectionCount(),
                'memory_mb' => round(memory_get_usage(true) / 1024 / 1024, 2),
            ],
        ];
    }

    private function checkDatabase(): array
    {
        try {
            Db::select('SELECT 1');
            return ['status' => 'UP', 'latency_ms' => 5];
        } catch (\Throwable $e) {
            $this->logger->error('Database health check failed', ['error' => $e->getMessage()]);
            return ['status' => 'DOWN', 'error' => $e->getMessage()];
        }
    }

    private function checkRedis(): array
    {
        try {
            $start = microtime(true);
            $this->redis->ping();
            $latency = round((microtime(true) - $start) * 1000, 2);

            return ['status' => 'UP', 'latency_ms' => $latency];
        } catch (\Throwable $e) {
            $this->logger->error('Redis health check failed', ['error' => $e->getMessage()]);
            return ['status' => 'DOWN', 'error' => $e->getMessage()];
        }
    }

    private function checkNacos(): array
    {
        $nacosHost = env('NACOS_HOST', '127.0.0.1');
        $nacosPort = env('NACOS_PORT', 8848);
        $url = "http://{$nacosHost}:{$nacosPort}/nacos/v1/console/health/readiness";

        $start = microtime(true);
        $response = @file_get_contents($url);
        $latency = round((microtime(true) - $start) * 1000, 2);

        if ($response === 'UP') {
            return ['status' => 'UP', 'latency_ms' => $latency];
        }

        return ['status' => 'DOWN', 'error' => 'Nacos not ready'];
    }

    private function checkDisk(): array
    {
        $free = disk_free_space('/');
        $total = disk_total_space('/');
        $usagePercent = round((1 - $free / $total) * 100, 2);

        $status = $usagePercent < 90 ? 'UP' : 'DOWN';

        return [
            'status' => $status,
            'usage_percent' => $usagePercent,
            'free_gb' => round($free / 1024 / 1024 / 1024, 2),
        ];
    }

    private function getConnectionCount(): int
    {
        // Return active DB connections or coroutine count
        return 42; // Mock value
    }
}

自定义健康检查事件监听器 — HealthCheckListener.php

<?php

declare(strict_types=1);

namespace App\Listener;

use Hyperf\Event\Annotation\Listener;
use Hyperf\Event\Contract\ListenerInterface;
use Hyperf\Framework\Event\BeforeServerStart;
use Psr\Log\LoggerInterface;

#[Listener]
class HealthCheckListener implements ListenerInterface
{
    public function __construct(
        private LoggerInterface $logger
    ) {}

    public function listen(): array
    {
        return [BeforeServerStart::class];
    }

    public function process(object $event): void
    {
        $this->logger->info('Starting health check pre-flight...');

        // 🟡 Verify all dependencies are healthy before starting
        $checks = [
            'database' => $this->checkDatabase(),
            'redis' => $this->checkRedis(),
            'nacos' => $this->checkNacos(),
        ];

        $allHealthy = !in_array(false, $checks, true);

        if (!$allHealthy) {
            $this->logger->error('Pre-flight health check failed', ['checks' => $checks]);
            // throw new \RuntimeException('Dependencies not ready');
        }

        $this->logger->info('Pre-flight health check passed', ['checks' => $checks]);
    }

    private function checkDatabase(): bool
    {
        try {
            \Hyperf\DbConnection\Db::select('SELECT 1');
            return true;
        } catch (\Throwable $e) {
            return false;
        }
    }

    private function checkRedis(): bool
    {
        try {
            $this->container->get(\Hyperf\Redis\Redis::class)->ping();
            return true;
        } catch (\Throwable $e) {
            return false;
        }
    }

    private function checkNacos(): bool
    {
        $url = 'http://127.0.0.1:8848/nacos/v1/console/health/readiness';
        return @file_get_contents($url) === 'UP';
    }
}

  name: user-service
spec:
  replicas: 3
  selector:
    matchLabels:
      app: user-service
  template:
    metadata:
      labels:
        app: user-service
    spec:
      containers:
        - name: user-service
          image: user-service:1.0.0
          ports:
            - containerPort: 9501

          # ✅ Liveness probe — restart if unhealthy
          livenessProbe:
            httpGet:
              path: /health/live
              port: 9501
            initialDelaySeconds: 30
            periodSeconds: 10
            timeoutSeconds: 5
            failureThreshold: 3

          # ✅ Readiness probe — stop traffic if not ready
          readinessProbe:
            httpGet:
              path: /health/ready
              port: 9501
            initialDelaySeconds: 5
            periodSeconds: 5
            timeoutSeconds: 3
            failureThreshold: 2

        }
        echo "\n";
    }
}

echo "\n3️⃣ Detailed Check (/health/detail):\n";
$response = @file_get_contents("{$baseUrl}/health/detail");
echo json_encode(json_decode($response) ?: [], JSON_PRETTY_PRINT | JSON_UNESCAPED_SLASHES) . "\n";

echo "\n✅ Test complete!\n";

注意事项

约束 说明 违反后果
🔴 liveness 和 readiness 区分 liveness 重启容器，readiness 停流量 混用导致误重启或流量泄露
🟡 检查间隔 不要太频繁（5-10s），也不要太长（>30s） 频繁检查增加负载，太长延迟发现故障
🟢 依赖检查超时 每个检查要有超时（1-3s） 单个依赖挂起导致整个检查卡死
🔴 优雅降级 次要依赖失败不影响整体状态 小问题导致整个服务不可用
🟡 日志级别 健康检查失败要记录警告或错误 无法排查故障原因

避坑指南

错误 现象 正确做法
liveness 响应慢 容器被频繁重启 liveness 只检查进程，不查依赖
readiness 不检查 DB 流量打到 DB 失败的实例 readiness 要检查所有关键依赖
检查超时太长 探测超时，误判不健康 单个检查超时设置 1-3 秒
忘记返回 JSON 探测失败 确保 Content-Type: application/json
没有缓存健康状态 每次请求都查所有依赖 缓存结果，定期刷新

          2. 知识点总结

          健康检查
├── 检查类型
│   ├── Liveness — 进程是否运行
│   ├── Readiness — 依赖是否就绪
│   └── Startup — 启动是否完成
├── Nacos 心跳机制
│   ├── 实例定时发送心跳
│   ├── 超时未收到心跳 → 标记不健康
│   └── 长时间不健康 → 自动摘除
├── Hyperf 实现
│   ├── HealthController
│   ├── 配置 heartbeat 间隔
│   └── 自定义健康检查端点
└── Kubernetes 集成
    ├── livenessProbe
    ├── readinessProbe
    └── startupProbe

          举一反三

          场景 适配方式 关键差异
          优雅停机 停机前先返回 readiness=DOWN 停止接收新流量，处理完现有请求
          滚动发布 新实例 readiness=UP 后才下线旧实例 零停机部署
          故障自愈 readiness=DOWN 自动触发重启 容器编排平台（K8s）支持
