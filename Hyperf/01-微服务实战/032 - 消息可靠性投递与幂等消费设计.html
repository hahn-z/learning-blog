消息可靠性投递与幂等消费设计

难度： ⭐⭐⭐⭐
前置知识： Kafka 基础（031篇）、数据库事务、唯一索引
预估用时： 50-70 分钟

1. 概念讲解

消息可靠性指消息从生产到消费的全链路不丢失、不重复、不乱序；幂等消费指同一条消息被消费多次，业务结果与消费一次完全一致。

类比理解：寄快递时，保价+签收确认 = 可靠投递；收件人根据快递单号去重，同一个单号的包裹签收一次就够 = 幂等消费。在用户服务中，用户注册消息可能因网络重试被投递多次，如果没有幂等机制，就会发送多封欢迎邮件、重复创建积分账户。

3. 完整代码

3.1 可靠生产者 app/Kafka/Producer/ReliableProducer.php

<?php
declare(strict_types=1);

namespace App\Kafka\Producer;

use Hyperf\Kafka\Producer;
use Hyperf\Di\Annotation\Inject;
use Psr\Log\LoggerInterface;

class ReliableProducer
{
    #[Inject]
    private Producer $producer;

    #[Inject]
    private LoggerInterface $logger;

    private const MAX_RETRY = 3;

    /**
     * ✅ Reliable send with retry mechanism
     */
    public function reliableSend(string $topic, array $payload, string $key = ''): bool
    {
        // ✅ Attach event ID and timestamp for tracing
        $payload['event_id'] = $payload['event_id'] ?? uniqid('evt_', true);
        $payload['sent_at'] = date('c');

        $body = json_encode($payload, JSON_THROW_ON_ERROR);

        for ($attempt = 1; $attempt <= self::MAX_RETRY; $attempt++) {
            try {
                $this->producer->send($topic, $body, $key);
                $this->logger->info('Message sent successfully', [
                    'topic' => $topic,
                    'event_id' => $payload['event_id'],
                    'attempt' => $attempt,
                ]);
                return true;
            } catch (\Throwable $e) {
                $this->logger->warning('Send failed, retrying', [
                    'topic' => $topic,
                    'attempt' => $attempt,
                    'error' => $e->getMessage(),
                ]);
                if ($attempt >= self::MAX_RETRY) {
                    // 🔴 Dead letter: persist to DB for manual processing
                    $this->persistToDeadLetter($topic, $body, $e->getMessage());
                    return false;
                }
                usleep(100000 * $attempt); // ✅ Linear backoff
            }
        }
        return false;
    }

    private function persistToDeadLetter(string $topic, string $body, string $error): void
    {
        $this->logger->error('Dead letter: message permanently failed', [
            'topic' => $topic,
            'body' => $body,
            'error' => $error,
        ]);
    }
}

3.2 幂等消费者 app/Kafka/Consumer/IdempotentConsumer.php

<?php
declare(strict_types=1);

namespace App\Kafka\Consumer;

use Hyperf\Kafka\Annotation\Consumer;
use Hyperf\Kafka\AbstractConsumer;
use Hyperf\Di\Annotation\Inject;
use App\Repository\EventLogRepository;
use Psr\Log\LoggerInterface;
use longlang\phpkafka\Consumer\ConsumeMessage;

#[Consumer(
    topic: 'user.registered',
    groupId: 'idempotent-email-group',
    autoCommit: false
)]
class IdempotentConsumer extends AbstractConsumer
{
    #[Inject]
    private EventLogRepository $eventLog;

    #[Inject]
    private LoggerInterface $logger;

    public function consume(ConsumeMessage $message): void
    {
        $data = json_decode($message->getValue(), true, 512, JSON_THROW_ON_ERROR);
        $eventId = $data['event_id'] ?? throw new \InvalidArgumentException('Missing event_id');

        // ✅ Idempotency gate: check if already processed
        if ($this->eventLog->isProcessed($eventId)) {
            $this->logger->info('Duplicate event skipped', ['event_id' => $eventId]);
            $message->getConsumer()->ack($message);
            return;
        }

        try {
            // ✅ Process within a DB transaction for atomicity
            $this->eventLog->processInTransaction($eventId, function () use ($data) {
                $this->handleUserRegistered($data);
            });

            $message->getConsumer()->ack($message);
            $this->logger->info('Event processed', ['event_id' => $eventId]);
        } catch (\Throwable $e) {
            $this->logger->error('Event processing failed', [
                'event_id' => $eventId,
                'error' => $e->getMessage(),
            ]);
            throw $e;
        }
    }

    private function handleUserRegistered(array $data): void
    {
        $this->logger->info("Welcome email sent to {$data['email']}", [
            'user_id' => $data['user_id'],
        ]);
    }
}

3.3 事件日志仓储 app/Repository/EventLogRepository.php

<?php
declare(strict_types=1);

namespace App\Repository;

use Hyperf\DbConnection\Db;
use Psr\Log\LoggerInterface;
use Hyperf\Di\Annotation\Inject;

class EventLogRepository
{
    #[Inject]
    private Db $db;

    #[Inject]
    private LoggerInterface $logger;

    /**
     * ✅ Check if event has been processed (idempotency key)
     */
    public function isProcessed(string $eventId): bool
    {
        return $this->db->table('event_log')
            ->where('event_id', $eventId)
            ->where('status', 'processed')
            ->exists();
    }

    /**
     * ✅ Atomic: mark event + execute business logic in one transaction
     */
    public function processInTransaction(string $eventId, callable $businessLogic): void
    {
        $this->db->transaction(function () use ($eventId, $businessLogic) {
            // ✅ INSERT with UNIQUE constraint prevents duplicate
            $this->db->table('event_log')->insertOrIgnore([
                'event_id' => $eventId,
                'status' => 'processed',
                'processed_at' => date('c'),
            ]);

            $businessLogic();
        });
    }
}

5. 注意事项

约束 说明 违反后果
event_id 必须全局唯一 用 uniqid 或 UUID 幂等检查失效，重复处理
event_log 表必须有 UNIQUE 约束 防止并发写入重复行 并发场景下同一事件被处理两次
业务逻辑和标记必须同一事务 原子性保证 标记了但业务未执行，或反之
死信记录不可丢 持久化到 DB 或死信队列 消息永久丢失无迹可查
重试要有退避策略 线性或指数退避 立即重试加重系统负载

6. 避坑指南

错误 现象 正确做法
先 ack 再处理 处理失败但 offset 已提交，消息丢失 先处理，成功后再 ack
SELECT 再 INSERT 无事务 并发时两个消费者都通过 SELECT 检查 INSERT + UNIQUE 约束 + 事务
event_id 用自增 ID 分布式环境不唯一 用 UUID 或 snowflake ID
不设最大重试 无限重试导致消费阻塞 设 MAX_RETRY + 死信队列
只记日志不持久化死信 服务重启后死信丢失 持久化到 DB 或独立 Topic

8. 知识点总结

Message Reliability
├── Producer Side
│   ├── Retry mechanism (3 attempts)
│   ├── Backoff strategy (linear/exponential)
│   └── Dead letter persistence
├── Consumer Side
│   ├── Manual commit (autoCommit=false)
│   ├── Idempotency check (event_id)
│   └── Transactional processing
├── Event Log Table
│   ├── UNIQUE constraint on event_id
│   ├── INSERT ... ON DUPLICATE KEY IGNORE
│   └── Atomic mark + business logic
└── Reliability Patterns
    ├── At-Least-Once + Idempotent
    ├── Dead Letter Queue
    └── Monitoring & Alerting

    v1 ❌ 无任何保障

    // ❌ No retry, no idempotency, auto commit
$this->producer->send($topic, json_encode($data));
// Consumer: process then auto-ack, no error handling

    v2 ✅ 重试 + 手动提交

    // ✅ Retry mechanism + manual commit
for ($i = 1; $i <= 3; $i++) {
    try {
        $this->producer->send($topic, $body, $key);
        break;
    } catch (\Throwable $e) {
        usleep(100000 * $i);
    }
}
// Consumer: try { process(); ack(); } catch { throw; }

    v3 🟢 完整可靠方案

    // 🟢 Retry + dead letter + idempotent + transactional
$this->reliableSend($topic, $payload, $key);
// Consumer: eventLog->isProcessed() → processInTransaction() → ack()
