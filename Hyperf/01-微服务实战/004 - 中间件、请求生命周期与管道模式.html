004 - 中间件、请求生命周期与管道模式

📂 Hyperf微服务实战 🏷️ Hyperf

中间件、请求生命周期与管道模式

难度： ⭐⭐⭐
前置知识： DI 容器（003）、HTTP 基础、PSR-7/PSR-15 规范
预估用时： 50 分钟

一、概念讲解

一句话定义： 中间件是在请求到达控制器之前和响应返回之后执行的过滤器链，采用管道模式（Pipeline）依次处理。

现实类比： 你去银行办业务——先取号（认证中间件）→ 过安检（安全中间件）→ 到窗口办业务（Controller）→ 拿回执离开（日志中间件）。每个环节都是一个中间件，可以放行也可以拒绝。

技术场景： 用户服务中，每个 API 请求都需要经过：CORS 处理 → 请求日志 → JWT 鉴权 → 速率限制 → 到达 Controller。这些横切关注点用中间件统一管理，避免在每个 Controller 里重复编写。

二、实时脑图

Request → [CORS] → [Log] → [Auth] → [RateLimit] → Controller
                                                      │
Response ← [CORS] ← [Log] ← [Auth] ← [RateLimit] ← ┘

Pipeline Flow:
┌──────────┐    ┌──────────┐    ┌──────────┐    ┌──────────┐
│Middleware │───→│Middleware │───→│Middleware │───→│Controller│

     └──────────── Response flows back ──────────────┘

三、完整代码

3.1 CORS 中间件

<?php
// ✅ app/Middleware/CorsMiddleware.php

declare(strict_types=1);

namespace App\Middleware;

use Psr\Http\Message\ResponseInterface;
use Psr\Http\Message\ServerRequestInterface;
use Psr\Http\Server\MiddlewareInterface;
use Psr\Http\Server\RequestHandlerInterface;

class CorsMiddleware implements MiddlewareInterface
{
    public function process(
        ServerRequestInterface $request,
        RequestHandlerInterface $handler
    ): ResponseInterface {
        // ✅ Handle preflight OPTIONS request
        if ($request->getMethod() === 'OPTIONS') {
            $response = (new \Hyperf\HttpServer\Response())
                ->withStatus(204);
        } else {
            $response = $handler->handle($request);
        }

        // ✅ Add CORS headers
        return $response
            ->withHeader('Access-Control-Allow-Origin', '*')
            ->withHeader('Access-Control-Allow-Methods', 'GET,POST,PUT,DELETE,OPTIONS')
            ->withHeader('Access-Control-Allow-Headers', 'Content-Type,Authorization')
            ->withHeader('Access-Control-Max-Age', '86400');
    }
}

3.2 请求日志中间件

<?php
// ✅ app/Middleware/RequestLogMiddleware.php

declare(strict_types=1);

namespace App\Middleware;

use Hyperf\Di\Annotation\Inject;
use Psr\Http\Message\ResponseInterface;
use Psr\Http\Message\ServerRequestInterface;
use Psr\Http\Server\MiddlewareInterface;
use Psr\Http\Server\RequestHandlerInterface;
use Psr\Log\LoggerInterface;

class RequestLogMiddleware implements MiddlewareInterface
{
    #[Inject]
    private LoggerInterface $logger;

    public function process(
        ServerRequestInterface $request,
        RequestHandlerInterface $handler
    ): ResponseInterface {
        $startTime = microtime(true);
        $method = $request->getMethod();
        $uri = (string) $request->getUri();

        $this->logger->info("-> {$method} {$uri}");

        // ✅ Pass to next middleware
        $response = $handler->handle($request);

        $elapsed = round((microtime(true) - $startTime) * 1000, 2);
        $status = $response->getStatusCode();

        $this->logger->info("<- {$method} {$uri} [{$status}] {$elapsed}ms");

        return $response;
    }
}

3.3 JWT 认证中间件

<?php
// ✅ app/Middleware/JwtAuthMiddleware.php

declare(strict_types=1);

namespace App\Middleware;

use Hyperf\HttpServer\Contract\RequestInterface;
use Psr\Http\Message\ResponseInterface;
use Psr\Http\Message\ServerRequestInterface;
use Psr\Http\Server\MiddlewareInterface;
use Psr\Http\Server\RequestHandlerInterface;

class JwtAuthMiddleware implements MiddlewareInterface
{
    public function process(
        ServerRequestInterface $request,
        RequestHandlerInterface $handler
    ): ResponseInterface {
        // ✅ Extract token from Authorization header
        $header = $request->getHeaderLine('Authorization');

        if (!$header || !str_starts_with($header, 'Bearer ')) {
            return $this->unauthorized('Missing or invalid token');
        }

        $token = substr($header, 7);

        // 🔴 TODO: Implement actual JWT validation
        // For now, just check token is not empty
        if (empty($token)) {
            return $this->unauthorized('Empty token');
        }

        // ✅ Token valid, pass to next middleware
        return $handler->handle($request);
    }

    private function unauthorized(string $message): ResponseInterface
    {
        return (new \Hyperf\HttpServer\Response())
            ->withStatus(401)
            ->withHeader('Content-Type', 'application/json')
            ->withBody(
                \Hyperf\HttpMessage\Stream\SwooleStream::create(
                    json_encode(['code' => 401, 'message' => $message])
                )
            );
    }
}

3.5 路由级中间件

<?php
// ✅ config/routes.php

use Hyperf\HttpServer\Router\Router;

// 🟢 Public routes (no auth required)
Router::addGroup('/user', function () {
    Router::post('/register', 'App\Controller\UserController@register');
    Router::post('/login', 'App\Controller\UserController@login');
});

// 🟢 Protected routes (auth required)
Router::addGroup('/user', function () {
    Router::get('/profile', 'App\Controller\UserController@profile');
    Router::put('/update', 'App\Controller\UserController@update');
    Router::post('/logout', 'App\Controller\UserController@logout');
}, ['middleware' => [App\Middleware\JwtAuthMiddleware::class]]);

五、注意事项

约束 说明 违反后果
中间件顺序很重要 全局中间件按注册顺序执行 CORS 拦截在 Auth 后面导致预检失败
必须调用 $handler->handle() 否则请求链断裂 后续中间件和 Controller 都不执行
响应对象不可变 PSR-7 是 immutable 的 修改 header 不生效（需要 withHeader 返回新对象）
不要在中间件做重操作 每个请求都经过中间件 性能瓶颈
异常要被捕获 中间件内异常会中断管道 返回 500 而非友好提示

六、避坑指南

典型错误 现象 正确做法
❌ 忘记调用 handle() 请求直接返回空响应 始终在中间件中调用 $handler->handle($request)
❌ 全局注册 Auth 中间件 登录/注册也需要 token 使用路由级中间件，区分公开/私有路由
❌ 直接修改 $response Header 修改不生效 用 withHeader() 返回新对象
❌ 中间件间传递数据用全局变量 协程间数据串扰 用 $request->withAttribute() 传递

十一、代码演进

v1 ❌ 控制器内处理横切关注点

<?php
// ❌ Cross-cutting concerns mixed into controller
class UserController
{
    public function profile(RequestInterface $request): array
    {
        // ❌ Auth check in controller - repeated everywhere
        $token = $request->getHeaderLine('Authorization');
        if (!$token) return ['code' => 401, 'message' => 'Unauthorized'];
        
        // ❌ Logging in controller
        error_log("User profile requested");
        
        // ❌ Manual JSON header
        header('Content-Type: application/json');
        
        return ['data' => $user];
    }
}

v2 ✅ 中间件分离横切关注点

<?php
// ✅ Auth in middleware, controller focuses on business logic
class UserController
{
    #[Inject]
    private UserService $userService;

    public function profile(RequestInterface $request): array
    {
        $id = (int) $request->input('id');
        return ['code' => 0, 'data' => $this->userService->getUser($id)];
    }
}
// Auth, logging, CORS handled in middleware pipeline

v3 🟢 注解驱动的中间件配置

<?php
// 🟢 Annotation-based middleware assignment
#[AutoController(prefix: "user")]
#[Middleware(CorsMiddleware::class)]

{
    #[Middleware(JwtAuthMiddleware::class)]
    #[RateLimit(maxRequests: 100)]

        return ['code' => 0, 'data' => $this->userService->getUser(
            (int) $request->input('id')
        )];
    }

    // ✅ No auth middleware - public endpoint
    public function register(RequestInterface $request): array
    {
        // Registration logic...
    }
}
