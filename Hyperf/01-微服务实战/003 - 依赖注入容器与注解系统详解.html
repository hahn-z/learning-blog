一、概念讲解

一句话定义： 依赖注入（DI）是一种设计模式，由容器自动解析并注入对象依赖，而非手动 new 创建。

现实类比： 你去餐厅点餐，不需要自己去厨房找厨师、找食材、找锅——服务员（容器）会自动把一切安排好端到你面前。你只管"点菜"（声明依赖），不关心"怎么做"（对象创建）。

技术场景： 在用户服务中，UserController 需要 UserService，UserService 需要 UserRepository 和 Logger。手动创建这些对象并传递依赖会让代码耦合度爆表。Hyperf 的 DI 容器通过注解（如 #[Inject]）自动完成依赖解析和注入，让代码解耦、可测试、可维护。

三、完整代码

3.1 Repository 层

<?php
// ✅ app/Repository/UserRepository.php

declare(strict_types=1);

namespace App\Repository;

use Hyperf\DbConnection\Db;

class UserRepository
{
    // ✅ Find user by ID
    public function findById(int $id): ?object
    {
        return Db::table('users')->where('id', $id)->first();
    }

    // ✅ Find user by email
    public function findByEmail(string $email): ?object
    {
        return Db::table('users')->where('email', $email)->first();
    }

    // ✅ Create a new user record
    public function create(array $data): int
    {
        return Db::table('users')->insertGetId($data);
    }
}

3.2 Service 层（使用依赖注入）

<?php
// ✅ app/Service/UserService.php

declare(strict_types=1);

namespace App\Service;

use App\Repository\UserRepository;
use Hyperf\Di\Annotation\Inject;
use Psr\Log\LoggerInterface;

class UserService
{
    // 🟢 Constructor injection (recommended for required dependencies)
    public function __construct(
        private UserRepository $userRepository,
    ) {}

    // 🟢 Property injection via annotation (alternative approach)
    #[Inject]
    private LoggerInterface $logger;

    // ✅ Get user by ID with logging
    public function getUser(int $id): ?array
    {
        $this->logger->info("Fetching user by ID", ['id' => $id]);
        $user = $this->userRepository->findById($id);

        if (!$user) {
            $this->logger->warning("User not found", ['id' => $id]);
            return null;
        }

        return (array) $user;
    }
}

3.3 Controller 层

<?php
// ✅ app/Controller/UserController.php

declare(strict_types=1);

namespace App\Controller;

use App\Service\UserService;
use Hyperf\HttpServer\Annotation\AutoController;
use Hyperf\HttpServer\Contract\RequestInterface;
use Hyperf\Di\Annotation\Inject;

#[AutoController(prefix: "user")]
class UserController
{
    // ✅ Inject service via annotation
    #[Inject]
    private UserService $userService;

    // ✅ Get user profile
    public function profile(RequestInterface $request): array
    {
        $id = (int) $request->input('id', 0);

        if ($id <= 0) {
            return ['code' => 400, 'message' => 'Invalid user ID'];
        }

        $user = $this->userService->getUser($id);

        if (!$user) {
            return ['code' => 404, 'message' => 'User not found'];
        }

        return ['code' => 0, 'data' => $user];
    }
}

3.4 自定义注解

<?php
// ✅ app/Annotation/RateLimit.php

declare(strict_types=1);

namespace App\Annotation;

use Attribute;
use Hyperf\Di\Annotation\AbstractAnnotation;

// 🟢 Custom annotation for rate limiting
#[Attribute(Attribute::TARGET_METHOD)]
class RateLimit extends AbstractAnnotation
{
    public function __construct(
        public int $maxRequests = 60,
        public int $window = 60,
    ) {}
}
<?php
// ✅ app/Aspect/RateLimitAspect.php

declare(strict_types=1);

namespace App\Aspect;

use App\Annotation\RateLimit;
use Hyperf\Di\Aop\AbstractAspect;
use Hyperf\Di\Aop\ProceedingJoinPoint;
use Hyperf\Redis\Redis;
use Hyperf\Di\Annotation\Aspect;
use Hyperf\Context\ApplicationContext;

#[Aspect]
class RateLimitAspect extends AbstractAspect
{
    // ✅ Define which annotations this aspect handles
    public array $annotations = [
        RateLimit::class,
    ];

    public function process(ProceedingJoinPoint $proceedingJoinPoint)
    {
        $metadata = $proceedingJoinPoint->getAnnotationMetadata();
        /** @var RateLimit $rateLimit */
        $rateLimit = $metadata->method[RateLimit::class] ?? null;

        if (!$rateLimit) {
            return $proceedingJoinPoint->process();
        }

        $container = ApplicationContext::getContainer();
        $redis = $container->get(Redis::class);
        $key = 'rate_limit:' . md5($proceedingJoinPoint->className . '::' . $proceedingJoinPoint->methodName);

        $current = $redis->incr($key);
        if ($current === 1) {
            $redis->expire($key, $rateLimit->window);
        }

        if ($current > $rateLimit->maxRequests) {
            return ['code' => 429, 'message' => 'Too many requests'];
        }

        return $proceedingJoinPoint->process();
    }
}

五、注意事项

约束 说明 违反后果
不要 new Service 对象 绕过容器导致依赖不被注入 #[Inject] 属性为 null，NPE 崩溃
注解类必须被 Scanner 扫描 需在 config/autoload/annotations.php 配置扫描路径 注解不生效
AOP 代理类限制 final 类和 private 方法不支持 AOP Aspect 不生效，无报错
单例是默认行为 容器中对象默认单例 请求间数据串扰（需注意状态管理）
构造函数不要做重操作 容器启动时就会解析 启动变慢，连接池未就绪报错

六、避坑指南

典型错误 现象 正确做法
❌ new UserService() 手动创建 注入的属性为 null 用容器 make() 或构造函数注入
❌ 在 __construct 中用 #[Inject] 属性 属性尚未注入，值为 null 用构造函数参数注入依赖
❌ 忘记配置 scan.enable 自定义注解/Aspect 不生效 检查 annotations.php 配置
❌ 在注解属性中写业务逻辑 逻辑不执行 注解只做元数据，逻辑放 Aspect

十一、代码演进

v1 ❌ 手动创建依赖

<?php
// ❌ Tight coupling: manual dependency creation
class UserController
{
    public function profile(RequestInterface $request): array
    {
        // ❌ Manual instantiation, no logging, no DI
        $db = new \PDO('mysql:host=127.0.0.1;dbname=test', 'root', '');
        $repo = new UserRepository($db);
        $service = new UserService($repo);
        $user = $service->getUser(1);
        return (array) $user;
    }
}

v2 ✅ 构造函数注入

<?php
// ✅ Constructor injection with DI container
class UserController
{
    public function __construct(
        private UserService $userService,
    ) {}

    public function profile(RequestInterface $request): array
    {
        $user = $this->userService->getUser(1);
        return ['code' => 0, 'data' => $user];
    }
}

v3 🟢 注解驱动 + AOP

<?php
// 🟢 Annotation-driven with rate limiting AOP
#[AutoController(prefix: "user")]
class UserController
{
    #[Inject]
    private UserService $userService;

    #[RateLimit(maxRequests: 100, window: 60)]
    public function profile(RequestInterface $request): array
    {
        $id = (int) $request->input('id', 0);
        $user = $this->userService->getUser($id);
        return $user ? ['code' => 0, 'data' => $user] : ['code' => 404];
    }
}
