一、概念讲解

一句话定义： JWT（JSON Web Token）是无状态认证方案，用户登录后获得包含身份信息的加密令牌，后续请求携带令牌即可通过鉴权，无需服务端存储会话。

现实类比： 你进迪士尼乐园——门票检票后给你一个腕带（JWT），你去任何游乐项目时扫描腕带（验证 token）即可进入，不需要每次都回检票口查数据库。腕带过期后（refresh expired）你需要去服务台换新腕带。

技术场景： 用户服务登录成功后签发 Access Token（2 小时有效）和 Refresh Token（7 天有效）。客户端将 Token 存在 LocalStorage 或 HttpOnly Cookie，后续 API 请求在 Header 中携带 Authorization: Bearer 。服务端验证 Token 签名和有效期，返回用户信息。

3.2 JWT Service

<?php
// ✅ app/Service/JwtService.php

declare(strict_types=1);

namespace App\Service;

use Firebase\JWT\JWT;
use Firebase\JWT\Key;

class JwtService
{
    private string $secret;
    private int $accessExpire;
    private int $refreshExpire;
    private string $issuer;
    private string $algorithm;

    public function __construct()
    {
        $config = config('jwt');
        $this->secret = $config['secret'];
        $this->accessExpire = $config['access_expire'];
        $this->refreshExpire = $config['refresh_expire'];
        $this->issuer = $config['issuer'];
        $this->algorithm = $config['algorithm'];
    }

    // ✅ Generate access token (short-lived)
    public function generateAccessToken(int $userId, array $payload = []): string
    {
        $now = time();

        $data = array_merge($payload, [
            'iss' => $this->issuer,
            'iat' => $now,
            'exp' => $now + $this->accessExpire,
            'sub' => $userId,
            'type' => 'access',
        ]);

        return JWT::encode($data, $this->secret, $this->algorithm);
    }

    // ✅ Generate refresh token (long-lived)
    public function generateRefreshToken(int $userId, array $payload = []): string
    {
        $now = time();

        $data = array_merge($payload, [
            'iss' => $this->issuer,
            'iat' => $now,
            'exp' => $now + $this->refreshExpire,
            'sub' => $userId,
            'type' => 'refresh',
        ]);

        return JWT::encode($data, $this->secret, $this->algorithm);
    }

    // ✅ Decode and validate token
    public function decode(string $token): ?object
    {
        try {
            $decoded = JWT::decode($token, new Key($this->secret, $this->algorithm));
            return (object) $decoded;
        } catch (\Firebase\JWT\ExpiredException $e) {
            throw new \Exception('Token expired');
        } catch (\Firebase\JWT\SignatureInvalidException $e) {
            throw new \Exception('Invalid token signature');
        } catch (\Exception $e) {
            throw new \Exception('Invalid token');
        }
    }

    // ✅ Verify token and return user ID
    public function verify(string $token): ?int
    {
        try {
            $decoded = $this->decode($token);
            return (int) $decoded->sub;
        } catch (\Exception $e) {
            return null;
        }
    }

    // ✅ Get token expiration time
    public function getExpiration(string $token): ?int
    {
        try {
            $decoded = $this->decode($token);
            return $decoded->exp;
        } catch (\Exception $e) {
            return null;
        }
    }

    // ✅ Check if token is expired
    public function isExpired(string $token): bool
    {
        $exp = $this->getExpiration($token);
        return $exp === null || $exp < time();
    }
}

3.3 登录 Service

<?php
// ✅ app/Service/AuthService.php

declare(strict_types=1);

namespace App\Service;

use App\Model\User;
use App\Model\UserToken;
use App\Service\JwtService;
use Hyperf\DbConnection\Db;
use Hyperf\Context\Context;
use Hyperf\Di\Annotation\Inject;
use Psr\Log\LoggerInterface;

class AuthService
{
    #[Inject]
    private JwtService $jwt;

    #[Inject]
    private LoggerInterface $logger;

    // ✅ Login with email/phone and password
    public function login(string $account, string $password): array
    {
        // 🟢 Find user by email or phone
        $user = User::where('email', $account)
            ->orWhere('phone', $account)
            ->first();

        // 🔴 Check user exists and password is correct
        if (!$user || !$user->verifyPassword($password)) {
            return ['code' => 401, 'message' => 'Invalid credentials'];
        }

        // 🔴 Check user status
        if (!$user->isActive()) {
            return ['code' => 403, 'message' => 'Account is disabled'];
        }

        // 🟢 Generate tokens
        $accessToken = $this->jwt->generateAccessToken($user->id);
        $refreshToken = $this->jwt->generateRefreshToken($user->id);

        // 🟢 Store token in database (for revocation support)
        UserToken::createAccessToken($user->id, $accessToken);
        UserToken::createRefreshToken($user->id, $refreshToken);

        // 🟢 Update login info
        $user->update([
            'last_login_at' => now(),
            'last_login_ip' => Context::get('request_ip', '0.0.0.0'),
        ]);

        $this->logger->info('User logged in', ['user_id' => $user->id]);

        return [
            'code' => 0,
            'message' => 'Login successful',
            'data' => [
                'access_token' => $accessToken,
                'refresh_token' => $refreshToken,
                'expires_in' => config('jwt.access_expire'),
                'user' => [
                    'id' => $user->id,
                    'username' => $user->username,
                    'email' => $user->email,
                ],
            ],
        ];
    }

    // ✅ Refresh access token using refresh token
    public function refresh(string $refreshToken): array
    {
        try {
            $userId = $this->jwt->verify($refreshToken);
            if (!$userId) {
                return ['code' => 401, 'message' => 'Invalid refresh token'];
            }

            // 🟢 Verify token is stored in database (not revoked)
            $tokenRecord = UserToken::where('token_hash', hash('sha256', $refreshToken))
                ->where('type', 'refresh')
                ->where('revoked', false)
                ->first();

            if (!$tokenRecord || $tokenRecord->isExpired()) {
                return ['code' => 401, 'message' => 'Refresh token expired or revoked'];
            }

            // 🟢 Generate new access token
            $newAccessToken = $this->jwt->generateAccessToken($userId);
            UserToken::createAccessToken($userId, $newAccessToken);

            $this->logger->info('Token refreshed', ['user_id' => $userId]);

            return [
                'code' => 0,
                'message' => 'Token refreshed',
                'data' => [
                    'access_token' => $newAccessToken,
                    'expires_in' => config('jwt.access_expire'),
                ],
            ];
        } catch (\Exception $e) {
            $this->logger->error('Token refresh failed', ['error' => $e->getMessage()]);
            return ['code' => 500, 'message' => 'Token refresh failed'];
        }
    }

    // ✅ Logout (revoke token)
    public function logout(string $accessToken): array
    {
        try {
            $userId = $this->jwt->verify($accessToken);
            if (!$userId) {
                return ['code' => 401, 'message' => 'Invalid token'];
            }

            // 🔴 Revoke all tokens for this user
            UserToken::where('user_id', $userId)->update(['revoked' => true]);

            $this->logger->info('User logged out', ['user_id' => $userId]);

            return ['code' => 0, 'message' => 'Logout successful'];
        } catch (\Exception $e) {
            return ['code' => 500, 'message' => 'Logout failed'];
        }
    }
}

3.4 JWT 认证中间件

<?php
// ✅ app/Middleware/JwtAuthMiddleware.php

declare(strict_types=1);

namespace App\Middleware;

use App\Service\JwtService;
use Hyperf\Context\Context;
use Hyperf\Di\Annotation\Inject;
use Psr\Http\Message\ResponseInterface;
use Psr\Http\Message\ServerRequestInterface;
use Psr\Http\Server\MiddlewareInterface;
use Psr\Http\Server\RequestHandlerInterface;

class JwtAuthMiddleware implements MiddlewareInterface
{
    #[Inject]
    private JwtService $jwt;

    public function process(
        ServerRequestInterface $request,
        RequestHandlerInterface $handler
    ): ResponseInterface {
        // 🟢 Extract token from Authorization header
        $authHeader = $request->getHeaderLine('Authorization');

        if (empty($authHeader) || !str_starts_with($authHeader, 'Bearer ')) {
            return $this->unauthorized('Missing or invalid authorization header');
        }

        $token = substr($authHeader, 7);

        // 🟢 Verify token
        $userId = $this->jwt->verify($token);
        if (!$userId) {
            return $this->unauthorized('Invalid or expired token');
        }

        // 🟢 Store user ID in context for later use
        Context::set('user_id', $userId);
        Context::set('auth_token', $token);

        // ✅ Pass to next middleware
        return $handler->handle($request);
    }

    private function unauthorized(string $message): ResponseInterface
    {
        return (new \Hyperf\HttpServer\Response())
            ->withStatus(401)
            ->withHeader('Content-Type', 'application/json')
            ->withBody(
                \Hyperf\HttpMessage\Stream\SwooleStream::create(
                    json_encode(['code' => 401, 'message' => $message])
                )
            );
    }
}

3.5 登录 Controller

<?php
// ✅ app/Controller/AuthController.php

declare(strict_types=1);

namespace App\Controller;

use App\Service\AuthService;
use Hyperf\HttpServer\Annotation\AutoController;
use Hyperf\HttpServer\Contract\RequestInterface;
use Hyperf\Di\Annotation\Inject;

#[AutoController(prefix: "auth")]
class AuthController
{
    #[Inject]
    private AuthService $authService;

    // ✅ Login endpoint
    public function login(RequestInterface $request): array
    {
        $account = $request->input('account');
        $password = $request->input('password');

        if (empty($account) || empty($password)) {
            return ['code' => 400, 'message' => 'Account and password are required'];
        }

        return $this->authService->login($account, $password);
    }

    // ✅ Refresh token endpoint
    public function refresh(RequestInterface $request): array
    {
        $refreshToken = $request->input('refresh_token');

        if (empty($refreshToken)) {
            return ['code' => 400, 'message' => 'Refresh token is required'];
        }

        return $this->authService->refresh($refreshToken);
    }

    // ✅ Logout endpoint
    public function logout(RequestInterface $request): array
    {
        $token = $request->getHeaderLine('Authorization');
        $token = substr($token, 7);

        return $this->authService->logout($token);
    }
}

五、注意事项

约束 说明 违反后果
JWT Secret 必须强随机 至少 32 字节随机字符串 Token 可被伪造
Token 存储在 HttpOnly Cookie 防止 XSS 攻击窃取 Token Token 泄露
Refresh Token 只能用一次 刷新后旧 Token 失效 重放攻击
Logout 必须 Revoke Token 从数据库标记 revoked Token 仍可使用
时钟容忍度配置 处理客户端时间不准 Token 验证失败

六、避坑指南

典型错误 现象 正确做法
❌ JWT Secret 写死在代码 密钥泄露，可伪造 Token 用环境变量 env('JWT_SECRET')
❌ Access Token 有效期太长 Token 被盗后长时间有效 2 小时以内
❌ 不验证 Token 存储 Token 被盗仍可用 数据库验证 + Revoke 机制
❌ Header 写错 Authorization 中间件拿不到 Token Authorization: Bearer 格式
❌ 忽略 Token 过期检查 过期 Token 仍通过 JWT::decode 自动检查

十一、代码演进

v1 ❌ Session 认证（不适用微服务）

<?php
// ❌ Session-based auth (not suitable for microservices)
session_start();

public function login($email, $password) {
    $user = User::where('email', $email)->first();
    if ($user && password_verify($password, $user->password)) {
        $_SESSION['user_id'] = $user->id;  // ❌ Session state on server
        return ['success' => true];
    }
    return ['success' => false];
}

v2 ✅ 基本 JWT 认证

<?php
// ✅ JWT without storage (stateless but can't revoke)
public function login($email, $password) {
    $user = User::where('email', $email)->first();
    if ($user && password_verify($password, $user->password)) {
        $token = JWT::encode([
            'sub' => $user->id,
            'exp' => time() + 7200,
        ], 'secret', 'HS256');
        return ['token' => $token];  // ❌ Can't revoke stolen tokens
    }
}

v3 🟢 生产级 JWT 认证（可撤销）

<?php
// 🟢 JWT with database storage and revocation support
public function login(string $account, string $password): array {
    // Find user
    $user = User::where('email', $account)->orWhere('phone', $account)->first();
    if (!$user || !$user->verifyPassword($password)) {
        return ['code' => 401, 'message' => 'Invalid credentials'];
    }

    // Generate tokens
    $accessToken = $this->jwt->generateAccessToken($user->id);
    $refreshToken = $this->jwt->generateRefreshToken($user->id);

    // Store token hashes for revocation support
    UserToken::createAccessToken($user->id, $accessToken);
    UserToken::createRefreshToken($user->id, $refreshToken);

    // Update login info
    $user->update([
        'last_login_at' => now(),
        'last_login_ip' => $this->getClientIp(),
    ]);

    return [
        'code' => 0,
        'data' => [
            'access_token' => $accessToken,
            'refresh_token' => $refreshToken,
            'expires_in' => config('jwt.access_expire'),
        ],
    ];
}

// ✅ Middleware verifies token and checks revocation status
public function process(RequestInterface $request, RequestHandlerInterface $handler) {
    $token = $this->extractToken($request);
    $userId = $this->jwt->verify($token);
    
    if (!$userId) {
        return $this->unauthorized('Invalid token');
    }

    // Check if token is revoked in database
    $isRevoked = UserToken::where('token_hash', hash('sha256', $token))
        ->where('revoked', true)
        ->exists();
    
    if ($isRevoked) {
        return $this->unauthorized('Token revoked');
    }

    Context::set('user_id', $userId);
    return $handler->handle($request);
}
