    <!-- ConditionalListDemo.vue -->
<template>
  <div class="demo">
    <!-- 1. v-if / v-else-if / v-else -->
    <h2>Conditional Rendering</h2>
    <input v-model.number="score" type="number" min="0" max="100" />
    <p v-if="score >= 90">рҹҸҶ Excellent! Grade: A</p>
    <p v-else-if="score >= 80">рҹ‘Қ Good! Grade: B</p>
    <p v-else-if="score >= 60">вң… Pass! Grade: C</p>
    <p v-else>вқҢ Fail. Keep trying!</p>

    <!-- v-show for frequent toggle -->
    <button @click="showDetails = !showDetails">
      {{ showDetails ? 'Hide' : 'Show' }} Details
    </button>
    <div v-show="showDetails" class="details">
      This content uses v-show (CSS toggle, frequent switching)
    </div>

    <!-- template + v-if for multiple elements -->
    <template v-if="showDetails">
      <p>Detail line 1</p>
      <p>Detail line 2</p>
    </template>

    <!-- 2. v-for list rendering -->
    <h2>List Rendering - Task Manager</h2>
    <input v-model="newTask" @keyup.enter="addTask" placeholder="Add a task" />
    <button @click="addTask">Add</button>

    <!-- Filter buttons using computed -->
    <div class="filters">
      <button v-for="f in filters" :key="f.value"
        :class="{ active: currentFilter === f.value }"
        @click="currentFilter = f.value">
        {{ f.label }}
      </button>
    </div>

    <!-- Task list with v-for -->
    <ul>
      <li v-for="task in filteredTasks" :key="task.id"
        :class="{ done: task.done }">
        <input type="checkbox" v-model="task.done" />
        <span>{{ task.text }}</span>
        <button @click="removeTask(task.id)">вң•</button>
      </li>
    </ul>
    <p>Total: {{ tasks.length }}, Done: {{ doneCount }}, Left: {{ leftCount }}</p>

    <!-- 3. v-for with objects -->
    <h2>Object Iteration</h2>
    <div v-for="(value, key) in userInfo" :key="key">
      <strong>{{ key }}:</strong> {{ value }}
    </div>

    <!-- 4. v-for range -->
    <h2>Star Rating</h2>
    <div>
      <span v-for="n in 5" :key="n"
        @click="rating = n"
        :style="{ cursor: 'pointer', opacity: n <= rating ? 1 : 0.3 }">
        вӯҗ
      </span>
      <span> {{ rating }}/5</span>
    </div>
  </div>
</template>

<script setup>
import { ref, computed } from 'vue'

// --- Conditional rendering ---
const score = ref(85)
const showDetails = ref(false)

// --- List rendering ---
let nextId = 4
const newTask = ref('')
const tasks = ref([
  { id: 1, text: 'Learn Vue 3 basics', done: true },
  { id: 2, text: 'Build a project', done: false },
  { id: 3, text: 'Deploy to production', done: false }
])

const filters = [
  { value: 'all', label: 'All' },
  { value: 'active', label: 'Active' },
  { value: 'done', label: 'Done' }
]
const currentFilter = ref('all')

// Computed: filter tasks based on current filter
const filteredTasks = computed(() => {
  switch (currentFilter.value) {
    case 'active': return tasks.value.filter(t => !t.done)
    case 'done': return tasks.value.filter(t => t.done)
    default: return tasks.value
  }
})

const doneCount = computed(() => tasks.value.filter(t => t.done).length)
const leftCount = computed(() => tasks.value.filter(t => !t.done).length)

function addTask() {
  const text = newTask.value.trim()
  if (!text) return
  tasks.value.push({ id: nextId++, text, done: false })
  newTask.value = ''
}

function removeTask(id) {
  tasks.value = tasks.value.filter(t => t.id !== id)
}

// --- Object iteration ---
const userInfo = ref({
  name: 'Vue Developer',
  level: 'Intermediate',
  framework: 'Vue 3',
  experience: '2 years'
})

// --- Star rating ---
const rating = ref(3)
</script>
