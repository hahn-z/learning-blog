概念讲解

什么是 Vue 插件？

插件是自包含的功能模块，通过 app.use() 注册时向 Vue 应用添加全局功能。插件可以：

  • 注册全局组件
  • 注册全局指令
  • 提供 provide/inject 服务
  • 添加全局属性
  • 执行初始化逻辑

    • 插件的结构

    // Plugin pattern
const myPlugin = {
  install(app, options) {
    // app: Vue app instance
    // options: user-provided config
  }
}

// Usage: app.use(myPlugin, { theme: 'dark' })

    插件 vs 组件库

    概念 范围 例子
    插件 添加功能（组件+指令+服务） vuetify, vue-router
    组件库 主要是 UI 组件 element-plus
    指令库 主要是指令 @vueuse/integrations

    完整代码

    // plugins/toast/index.js — Toast notification plugin
import ToastContainer from './ToastContainer.vue'
import { createApp, ref } from 'vue'

// Plugin state (shared across all toast instances)
const toasts = ref([])
let toastId = 0

function addToast(message, options = {}) {
  const id = toastId++
  const toast = {
    id,
    message,
    type: options.type || 'info',
    duration: options.duration || 3000
  }
  toasts.value.push(toast)

  // Auto-remove after duration
  if (toast.duration > 0) {
    setTimeout(() => removeToast(id), toast.duration)
  }

  return id
}

function removeToast(id) {
  toasts.value = toasts.value.filter(t => t.id !== id)
}

// Exported toast service
export function useToast() {
  return {
    success: (msg, opts) => addToast(msg, { ...opts, type: 'success' }),
    error: (msg, opts) => addToast(msg, { ...opts, type: 'error' }),
    info: (msg, opts) => addToast(msg, { ...opts, type: 'info' }),
    warning: (msg, opts) => addToast(msg, { ...opts, type: 'warning' }),
    remove: removeToast
  }
}

// Plugin install function
export default {
  install(app, options = {}) {
    // 1. Register toast container component
    app.component('ToastContainer', ToastContainer)

    // 2. Provide toast service globally
    app.provide('toast', useToast())

    // 3. Global property (for Options API)
    app.config.globalProperties.$toast = useToast()
  }
}

    // plugins/loading/index.js — Loading directive plugin
export default {
  install(app) {
    app.directive('loading', {
      mounted(el, binding) {
        // Create loading overlay
        const overlay = document.createElement('div')
        overlay.className = 'v-loading-overlay'
        overlay.innerHTML = '<div class="spinner"></div>'
        el.style.position = 'relative'
        el._loadingOverlay = overlay

        if (binding.value) {
          el.appendChild(overlay)
        }
      },

      updated(el, binding) {
        if (binding.value && !el.contains(el._loadingOverlay)) {
          el.appendChild(el._loadingOverlay)
        } else if (!binding.value && el.contains(el._loadingOverlay)) {
          el.removeChild(el._loadingOverlay)
        }
      },

      unmounted(el) {
        el._loadingOverlay = null
      }
    })
  }
}
    // plugins/i18n/index.js — Simple i18n plugin
import { ref, computed } from 'vue'

const locale = ref('zh')
const messages = ref({})

export function useI18n() {
  const t = (key) => {
    return messages.value[locale.value]?.[key] || key
  }
  const setLocale = (newLocale) => {
    locale.value = newLocale
  }
  return { t, locale: computed(() => locale.value), setLocale }
}

export default {
  install(app, options = {}) {
    // Set initial messages
    if (options.messages) {
      messages.value = options.messages
    }
    if (options.locale) {
      locale.value = options.locale
    }

    // Provide globally
    app.provide('i18n', useI18n())
    app.config.globalProperties.$t = (key) =>
      messages.value[locale.value]?.[key] || key
  }
}
    // main.js — Using all plugins
import { createApp } from 'vue'
import App from './App.vue'

// Plugins
import ToastPlugin from './plugins/toast'
import LoadingPlugin from './plugins/loading'
import I18nPlugin from './plugins/i18n'

const app = createApp(App)

app.use(ToastPlugin)
app.use(LoadingPlugin)
app.use(I18nPlugin, {
  locale: 'zh',
  messages: {
    zh: { hello: '你好', goodbye: '再见' },
    en: { hello: 'Hello', goodbye: 'Goodbye' }
  }
})

app.mount('#app')
    <!-- App.vue — Using plugins in components -->
<script setup>
import { useToast } from './plugins/toast'
import { useI18n } from './plugins/i18n'
import { ref } from 'vue'

const toast = useToast()
const { t, setLocale } = useI18n()
const loading = ref(false)

async function saveData() {
  loading.value = true
  try {
    await new Promise(r => setTimeout(r, 2000))
    toast.success('保存成功！')
  } catch {
    toast.error('保存失败')
  } finally {
    loading.value = false
  }
}
</script>

<template>
  <div v-loading="loading">
    <h1>{{ t('hello') }}</h1>
    <button @click="setLocale('en')">English</button>
    <button @click="setLocale('zh')">中文</button>
    <button @click="saveData">Save</button>
    <button @click="toast.info('Just a message')">Show Toast</button>
  </div>
  <ToastContainer />
</template>

    注意事项

    注意点 说明
    install 必须有 插件必须有 install 方法才能用 app.use()
    app.use 自动防重 同一插件多次 use 只执行一次 install
    provide/inject 优先 比 globalProperties 更适合 Composition API
    Tree-shaking 导出单独的 composable（如 useToast）便于按需引入
    peerDependencies 插件包的 vue 应设为 peerDependency，不打包进产物

    避坑指南

    ❌ 错误做法 ✅ 正确做法
    所有功能塞进一个插件 按功能拆分多个插件，按需注册
    用 globalProperties 挂载所有东西 provide/inject + composable 更类型安全
    插件中硬编码配置 install(app, options) 接受用户配置
    不清理副作用 提供卸载机制或文档说明
    打包时包含 vue vue 设为 peerDependency，external 处理
