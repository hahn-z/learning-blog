## 概念讲解

### 后台管理系统架构

后台管理系统是前端开发中最常见的项目类型，核心特征：

- **布局固定**：侧边栏 + 顶部导航 + 主内容区
- **权限驱动**：菜单、路由、按钮按角色动态显示
- **数据密集**：大量表格、表单、图表
- **CRUD 为主**：增删改查是核心操作

### 核心模块

| 模块 | 功能 | 技术 |
|------|------|------|
| 布局框架 | 侧边栏 + 顶栏 + 内容区 | CSS Grid / Flex |
| 动态菜单 | 按权限渲染菜单 | 递归组件 + 路由元信息 |
| 权限路由 | 动态添加路由 | router.addRoute() |
| 数据表格 | 分页/排序/筛选 | Element Plus Table |
| 表单 | 验证/联动/动态表单 | Element Plus Form |
| 数据看板 | 图表 + 统计卡片 | ECharts |
| 全局状态 | 用户/菜单/设置 | Pinia |

### 动态路由注册

```ts
// src/router/dynamic.ts
import type { RouteRecordRaw } from 'vue-router'
import AdminLayout from '@/layouts/AdminLayout.vue'

// All available admin routes (lazy loaded)
const adminRoutes: RouteRecordRaw[] = [
  {
    path: '/admin',
    component: AdminLayout,
    redirect: '/admin/dashboard',
    children: [
      { path: 'dashboard', name: 'Dashboard', component: () => import('@/views/admin/Dashboard.vue'), meta: { title: 'Dashboard', icon: 'DataLine' } },
      { path: 'users', name: 'Users', component: () => import('@/views/admin/Users.vue'), meta: { title: 'User Mgmt', icon: 'User', role: 'admin' } },
      { path: 'posts', name: 'Posts', component: () => import('@/views/admin/Posts.vue'), meta: { title: 'Post Mgmt', icon: 'Document' } },
      { path: 'settings', name: 'Settings', component: () => import('@/views/admin/Settings.vue'), meta: { title: 'Settings', icon: 'Setting', role: 'admin' } },
    ]
  }
]

// Register routes based on user role
export function registerRoutes(role: string) {
  const router = useRouter()

  function filterRoutes(routes: RouteRecordRaw[]): RouteRecordRaw[] {
    return routes.map(route => {
      const filtered = { ...route }
      if (route.children) {
        filtered.children = route.children.filter(child => {
          const requiredRole = child.meta?.role
          return !requiredRole || role === requiredRole
        })
      }
      return filtered
    })
  }

  const allowed = filterRoutes(adminRoutes)
  allowed.forEach(route => router.addRoute(route))
}
```

<template>
  <div class="user-management">
    <!-- Search bar -->
    <el-card class="search-bar">
      <el-form :model="searchForm" inline>
        <el-form-item label="Keyword">
          <el-input v-model="searchForm.keyword" placeholder="Name or email" clearable />

          <span>User List</span>

        </div>
      </template>

      <el-table :data="tableData" v-loading="loading" stripe border>
        <el-table-column prop="id" label="ID" width="80" />
        <el-table-column prop="name" label="Name" />
        <el-table-column prop="email" label="Email" />
        <el-table-column prop="role" label="Role" width="100">
          <template #default="{ row }">
            <el-tag :type="row.role === 'admin' ? 'danger' : 'info'">{{ row.role }}</el-tag>
          </template>
        </el-table-column>
        <el-table-column prop="createdAt" label="Created" width="180" />
        <el-table-column label="Actions" width="200" fixed="right">
          <template #default="{ row }">
            <el-button size="small" @click="openDialog(row)">Edit</el-button>
            <el-button size="small" type="danger" @click="handleDelete(row.id)">Delete</el-button>
          </template>
        </el-table-column>
      </el-table>

      <el-pagination
        v-model:current-page="page"
        v-model:page-size="pageSize"
        :total="total"
        :page-sizes="[10, 20, 50]"
        layout="total, sizes, prev, pager, next"
        @current-change="fetchData"
        @size-change="() => { page = 1; fetchData() }"
      />
    </el-card>

    <!-- Create/Edit Dialog -->
    <el-dialog v-model="dialogVisible" :title="editingUser ? 'Edit User' : 'Add User'" width="500px">
      <el-form ref="formRef" :model="form" :rules="rules" label-width="80px">
        <el-form-item label="Name" prop="name">
          <el-input v-model="form.name" />
        </el-form-item>
        <el-form-item label="Email" prop="email">
          <el-input v-model="form.email" />
        </el-form-item>
        <el-form-item label="Role" prop="role">
          <el-select v-model="form.role">
            <el-option label="Admin" value="admin" />
            <el-option label="User" value="user" />
          </el-select>
        </el-form-item>
        <el-form-item v-if="!editingUser" label="Password" prop="password">
          <el-input v-model="form.password" type="password" />
        </el-form-item>
      </el-form>
      <template #footer>
        <el-button @click="dialogVisible = false">Cancel</el-button>
        <el-button type="primary" :loading="submitting" @click="handleSubmit">Confirm</el-button>
      </template>
    </el-dialog>
  </div>
</template>

<script setup lang="ts">
import { ref, reactive, onMounted } from 'vue'
import { ElMessage, ElMessageBox } from 'element-plus'
import type { FormInstance } from 'element-plus'
import { userApi } from '@/api/user'

const loading = ref(false)
const tableData = ref<any[]>([])
const page = ref(1)
const pageSize = ref(10)
const total = ref(0)
const searchForm = reactive({ keyword: '', role: '' })

const dialogVisible = ref(false)
const submitting = ref(false)
const editingUser = ref<any>(null)
const formRef = ref<FormInstance>()
const form = reactive({ name: '', email: '', role: 'user', password: '' })
const rules = {
  name: [{ required: true, message: 'Required', trigger: 'blur' }],
  email: [{ required: true, type: 'email', message: 'Valid email required', trigger: 'blur' }],
  role: [{ required: true, message: 'Required', trigger: 'change' }],
  password: [{ required: true, min: 6, message: 'Min 6 chars', trigger: 'blur' }]
}

async function fetchData() {
  loading.value = true
  try {
    const res = await userApi.getList({ page: page.value, pageSize: pageSize.value, ...searchForm })
    tableData.value = res.list
    total.value = res.total
  } finally {
    loading.value = false
  }
}

function handleSearch() { page.value = 1; fetchData() }
function handleReset() { searchForm.keyword = ''; searchForm.role = ''; handleSearch() }

function openDialog(user?: any) {
  editingUser.value = user || null
  if (user) {
    Object.assign(form, { name: user.name, email: user.email, role: user.role, password: '' })
  } else {
    Object.assign(form, { name: '', email: '', role: 'user', password: '' })
  }
  dialogVisible.value = true
}

async function handleSubmit() {
  await formRef.value?.validate()
  submitting.value = true
  try {
    if (editingUser.value) {
      await userApi.update(editingUser.value.id, form)
      ElMessage.success('Updated')
    } else {
      await userApi.create(form)
      ElMessage.success('Created')
    }
    dialogVisible.value = false
    fetchData()
  } finally {
    submitting.value = false
  }
}

async function handleDelete(id: number) {
  await ElMessageBox.confirm('Confirm delete?', 'Warning', { type: 'warning' })
  await userApi.delete(id)
  ElMessage.success('Deleted')
  fetchData()
}

onMounted(fetchData)
</script>

## 注意事项

| 事项 | 说明 | 建议 |
|------|------|------|
| 路由注册 | addRoute 只能注册一次 | 用 flag 防止重复注册 |
| 菜单缓存 | keep-alive 缓存页面 | 表格页需要刷新策略 |
| 大数据量 | 表格数据多时卡顿 | 虚拟滚动或分页 |
| 表单重置 | Dialog 关闭时重置 | @closed 事件中 resetFields |
| 权限粒度 | 菜单 vs 按钮 | 菜单用路由，按钮用指令 |
| 响应式 | 后台也需要适配 | 侧边栏可折叠 |
