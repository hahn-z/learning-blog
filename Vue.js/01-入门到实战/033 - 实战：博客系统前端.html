博客系统前端架构

博客前端是一个经典的 CRUD 应用，但涉及的技术点非常全面：

  • 内容展示：Markdown 渲染、代码高亮、目录导航
  • 数据筛选：分类、标签、搜索、分页
  • 状态管理：文章缓存、用户状态、UI 状态
  • 性能优化：组件懒加载、图片懒加载、SEO

    • 技术选型

    模块 技术 理由
    框架 Vue 3 + TypeScript 类型安全、Composition API

    UI 组件 Element Plus 后台管理场景
    Markdown markdown-it + highlight.js 可扩展、高性能

    文章 Store

    // src/stores/post.ts
import { defineStore } from 'pinia'
import { ref } from 'vue'
import { postApi, type Post, type PostListParams } from '@/api/post'

export const usePostStore = defineStore('post', () => {
  // Cache fetched posts by slug
  const cache = ref<Map<string, Post>>(new Map())
  const currentPost = ref<Post | null>(null)

  async function fetchList(params: PostListParams) {
    const res = await postApi.getList(params)
    return res
  }

  async function fetchBySlug(slug: string) {
    // Check cache first
    if (cache.value.has(slug)) {
      currentPost.value = cache.value.get(slug)!
      return currentPost.value
    }
    const post = await postApi.getBySlug(slug)
    cache.value.set(slug, post)
    currentPost.value = post
    return post
  }

  function clearCache() {
    cache.value.clear()
  }

  return { currentPost, fetchList, fetchBySlug, clearCache }
})

    Markdown 渲染工具

    // src/utils/markdown.ts
import MarkdownIt from 'markdown-it'
import hljs from 'highlight.js'
import toc from 'markdown-it-table-of-contents'

const md = new MarkdownIt({
  html: true,
  linkify: true,
  // Custom code highlighting
  highlight(str: string, lang: string) {
    if (lang && hljs.getLanguage(lang)) {
      try {
        return `<pre class="hljs"><code>${hljs.highlight(str, { language: lang }).value}</code></pre>`
      } catch (_) { /* fallback */ }
    }
    return `<pre class="hljs"><code>${md.utils.escapeHtml(str)}</code></pre>`
  }
})

// Add heading IDs for TOC
md.use(toc)

// Render markdown to HTML + extract headings for TOC
export function renderMarkdown(content: string) {
  const html = md.render(content)
  const headings = extractHeadings(html)
  return { html, headings }
}

// Extract h2/h3 headings for sidebar navigation
function extractHeadings(html: string) {
  const regex = /<h([2-3])[^>]*id="([^"]*)"[^>]*>(.*?)<\/h\1>/g
  const headings: { level: number; id: string; text: string }[] = []
  let match
  while ((match = regex.exec(html)) !== null) {
    headings.push({
      level: parseInt(match[1]),
      id: match[2],
      text: match[3].replace(/<[^>]*>/g, '') // strip HTML tags
    })
  }
  return headings
}

    文章详情页

    <!-- src/views/PostDetail.vue -->
<template>
  <div class="post-detail" v-loading="loading">
    <article v-if="post" class="post-content">
      <header class="post-header">
        <h1 class="post-title">{{ post.title }}</h1>
        <div class="post-meta">
          <span>{{ formatDate(post.createdAt) }}</span>
          <span>·</span>
          <router-link :to="`/category/${post.category.slug}`">
            {{ post.category.name }}
          </router-link>
          <span>·</span>
          <span>{{ post.readingTime }} min read</span>
        </div>
        <div class="post-tags">
          <router-link
            v-for="tag in post.tags"
            :key="tag.slug"
            :to="`/tag/${tag.slug}`"
            class="tag"
          >
            #{{ tag.name }}
          </router-link>
        </div>
      </header>

      <!-- Markdown rendered content -->
      <div class="post-body markdown-body" v-html="renderedHtml" />

      <footer class="post-footer">
        <h3>Related Posts</h3>
        <div class="related-posts">
          <router-link
            v-for="related in relatedPosts"
            :key="related.slug"
            :to="`/post/${related.slug}`"
            class="related-card"
          >
            <h4>{{ related.title }}</h4>
            <p>{{ related.excerpt }}</p>
          </router-link>
        </div>
      </footer>
    </article>

    <!-- TOC Sidebar -->
    <aside v-if="headings.length" class="toc-sidebar">
      <h3>Table of Contents</h3>
      <nav>
        <a
          v-for="h in headings"
          :key="h.id"
          :href="`#${h.id}`"
          :class="['toc-link', `toc-h${h.level}`, { active: activeHeading === h.id }]"
        >
          {{ h.text }}
        </a>
      </nav>
    </aside>
  </div>
</template>

<script setup lang="ts">
import { ref, computed, watch, onMounted, onUnmounted } from 'vue'
import { useRoute } from 'vue-router'
import { usePostStore } from '@/stores/post'
import { renderMarkdown } from '@/utils/markdown'
import { postApi } from '@/api/post'

const route = useRoute()
const postStore = usePostStore()

const loading = ref(false)
const relatedPosts = ref<any[]>([])
const activeHeading = ref('')

// Fetch post data
const post = computed(() => postStore.currentPost)

// Render markdown
const renderedHtml = computed(() => {
  if (!post.value?.content) return ''
  return renderMarkdown(post.value.content).html
})

const headings = computed(() => {
  if (!post.value?.content) return []
  return renderMarkdown(post.value.content).headings
})

// Load post when slug changes
async function loadPost(slug: string) {
  loading.value = true
  try {
    await postStore.fetchBySlug(slug)
    // Load related posts
    const related = await postApi.getRelated(slug)
    relatedPosts.value = related
  } finally {
    loading.value = false
  }
}

watch(() => route.params.slug, (slug) => {
  if (slug) loadPost(slug as string)
}, { immediate: true })

// Track active heading on scroll
function handleScroll() {
  for (const h of headings.value) {
    const el = document.getElementById(h.id)
    if (el && el.getBoundingClientRect().top > 100) break
    activeHeading.value = h.id
  }
}

onMounted(() => window.addEventListener('scroll', handleScroll))
onUnmounted(() => window.removeEventListener('scroll', handleScroll))

function formatDate(date: string) {
  return new Date(date).toLocaleDateString('en-US', {
    year: 'numeric', month: 'long', day: 'numeric'
  })
}
</script>

    注意事项

    事项 说明 建议
    XSS 安全 v-html 有注入风险 后端对 Markdown 内容做 sanitize
    SEO SPA 不利 SEO 考虑 SSR 或预渲染
    图片优化 博客图片可能很大 懒加载 + WebP 格式
    代码高亮 highlight.js 包较大 按需引入语言包
    TOC 定位 标题可能动态变化 用 IntersectionObserver
    缓存策略 文章内容不常变 Pinia 缓存 + localStorage

    避坑指南

    • ❌ 每次进入文章都重新请求 → ✅ Pinia 缓存已加载的文章
    • ❌ v-html 渲染未过滤的内容 → ✅ 后端 sanitize 或使用 DOMPurify
    • ❌ 引入全部 highlight.js 语言 → ✅ 只引入需要的语言包
    • ❌ TOC 用 scroll 事件监听 → ✅ IntersectionObserver 性能更好
    • ❌ 文章图片不处理 → ✅ 懒加载 + 点击放大
