阶段二：进阶（2-4 周）

// Composable with TypeScript
import { ref, computed, onMounted, onUnmounted } from 'vue'

interface MousePosition {
  x: number
  y: number
}

export function useMouse(): { pos: MousePosition; isInViewport: boolean } {
  const pos = ref<MousePosition>({ x: 0, y: 0 })

  const isInViewport = computed(() =>
    pos.value.x >= 0 && pos.value.x <= window.innerWidth &&
    pos.value.y >= 0 && pos.value.y <= window.innerHeight
  )

  function update(event: MouseEvent) {
    pos.value = { x: event.clientX, y: event.clientY }
  }

  onMounted(() => window.addEventListener('mousemove', update))
  onUnmounted(() => window.removeEventListener('mousemove', update))

  return { pos: pos.value, isInViewport: isInViewport.value }
}

阶段三：高级（4-8 周）

// Generic composable with advanced patterns
import { ref, watch, type Ref, computed } from 'vue'

export function useAsyncState<T>(
  promise: Ref<Promise<T>> | (() => Promise<T>),
  defaultValue: T
) {
  const state = ref<T>(defaultValue) as Ref<T>
  const isLoading = ref(false)
  const error = ref<Error | null>(null)

  async function execute() {
    isLoading.value = true
    error.value = null
    try {
      const p = typeof promise === 'function' ? promise() : promise.value
      state.value = await p
    } catch (e: any) {
      error.value = e
    } finally {
      isLoading.value = false
    }
  }

  const isEmpty = computed(() => {
    if (Array.isArray(state.value)) return state.value.length === 0
    return state.value === null || state.value === undefined
  })

  return { state, isLoading, error, isEmpty, execute }
}

阶段四：架构师（持续）

// Plugin architecture pattern
import type { App, Plugin } from 'vue'

interface BlogModuleOptions {
  apiBase: string
  theme: 'light' | 'dark'
  locale: string
}

// Plugin pattern for module registration
export const BlogModule: Plugin = {
  install(app: App, options: BlogModuleOptions) {
    // Provide global config
    app.provide('blogConfig', options)

    // Register global components
    app.component('BlogPost', defineAsyncComponent(() =>
      import('./components/BlogPost.vue')
    ))
    app.component('BlogList', defineAsyncComponent(() =>
      import('./components/BlogList.vue')
    ))

    // Register directives
    app.directive('lazy', {
      mounted(el, binding) {
        const observer = new IntersectionObserver(([entry]) => {
          if (entry.isIntersecting) {
            el.src = binding.value
            observer.disconnect()
          }
        })
        observer.observe(el)
      }
    })
  }
}

// Usage: app.use(BlogModule, { apiBase: '/api', theme: 'light' })
