  return value.toFixed(2);
}

// ============================================
// Function Types and Overloads
// ============================================

// --- 1. Basic Function Types ---
function greet(name: string): string {
  return `Hello, ${name}!`;
}

// Function type alias
type Callback = (data: string) => void;
type AsyncFn = (url: string) => Promise<Response>;

// --- 2. Optional & Default Parameters ---
function search(
  query: string,
  limit: number = 10,
  offset?: number
): string[] {
  const results: string[] = [];
  // Simulate search results
  for (let i = 0; i < limit; i++) {
    results.push(`${query}-result-${i}`);
  }
  return results;
}

console.log(search('ts'));           // 10 results
console.log(search('ts', 5));        // 5 results
console.log(search('ts', 5, 10));    // 5 results with offset

// --- 3. Rest Parameters ---
function log(level: string, ...messages: string[]): void {
  console.log(`[${level}]`, messages.join(' '));
}

log('INFO', 'Server', 'started', 'on', 'port', '3000');

// --- 4. Function Overloading ---

// Example 1: Type-safe event handler
function on(event: 'click', handler: (e: MouseEvent) => void): void;
function on(event: 'input', handler: (e: InputEvent) => void): void;
function on(event: string, handler: (e: Event) => void): void {
  console.log(`Registered handler for ${event}`);
}

on('click', (e) => {
  console.log('Click at', e.clientX, e.clientY); // e is MouseEvent
});

// Example 2: Utility function with overloads
function pluck<T, K extends keyof T>(obj: T, key: K): T[K];
function pluck<T>(obj: T, keys: (keyof T)[]): unknown[];
function pluck<T>(obj: T, keyOrKeys: keyof T | (keyof T)[]): unknown {
  if (Array.isArray(keyOrKeys)) {
    return keyOrKeys.map(k => obj[k]);
  }
  return obj[keyOrKeys];
}

const user = { name: 'Alice', age: 30, email: 'a@b.com' };
const name = pluck(user, 'name');      // type: string
const vals = pluck(user, ['name', 'age']); // type: unknown[]

// --- 5. 'this' Parameter ---
interface DB {
  users: string[];
  addUser(this: DB, name: string): void;
}

const db: DB = {
  users: [],
  addUser(name: string) {
    this.users.push(name);
  }
};

db.addUser('Alice');
console.log(db.users); // ['Alice']

// --- 6. Assertion Functions ---
function assertDefined<T>(value: T | undefined | null, msg?: string): asserts value is T {
  if (value == null) throw new Error(msg ?? 'Value is null/undefined');
}

let maybeName: string | undefined = 'Alice';
assertDefined(maybeName);
console.log(maybeName.toUpperCase()); // TS knows it's string now

// --- 7. Predicate Functions ---
function isString(value: unknown): value is string {
  return typeof value === 'string';
}

const items: unknown[] = ['hello', 42, 'world', true];
const strings = items.filter(isString); // type: string[]
console.log(strings); // ['hello', 'world']

// --- 8. Constructor Type ---
type Constructor<T> = new (...args: any[]) => T;

function createInstance<T>(Ctor: Constructor<T>, ...args: any[]): T {
  return new Ctor(...args);
}

class Person {
  constructor(public name: string) {}
}

const p = createInstance(Person, 'Alice');
console.log(p.name); // "Alice"

console.log('All function type examples completed!');
