一、概念讲解

设计一个类型安全的工具库，需要将TypeScript的高级类型特性发挥到极致。本文通过实际案例，展示如何设计既灵活又类型安全的工具函数。

设计原则

  1. 类型推导优先：让用户不需要手动标注类型
    1. 错误在编译时暴露：不可能的用法应该编译失败
      1. 类型守卫结合：编译时+运行时双重保障
        1. API最小化：少量精心设计的函数胜过大量同质函数

// --- 1. Deep Utility Types ---
type DeepPartial<T> = T extends object
  ? { [K in keyof T]?: DeepPartial<T[K]> }
  : T;

type DeepReadonly<T> = T extends object
  ? { readonly [K in keyof T]: DeepReadonly<T[K]> }
  : T;

type DeepRequired<T> = T extends object
  ? { [K in keyof T]-?: DeepRequired<T[K]> }
  : T;

// --- 2. Key/Value Filtering ---
type PickByValue<T, V> = {
  [K in keyof T as T[K] extends V ? K : never]: T[K];
};

type OmitByValue<T, V> = {
  [K in keyof T as T[K] extends V ? never : K]: T[K];
};

type RequiredKeys<T> = {
  [K in keyof T]-?: {} extends Pick<T, K> ? never : K;
}[keyof T];

type OptionalKeys<T> = {
  [K in keyof T]-?: {} extends Pick<T, K> ? K : never;
}[keyof T];

// --- 3. Type-safe Object Functions ---
function typedPick<T extends object, K extends keyof T>(
  obj: T,
  keys: K[]
): Pick<T, K> {
  const result = {} as Pick<T, K>;
  for (const key of keys) {
    result[key] = obj[key];
  }
  return result;
}

function typedOmit<T extends object, K extends keyof T>(
  obj: T,
  keys: K[]
): Omit<T, K> {
  const result = { ...obj };
  for (const key of keys) {
    delete result[key];
  }
  return result;
}

// --- 4. Type-safe groupBy ---
function groupBy<T, K extends keyof T>(
  arr: T[],
  key: K
): Record<string, T[]> {
  return arr.reduce(
    (groups, item) => {
      const groupKey = String(item[key]);
      if (!groups[groupKey]) {
        groups[groupKey] = [];
      }
      groups[groupKey].push(item);
      return groups;
    },
    {} as Record<string, T[]>
  );
}

// --- 5. Type-safe compose/pipe ---
function compose<A, B, C>(
  f: (a: B) => C,
  g: (a: A) => B
): (a: A) => C;
function compose<A, B, C, D>(
  f: (a: C) => D,
  g: (b: B) => C,
  h: (a: A) => B
): (a: A) => D;
function compose(...fns: Function[]): Function {
  return (x: unknown) => fns.reduceRight((acc, fn) => fn(acc), x);
}

function pipe<A>(a: A): A;
function pipe<A, B>(a: A, ab: (a: A) => B): B;
function pipe<A, B, C>(a: A, ab: (a: A) => B, bc: (b: B) => C): C;
function pipe<A, B, C, D>(
  a: A,
  ab: (a: A) => B,
  bc: (b: B) => C,
  cd: (c: C) => D
): D;
function pipe(initial: unknown, ...fns: Function[]): unknown {
  return fns.reduce((acc, fn) => fn(acc), initial);
}

// --- 6. Type-safe EventEmitter (Advanced) ---
type EventMap = Record<string, unknown>;
type EventKey<T extends EventMap> = string & keyof T;

interface TypedEventEmitter<Events extends EventMap> {
  on<K extends EventKey<Events>>(
    event: K,
    listener: (payload: Events[K]) => void
  ): this;
  off<K extends EventKey<Events>>(
    event: K,
    listener: (payload: Events[K]) => void
  ): this;
  emit<K extends EventKey<Events>(event: K, payload: Events[K]): boolean;
}

class EventEmitter<Events extends EventMap> implements TypedEventEmitter<Events> {
  private listeners = new Map<string, Set<Function>>();

  on<K extends EventKey<Events>>(
    event: K,
    listener: (payload: Events[K]) => void
  ): this {
    if (!this.listeners.has(event)) {
      this.listeners.set(event, new Set());
    }
    this.listeners.get(event)!.add(listener);
    return this;
  }

  off<K extends EventKey<Events>>(
    event: K,
    listener: (payload: Events[K]) => void
  ): this {
    this.listeners.get(event)?.delete(listener);
    return this;
  }

  emit<K extends EventKey<Events>>(event: K, payload: Events[K]): boolean {
    const handlers = this.listeners.get(event);
    if (!handlers?.size) return false;
    handlers.forEach((fn) => fn(payload));
    return true;
  }
}

// --- 7. Type-safe Schema Validator ---
type SchemaType<S extends Schema> = S extends SchemaString
  ? string
  : S extends SchemaNumber
    ? number
    : S extends SchemaBoolean
      ? boolean
      : S extends SchemaObject<infer T>
        ? { [K in keyof T]: SchemaType<T[K]> }
        : never;

interface SchemaString {
  type: "string";
  min?: number;
  max?: number;
}

interface SchemaNumber {
  type: "number";
  min?: number;
  max?: number;
}

interface SchemaBoolean {
  type: "boolean";
}

interface SchemaObject<T extends Record<string, Schema>> {
  type: "object";
  properties: T;
  required?: (keyof T)[];
}

type Schema = SchemaString | SchemaNumber | SchemaBoolean | SchemaObject<Record<string, Schema>>;

function validate<S extends Schema>(
  value: unknown,
  schema: S
): value is SchemaType<S> {
  if (schema.type === "string") {
    if (typeof value !== "string") return false;
    if (schema.min !== undefined && value.length < schema.min) return false;
    if (schema.max !== undefined && value.length > schema.max) return false;
    return true;
  }
  if (schema.type === "number") {
    if (typeof value !== "number") return false;
    if (schema.min !== undefined && value < schema.min) return false;
    if (schema.max !== undefined && value > schema.max) return false;
    return true;
  }
  if (schema.type === "boolean") {
    return typeof value === "boolean";
  }
  if (schema.type === "object") {
    if (typeof value !== "object" || value === null) return false;
    const obj = value as Record<string, unknown>;
    for (const [key, propSchema] of Object.entries(schema.properties)) {
      if (!(key in obj)) {
        if (schema.required?.includes(key)) return false;
        continue;
      }
      if (!validate(obj[key], propSchema as Schema)) return false;
    }
    return true;
  }
  return false;
}

// --- 8. Usage Example ---
interface Product {
  id: string;
  name: string;
  price: number;
  category: string;
}

const products: Product[] = [
  { id: "1", name: "TypeScript Guide", price: 49.99, category: "Books" },
  { id: "2", name: "React Handbook", price: 39.99, category: "Books" },
  { id: "3", name: "VS Code", price: 0, category: "Tools" },
];

// groupBy
const byCategory = groupBy(products, "category");
console.log("Books:", byCategory["Books"]?.length); // 2

// typedPick
const preview = typedPick(products[0], ["id", "name"]);
console.log("Preview:", preview);

// Schema validation
const userSchema: SchemaObject<{
  name: SchemaString;
  age: SchemaNumber;
}> = {
  type: "object",
  properties: {
    name: { type: "string", min: 1 },
    age: { type: "number", min: 0, max: 150 },
  },
  required: ["name", "age"],
};

const testData = { name: "Alice", age: 30 };
if (validate(testData, userSchema)) {
  // TypeScript knows testData has { name: string; age: number }
  console.log(`Valid: ${testData.name}, age ${testData.age}`);
}

          五、注意事项

          要点 说明 建议
          递归类型 DeepPartial 等可能很慢 限制递归深度，或用具体类型
          函数重载 compose/pipe需要大量重载 考虑用更简单的API设计
          类型推导 复杂类型可能导致推导失败 适当添加手动注解
          运行时开销 类型复杂不代表运行时慢 类型在编译时擦除
          Schema验证 手写验证器容易有bug 生产环境用Zod等成熟库
