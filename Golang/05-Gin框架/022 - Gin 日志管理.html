022 - Gin 日志管理

难度： ⭐⭐⭐（较高）
前置知识： Gin 中间件、Go io 接口、结构化日志概念
预计阅读时间： 25 分钟

一、概念讲解

日志是生产环境的"黑匣子"。Gin 默认的 gin.Logger() 只输出文本格式日志，在生产环境中远远不够。我们需要结构化日志、请求追踪、日志轮转和性能分级。

日志层次体系：

层级 用途 示例
请求日志 每个 HTTP 请求的进出记录 method, path, status, latency
业务日志 业务逻辑关键节点 order_created, payment_success
错误日志 异常和错误堆栈 db_connection_failed
审计日志 安全敏感操作 user_login, permission_change

技术选型对比：

方案 性能 结构化 适用场景
log/slog (Go 1.21+) 中 ✅ 原生 标准库优先、新项目推荐
uber-go/zap 极高 ✅ 高并发、高性能要求
gin.Logger() 低 ❌ 仅开发环境

三、完整代码

v1：slog 结构化日志 + 请求 ID

// main.go - v1 slog integration with request ID
package main

import (
	"log/slog"
	"net/http"
	"os"
	"time"

	"github.com/gin-gonic/gin"
	"github.com/google/uuid"
)

func main() {
	// Configure structured JSON logger
	logger := slog.New(slog.NewJSONHandler(os.Stdout, &slog.HandlerOptions{
		Level: slog.LevelInfo,
	}))
	slog.SetDefault(logger)

	r := gin.New() // no default middleware
	r.Use(gin.Recovery())
	r.Use(requestIDMiddleware())
	r.Use(slogRequestMiddleware())

	r.GET("/ping", func(c *gin.Context) {
		slog.InfoContext(c.Request.Context(), "ping received",
			"path", c.Request.URL.Path)
		c.JSON(http.StatusOK, gin.H{"message": "pong"})
	})

	r.GET("/users/:id", func(c *gin.Context) {
		slog.InfoContext(c.Request.Context(), "fetching user",
			"user_id", c.Param("id"))
		c.JSON(http.StatusOK, gin.H{
			"id": c.Param("id"), "name": "Alice",
		})
	})

	r.Run(":8080")
}

// requestIDMiddleware generates and propagates a request ID
func requestIDMiddleware() gin.HandlerFunc {
	return func(c *gin.Context) {
		requestID := c.GetHeader("X-Request-ID")
		if requestID == "" {
			requestID = uuid.New().String()
		}
		c.Set("request_id", requestID)
		c.Header("X-Request-ID", requestID)
		c.Next()
	}
}

// slogRequestMiddleware logs every request with slog
func slogRequestMiddleware() gin.HandlerFunc {
	return func(c *gin.Context) {
		start := time.Now()
		path := c.Request.URL.Path
		query := c.Request.URL.RawQuery

		c.Next()

		latency := time.Since(start)
		slog.Info("request",
			"request_id", c.GetString("request_id"),
			"status", c.Writer.Status(),
			"method", c.Request.Method,
			"path", path,
			"query", query,
			"latency_ms", latency.Milliseconds(),
			"client_ip", c.ClientIP(),
			"user_agent", c.Request.UserAgent(),
			"body_size", c.Writer.Size(),
			"errors", c.Errors.ByType(gin.ErrorTypePrivate).String(),
		)
	}
}

v2：zap + lumberjack 日志轮转

// main.go - v2 zap + lumberjack with rotation
package main

import (
	"net/http"
	"time"

	"github.com/gin-gonic/gin"
	"github.com/google/uuid"
	"go.uber.org/zap"
	"go.uber.org/zap/zapcore"
	"gopkg.in/natefinch/lumberjack.v2"
)

var logger *zap.Logger

func main() {
	initLogger()
	defer logger.Sync()

	r := gin.New()
	r.Use(gin.Recovery())
	r.Use(requestIDMiddleware())
	r.Use(zapRequestMiddleware())
	r.Use(slowRequestMiddleware(500*time.Millisecond))

	r.GET("/ping", func(c *gin.Context) {
		logger.Info("ping",
			zap.String("request_id", c.GetString("request_id")))
		c.JSON(http.StatusOK, gin.H{"message": "pong"})
	})

	r.GET("/heavy", func(c *gin.Context) {
		time.Sleep(800 * time.Millisecond) // simulate slow
		c.JSON(http.StatusOK, gin.H{"done": true})
	})

	r.Run(":8080")
}

// initLogger sets up zap with lumberjack rotation
func initLogger() {
	// Lumberjack for log rotation
	writer := &lumberjack.Logger{
		Filename:   "logs/app.log",
		MaxSize:    100, // MB per file
		MaxBackups: 10,  // keep 10 old files
		MaxAge:     30,  // days
		Compress:   true,
	}

	core := zapcore.NewCore(
		zapcore.NewJSONEncoder(zap.NewProductionEncoderConfig()),
		zapcore.AddSync(writer),
		zapcore.InfoLevel,
	)

	logger = zap.New(core, zap.AddCaller(), zap.AddStacktrace(zap.ErrorLevel))
}

func requestIDMiddleware() gin.HandlerFunc {
	return func(c *gin.Context) {
		requestID := c.GetHeader("X-Request-ID")
		if requestID == "" {
			requestID = uuid.New().String()
		}
		c.Set("request_id", requestID)
		c.Header("X-Request-ID", requestID)
		c.Next()
	}
}

func zapRequestMiddleware() gin.HandlerFunc {
	return func(c *gin.Context) {
		start := time.Now()
		c.Next()
		latency := time.Since(start)

		logger.Info("request",
			zap.String("request_id", c.GetString("request_id")),
			zap.String("method", c.Request.Method),
			zap.String("path", c.Request.URL.Path),
			zap.Int("status", c.Writer.Status()),
			zap.Duration("latency", latency),
			zap.String("client_ip", c.ClientIP()),
		)
	}
}

// slowRequestMiddleware logs warnings for requests exceeding threshold
func slowRequestMiddleware(threshold time.Duration) gin.HandlerFunc {
	return func(c *gin.Context) {
		start := time.Now()
		c.Next()
		latency := time.Since(start)

		if latency > threshold {
			logger.Warn("slow request detected",
				zap.String("request_id", c.GetString("request_id")),
				zap.String("method", c.Request.Method),
				zap.String("path", c.Request.URL.Path),
				zap.Duration("latency", latency),
				zap.Duration("threshold", threshold),
			)
		}
	}
}

v3：slog + zap 双轨 + 分级输出

// main.go - v3 dual logger with level-based output
package main

import (
	"io"
	"log/slog"
	"net/http"
	"os"
	"time"

	"github.com/gin-gonic/gin"
	"github.com/google/uuid"
	"go.uber.org/zap"
	"go.uber.org/zap/zapcore"
	"gopkg.in/natefinch/lumberjack.v2"
)

var zapLogger *zap.Logger

func main() {
	// Setup slog to stdout for development visibility
	slog.SetDefault(slog.New(slog.NewJSONHandler(os.Stdout, &slog.HandlerOptions{
		Level: slog.LevelDebug,
	})))

	// Setup zap to file for production persistence
	initZapLogger()
	defer zapLogger.Sync()

	r := gin.New()
	r.Use(gin.Recovery())
	r.Use(requestIDMiddleware())
	r.Use(dualLogMiddleware())
	r.Use(slowRequestMiddleware(500*time.Millisecond))

	r.GET("/ping", func(c *gin.Context) {
		slog.InfoContext(c.Request.Context(), "ping")
		c.JSON(http.StatusOK, gin.H{"message": "pong"})
	})

	r.GET("/error", func(c *gin.Context) {
		slog.ErrorContext(c.Request.Context(), "simulated error",
			"detail", "something went wrong")
		c.JSON(http.StatusInternalServerError, gin.H{"error": "internal"})
	})

	r.Run(":8080")
}

func initZapLogger() {
	infoWriter := &lumberjack.Logger{
		Filename: "logs/info.log", MaxSize: 100, MaxBackups: 5, Compress: true,
	}
	errorWriter := &lumberjack.Logger{
		Filename: "logs/error.log", MaxSize: 50, MaxBackups: 10, Compress: true,
	}

	// Info and above → info.log, Error and above → error.log
	infoCore := zapcore.NewCore(
		zapcore.NewJSONEncoder(zap.NewProductionEncoderConfig()),
		zapcore.AddSync(infoWriter),
		zapcore.InfoLevel,
	)
	errorCore := zapcore.NewCore(
		zapcore.NewJSONEncoder(zap.NewProductionEncoderConfig()),
		zapcore.AddSync(errorWriter),
		zapcore.ErrorLevel,
	)

	zapLogger = zap.New(zapcore.NewTee(infoCore, errorCore),
		zap.AddCaller(), zap.AddStacktrace(zap.ErrorLevel))
}

func requestIDMiddleware() gin.HandlerFunc {
	return func(c *gin.Context) {
		rid := c.GetHeader("X-Request-ID")
		if rid == "" {
			rid = uuid.New().String()
		}
		c.Set("request_id", rid)
		c.Header("X-Request-ID", rid)
		c.Next()
	}
}

func dualLogMiddleware() gin.HandlerFunc {
	return func(c *gin.Context) {
		start := time.Now()
		c.Next()
		latency := time.Since(start)
		rid := c.GetString("request_id")

		// slog to stdout
		slog.Info("request", "request_id", rid, "status", c.Writer.Status(),
			"latency_ms", latency.Milliseconds(), "path", c.Request.URL.Path)

		// zap to file
		zapLogger.Info("request",
			zap.String("request_id", rid),
			zap.String("method", c.Request.Method),
			zap.String("path", c.Request.URL.Path),
			zap.Int("status", c.Writer.Status()),
			zap.Duration("latency", latency),
			zap.String("client_ip", c.ClientIP()),
		)
	}
}

func slowRequestMiddleware(threshold time.Duration) gin.HandlerFunc {
	return func(c *gin.Context) {
		start := time.Now()
		c.Next()
		if d := time.Since(start); d > threshold {
			slog.Warn("slow request", "path", c.Request.URL.Path, "latency", d)
			zapLogger.Warn("slow request",
				zap.String("path", c.Request.URL.Path),
				zap.Duration("latency", d))
		}
	}
}

五、注意事项

事项 说明
日志不要记敏感信息 密码、Token、身份证号必须脱敏或排除
请求体日志要谨慎 大文件上传、JSON 请求体可能很大，设上限
zap Logger vs Sugar Logger 性能更高（强类型），Sugar 更灵活，按需选择
lumberjack 并发安全 内部有 mutex，多 goroutine 写入没问题
请求 ID 传播 微服务场景需在 HTTP/gRPC 调用间透传 X-Request-ID
slog 最低版本 需要 Go 1.21+，老项目用 zap

六、避坑指南

❌ 错误做法 ✅ 正确做法
用 fmt.Println 做日志 用 slog/zap 结构化输出，方便日志平台检索
日志打到 stdout 不落盘 配合 lumberjack 写文件，或发送到日志平台
每个请求打印完整请求体 设置大小上限，超过只记录前 N 字节
zap 不调用 Sync() 程序退出前 defer logger.Sync() 确保刷盘
错误日志不含堆栈 zap.AddStacktrace(zap.ErrorLevel) 自动附加
日志文件不轮转 lumberjack 自动按大小/时间切割，避免磁盘爆满
