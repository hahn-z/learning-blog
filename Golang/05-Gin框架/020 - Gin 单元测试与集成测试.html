020 - Gin 单元测试与集成测试

难度：⭐⭐⭐ | 预计阅读：18 分钟

使用 Go 的 net/http/httptest 包测试 Gin Handler，包括 JSON 请求/响应、Mock 数据库、中间件测试、测试覆盖率和最佳实践。

一、概念讲解

Go 的测试哲学是"内置优先"——标准库 testing + httptest 就能满足大部分 Web 测试需求。

测试分层：

单元测试     → 测试单个函数/方法（最快，无外部依赖）
  ↓
Handler 测试  → 测试 HTTP 接口（httptest 模拟请求/响应）
  ↓
集成测试     → 测试多个组件协作（数据库、缓存等）
  ↓
E2E 测试     → 测试完整用户流程（真实 HTTP 服务器）

核心工具：

  • httptest.NewRecorder() — 模拟 http.ResponseWriter
  • httptest.NewServer() — 创建真实测试服务器
  • gin.CreateTestContext() — 创建测试用 Gin Context

    repository/user_mock.go

    package repository

import "myapp/model"

// MockUserRepo is a mock implementation for testing
type MockUserRepo struct {
	Users map[int]*model.User
	NextID int
	Err    error // inject error for testing
}

func NewMockUserRepo() *MockUserRepo {
	return &MockUserRepo{
		Users: map[int]*model.User{
			1: {ID: 1, Username: "alice", Email: "alice@test.com"},
			2: {ID: 2, Username: "bob", Email: "bob@test.com"},
		},
		NextID: 3,
	}
}

func (m *MockUserRepo) GetByID(id int) (*model.User, error) {
	if m.Err != nil {
		return nil, m.Err
	}
	u, ok := m.Users[id]
	if !ok {
		return nil, fmt.Errorf("user not found")
	}
	return u, nil
}

func (m *MockUserRepo) Create(req *model.CreateUserRequest) (*model.User, error) {
	if m.Err != nil {
		return nil, m.Err
	}
	u := &model.User{ID: m.NextID, Username: req.Username, Email: req.Email}
	m.Users[m.NextID] = u
	m.NextID++
	return u, nil
}

func (m *MockUserRepo) List() ([]model.User, error) {
	if m.Err != nil {
		return nil, m.Err
	}
	var users []model.User
	for _, u := range m.Users {
		users = append(users, *u)
	}
	return users, nil
}

    handler/user.go

    package handler

import (
	"net/http"
	"strconv"

	"github.com/gin-gonic/gin"
	"myapp/model"
	"myapp/repository"
)

type UserHandler struct {
	repo repository.UserRepository
}

func NewUserHandler(repo repository.UserRepository) *UserHandler {
	return &UserHandler{repo: repo}
}

func (h *UserHandler) GetUser(c *gin.Context) {
	id, err := strconv.Atoi(c.Param("id"))
	if err != nil {
		c.JSON(http.StatusBadRequest, gin.H{"error": "invalid id"})
		return
	}

	user, err := h.repo.GetByID(id)
	if err != nil {
		c.JSON(http.StatusNotFound, gin.H{"error": err.Error()})
		return
	}

	c.JSON(http.StatusOK, user)
}

func (h *UserHandler) CreateUser(c *gin.Context) {
	var req model.CreateUserRequest
	if err := c.ShouldBindJSON(&req); err != nil {
		c.JSON(http.StatusBadRequest, gin.H{"error": err.Error()})
		return
	}

	user, err := h.repo.Create(&req)
	if err != nil {
		c.JSON(http.StatusInternalServerError, gin.H{"error": err.Error()})
		return
	}

	c.JSON(http.StatusCreated, user)
}

func (h *UserHandler) ListUsers(c *gin.Context) {
	users, err := h.repo.List()
	if err != nil {
		c.JSON(http.StatusInternalServerError, gin.H{"error": err.Error()})
		return
	}
	c.JSON(http.StatusOK, gin.H{"users": users})
}

// SetupRouter creates a router with all user routes
func SetupRouter(h *UserHandler) *gin.Engine {
	r := gin.Default()
	api := r.Group("/api")
	{
		api.GET("/users/:id", h.GetUser)
		api.POST("/users", h.CreateUser)
		api.GET("/users", h.ListUsers)
	}
	return r
}

func TestGetUser(t *testing.T) {
	router, _ := setupTestRouter()

	tests := []struct {
		name       string
		id         string
		wantStatus int
		wantBody   string
	}{
		{
			name:       "existing user",
			id:         "1",
			wantStatus: http.StatusOK,
			wantBody:   `"username":"alice"`,
		},
		{
			name:       "non-existing user",
			id:         "999",
			wantStatus: http.StatusNotFound,
			wantBody:   `"error"`,
		},
		{
			name:       "invalid id",
			id:         "abc",
			wantStatus: http.StatusBadRequest,
			wantBody:   `"error"`,
		},
	}

	for _, tt := range tests {
		t.Run(tt.name, func(t *testing.T) {
			w := httptest.NewRecorder()
			req, _ := http.NewRequest("GET", "/api/users/"+tt.id, nil)
			router.ServeHTTP(w, req)

			assert.Equal(t, tt.wantStatus, w.Code)
			assert.Contains(t, w.Body.String(), tt.wantBody)
		})
	}
}

// ===== Test POST with JSON body =====

func TestCreateUser(t *testing.T) {
	router, _ := setupTestRouter()

	body := model.CreateUserRequest{
		Username: "charlie",
		Email:    "charlie@test.com",
	}
	jsonBody, _ := json.Marshal(body)

	w := httptest.NewRecorder()
	req, _ := http.NewRequest("POST", "/api/users", bytes.NewBuffer(jsonBody))
	req.Header.Set("Content-Type", "application/json")
	router.ServeHTTP(w, req)

	assert.Equal(t, http.StatusCreated, w.Code)

	var response map[string]interface{}
	json.Unmarshal(w.Body.Bytes(), &response)
	assert.Equal(t, "charlie", response["username"])
	assert.Equal(t, "charlie@test.com", response["email"])
	assert.Equal(t, float64(3), response["id"]) // NextID = 3
}

    middleware/auth.go

    package middleware

import (
	"net/http"
	"strings"

	"github.com/gin-gonic/gin"
)

// AuthMiddleware validates Bearer token
func AuthMiddleware(validToken string) gin.HandlerFunc {
	return func(c *gin.Context) {
		auth := c.GetHeader("Authorization")
		if auth == "" {
			c.JSON(http.StatusUnauthorized, gin.H{"error": "missing authorization header"})
			c.Abort()
			return
		}

		if !strings.HasPrefix(auth, "Bearer ") {
			c.JSON(http.StatusUnauthorized, gin.H{"error": "invalid authorization format"})
			c.Abort()
			return
		}

		token := strings.TrimPrefix(auth, "Bearer ")
		if token != validToken {
			c.JSON(http.StatusForbidden, gin.H{"error": "invalid token"})
			c.Abort()
			return
		}

		c.Set("userID", "user-from-token")
		c.Next()
	}
}

    middleware/auth_test.go

    package middleware

import (
	"net/http"
	"net/http/httptest"
	"testing"

	"github.com/gin-gonic/gin"
	"github.com/stretchr/testify/assert"
)

func setupAuthRouter(token string) *gin.Engine {
	gin.SetMode(gin.TestMode)
	r := gin.New()
	r.Use(AuthMiddleware(token))
	r.GET("/protected", func(c *gin.Context) {
		c.JSON(200, gin.H{"message": "ok"})
	})
	return r
}

func TestAuthMiddleware(t *testing.T) {
	tests := []struct {
		name       string
		token      string
		authHeader string
		wantStatus int
	}{
		{
			name:       "valid token",
			token:      "secret123",
			authHeader: "Bearer secret123",
			wantStatus: http.StatusOK,
		},
		{
			name:       "missing header",
			token:      "secret123",
			authHeader: "",
			wantStatus: http.StatusUnauthorized,
		},
		{
			name:       "wrong token",
			token:      "secret123",
			authHeader: "Bearer wrong",
			wantStatus: http.StatusForbidden,
		},
		{
			name:       "invalid format",
			token:      "secret123",
			authHeader: "Basic secret123",
			wantStatus: http.StatusUnauthorized,
		},
	}

	for _, tt := range tests {
		t.Run(tt.name, func(t *testing.T) {
			router := setupAuthRouter(tt.token)
			w := httptest.NewRecorder()
			req, _ := http.NewRequest("GET", "/protected", nil)
			if tt.authHeader != "" {
				req.Header.Set("Authorization", tt.authHeader)
			}
			router.ServeHTTP(w, req)
			assert.Equal(t, tt.wantStatus, w.Code)
		})
	}
}

    集成测试（httptest.NewServer）

    package handler

import (
	"encoding/json"
	"net/http"
	"net/http/httptest"
	"testing"

	"github.com/stretchr/testify/assert"
	"myapp/repository"
)

func TestIntegration_UserFlow(t *testing.T) {
	gin.SetMode(gin.TestMode)
	mockRepo := repository.NewMockUserRepo()
	handler := NewUserHandler(mockRepo)
	router := SetupRouter(handler)

	// Create a real test server
	server := httptest.NewServer(router)
	defer server.Close()
	client := server.Client()

	// Step 1: List users (should have 2)
	resp, _ := client.Get(server.URL + "/api/users")
	assert.Equal(t, 200, resp.StatusCode)
	var listResp map[string]interface{}
	json.NewDecoder(resp.Body).Decode(&listResp)
	assert.Len(t, listResp["users"], 2)

	// Step 2: Create a new user
	body := `{"username":"newuser","email":"new@test.com"}`
	resp, _ = client.Post(server.URL+"/api/users", "application/json", bytes.NewBufferString(body))
	assert.Equal(t, 201, resp.StatusCode)

	// Step 3: Get the new user
	resp, _ = client.Get(server.URL + "/api/users/3")
	assert.Equal(t, 200, resp.StatusCode)
	var user map[string]interface{}
	json.NewDecoder(resp.Body).Decode(&user)
	assert.Equal(t, "newuser", user["username"])
}

    五、注意事项

    项目 说明
    gin.TestMode 测试中设置，减少日志输出
    httptest.NewRecorder 不启动真实服务器，最快
    httptest.NewServer 启动真实 HTTP 服务器，适合集成测试
    接口设计 Repository 用接口，方便 Mock 替换
    表驱动测试 Go 推荐模式，一个函数覆盖多种场景
    测试文件 xxx_test.go 必须在同一包，go test 自动发现

    六、避坑指南

    ❌ 错误做法 ✅ 正确做法
    测试中连真实数据库 用接口 + Mock 或测试专用 SQLite
    gin.Default() 跑测试 gin.New() 或 gin.SetMode(TestMode)
    测试间共享可变状态 每个测试函数独立初始化
    只测试 Happy Path 必须测试错误情况（404, 400, 500）
    忽略测试覆盖率 核心代码覆盖率目标 > 80%
    手动拼 JSON 字符串 用 json.Marshal 构造请求体
