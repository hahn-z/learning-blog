响应处理与渲染

难度：⭐⭐ 进阶级 | 预计阅读：25分钟 | 适合：掌握参数绑定，需要规范输出响应、支持多种内容格式的开发者

一、概念讲解

1.1 Gin的响应体系

Gin提供了丰富的响应方法，覆盖常见的内容格式：

响应方法族
├── c.JSON()       → application/json     (最常用)
├── c.String()     → text/plain
├── c.HTML()       → text/html            (模板渲染)
├── c.XML()        → application/xml
├── c.YAML()       → application/yaml
├── c.File()       → 文件下载
├── c.Data()       → 自定义二进制数据
├── c.SSEvent()    → Server-Sent Events
└── c.Redirect()   → HTTP重定向

1.2 统一响应结构

REST API最佳实践：所有响应遵循统一格式。

// Success
{
    "code": 0,
    "message": "success",
    "data": { ... }
}

// Error
{
    "code": 40001,
    "message": "Invalid parameter",
    "data": null
}

// List with pagination
{
    "code": 0,
    "message": "success",
    "data": {
        "items": [...],
        "total": 100,
        "page": 1,
        "page_size": 10
    }
}

1.3 HTTP状态码选择

场景 状态码 说明
创建成功 201 Created POST成功
查询成功 200 OK GET成功
更新成功 200 OK PUT/PATCH成功
删除成功 200 OK / 204 No Content DELETE成功
参数错误 400 Bad Request 验证失败
未认证 401 Unauthorized 未登录
无权限 403 Forbidden 权限不足
资源不存在 404 Not Found 找不到
重复创建 409 Conflict 资源已存在
服务器错误 500 Internal Server Error 未捕获异常

三、完整代码

3.1 统一响应结构封装

package response

import (
    "net/http"
    "github.com/gin-gonic/gin"
)

// Response is the unified API response structure
type Response struct {
    Code    int         `json:"code"`
    Message string      `json:"message"`
    Data    interface{} `json:"data,omitempty"`
}

// PageData wraps paginated list results
type PageData struct {
    Items    interface{} `json:"items"`
    Total    int64       `json:"total"`
    Page     int         `json:"page"`
    PageSize int         `json:"page_size"`
}

// Success sends a successful response
func Success(c *gin.Context, data interface{}) {
    c.JSON(http.StatusOK, Response{
        Code:    0,
        Message: "success",
        Data:    data,
    })
}

// Created sends a 201 response for resource creation
func Created(c *gin.Context, data interface{}) {
    c.JSON(http.StatusCreated, Response{
        Code:    0,
        Message: "created",
        Data:    data,
    })
}

// Error sends an error response
func Error(c *gin.Context, httpStatus int, code int, message string) {
    c.JSON(httpStatus, Response{
        Code:    code,
        Message: message,
    })
}

// PageResult sends a paginated list response
func PageResult(c *gin.Context, items interface{}, total int64, page, pageSize int) {
    Success(c, PageData{
        Items:    items,
        Total:    total,
        Page:     page,
        PageSize: pageSize,
    })
}

3.2 多种响应格式

package main

import (
    "net/http"
    "github.com/gin-gonic/gin"
)

type User struct {
    ID   int    `json:"id" xml:"id"`
    Name string `json:"name" xml:"name"`
}

func main() {
    r := gin.Default()

    // JSON response (most common)
    r.GET("/json", func(c *gin.Context) {
        c.JSON(http.StatusOK, gin.H{
            "framework": "Gin",
            "version":   "1.9",
        })
    })

    // Pretty JSON (indented, for debugging)
    r.GET("/json/pretty", func(c *gin.Context) {
        c.IndentedJSON(http.StatusOK, gin.H{
            "framework": "Gin",
            "features":  []string{"fast", "middleware", "binding"},
        })
    })

    // String response
    r.GET("/string", func(c *gin.Context) {
        c.String(http.StatusOK, "Hello, %s! You are visitor #%d", "Gopher", 42)
    })

    // XML response
    r.GET("/xml", func(c *gin.Context) {
        user := User{ID: 1, Name: "Alice"}
        c.XML(http.StatusOK, user)
    })

    // YAML response
    r.GET("/yaml", func(c *gin.Context) {
        c.YAML(http.StatusOK, gin.H{
            "name":    "Gin",
            "version": "1.9",
        })
    })

    // File download
    r.GET("/download", func(c *gin.Context) {
        c.File("./files/report.pdf")
    })

    // File with custom filename
    r.GET("/download/custom", func(c *gin.Context) {
        c.Header("Content-Disposition", "attachment; filename=report-2026.pdf")
        c.File("./files/report.pdf")
    })

    // Redirect
    r.GET("/redirect", func(c *gin.Context) {
        c.Redirect(http.StatusMovedPermanently, "/json")
    })

    // External redirect
    r.GET("/external", func(c *gin.Context) {
        c.Redirect(http.StatusFound, "https://gin-gonic.com")
    })

    r.Run(":8080")
}

3.4 Server-Sent Events (SSE) 流式响应

package main

import (
    "fmt"
    "math/rand"
    "net/http"
    "time"
    "github.com/gin-gonic/gin"
)

func main() {
    r := gin.Default()

    // SSE endpoint for real-time updates
    r.GET("/stream", func(c *gin.Context) {
        // Set SSE headers
        c.Header("Content-Type", "text/event-stream")
        c.Header("Cache-Control", "no-cache")
        c.Header("Connection", "keep-alive")

        // Stream events
        for i := 0; i < 10; i++ {
            // Use c.SSEvent for structured SSE
            c.SSEvent("message", gin.H{
                "id":    i,
                "value": rand.Intn(100),
                "time":  time.Now().Format("15:04:05"),
            })
            c.Writer.Flush() // Important: flush each event
            
            select {
            case <-time.After(time.Second):
                // Wait 1 second between events
            case <-c.Request.Context().Done():
                // Client disconnected
                return
            }
        }
    })

    // Stream using c.Stream for more control
    r.GET("/stream/custom", func(c *gin.Context) {
        c.Header("Content-Type", "text/event-stream")
        c.Header("Cache-Control", "no-cache")

        c.Stream(func(w io.Writer) bool {
            fmt.Fprintf(w, "data: %s\n\n", time.Now().Format("15:04:05"))
            return true // return false to stop streaming
        })
    })

    r.Run(":8080")
}

3.5 完整示例：统一响应+多种格式

package main

import (
    "io"
    "math/rand"
    "net/http"
    "time"
    "github.com/gin-gonic/gin"
)

// Unified response helpers
func OK(c *gin.Context, data interface{}) {
    c.JSON(http.StatusOK, gin.H{"code": 0, "message": "success", "data": data})
}

func Fail(c *gin.Context, code int, msg string) {
    c.JSON(code, gin.H{"code": code * 100, "message": msg})
}

func main() {
    r := gin.Default()
    r.LoadHTMLGlob("templates/*")

    // API routes with unified JSON responses
    api := r.Group("/api")
    {
        api.GET("/users", func(c *gin.Context) {
            users := []gin.H{
                {"id": 1, "name": "Alice"},
                {"id": 2, "name": "Bob"},
            }
            OK(c, gin.H{"items": users, "total": len(users)})
        })

        api.GET("/users/:id", func(c *gin.Context) {
            // ... lookup user
            OK(c, gin.H{"id": 1, "name": "Alice"})
        })

        api.POST("/users", func(c *gin.Context) {
            var input struct {
                Name string `json:"name" binding:"required"`
            }
            if err := c.ShouldBindJSON(&input); err != nil {
                Fail(c, http.StatusBadRequest, err.Error())
                return
            }
            c.JSON(http.StatusCreated, gin.H{
                "code": 0, "message": "created",
                "data": gin.H{"id": rand.Intn(1000), "name": input.Name},
            })
        })

        api.DELETE("/users/:id", func(c *gin.Context) {
            c.Status(http.StatusNoContent) // 204, no body
        })
    }

    // Page routes with HTML rendering
    r.GET("/", func(c *gin.Context) {
        c.HTML(http.StatusOK, "index.html", gin.H{"title": "Home"})
    })

    // SSE stream
    r.GET("/events", func(c *gin.Context) {
        c.Header("Content-Type", "text/event-stream")
        c.Header("Cache-Control", "no-cache")
        c.Header("Connection", "keep-alive")
        for i := 0; i < 5; i++ {
            c.SSEvent("tick", gin.H{"count": i, "ts": time.Now().Unix()})
            c.Writer.Flush()
            time.Sleep(time.Second)
        }
    })

    r.Run(":8080")
}

五、注意事项

事项 说明 建议
JSON序列化 结构体字段必须导出(大写) 使用json标签控制字段名
模板热重载 生产环境不要热重载 开发用LoadHTMLGlob，生产用LoadHTMLFiles
SSE断开 客户端可能随时断开 监听c.Request.Context().Done()
204响应 没有响应体 用c.Status(204)而非c.JSON(204, ...)
中文编码 String默认UTF-8 XML需声明

六、避坑指南

❌ 响应后继续操作

r.GET("/bad", func(c *gin.Context) {
    c.JSON(200, gin.H{"msg": "ok"})
    // Continuing to write response after sending is dangerous
    c.JSON(200, gin.H{"msg": "another"}) // Ignored or corrupts response
})

✅ 响应后立即return

r.GET("/good", func(c *gin.Context) {
    c.JSON(200, gin.H{"msg": "ok"})
    return // Clear exit after response
})

❌ 不统一的响应格式

// Sometimes {"error": "..."}, sometimes {"msg": "..."}, sometimes raw string
r.GET("/a", func(c *gin.Context) { c.JSON(200, "ok") })
r.GET("/b", func(c *gin.Context) { c.JSON(400, gin.H{"error": "bad"}) })
r.GET("/c", func(c *gin.Context) { c.String(500, "server error") })

✅ 封装统一响应函数

func OK(c *gin.Context, data interface{}) {
    c.JSON(200, gin.H{"code": 0, "message": "success", "data": data})
}
func Fail(c *gin.Context, code int, msg string) {
    c.JSON(code, gin.H{"code": code, "message": msg})
}

❌ SSE不Flush

r.GET("/stream", func(c *gin.Context) {
    c.SSEvent("msg", "data")
    // Without Flush, data stays in buffer, client sees nothing!
})

✅ 每次SSEvent后Flush

r.GET("/stream", func(c *gin.Context) {
    c.SSEvent("msg", "data")
    c.Writer.Flush() // Push data to client immediately
})

              package main

import (
    "fmt"
    "io"
    "log"
    "net/http"
    "time"
    "github.com/gin-gonic/gin"
)

// --- Unified response package ---
type R struct {
    Code    int         `json:"code"`
    Message string      `json:"message"`
    Data    interface{} `json:"data,omitempty"`
}

func Ok(c *gin.Context, data interface{}) {
    c.JSON(http.StatusOK, R{Code: 0, Message: "success", Data: data})
}

func Created(c *gin.Context, data interface{}) {
    c.JSON(http.StatusCreated, R{Code: 0, Message: "created", Data: data})
}

func Err(c *gin.Context, status int, msg string) {
    c.JSON(status, R{Code: status * 100, Message: msg})
}

func NoContent(c *gin.Context) {
    c.Status(http.StatusNoContent)
}

// --- Application ---
type User struct {
    ID   int    `json:"id"`
    Name string `json:"name"`
}

var users = []User{{1, "Alice"}, {2, "Bob"}}

func main() {
    gin.SetMode(gin.ReleaseMode)
    r := gin.Default()
    r.LoadHTMLGlob("templates/*")

    // JSON API
    api := r.Group("/api")
    {
        api.GET("/users", func(c *gin.Context) {
            Ok(c, gin.H{"items": users, "total": len(users)})
        })
        api.GET("/users/:id", func(c *gin.Context) {
            Ok(c, users[0])
        })
        api.POST("/users", func(c *gin.Context) {
            var input struct {
                Name string `json:"name" binding:"required"`
            }
            if err := c.ShouldBindJSON(&input); err != nil {
                Err(c, http.StatusBadRequest, err.Error())
                return
            }
            Created(c, User{ID: 3, Name: input.Name})
        })
        api.DELETE("/users/:id", func(c *gin.Context) {
            NoContent(c)
        })

        // XML format
        api.GET("/users.xml", func(c *gin.Context) {
            c.XML(http.StatusOK, gin.H{"users": users})
        })
    }

    // HTML page
    r.GET("/", func(c *gin.Context) {
        c.HTML(http.StatusOK, "index.html", gin.H{
            "title": "Gin Demo",
            "users": users,
        })
    })

    // SSE stream
    r.GET("/events", func(c *gin.Context) {
        c.Header("Content-Type", "text/event-stream")
        c.Header("Cache-Control", "no-cache")
        c.Header("Connection", "keep-alive")
        
        ctx := c.Request.Context()
        for i := 0; i < 20; i++ {
            select {
            case <-ctx.Done():
                log.Println("Client disconnected")
                return
            case <-time.After(time.Second):
                c.SSEvent("tick", fmt.Sprintf("event #%d at %s", i, time.Now().Format("15:04:05")))
                c.Writer.Flush()
            }
        }
    })

    // File download with custom name
    r.GET("/download/report", func(c *gin.Context) {
        c.Header("Content-Disposition", "attachment; filename=report.csv")
        c.Data(http.StatusOK, "text/csv", []byte("id,name\n1,Alice\n2,Bob\n"))
    })

    log.Println("Server on :8080")
    r.Run(":8080")
}
