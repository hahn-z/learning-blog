010 - 数据库事务处理

难度标注： ⭐⭐⭐（中级）

前置知识： GORM CRUD、Service 层架构

学习目标： 掌握 GORM 事务管理、手动事务、嵌套事务、事务最佳实践

一、概念讲解

事务（Transaction） 是数据库操作的最小执行单元，具有 ACID 四大特性：

特性 英文 说明 示例
原子性 Atomicity 全部成功或全部回滚 转账：扣款+到账必须同时成功
一致性 Consistency 操作前后数据一致 总金额不变
隔离性 Isolation 并发事务互不干扰 两人同时转账不冲突
持久性 Durability 提交后永久保存 断电不丢数据

什么时候需要事务？

  • 转账操作：扣款 + 加款必须原子性
  • 订单创建：创建订单 + 扣库存 + 记录日志
  • 批量操作：插入多条关联数据
  • 状态变更：多个表联动更新

    • GORM 事务三种方式

    1. 自动事务：db.Transaction(func(tx *gorm.DB) error {...})（推荐）
  1. 手动事务：db.Begin() + tx.Commit() + tx.Rollback()
    1. 嵌套事务：tx.SavePoint() + tx.RollbackTo()

      v1：三种事务方式基础示例

      // main.go - Three ways to use transactions in GORM
package main

import (
	"errors"
	"fmt"
	"log"

	"gorm.io/driver/sqlite"
	"gorm.io/gorm"
)

// Account model for transfer demo
type Account struct {
	ID      uint    `gorm:"primaryKey" json:"id"`
	Name    string  `gorm:"size:100;not null" json:"name"`
	Balance float64 `gorm:"not null;default:0" json:"balance"`
}

// TransferLog records each transfer
type TransferLog struct {
	ID       uint    `gorm:"primaryKey" json:"id"`
	FromID   uint    `gorm:"not null;index" json:"from_id"`
	ToID     uint    `gorm:"not null;index" json:"to_id"`
	Amount   float64 `gorm:"not null" json:"amount"`
	Status   string  `gorm:"size:20;default:pending" json:"status"`
}

var db *gorm.DB

func main() {
	var err error
	db, err = gorm.Open(sqlite.Open("bank.db"), &gorm.Config{})
	if err != nil {
		log.Fatal(err)
	}
	db.AutoMigrate(&Account{}, &TransferLog{})

	// Seed data
	db.Where("1=1").Delete(&Account{})
	db.Where("1=1").Delete(&TransferLog{})
	db.Create(&Account{Name: "Alice", Balance: 1000})
	db.Create(&Account{Name: "Bob", Balance: 500})

	fmt.Println("=== Method 1: Auto Transaction ===")
	transferAuto(1, 2, 200)

	fmt.Println("\n=== Method 2: Manual Transaction ===")
	transferManual(1, 2, 100)

	fmt.Println("\n=== Method 3: Nested Transaction (SavePoint) ===")
	transferWithSavePoint(1, 2, 50)

	showBalances()
}

// ========== Method 1: Auto Transaction (Recommended) ==========
func transferAuto(fromID, toID uint, amount float64) error {
	return db.Transaction(func(tx *gorm.DB) error {
		// Step 1: Deduct from sender
		var from Account
		if err := tx.First(&from, fromID).Error; err != nil {
			return err // Will auto rollback
		}
		if from.Balance < amount {
			return errors.New("insufficient balance") // Will auto rollback
		}
		if err := tx.Model(&from).Update("balance", gorm.Expr("balance - ?", amount)).Error; err != nil {
			return err // Will auto rollback
		}

		// Step 2: Add to receiver
		if err := tx.Model(&Account{}).Where("id = ?", toID).Update("balance", gorm.Expr("balance + ?", amount)).Error; err != nil {
			return err // Will auto rollback
		}

		// Step 3: Log transfer
		log := TransferLog{FromID: fromID, ToID: toID, Amount: amount, Status: "completed"}
		if err := tx.Create(&log).Error; err != nil {
			return err // Will auto rollback
		}

		fmt.Printf("Transfer %.2f from Account%d to Account%d (auto)\n", amount, fromID, toID)
		return nil // Commit
	})
}

// ========== Method 2: Manual Transaction ==========
func transferManual(fromID, toID uint, amount float64) error {
	tx := db.Begin() // Start transaction
	defer func() {
		if r := recover(); r != nil {
			tx.Rollback()
			panic(r) // Re-throw after rollback
		}
	}()

	// Step 1: Check and deduct
	var from Account
	if err := tx.First(&from, fromID).Error; err != nil {
		tx.Rollback()
		return err
	}
	if from.Balance < amount {
		tx.Rollback()
		return errors.New("insufficient balance")
	}
	tx.Model(&from).Update("balance", gorm.Expr("balance - ?", amount))

	// Step 2: Add to receiver
	tx.Model(&Account{}).Where("id = ?", toID).Update("balance", gorm.Expr("balance + ?", amount))

	// Step 3: Log
	tx.Create(&TransferLog{FromID: fromID, ToID: toID, Amount: amount, Status: "completed"})

	fmt.Printf("Transfer %.2f from Account%d to Account%d (manual)\n", amount, fromID, toID)
	return tx.Commit().Error // Commit
}

// ========== Method 3: Nested Transaction with SavePoint ==========
func transferWithSavePoint(fromID, toID uint, amount float64) error {
	return db.Transaction(func(tx *gorm.DB) error {
		// Outer transaction: deduct from sender
		var from Account
		if err := tx.First(&from, fromID).Error; err != nil {
			return err
		}
		tx.Model(&from).Update("balance", gorm.Expr("balance - ?", amount))

		// Inner transaction: add to receiver (SavePoint)
		err := tx.Transaction(func(tx2 *gorm.DB) error {
			// Create savepoint automatically
			if err := tx2.Model(&Account{}).Where("id = ?", toID).
				Update("balance", gorm.Expr("balance + ?", amount)).Error; err != nil {
				return err // Rollback to savepoint
			}

			// Log with nested check
			logEntry := TransferLog{FromID: fromID, ToID: toID, Amount: amount, Status: "completed"}
			return tx2.Create(&logEntry).Error
		})
		if err != nil {
			// Inner failed, but outer can decide what to do
			fmt.Println("Inner transaction failed:", err)
			return err // Rollback everything
		}

		fmt.Printf("Transfer %.2f from Account%d to Account%d (nested)\n", amount, fromID, toID)
		return nil
	})
}

func showBalances() {
	var accounts []Account
	db.Find(&accounts)
	for _, a := range accounts {
		fmt.Printf("  %s: %.2f\n", a.Name, a.Balance)
	}
}

      // repository/account_repository.go
package repository

import (
	"myapp/models"

	"gorm.io/gorm"
)

type AccountRepository interface {
	FindByID(tx *gorm.DB, id uint) (*models.Account, error)
	UpdateBalance(tx *gorm.DB, id uint, amount float64) error
	CreateLog(tx *gorm.DB, log *models.TransferLog) error
	GetDB() *gorm.DB
}

type accountRepository struct {
	db *gorm.DB
}

func NewAccountRepository(db *gorm.DB) AccountRepository {
	return &accountRepository{db: db}
}

func (r *accountRepository) FindByID(tx *gorm.DB, id uint) (*models.Account, error) {
	if tx == nil {
		tx = r.db
	}
	var account models.Account
	if err := tx.First(&account, id).Error; err != nil {
		return nil, err
	}
	return &account, nil
}

func (r *accountRepository) UpdateBalance(tx *gorm.DB, id uint, delta float64) error {
	return tx.Model(&models.Account{}).Where("id = ?", id).
		Update("balance", gorm.Expr("balance + ?", delta)).Error
}

func (r *accountRepository) CreateLog(tx *gorm.DB, log *models.TransferLog) error {
	return tx.Create(log).Error
}

func (r *accountRepository) GetDB() *gorm.DB {
	return r.db
}
      // service/transfer_service.go - Business logic with transaction
package service

import (
	"errors"
	"fmt"

	"myapp/models"
	"myapp/repository"
)

type TransferService interface {
	Transfer(fromID, toID uint, amount float64) (*models.TransferLog, error)
}

type transferService struct {
	accountRepo repository.AccountRepository
}

func NewTransferService(accountRepo repository.AccountRepository) TransferService {
	return &transferService{accountRepo: accountRepo}
}

func (s *transferService) Transfer(fromID, toID uint, amount float64) (*models.TransferLog, error) {
	// Validate
	if amount <= 0 {
		return nil, errors.New("amount must be positive")
	}
	if fromID == toID {
		return nil, errors.New("cannot transfer to self")
	}

	// Execute in transaction
	var transferLog models.TransferLog
	err := s.accountRepo.GetDB().Transaction(func(tx *gorm.DB) error {
		// Lock and check sender balance
		from, err := s.accountRepo.FindByID(tx, fromID)
		if err != nil {
			return fmt.Errorf("sender not found: %w", err)
		}
		if from.Balance < amount {
			return errors.New("insufficient balance")
		}

		// Check receiver exists
		if _, err := s.accountRepo.FindByID(tx, toID); err != nil {
			return fmt.Errorf("receiver not found: %w", err)
		}

		// Deduct
		if err := s.accountRepo.UpdateBalance(tx, fromID, -amount); err != nil {
			return fmt.Errorf("deduct failed: %w", err)
		}

		// Add
		if err := s.accountRepo.UpdateBalance(tx, toID, amount); err != nil {
			return fmt.Errorf("credit failed: %w", err)
		}

		// Log
		transferLog = models.TransferLog{
			FromID: fromID,
			ToID:   toID,
			Amount: amount,
			Status: "completed",
		}
		return s.accountRepo.CreateLog(tx, &transferLog)
	})

	if err != nil {
		return nil, err
	}
	return &transferLog, nil
}

      v3：重试机制 + 超时控制 + 死锁处理

      // pkg/dbutil/transaction.go - Reusable transaction utilities
package dbutil

import (
	"errors"
	"log"
	"time"

	"gorm.io/gorm"
)

// RetryOption configures retry behavior
type RetryOption struct {
	MaxRetries int           // Max retry attempts (default: 3)
	Delay      time.Duration // Delay between retries (default: 100ms)
	MaxDelay   time.Duration // Max delay for exponential backoff
}

// RetryTransaction executes fn with automatic retry on deadlock
func RetryTransaction(db *gorm.DB, opt RetryOption, fn func(tx *gorm.DB) error) error {
	if opt.MaxRetries == 0 {
		opt.MaxRetries = 3
	}
	if opt.Delay == 0 {
		opt.Delay = 100 * time.Millisecond
	}
	if opt.MaxDelay == 0 {
		opt.MaxDelay = 5 * time.Second
	}

	var lastErr error
	for i := 0; i < opt.MaxRetries; i++ {
		err := db.Transaction(fn)
		if err == nil {
			return nil // Success
		}

		lastErr = err

		// Check if error is retryable (deadlock, lock timeout)
		if isDeadlockError(err) {
			delay := time.Duration(1<<uint(i)) * opt.Delay // Exponential backoff
			if delay > opt.MaxDelay {
				delay = opt.MaxDelay
			}
			log.Printf("Transaction retry %d/%d after %v: %v", i+1, opt.MaxRetries, delay, err)
			time.Sleep(delay)
			continue
		}

		// Non-retryable error
		return err
	}

	return lastErr
}

// isDeadlockError checks if the error is a deadlock or lock timeout
func isDeadlockError(err error) bool {
	if err == nil {
		return false
	}
	errMsg := err.Error()
	// MySQL: Error 1213 (Deadlock found)
	// PostgreSQL: could not serialize access
	deadlockMessages := []string{
		"Deadlock",
		"deadlock",
		"lock timeout",
		"try restarting transaction",
		"could not serialize access",
	}
	for _, msg := range deadlockMessages {
		if contains(errMsg, msg) {
			return true
		}
	}
	return false
}

func contains(s, substr string) bool {
	return len(s) >= len(substr) && (s == substr || len(s) > 0 && containsStr(s, substr))
}

func containsStr(s, substr string) bool {
	for i := 0; i <= len(s)-len(substr); i++ {
		if s[i:i+len(substr)] == substr {
			return true
		}
	}
	return false
}

// WithTimeout executes a transaction with a context timeout
func WithTimeout(db *gorm.DB, timeout time.Duration, fn func(tx *gorm.DB) error) error {
	tx := db.Begin()
	done := make(chan error, 1)

	go func() {
		done <- fn(tx)
	}()

	select {
	case err := <-done:
		if err != nil {
			tx.Rollback()
			return err
		}
		return tx.Commit().Error
	case <-time.After(timeout):
		tx.Rollback()
		return errors.New("transaction timeout")
	}
}

      四、注意事项

      项目 说明 建议
      事务粒度 事务内操作越少越好 长事务锁住资源，影响并发
      错误处理 返回 error 自动回滚 自动事务中 return err 即回滚
      手动事务 忘记 Rollback 会泄漏连接 用 defer 确保 Rollback
      嵌套事务 GORM 用 SavePoint 实现 不是真正的嵌套，注意语义
      超时 长事务可能导致死锁 设置合理超时
      并发 高并发下注意隔离级别 MySQL 默认 REPEATABLE READ

      五、避坑指南

      ❌ 手动事务忘记 Rollback

      tx := db.Begin()
tx.Create(&order)
tx.Create(&orderItem)
// If panic or error, connection leaks!
tx.Commit()

      ✅ 使用 defer + recover

      tx := db.Begin()
defer func() {
    if r := recover(); r != nil {
        tx.Rollback()
        panic(r)
    }
}()
// Or better: use db.Transaction() which handles this automatically

      ❌ 事务中做耗时操作

      db.Transaction(func(tx *gorm.DB) error {
    tx.Create(&order)
    time.Sleep(5 * time.Second) // Holding lock for 5s!
    emailService.Send(...)       // External call in transaction
    return nil
})

      ✅ 事务只做数据库操作

      // 1. Validate outside transaction
// 2. Execute DB operations in transaction
err := db.Transaction(func(tx *gorm.DB) error {
    tx.Create(&order)
    tx.Create(&orderItem)
    return nil // Keep it short
})
// 3. Send notification AFTER transaction commits
if err == nil {
    emailService.Send(...)
}

      ❌ 不处理死锁

      db.Transaction(func(tx *gorm.DB) error {
    // Two concurrent transactions may deadlock
    // User A: update account 1 then 2
    // User B: update account 2 then 1
    return nil
})
// Deadlock error goes unhandled!

      ✅ 死锁重试

      dbutil.RetryTransaction(db, dbutil.RetryOption{
    MaxRetries: 3,
}, func(tx *gorm.DB) error {
    // Always lock in consistent order (e.g., by ID)
    return nil
})

                  2. 七、知识点总结

                  数据库事务处理
├── ACID 特性
│   ├── 原子性 (Atomicity) - 全部成功或全部失败
│   ├── 一致性 (Consistency) - 数据状态一致
│   ├── 隔离性 (Isolation) - 并发互不干扰
│   └── 持久性 (Durability) - 提交后永久保存
├── GORM 事务方式
│   ├── 自动事务: db.Transaction(fn) ← 推荐
│   ├── 手动事务: Begin/Commit/Rollback
│   └── 嵌套事务: SavePoint/RollbackTo
├── 事务场景
│   ├── 转账: 扣款+到账
│   ├── 订单: 创建+扣库存+日志
│   └── 批量: 多表联动更新
├── 最佳实践
│   ├── 事务粒度小（只含 DB 操作）
│   ├── 不在事务中做外部调用
│   ├── 一致的锁顺序（防死锁）
│   └── 超时控制 + 重试机制
└── 高级话题
    ├── 隔离级别: Read Uncommitted / RC / RR / Serializable
    ├── 分布式事务: 2PC / TCC / Saga
    └── 行锁: SELECT ... FOR UPDATE
