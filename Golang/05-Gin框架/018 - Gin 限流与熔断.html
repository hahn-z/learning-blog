018 - Gin 限流与熔断

难度：⭐⭐⭐⭐ | 预计阅读：20 分钟

深入理解限流算法（令牌桶/漏桶/滑动窗口），在 Gin 中实现单机/分布式限流中间件，以及熔断降级策略。

一、概念讲解

限流（Rate Limiting）和熔断（Circuit Breaker）是微服务中保护系统的两道防线。

限流：控制单位时间内的请求量，防止流量突增压垮服务。

熔断：当下游服务异常率过高时，主动断开请求，避免级联故障。

三大限流算法对比：

算法 特点 适合场景
令牌桶 允许突发流量，匀速生成令牌 API 限流（允许短时突发）
漏桶 严格匀速输出 流量整形（严格控制速率）
滑动窗口 精确统计时间窗口内请求数 精确限流（无突发）

v1：基于令牌桶的单机限流中间件

package main

import (
	"net/http"
	"sync"
	"time"

	"github.com/gin-gonic/gin"
	"golang.org/x/time/rate"
)

// IPRateLimiter stores rate limiters per IP
type IPRateLimiter struct {
	ips     map[string]*rate.Limiter
	mu      sync.RWMutex
	rate    rate.Limit
	burst   int
}

func NewIPRateLimiter(r rate.Limit, burst int) *IPRateLimiter {
	return &IPRateLimiter{
		ips:   make(map[string]*rate.Limiter),
		rate:  r,
		burst: burst,
	}
}

func (l *IPRateLimiter) GetLimiter(ip string) *rate.Limiter {
	l.mu.Lock()
	defer l.mu.Unlock()

	limiter, exists := l.ips[ip]
	if !exists {
		limiter = rate.NewLimiter(l.rate, l.burst)
		l.ips[ip] = limiter
	}
	return limiter
}

// RateLimitMiddleware creates a per-IP rate limiter
func RateLimitMiddleware(r rate.Limit, burst int) gin.HandlerFunc {
	limiter := NewIPRateLimiter(r, burst)

	return func(c *gin.Context) {
		ip := c.ClientIP()
		if !limiter.GetLimiter(ip).Allow() {
			c.JSON(http.StatusTooManyRequests, gin.H{
				"error": "rate limit exceeded",
				"retry": time.Now().Add(time.Second).Unix(),
			})
			c.Abort()
			return
		}
		c.Next()
	}
}

func main() {
	r := gin.Default()

	// 10 requests per second, burst 20
	r.Use(RateLimitMiddleware(10, 20))

	r.GET("/ping", func(c *gin.Context) {
		c.JSON(200, gin.H{"message": "pong"})
	})

	r.Run(":8080")
}

v2：滑动窗口限流 + 用户级限流

package main

import (
	"net/http"
	"sync"
	"time"

	"github.com/gin-gonic/gin"
)

// SlidingWindowCounter implements sliding window rate limiting
type SlidingWindowCounter struct {
	windows map[string]*window
	mu      sync.Mutex
	limit   int
	period  time.Duration
}

type window struct {
	count    int
	startAt  time.Time
}

func NewSlidingWindow(limit int, period time.Duration) *SlidingWindowCounter {
	return &SlidingWindowCounter{
		windows: make(map[string]*window),
		limit:   limit,
		period:  period,
	}
}

func (s *SlidingWindowCounter) Allow(key string) bool {
	s.mu.Lock()
	defer s.mu.Unlock()

	now := time.Now()
	w, exists := s.windows[key]
	if !exists || now.Sub(w.startAt) > s.period {
		s.windows[key] = &window{count: 1, startAt: now}
		return true
	}

	if w.count >= s.limit {
		return false
	}
	w.count++
	return true
}

// UserRateLimit middleware checks rate by user ID from header
func UserRateLimit(limit int, period time.Duration) gin.HandlerFunc {
	limiter := NewSlidingWindow(limit, period)
	return func(c *gin.Context) {
		userID := c.GetHeader("X-User-ID")
		if userID == "" {
			userID = c.ClientIP()
		}
		if !limiter.Allow(userID) {
			c.JSON(http.StatusTooManyRequests, gin.H{
				"error": "rate limit exceeded",
				"limit": limit,
				"window": int(period.Seconds()) + "s",
			})
			c.Abort()
			return
		}
		c.Next()
	}
}

// GlobalRateLimit limits total requests to the service
func GlobalRateLimit(limit int, period time.Duration) gin.HandlerFunc {
	limiter := NewSlidingWindow(limit, period)
	return func(c *gin.Context) {
		if !limiter.Allow("global") {
			c.JSON(http.StatusTooManyRequests, gin.H{"error": "server busy"})
			c.Abort()
			return
		}
		c.Next()
	}
}

func main() {
	r := gin.Default()

	// Global: 1000 req/min
	r.Use(GlobalRateLimit(1000, time.Minute))

	api := r.Group("/api")
	// Per user: 100 req/min
	api.Use(UserRateLimit(100, time.Minute))
	{
		api.GET("/data", func(c *gin.Context) {
			c.JSON(200, gin.H{"data": "success"})
		})
	}

	r.Run(":8080")
}

v3：Redis 分布式限流 + 熔断器

package main

import (
	"context"
	"fmt"
	"net/http"
	"time"

	"github.com/gin-gonic/gin"
	"github.com/redis/go-redis/v9"
)

// ===== Redis Sliding Window Rate Limiter =====

type RedisRateLimiter struct {
	rdb    *redis.Client
	limit  int
	period time.Duration
}

func NewRedisRateLimiter(rdb *redis.Client, limit int, period time.Duration) *RedisRateLimiter {
	return &RedisRateLimiter{rdb: rdb, limit: limit, period: period}
}

// Lua script for atomic sliding window check
var luaScript = redis.NewScript(`
	local key = KEYS[1]
	local now = tonumber(ARGV[1])
	local window = tonumber(ARGV[2])
	local limit = tonumber(ARGV[3])

	redis.call('ZREMRANGEBYSCORE', key, '-inf', now - window)
	local count = redis.call('ZCARD', key)

	if count < limit then
		redis.call('ZADD', key, now, now .. '-' .. math.random(1000000))
		redis.call('EXPIRE', key, window / 1000)
		return 1
	end
	return 0
`)

func (r *RedisRateLimiter) Allow(ctx context.Context, key string) bool {
	now := time.Now().UnixMilli()
	result, err := luaScript.Run(ctx, r.rdb,
		[]string{"ratelimit:" + key},
		now, r.period.Milliseconds(), r.limit,
	).Int()
	if err != nil {
		// Fail open on Redis error
		return true
	}
	return result == 1
}

func RedisRateLimitMiddleware(rdb *redis.Client, limit int, period time.Duration) gin.HandlerFunc {
	limiter := NewRedisRateLimiter(rdb, limit, period)
	return func(c *gin.Context) {
		key := c.ClientIP()
		if !limiter.Allow(c.Request.Context(), key) {
			c.JSON(http.StatusTooManyRequests, gin.H{
				"error":  "rate limit exceeded",
				"limit":  limit,
				"window": fmt.Sprintf("%ds", int(period.Seconds())),
			})
			c.Abort()
			return
		}
		c.Next()
	}
}

// ===== Circuit Breaker =====

type State int

const (
	Closed   State = iota
	Open
	HalfOpen
)

type CircuitBreaker struct {
	name          string
	state         State
	failures      int
	threshold     int
	resetTimeout  time.Duration
	lastFailure   time.Time
	successes     int
	halfOpenReq   int
}

func NewCircuitBreaker(name string, threshold int, resetTimeout time.Duration) *CircuitBreaker {
	return &CircuitBreaker{
		name:         name,
		state:        Closed,
		threshold:    threshold,
		resetTimeout: resetTimeout,
	}
}

func (cb *CircuitBreaker) Allow() bool {
	switch cb.state {
	case Closed:
		return true
	case Open:
		if time.Since(cb.lastFailure) > cb.resetTimeout {
			cb.state = HalfOpen
			cb.halfOpenReq = 0
			return true
		}
		return false
	case HalfOpen:
		// Allow limited requests in half-open state
		if cb.halfOpenReq < 3 {
			cb.halfOpenReq++
			return true
		}
		return false
	}
	return false
}

func (cb *CircuitBreaker) RecordSuccess() {
	if cb.state == HalfOpen {
		cb.successes++
		if cb.successes >= 3 {
			cb.state = Closed
			cb.failures = 0
			cb.successes = 0
		}
	}
}

func (cb *CircuitBreaker) RecordFailure() {
	cb.failures++
	cb.lastFailure = time.Now()
	if cb.state == HalfOpen || cb.failures >= cb.threshold {
		cb.state = Open
		cb.successes = 0
	}
}

// CircuitBreakerMiddleware wraps a downstream call with circuit breaker
func CircuitBreakerMiddleware(cb *CircuitBreaker) gin.HandlerFunc {
	return func(c *gin.Context) {
		if !cb.Allow() {
			c.JSON(http.StatusServiceUnavailable, gin.H{
				"error":   "circuit breaker open",
				"service": cb.name,
			})
			c.Abort()
			return
		}

		c.Next()

		if c.Writer.Status() >= 500 {
			cb.RecordFailure()
		} else {
			cb.RecordSuccess()
		}
	}
}

// DegradationMiddleware returns cached or default data when service is degraded
func DegradationMiddleware(cb *CircuitBreaker, fallbackData interface{}) gin.HandlerFunc {
	return func(c *gin.Context) {
		if !cb.Allow() {
			c.JSON(http.StatusOK, gin.H{
				"data":    fallbackData,
				"degraded": true,
				"message": "service temporarily unavailable, using cached data",
			})
			c.Abort()
			return
		}
		c.Next()
	}
}

func main() {
	rdb := redis.NewClient(&redis.Options{Addr: "localhost:6379"})
	cb := NewCircuitBreaker("downstream-api", 5, 30*time.Second)

	r := gin.Default()

	// Redis distributed rate limit: 100 req/min per IP
	r.Use(RedisRateLimitMiddleware(rdb, 100, time.Minute))

	api := r.Group("/api")
	api.Use(CircuitBreakerMiddleware(cb))
	{
		api.GET("/data", func(c *gin.Context) {
			// Simulate downstream call
			c.JSON(200, gin.H{"data": "live data"})
		})

		// With degradation fallback
		api.GET("/recommendations",
			DegradationMiddleware(cb, []string{"popular item 1", "popular item 2"}),
			func(c *gin.Context) {
				c.JSON(200, gin.H{"data": "personalized recommendations"})
			},
		)
	}

	r.Run(":8080")
}

五、注意事项

项目 说明
golang.org/x/time/rate 官方令牌桶实现，生产推荐
Redis Lua 原子性 分布式限流必须用 Lua 保证原子操作
Fail Open Redis 不可用时应放行请求，避免服务全挂
熔断阈值 根据服务 SLA 设定，一般 50% 错误率触发
限流粒度 IP 限流易被代理绕过，用户级更可靠
响应头 建议返回 X-RateLimit-Remaining 等头部

六、避坑指南

❌ 错误做法 ✅ 正确做法
用 sync.Map 存所有限流状态 按需创建 + 定期清理过期 key
Redis 限流不用 Lua Lua 脚本保证检查+计数原子性
熔断器永不恢复 设置 resetTimeout，HalfOpen 试探恢复
限流返回 200 + 错误信息 返回 429 Too Many Requests
全局限流用 map 无上限 加 TTL 或 LRU 淘汰过期 key
