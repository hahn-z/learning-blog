一、概念讲解

Swagger（OpenAPI）是 RESTful API 的标准文档规范。在 Gin 项目中，我们使用 swaggo/swag 自动从代码注解生成美观的 API 文档页面。

为什么需要 Swagger？

  • 📖 前后端协作的"契约"，告别口头沟通
  • 🧪 内置在线测试面板，直接在文档里调接口
  • 🔄 代码即文档，注解和实现永远同步
  • 🎨 自动生成，不用手写文档页面

    • 核心组件：

    组件 作用
    swaggo/swag 注解解析器，生成 JSON/YAML
    swaggo/gin-swagger Gin 中间件，托管文档页面
    swaggo/swag/files 静态文件嵌入

    工作流程： 在 handler 函数上方写注解 → swag init 解析生成 docs/ 目录 → Gin 路由挂载 Swagger UI → 浏览器访问 /swagger/index.html。

    三、完整代码

    v1：基础集成

    // main.go - Basic Swagger integration
package main

import (
	"net/http"

	"github.com/gin-gonic/gin"
	_ "myapp/docs" // auto-generated by swag init
	swaggerFiles "github.com/swaggo/files"
	ginSwagger "github.com/swaggo/gin-swagger"
)

// @title           Gin Swagger Demo API
// @version         1.0
// @description     A demo API with Swagger documentation.
// @host            localhost:8080
// @BasePath        /api
func main() {
	r := gin.Default()

	// Mount Swagger UI at /swagger/*
	r.GET("/swagger/*any", ginSwagger.WrapHandler(swaggerFiles.Handler))

	api := r.Group("/api")
	{
		api.GET("/users", listUsers)
		api.GET("/users/:id", getUser)
		api.POST("/users", createUser)
	}

	r.Run(":8080")
}

// User represents a user model
type User struct {
	ID    int    `json:"id" example:"1"`
	Name  string `json:"name" example:"John Doe"`
	Email string `json:"email" example:"john@example.com"`
}

// @Summary      List all users
// @Description  Get a list of all users
// @Tags         users
// @Accept       json
// @Produce      json
// @Success      200  {array}   User
// @Router       /users [get]
func listUsers(c *gin.Context) {
	users := []User{
		{ID: 1, Name: "Alice", Email: "alice@example.com"},
		{ID: 2, Name: "Bob", Email: "bob@example.com"},
	}
	c.JSON(http.StatusOK, users)
}

// @Summary      Get user by ID
// @Description  Fetch a single user by their ID
// @Tags         users
// @Accept       json
// @Produce      json
// @Param        id   path      int  true  "User ID"
// @Success      200  {object}  User
// @Failure      404  {object}  map[string]string
// @Router       /users/{id} [get]
func getUser(c *gin.Context) {
	id := c.Param("id")
	c.JSON(http.StatusOK, gin.H{
		"id": id, "name": "Alice", "email": "alice@example.com",
	})
}

// @Summary      Create a new user
// @Description  Create a new user with the provided data
// @Tags         users
// @Accept       json
// @Produce      json
// @Param        user  body      User  true  "User data"
// @Success      201   {object}  User
// @Failure      400   {object}  map[string]string
// @Router       /users [post]
func createUser(c *gin.Context) {
	var user User
	if err := c.ShouldBindJSON(&user); err != nil {
		c.JSON(http.StatusBadRequest, gin.H{"error": err.Error()})
		return
	}
	user.ID = 3
	c.JSON(http.StatusCreated, user)
}

    v2：添加认证文档 + 模型验证

    // main.go - v2 with authentication docs
package main

import (
	"net/http"

	"github.com/gin-gonic/gin"
	_ "myapp/docs"
	swaggerFiles "github.com/swaggo/files"
	ginSwagger "github.com/swaggo/gin-swagger"
)

// @title           Gin Swagger Demo API
// @version         2.0
// @description     A demo API with authentication.
// @host            localhost:8080
// @BasePath        /api
// @securityDefinitions.apikey BearerAuth
// @in header
// @name Authorization
func main() {
	r := gin.Default()
	r.GET("/swagger/*any", ginSwagger.WrapHandler(swaggerFiles.Handler))

	api := r.Group("/api")
	api.Use(authMiddleware())
	{
		api.GET("/profile", getProfile)
		api.PUT("/profile", updateProfile)
	}

	r.Run(":8080")
}

func authMiddleware() gin.HandlerFunc {
	return func(c *gin.Context) {
		token := c.GetHeader("Authorization")
		if token == "" {
			c.AbortWithStatusJSON(http.StatusUnauthorized,
				gin.H{"error": "missing token"})
			return
		}
		c.Set("user_id", 1)
		c.Next()
	}
}

// ProfileResponse wraps profile data
type ProfileResponse struct {
	ID    int    `json:"id" example:"1"`
	Name  string `json:"name" example:"Admin"`
	Email string `json:"email" example:"admin@example.com"`
	Role  string `json:"role" example:"admin"`
}

// @Summary      Get current user profile
// @Description  Retrieve the authenticated user profile
// @Tags         auth
// @Accept       json
// @Produce      json
// @Security     BearerAuth
// @Success      200  {object}  ProfileResponse
// @Failure      401  {object}  map[string]string
// @Router       /profile [get]
func getProfile(c *gin.Context) {
	c.JSON(http.StatusOK, ProfileResponse{
		ID: 1, Name: "Admin", Email: "admin@example.com", Role: "admin",
	})
}

// UpdateProfileRequest defines the update payload
type UpdateProfileRequest struct {
	Name  string `json:"name" example:"New Name" binding:"required"`
	Email string `json:"email" example:"new@example.com" binding:"required,email"`
}

// @Summary      Update user profile
// @Description  Update the authenticated user's profile
// @Tags         auth
// @Accept       json
// @Produce      json
// @Security     BearerAuth
// @Param        body  body      UpdateProfileRequest  true  "Update data"
// @Success      200   {object}  ProfileResponse
// @Failure      400   {object}  map[string]string
// @Failure      401   {object}  map[string]string
// @Router       /profile [put]
func updateProfile(c *gin.Context) {
	var req UpdateProfileRequest
	if err := c.ShouldBindJSON(&req); err != nil {
		c.JSON(http.StatusBadRequest, gin.H{"error": err.Error()})
		return
	}
	c.JSON(http.StatusOK, ProfileResponse{
		ID: 1, Name: req.Name, Email: req.Email, Role: "admin",
	})
}

    v3：多版本 API + 自定义 Swagger 配置

    // main.go - v3 multi-version with custom config
package main

import (
	"net/http"

	"github.com/gin-gonic/gin"
	_ "myapp/docs"
	swaggerFiles "github.com/swaggo/files"
	ginSwagger "github.com/swaggo/gin-swagger"
)

// @title           Gin Multi-Version API
// @version         3.0
// @description     Multi-version API with custom Swagger config.
// @host            localhost:8080
// @BasePath        /
// @securityDefinitions.apikey BearerAuth
// @in header
// @name Authorization
// @externalDocs.description  OpenAPI Specification
// @externalDocs.url          https://swagger.io/specification/
func main() {
	r := gin.Default()

	// Custom Swagger configuration
	r.GET("/swagger/*any", ginSwagger.WrapHandler(
		swaggerFiles.Handler,
		ginSwagger.URL("/swagger/doc.json"),
		ginSwagger.DefaultModelsExpandDepth(1), // expand models 1 level
	))

	v1 := r.Group("/api/v1")
	{
		v1.GET("/products", listProductsV1)
	}

	v2 := r.Group("/api/v2")
	{
		v2.GET("/products", listProductsV2)
	}

	r.Run(":8080")
}

// Product represents a basic product
type Product struct {
	ID    int     `json:"id" example:"1"`
	Name  string  `json:"name" example:"Laptop"`
	Price float64 `json:"price" example:"999.99"`
}

// @Summary      List products V1
// @Description  Get all products (v1 - basic info)
// @Tags         products-v1
// @Produce      json
// @Success      200  {array}  Product
// @Router       /api/v1/products [get]
func listProductsV1(c *gin.Context) {
	c.JSON(http.StatusOK, []Product{
		{ID: 1, Name: "Laptop", Price: 999.99},
		{ID: 2, Name: "Mouse", Price: 29.99},
	})
}

// ProductV2 extends Product with category field
type ProductV2 struct {
	ID       int     `json:"id" example:"1"`
	Name     string  `json:"name" example:"Laptop"`
	Price    float64 `json:"price" example:"999.99"`
	Category string  `json:"category" example:"Electronics"`
}

// @Summary      List products V2
// @Description  Get all products (v2 - with category)
// @Tags         products-v2
// @Produce      json
// @Success      200  {array}  ProductV2
// @Router       /api/v2/products [get]
func listProductsV2(c *gin.Context) {
	c.JSON(http.StatusOK, []ProductV2{
		{ID: 1, Name: "Laptop", Price: 999.99, Category: "Electronics"},
		{ID: 2, Name: "Mouse", Price: 29.99, Category: "Accessories"},
	})
}

    四、执行预览

    # Step 1: Install swag CLI tool
$ go install github.com/swaggo/swag/cmd/swag@latest

# Step 2: Initialize swagger docs (generates docs/ folder)
$ swag init -g main.go -o ./docs

# Step 3: Run the server
$ go run main.go

# Step 4: Open Swagger UI in browser
# http://localhost:8080/swagger/index.html

    浏览器打开 Swagger UI 后可以看到：

    • 所有 API 按 Tags 分组展示
    • 点击接口展开，查看参数、响应模型
    • 点击 "Try it out" 直接测试接口
    • v2 版本中，右上角有 "Authorize" 按钮，输入 Bearer Token 后所有需要认证的接口自动携带

      • 五、注意事项

      事项 说明
      注解位置 必须写在对应的 handler 函数上方，swag init 按函数名匹配路由
      @Router 路径 必须和 Gin 注册的路由完全一致，否则文档和代码不匹配
      生成步骤 每次修改注解后需重新 swag init，文档不会热更新
      docs 包导入 swag init 生成的包需要空白导入 _ "myapp/docs"
      中文描述 注解支持中文，但建议 Summary 用英文保持一致性
      生产环境 建议用构建标签 //go:build swagger 控制是否暴露文档路由

      六、避坑指南

      ❌ 错误做法 ✅ 正确做法
      手写 swagger.json 用 swag init 从注解自动生成
      忘记运行 swag init 就部署 在 Makefile / CI 流水线中自动化 swag init
      @Param 类型写错（query 写成 path） 仔细核对参数来源：path / query / header / body / formData
      导入 docs 包路径错误 确保 _ "your-module/docs" 和 -o 输出路径一致
      文档路由暴露在生产环境 用构建标签或环境变量控制
      @Router 路径和 Gin 路由不一致 直接复制路由注册代码中的路径
