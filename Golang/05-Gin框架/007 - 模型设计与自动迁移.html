007 - 模型设计与自动迁移

难度标注： ⭐⭐⭐（中级）

前置知识： Go struct、GORM 基础 CRUD

学习目标： 掌握 GORM 模型定义、字段标签、自动迁移、钩子函数、软删除、索引与模型关联

一、概念讲解

在 GORM 中，模型（Model） 是 Go 的结构体，它通过标签（tags）来映射数据库表的字段。GORM 的模型设计直接决定了数据表的结构和行为。

模型 = Struct + Tags

type User struct {
    ID    uint   `gorm:"primaryKey"`           // 主键
    Name  string `gorm:"size:100;not null"`     // VARCHAR(100), NOT NULL
    Email string `gorm:"uniqueIndex"`           // 唯一索引
}

核心概念

  • 字段标签（Field Tags）：用 gorm:"..." 控制列属性，json:"..." 控制 JSON 输出
  • 自动迁移（AutoMigrate）：根据 struct 定义自动创建/修改表结构
  • 钩子（Hooks）：在 CRUD 操作前后自动执行的回调函数
  • 软删除（Soft Delete）：标记删除而非物理删除，使用 gorm.DeletedAt
  • 关联（Associations）：BelongsTo、HasMany、Many2Many 等表关系

    三、完整 Go 代码

    v1：基础模型 + AutoMigrate + Hooks

    // models/user.go - Basic model with hooks
package models

import (
	"fmt"
	"strings"
	"time"

	"gorm.io/gorm"
)

// User model with GORM conventions
type User struct {
	ID        uint           `gorm:"primaryKey" json:"id"`
	Name      string         `gorm:"size:100;not null" json:"name"`
	Email     string         `gorm:"size:255;uniqueIndex;not null" json:"email"`
	Password  string         `gorm:"size:255;not null" json:"-"` // json:"-" hides from API
	Age       int            `gorm:"default:18" json:"age"`
	Status    string         `gorm:"size:20;default:active" json:"status"`
	CreatedAt time.Time      `json:"created_at"`
	UpdatedAt time.Time      `json:"updated_at"`
	DeletedAt gorm.DeletedAt `gorm:"index" json:"-"` // Soft delete support
}

// TableName overrides default table name
func (User) TableName() string {
	return "users"
}

// BeforeCreate hook - runs before inserting a new record
func (u *User) BeforeCreate(tx *gorm.DB) error {
	u.Email = strings.ToLower(u.Email) // Normalize email
	fmt.Printf("About to create user: %s\n", u.Name)
	return nil
}

// AfterCreate hook - runs after inserting a new record
func (u *User) AfterCreate(tx *gorm.DB) error {
	fmt.Printf("Created user #%d: %s\n", u.ID, u.Name)
	return nil
}

// BeforeUpdate hook - validate before updating
func (u *User) BeforeUpdate(tx *gorm.DB) error {
	if u.Age < 0 || u.Age > 150 {
		return fmt.Errorf("invalid age: %d", u.Age)
	}
	return nil
}
    // main.go - AutoMigrate demo
package main

import (
	"fmt"
	"log"

	"gorm.io/driver/sqlite"
	"gorm.io/gorm"
	"myapp/models"
)

func main() {
	db, err := gorm.Open(sqlite.Open("app.db"), &gorm.Config{})
	if err != nil {
		log.Fatal(err)
	}

	// Auto migrate - creates/updates tables based on struct definitions
	err = db.AutoMigrate(
		&models.User{},
	)
	if err != nil {
		log.Fatal("Migration failed:", err)
	}
	fmt.Println("Migration completed successfully")

	// Test hooks
	user := models.User{
		Name:     "Alice",
		Email:    "Alice@Example.COM", // Will be lowercased by BeforeCreate
		Password: "hashed_password",
		Age:      25,
	}
	db.Create(&user)

	// Test soft delete
	db.Delete(&user)                            // Sets deleted_at, doesn't physically delete
	var count int64
	db.Model(&models.User{}).Count(&count)      // Excludes soft-deleted records
	fmt.Printf("Active users: %d\n", count)

	db.Unscoped().Model(&models.User{}).Count(&count) // Includes soft-deleted
	fmt.Printf("All users (including deleted): %d\n", count)
}

    v2：索引 + 复合主键 + 多模型关联

    // models/models.go - Complete model with associations
package models

import (
	"time"

	"gorm.io/gorm"
)

// ========== Base model (reusable) ==========
type BaseModel struct {
	ID        uint           `gorm:"primaryKey" json:"id"`
	CreatedAt time.Time      `json:"created_at"`
	UpdatedAt time.Time      `json:"updated_at"`
	DeletedAt gorm.DeletedAt `gorm:"index" json:"-"`
}

// ========== User model ==========
type User struct {
	BaseModel
	Name     string  `gorm:"size:100;not null;index" json:"name"` // index on name
	Email    string  `gorm:"size:255;uniqueIndex;not null" json:"email"`
	Age      int     `gorm:"default:18" json:"age"`
	Profile  Profile `json:"profile,omitempty"`   // HasOne
	Posts    []Post  `json:"posts,omitempty"`     // HasMany
}

// ========== Profile (HasOne) ==========
type Profile struct {
	ID     uint   `gorm:"primaryKey" json:"id"`
	UserID uint   `gorm:"uniqueIndex" json:"user_id"` // Foreign key
	Bio    string `gorm:"size:500" json:"bio"`
	Avatar string `gorm:"size:255" json:"avatar"`
}

// ========== Post (HasMany) ==========
type Post struct {
	ID       uint      `gorm:"primaryKey" json:"id"`
	UserID   uint      `gorm:"index;not null" json:"user_id"` // Foreign key + index
	Title    string    `gorm:"size:200;not null" json:"title"`
	Content  string    `gorm:"type:text" json:"content"`
	Published bool      `gorm:"default:false;index" json:"published"` // Index for filtering
	Tags     []Tag     `gorm:"many2many:post_tags;" json:"tags,omitempty"` // Many2Many
}

// ========== Tag (Many2Many) ==========
type Tag struct {
	ID   uint   `gorm:"primaryKey" json:"id"`
	Name string `gorm:"size:50;uniqueIndex;not null" json:"name"`
}

// ========== Composite index example ==========
type Order struct {
	ID        uint      `gorm:"primaryKey" json:"id"`
	UserID    uint      `gorm:"not null;uniqueIndex:idx_user_product" json:"user_id"`
	ProductID uint      `gorm:"not null;uniqueIndex:idx_user_product" json:"product_id"` // Composite unique
	Quantity  int       `gorm:"default:1" json:"quantity"`
	Status    string    `gorm:"size:20;default:pending" json:"status"`
	CreatedAt time.Time `json:"created_at"`
}
    // main.go - Association operations
package main

import (
	"fmt"
	"log"

	"gorm.io/driver/sqlite"
	"gorm.io/gorm"
	"myapp/models"
)

func main() {
	db, err := gorm.Open(sqlite.Open("blog.db"), &gorm.Config{})
	if err != nil {
		log.Fatal(err)
	}

	// AutoMigrate all models
	db.AutoMigrate(&models.User{}, &models.Profile{}, &models.Post{}, &models.Tag{}, &models.Order{})

	// Create user with profile (HasOne)
	user := models.User{
		Name:  "Bob",
		Email: "bob@example.com",
		Profile: models.Profile{
			Bio:    "Full-stack developer",
			Avatar: "bob.png",
		},
	}
	db.Create(&user) // Creates user + profile

	// Create posts for user (HasMany)
	post1 := models.Post{
		UserID:   user.ID,
		Title:    "GORM Guide",
		Content:  "Learn GORM...",
		Published: true,
	}
	post2 := models.Post{
		UserID:   user.ID,
		Title:    "Go Best Practices",
		Content:  "...",
		Published: false,
	}
	db.Create(&post1)
	db.Create(&post2)

	// Create tags and associate (Many2Many)
	tag1 := models.Tag{Name: "Go"}
	tag2 := models.Tag{Name: "GORM"}
	db.Create(&tag1)
	db.Create(&tag2)

	// Add tags to post
	db.Model(&post1).Association("Tags").Append([]models.Tag{tag1, tag2})

	// Preload - load user with associations
	var fullUser models.User
	db.Preload("Profile").Preload("Posts").Preload("Posts.Tags").First(&fullUser, user.ID)
	fmt.Printf("User: %s, Profile: %s, Posts: %d\n",
		fullUser.Name, fullUser.Profile.Bio, len(fullUser.Posts))

	// Joins query - find users with published posts
	var activeUsers []models.User
	db.Joins("JOIN posts ON posts.user_id = users.id").
		Where("posts.published = ?", true).
		Group("users.id").
		Find(&activeUsers)
	fmt.Printf("Users with published posts: %d\n", len(activeUsers))
}

    v3：模型方法 + 作用域 + 完整项目结构

    // models/user.go - Model with scopes and methods
package models

import "gorm.io/gorm"

// Scopes - reusable query conditions
func ActiveUsers(db *gorm.DB) *gorm.DB {
	return db.Where("status = ?", "active")
}

func UsersOlderThan(age int) func(db *gorm.DB) *gorm.DB {
	return func(db *gorm.DB) *gorm.DB {
		return db.Where("age >= ?", age)
	}
}

func Paginate(page, pageSize int) func(db *gorm.DB) *gorm.DB {
	return func(db *gorm.DB) *gorm.DB {
		if page < 1 {
			page = 1
		}
		if pageSize < 1 || pageSize > 100 {
			pageSize = 10
		}
		return db.Offset((page - 1) * pageSize).Limit(pageSize)
	}
}

// Model methods
func (u *User) IsAdult() bool {
	return u.Age >= 18
}

func (u *User) DisplayName() string {
	if u.Profile.Bio != "" {
		return u.Name + " - " + u.Profile.Bio
	}
	return u.Name
}

    四、注意事项

    项目 说明 建议
    json:"-" 敏感字段（密码）不输出到 API Password 字段必须加 json:"-"
    AutoMigrate 只加列不删列不改类型 生产环境用 migration 工具（如 golang-migrate）
    软删除 查询自动排除已删除记录 需要 Unscoped() 查看全部
    索引命名 GORM 自动命名，复合索引用 uniqueIndex:idx_name 显式命名便于维护
    钩子错误 返回 error 会回滚操作 钩子中避免耗时操作
    关联创建 db.Create(&user) 会同时创建关联 注意性能，大数组关联分批处理

    五、避坑指南

    ❌ 忘记添加 DeletedAt 字段

    type User struct {
    ID   uint
    Name string
}
db.Delete(&user) // Physically deletes the record!

    ✅ 使用 gorm.DeletedAt

    type User struct {
    ID        uint           `gorm:"primaryKey"`
    Name      string
    DeletedAt gorm.DeletedAt `gorm:"index"` // Enables soft delete
}
db.Delete(&user) // Only sets deleted_at timestamp

    ❌ 在钩子中查询同表导致死循环

    func (u *User) AfterFind(tx *gorm.DB) error {
    tx.Find(&u.RelatedUsers) // May trigger hooks again!
    return nil
}

    ✅ 使用 Session 跳过钩子

    func (u *User) AfterFind(tx *gorm.DB) error {
    tx.Session(&gorm.Session{SkipHooks: true}).Find(&u.RelatedUsers)
    return nil
}

    ❌ 关联外键不正确

    type Post struct {
    UserID uint // GORM guesses: UserID -> User.ID (works)
}
type Comment struct {
    PostID uint // GORM guesses: PostID -> Post.ID (works)
    Post   Post `gorm:"foreignKey:ID"` // WRONG! Should be foreignKey:PostID
}

    ✅ 明确指定外键

    type Comment struct {
    PostID uint
    Post   Post `gorm:"foreignKey:PostID"` // Explicit is clear
}

            2. 七、知识点总结

            GORM 模型设计与自动迁移
├── 模型定义
│   ├── 基础字段: uint / string / int / bool / time.Time
│   ├── 字段标签: gorm:"size;not null;default;uniqueIndex"
│   ├── JSON 标签: json:"name" / json:"-" / json:"name,omitempty"
│   └── 自定义表名: TableName() string
├── 自动迁移
│   ├── AutoMigrate(&Model{}) - 只加列不改列
│   └── 生产建议: 使用 golang-migrate 等工具
├── 钩子函数
│   ├── BeforeCreate / AfterCreate
│   ├── BeforeUpdate / AfterUpdate
│   ├── BeforeDelete / AfterDelete
│   └── AfterFind
├── 软删除
│   ├── DeletedAt gorm.DeletedAt
│   ├── db.Unscoped() 查看全部
│   └── db.Unscoped().Delete() 物理删除
├── 索引
│   ├── 单列索引: gorm:"index"
│   ├── 唯一索引: gorm:"uniqueIndex"
│   └── 复合索引: gorm:"uniqueIndex:idx_name"
└── 模型关联
    ├── HasOne: Profile belongs to User
    ├── HasMany: User has many Posts
    ├── BelongsTo: Post belongs to User
    └── Many2Many: Post <-> Tag (join table)
