## 概念讲解

### 为什么密码不能明文存储？

明文存储密码意味着一旦数据库泄露，所有用户密码暴露。**bcrypt** 是密码哈希的行业标准：

- 自动加盐（salt），相同密码生成不同哈希
- 可调成本因子（cost），抵御暴力破解
- 单向哈希，不可逆

### 注册登录流程

**注册流程：**
1. 用户提交 username + email + password
2. 后端验证输入（格式、唯一性）
3. bcrypt 哈希密码
4. 存储到数据库
5. 返回成功（可选自动登录）

**登录流程：**
1. 用户提交 username + password
2. 查询数据库获取用户记录
3. bcrypt 对比密码
4. 生成 JWT Token
5. 返回 Token 给客户端

```go
package models

import (
    "time"

    "gorm.io/gorm"
)

// User represents a system user
type User struct {
    ID           uint           `gorm:"primaryKey" json:"id"`
    Username     string         `gorm:"uniqueIndex;size:50;not null" json:"username"`
    Email        string         `gorm:"uniqueIndex;size:100;not null" json:"email"`
    PasswordHash string         `gorm:"size:255;not null" json:"-"` // Never expose in JSON
    Nickname     string         `gorm:"size:50" json:"nickname"`
    Avatar       string         `gorm:"size:255" json:"avatar"`
    Role         string         `gorm:"size:20;default:user" json:"role"`
    Status       int            `gorm:"default:1" json:"status"` // 1=active, 0=disabled
    LastLoginAt  *time.Time     `json:"last_login_at"`
    CreatedAt    time.Time      `json:"created_at"`
    UpdatedAt    time.Time      `json:"updated_at"`
    DeletedAt    gorm.DeletedAt `gorm:"index" json:"-"`
}

// TableName specifies the database table name
func (User) TableName() string {
    return "users"
}

// RegisterRequest for user registration
type RegisterRequest struct {
    Username string `json:"username" binding:"required,min=3,max=50"`
    Email    string `json:"email" binding:"required,email"`
    Password string `json:"password" binding:"required,min=6,max=128"`
    Nickname string `json:"nickname" binding:"max=50"`
}

// LoginRequest for user login
type LoginRequest struct {
    Username string `json:"username" binding:"required"`
    Password string `json:"password" binding:"required"`
}

// UpdateProfileRequest for updating user profile
type UpdateProfileRequest struct {
    Nickname string `json:"nickname" binding:"max=50"`
    Avatar   string `json:"avatar" binding:"max=255"`
}

// UserResponse is the safe user data returned to clients
type UserResponse struct {
    ID        uint      `json:"id"`
    Username  string    `json:"username"`
    Email     string    `json:"email"`
    Nickname  string    `json:"nickname"`
    Avatar    string    `json:"avatar"`
    Role      string    `json:"role"`
```

```go
package utils

import "regexp"

// IsStrongPassword checks password strength
func IsStrongPassword(password string) bool {
    if len(password) < 6 {
        return false
    }
    // At least one letter and one number
    hasLetter := regexp.MustCompile(`[a-zA-Z]`).MatchString
    hasNumber := regexp.MustCompile(`[0-9]`).MatchString
    return hasLetter(password) && hasNumber(password)
}
```

### utils/jwt.go

```go
package utils

import (
    "time"

    "github.com/golang-jwt/jwt/v5"
)

var jwtSecret []byte

// InitJWT sets the secret key
func InitJWT(secret string) {
    jwtSecret = []byte(secret)
}

type JWTClaims struct {
    UserID   uint   `json:"user_id"`
    Username string `json:"username"`
    Role     string `json:"role"`
    jwt.RegisteredClaims
}

// GenerateToken creates a signed JWT token
func GenerateToken(userID uint, username, role string) (string, error) {
    claims := JWTClaims{
        UserID:   userID,
        Username: username,
        Role:     role,
        RegisteredClaims: jwt.RegisteredClaims{
            ExpiresAt: jwt.NewNumericDate(time.Now().Add(24 * time.Hour)),
            IssuedAt:  jwt.NewNumericDate(time.Now()),
            Issuer:    "gin-auth-demo",
        },
    }
    token := jwt.NewWithClaims(jwt.SigningMethodHS256, claims)
    return token.SignedString(jwtSecret)
}

// ParseToken validates and extracts claims
func ParseToken(tokenString string) (*JWTClaims, error) {
    claims := &JWTClaims{}
    _, err := jwt.ParseWithClaims(tokenString, claims, func(token *jwt.Token) (interface{}, error) {
        return jwtSecret, nil
    })
    return claims, err
}
```

### middleware/auth.go

```go
package middleware

import (
    "net/http"
    "strings"

    "gin-auth-demo/utils"

    "github.com/gin-gonic/gin"
)

// AuthRequired is JWT authentication middleware
func AuthRequired() gin.HandlerFunc {
    return func(c *gin.Context) {
        authHeader := c.GetHeader("Authorization")
        if authHeader == "" {
            c.JSON(http.StatusUnauthorized, gin.H{"error": "Missing authorization header"})
            c.Abort()
            return
        }

        parts := strings.SplitN(authHeader, " ", 2)
        if len(parts) != 2 || parts[0] != "Bearer" {
            c.JSON(http.StatusUnauthorized, gin.H{"error": "Invalid authorization format"})
            c.Abort()
            return
        }

        claims, err := utils.ParseToken(parts[1])
        if err != nil {
            c.JSON(http.StatusUnauthorized, gin.H{"error": "Invalid or expired token"})
            c.Abort()
            return
        }

        c.Set("userID", claims.UserID)
        c.Set("username", claims.Username)
        c.Set("role", claims.Role)
        c.Next()
    }
}
```

### handlers/auth.go

```go
package handlers

import (
    "errors"
    "net/http"
    "time"

    "gin-auth-demo/database"
    "gin-auth-demo/models"
    "gin-auth-demo/utils"

    "github.com/gin-gonic/gin"
    "golang.org/x/crypto/bcrypt"
    "gorm.io/gorm"
)

// Register handles user registration
// POST /auth/register
func Register(c *gin.Context) {
    var req models.RegisterRequest
    if err := c.ShouldBindJSON(&req); err != nil {
        c.JSON(http.StatusBadRequest, gin.H{"error": err.Error()})
        return
    }

    // Check username uniqueness
    var existingUser models.User
    if err := database.DB.Where("username = ?", req.Username).First(&existingUser).Error; err == nil {
        c.JSON(http.StatusConflict, gin.H{"error": "Username already exists"})
        return
    }

    // Check email uniqueness
    if err := database.DB.Where("email = ?", req.Email).First(&existingUser).Error; err == nil {
        c.JSON(http.StatusConflict, gin.H{"error": "Email already registered"})
        return
    }

    // Hash password with bcrypt
    hash, err := bcrypt.GenerateFromPassword([]byte(req.Password), bcrypt.DefaultCost)
    if err != nil {
        c.JSON(http.StatusInternalServerError, gin.H{"error": "Failed to hash password"})
        return
    }

    // Create user record
    nickname := req.Nickname
    if nickname == "" {
        nickname = req.Username
    }

    user := models.User{
        Username:     req.Username,
        Email:        req.Email,
        PasswordHash: string(hash),
        Nickname:     nickname,
        Role:         "user",
        Status:       1,
    }

    if err := database.DB.Create(&user).Error; err != nil {
        c.JSON(http.StatusInternalServerError, gin.H{"error": "Failed to create user"})
        return
    }

    c.JSON(http.StatusCreated, gin.H{
        "message": "Registration successful",
        "data": models.UserResponse{
            ID:       user.ID,
            Username: user.Username,
            Email:    user.Email,
            Nickname: user.Nickname,
            Role:     user.Role,
        },
    })
}

// Login handles user authentication
// POST /auth/login
func Login(c *gin.Context) {
    var req models.LoginRequest
    if err := c.ShouldBindJSON(&req); err != nil {
        c.JSON(http.StatusBadRequest, gin.H{"error": err.Error()})
        return
    }

    // Find user by username
    var user models.User
    if err := database.DB.Where("username = ?", req.Username).First(&user).Error; err != nil {
        if errors.Is(err, gorm.ErrRecordNotFound) {
            c.JSON(http.StatusUnauthorized, gin.H{"error": "Invalid credentials"})
            return
        }
        c.JSON(http.StatusInternalServerError, gin.H{"error": "Database error"})
        return
    }

    // Check user status
    if user.Status != 1 {
        c.JSON(http.StatusForbidden, gin.H{"error": "Account is disabled"})
        return
    }

    // Compare password hash
    if err := bcrypt.CompareHashAndPassword([]byte(user.PasswordHash), []byte(req.Password)); err != nil {
        c.JSON(http.StatusUnauthorized, gin.H{"error": "Invalid credentials"})
        return
    }

    // Update last login time
    now := time.Now()
    database.DB.Model(&user).Update("last_login_at", now)

    // Generate JWT token
    token, err := utils.GenerateToken(user.ID, user.Username, user.Role)
    if err != nil {
        c.JSON(http.StatusInternalServerError, gin.H{"error": "Failed to generate token"})
        return
    }

    c.JSON(http.StatusOK, gin.H{
        "message": "Login successful",
        "data": gin.H{
            "token": token,
            "user": models.UserResponse{
                ID:        user.ID,
                Username:  user.Username,
                Email:     user.Email,
                Nickname:  user.Nickname,
                Avatar:    user.Avatar,
                Role:      user.Role,
                CreatedAt: user.CreatedAt,
            },
        },
    })
}

// GetProfile returns current user info
// GET /api/profile
func GetProfile(c *gin.Context) {
    userID := c.GetUint("userID")

    var user models.User
    if err := database.DB.First(&user, userID).Error; err != nil {
        c.JSON(http.StatusNotFound, gin.H{"error": "User not found"})
        return
    }

    c.JSON(http.StatusOK, gin.H{
        "data": models.UserResponse{
            ID:        user.ID,
            Username:  user.Username,
            Email:     user.Email,
            Nickname:  user.Nickname,
            Avatar:    user.Avatar,
            Role:      user.Role,
            CreatedAt: user.CreatedAt,
        },
    })
}

// UpdateProfile updates current user info
// PUT /api/profile
func UpdateProfile(c *gin.Context) {
    userID := c.GetUint("userID")
    var req models.UpdateProfileRequest
    if err := c.ShouldBindJSON(&req); err != nil {
        c.JSON(http.StatusBadRequest, gin.H{"error": err.Error()})
        return
    }

    if err := database.DB.Model(&models.User{}).Where("id = ?", userID).Updates(req).Error; err != nil {
        c.JSON(http.StatusInternalServerError, gin.H{"error": "Failed to update profile"})
        return
    }

    c.JSON(http.StatusOK, gin.H{"message": "Profile updated"})
}

// Logout handles user logout
// POST /auth/logout
func Logout(c *gin.Context) {
    // In a simple implementation, just tell client to delete token.
    // Production: add token to blacklist (Redis) with TTL matching expiry.
    c.JSON(http.StatusOK, gin.H{"message": "Logged out successfully"})
}
```

### main.go

```go
package main

import (
    "log"

    "gin-auth-demo/config"
    "gin-auth-demo/database"
    "gin-auth-demo/handlers"
    "gin-auth-demo/middleware"
    "gin-auth-demo/utils"

    "github.com/gin-gonic/gin"
)

func main() {
    cfg := config.Load()
    database.Connect(cfg)
    utils.InitJWT(cfg.JWTSecret)

    r := gin.Default()

    // Public auth routes
    auth := r.Group("/auth")
    {
        auth.POST("/register", handlers.Register)
        auth.POST("/login", handlers.Login)
        auth.POST("/logout", handlers.Logout)
    }

    // Protected API routes
    api := r.Group("/api")
    api.Use(middleware.AuthRequired())
    {
        api.GET("/profile", handlers.GetProfile)
        api.PUT("/profile", handlers.UpdateProfile)
    }

    log.Printf("Server starting on :%s", cfg.ServerPort)
    r.Run(":" + cfg.ServerPort)
}
```

## 注意事项

| 事项 | 说明 | 严重度 |
|------|------|--------|
| bcrypt Cost | DefaultCost(10) 足够，太高影响性能 | 🟡 重要 |
| 登录限流 | 防止暴力破解，建议 5 次/分钟/IP | 🔴 严重 |
| HTTPS Only | 密码和 Token 必须加密传输 | 🔴 严重 |
| 软删除 | GORM DeletedAt 实现软删除，保留数据 | 🟢 建议 |
| 日志脱敏 | 日志中不要记录密码明文 | 🔴 严重 |
| 并发注册 | 同一用户名并发注册需数据库唯一索引保障 | 🟡 重要 |

## 避坑指南

### ❌ 明文存储密码
```go
// BAD: Storing plain text password
user.PasswordHash = req.Password
```
✅ **正确做法：** 使用 `bcrypt.GenerateFromPassword()` 哈希存储。

### ❌ 登录时提示"用户不存在"
```go
// BAD: Reveals whether username exists
if user == nil {
    return "User not found"  // Attacker can enumerate users
}
```
✅ **正确做法：** 统一返回 "用户名或密码错误"，不透露具体原因。

### ❌ 密码比较用字符串相等
```go
// BAD: String comparison
if user.PasswordHash != req.Password { ... }
```
✅ **正确做法：** 使用 `bcrypt.CompareHashAndPassword()`。

### ❌ 注册不检查唯一性
```go
// BAD: Direct insert without uniqueness check
db.Create(&user) // May fail with DB error or create duplicates
```
✅ **正确做法：** 先查询 + 数据库唯一索引双重保障。

### ❌ JSON 返回密码哈希
```go
// BAD: Password hash in API response
type User struct {
    Password string `json:"password"` // Leaked!
}
```
✅ **正确做法：** 使用 `json:"-"` 标签忽略字段，用专门的 Response 结构体。

### v2 - 数据库 + bcrypt

```go
// v2: GORM + bcrypt, proper validation
func Register(c *gin.Context) {
    var req RegisterRequest
    if err := c.ShouldBindJSON(&req); err != nil {
        c.JSON(400, gin.H{"error": err.Error()})
        return
    }
    // Uniqueness check
    var existing User
    if db.Where("username = ?", req.Username).First(&existing).Error == nil {
        c.JSON(409, gin.H{"error": "Username taken"})
        return
    }
    // bcrypt hash
    hash, _ := bcrypt.GenerateFromPassword([]byte(req.Password), bcrypt.DefaultCost)
    user := User{Username: req.Username, PasswordHash: string(hash)}
    db.Create(&user)
    c.JSON(201, gin.H{"message": "registered"})
}
```

### v3 - 完整生产级

```go
// v3: Config-driven, separate response models, logging, rate limiting
func Register(c *gin.Context) {
    req, err := validateRegisterRequest(c)
    if err != nil { respondValidationError(c, err); return }
    
    // Rate limit check
    if isRateLimited(c.ClientIP(), "register") {
        respondError(c, 429, "Too many requests"); return
    }
    
    user, err := userService.Register(req)
    if err != nil {
        handleError(c, err); return
    }
    
    logger.Info("User registered", "username", user.Username, "ip", c.ClientIP())
    respondSuccess(c, 201, toUserResponse(user))
}
```
