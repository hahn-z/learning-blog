# 参数绑定与验证

> **难度：⭐⭐ 进阶级** | 预计阅读：25分钟 | 适合：掌握基础路由和中间件，需要规范处理请求数据的开发者

## 一、概念讲解

### 1.1 什么是参数绑定？

参数绑定就是把HTTP请求中的数据**自动映射到Go结构体**。Gin基于`binding`包实现，底层使用`validator`库做验证。

```
HTTP请求 → Gin绑定 → Go结构体 → 业务逻辑
          ↓
   ShouldBindXXX() 系列方法
   ├── ShouldBindJSON()    → Body JSON
   ├── ShouldBindQuery()   → URL查询参数
   ├── ShouldBindURI()     → URL路径参数
   ├── ShouldBind()        → 表单(自动判断Content-Type)
   └── ShouldBindXML()     → Body XML
```

### 1.2 Bind vs ShouldBind

| 方法 | 验证失败时 | 建议 |
|------|-----------|------|
| `BindXXX()` | 自动返回400响应 | ❌ 不推荐，会自动设置错误响应格式 |
| `ShouldBindXXX()` | 返回error，由你决定如何响应 | ✅ 推荐，控制权在你手中 |

**始终使用`ShouldBind`系列**，这是Gin社区的最佳实践。

### 1.3 binding标签一览

```go
type User struct {
    Name  string `json:"name" binding:"required"`                      // 必填
    Email string `json:"email" binding:"required,email"`               // 必填+邮箱格式
    Age   int    `json:"age" binding:"required,gte=0,lte=150"`        // 范围: 0-150
    Role  string `json:"role" binding:"required,oneof=admin user guest"` // 枚举
    Page  int    `json:"page" binding:"omitempty,min=1"`              // 可选，但填了就要>=1
    Size  int    `json:"size" binding:"omitempty,min=1,max=100"`      // 可选范围
}
```

常用binding标签速查：

| 标签 | 含义 | 示例 |
|------|------|------|
| `required` | 必填 | `binding:"required"` |
| `omitempty` | 空值跳过后续验证 | `binding:"omitempty,min=1"` |
| `min/max` | 最小/最大值(数字) | `binding:"min=1,max=100"` |
| `gt/gte/lt/lte` | 大于/大于等于/小于/小于等于 | `binding:"gte=18"` |
| `len` | 字符串长度 | `binding:"len=11"` |
| `oneof` | 枚举值 | `binding:"oneof=a b c"` |
| `email` | 邮箱格式 | `binding:"email"` |
| `url` | URL格式 | `binding:"url"` |
| `ip` | IP地址格式 | `binding:"ip"` |
| `datetime` | 时间格式 | `binding:"datetime=2006-01-02"` |

```
Gin参数绑定体系
├── 绑定方法
│   ├── ShouldBindJSON   → JSON Body

│   ├── required / omitempty
```

## 三、完整代码

### 3.1 JSON绑定与验证

```go
package main

import (
    "net/http"
    "github.com/gin-gonic/gin"
)

// CreateUserRequest defines the expected JSON payload
type CreateUserRequest struct {
    Name     string `json:"name" binding:"required,min=2,max=50"`
    Email    string `json:"email" binding:"required,email"`
    Password string `json:"password" binding:"required,min=8"`
    Age      int    `json:"age" binding:"required,gte=1,lte=150"`
    Role     string `json:"role" binding:"omitempty,oneof=admin user guest"`
}

func main() {
    r := gin.Default()

    r.POST("/users", func(c *gin.Context) {
        var req CreateUserRequest
        if err := c.ShouldBindJSON(&req); err != nil {
            c.JSON(http.StatusBadRequest, gin.H{
                "error":   "Validation failed",
                "details": err.Error(),
            })
            return
        }
        // All validations passed, proceed with business logic
        c.JSON(http.StatusCreated, gin.H{
            "message": "User created successfully",
            "user":    req,
        })
    })

    r.Run(":8080")
}
```

### 3.2 Query与URI绑定

```go
// ListUsersQuery binds query parameters
type ListUsersQuery struct {
    Page     int    `form:"page" binding:"omitempty,min=1"`
    PageSize int    `form:"page_size" binding:"omitempty,min=1,max=100"`
    Sort     string `form:"sort" binding:"omitempty,oneof=asc desc"`
    Search   string `form:"search" binding:"omitempty,min=1"`
}

// GetUserURI binds path parameters
type GetUserURI struct {
    ID int `uri:"id" binding:"required,min=1"`
}

func main() {
    r := gin.Default()

    // Bind query parameters
    r.GET("/users", func(c *gin.Context) {
        var query ListUsersQuery
        // Set defaults before binding
        query.Page = 1
        query.PageSize = 10
        
        if err := c.ShouldBindQuery(&query); err != nil {
            c.JSON(http.StatusBadRequest, gin.H{"error": err.Error()})
            return
        }
        c.JSON(http.StatusOK, gin.H{
            "page":      query.Page,
            "page_size": query.PageSize,
            "sort":      query.Sort,
            "search":    query.Search,
        })
    })

    // Bind URI parameters
    r.GET("/users/:id", func(c *gin.Context) {
        var uri GetUserURI
        if err := c.ShouldBindUri(&uri); err != nil {
            c.JSON(http.StatusBadRequest, gin.H{"error": "Invalid user ID"})
            return
        }
        c.JSON(http.StatusOK, gin.H{"user_id": uri.ID})
    })

    r.Run(":8080")
}
```

### 3.3 自定义验证器

```go
package main

import (
    "net/http"
    "regexp"
    "github.com/gin-gonic/gin"
    "github.com/gin-gonic/gin/binding"
    "github.com/go-playground/validator/v10"
)

// RegisterRequest with custom validation tags
type RegisterRequest struct {
    Username string `json:"username" binding:"required,min=3,max=20,alphanum"`
    Phone    string `json:"phone" binding:"required,validPhone"`
}

// Custom phone validation function (Chinese mobile format)
var validPhone validator.Func = func(fl validator.FieldLevel) bool {
    phone, ok := fl.Field().Interface().(string)
    if !ok {
        return false
    }
    matched, _ := regexp.MatchString(`^1[3-9]\d{9}$`, phone)
    return matched
}

func main() {
    r := gin.Default()

    // Register custom validator
    if v, ok := binding.Validator.Engine().(*validator.Validate); ok {
        v.RegisterValidation("validPhone", validPhone)
    }

    r.POST("/register", func(c *gin.Context) {
        var req RegisterRequest
        if err := c.ShouldBindJSON(&req); err != nil {
            c.JSON(http.StatusBadRequest, gin.H{"error": err.Error()})
            return
        }
        c.JSON(http.StatusOK, gin.H{"message": "registered", "username": req.Username})
    })

    r.Run(":8080")
}
```

### 3.4 文件上传

```go
// File upload handlers
func main() {
    r := gin.Default()

    // Single file upload
    r.POST("/upload", func(c *gin.Context) {
        file, err := c.FormFile("file")
        if err != nil {
            c.JSON(http.StatusBadRequest, gin.H{"error": "No file uploaded"})
            return
        }
        
        // Save file
        dst := "./uploads/" + file.Filename
        if err := c.SaveUploadedFile(file, dst); err != nil {
            c.JSON(http.StatusInternalServerError, gin.H{"error": "Failed to save file"})
            return
        }
        
        c.JSON(http.StatusOK, gin.H{
            "filename": file.Filename,
            "size":     file.Size,
            "saved_to": dst,
        })
    })

    // Multiple files upload
    r.POST("/uploads", func(c *gin.Context) {
        form, err := c.MultipartForm()
        if err != nil {
            c.JSON(http.StatusBadRequest, gin.H{"error": err.Error()})
            return
        }
        
        files := form.File["files"]
        var saved []string
        for _, file := range files {
            dst := "./uploads/" + file.Filename
            if err := c.SaveUploadedFile(file, dst); err == nil {
                saved = append(saved, file.Filename)
            }
        }
        
        c.JSON(http.StatusOK, gin.H{
            "total": len(files),
            "saved": saved,
        })
    })

    r.Run(":8080")
}
```

## 五、注意事项

| 事项 | 说明 | 建议 |
|------|------|------|
| 指针传参 | `ShouldBind`必须传指针 | `&req`，否则修改不生效 |
| json vs form | json标签用于JSON，form标签用于Query/Form | 按需添加两种标签 |
| 默认值 | binding不设默认值 | 在绑定前手动设置默认值 |
| 文件大小 | 默认32MB | `r.MaxMultipartMemory = 8 << 20` (8MB) |
| 多标签组合 | 用逗号分隔 | `binding:"required,min=1,max=100"` |

## 六、避坑指南

### ❌ 用Bind而不是ShouldBind
```go
// Bad: Bind auto-writes 400 response, you lose control
var req CreateUserRequest
if err := c.Bind(&req); err != nil {
    // Error response already sent! You can't customize it.
    return
}
```
### ✅ 始终用ShouldBind系列
```go
var req CreateUserRequest
if err := c.ShouldBindJSON(&req); err != nil {
    c.JSON(http.StatusBadRequest, gin.H{
        "error":   "Invalid request",
        "details": err.Error(), // You control the format
    })
    return
}
```

### ❌ 忘记设置默认值
```go
type Query struct {
    Page int `form:"page" binding:"omitempty,min=1"`
}
// If user doesn't send ?page=, Page is 0 (zero value!)
```
### ✅ 绑定前设置默认值
```go
var query Query
query.Page = 1 // Set default before binding
query.PageSize = 10
if err := c.ShouldBindQuery(&query); err != nil {
    // ...
}
```

### ❌ 文件上传不限制大小和类型
```go
// Bad: Accepts any file of any size
file, _ := c.FormFile("file")
c.SaveUploadedFile(file, dst)
```
### ✅ 校验文件大小和类型
```go
file, err := c.FormFile("file")
if err != nil {
    c.JSON(400, gin.H{"error": "upload failed"})
    return
}
// Limit size (e.g., 5MB)
if file.Size > 5<<20 {
    c.JSON(400, gin.H{"error": "file too large (max 5MB)"})
    return
}
// Validate extension
ext := filepath.Ext(file.Filename)
allowed := map[string]bool{".jpg": true, ".png": true, ".pdf": true}
if !allowed[ext] {
    c.JSON(400, gin.H{"error": "file type not allowed"})
    return
}
```

## 九、举一反三

| Gin绑定 | 其他框架 | 关键差异 |
|---------|---------|---------|
| `ShouldBindJSON` | Fiber: `BodyParser` | Fiber自动检测Content-Type |
| `binding:"required"` | NestJS: class-validator | Go用标签，TS用装饰器 |
| `ShouldBindQuery` | Echo: `Bind` | Echo绑定需要实现Binder接口 |
| 自定义验证器 | Express: Joi/custom | Gin底层用go-playground/validator |
| `FormFile` | Multer(Express) | Gin内置，无需额外包 |

### v3: 完整验证+自定义验证器+统一错误处理
```go
package main

import (
    "net/http"
    "regexp"
    "strings"
    "github.com/gin-gonic/gin"
    "github.com/gin-gonic/gin/binding"
    "github.com/go-playground/validator/v10"
)

type ErrorResponse struct {
    Field   string `json:"field"`
    Message string `json:"message"`
}

// Unified response structure
type APIResponse struct {
    Code    int             `json:"code"`
    Message string          `json:"message"`
    Data    interface{}     `json:"data,omitempty"`
    Errors  []ErrorResponse `json:"errors,omitempty"`
}

func formatValidationErrors(err error) []ErrorResponse {
    var errors []ErrorResponse
    if ve, ok := err.(validator.ValidationErrors); ok {
        for _, fe := range ve {
            msg := getErrorMsg(fe.Tag(), fe.Field(), fe.Param())
            errors = append(errors, ErrorResponse{
                Field:   strings.ToLower(fe.Field()),
                Message: msg,
            })
        }
    } else {
        errors = append(errors, ErrorResponse{
            Field:   "_",
            Message: err.Error(),
        })
    }
    return errors
}

func getErrorMsg(tag, field, param string) string {
    msgs := map[string]string{
        "required": field + " is required",
        "email":    field + " must be a valid email",
        "min":      field + " must be at least " + param,
        "max":      field + " must be at most " + param,
        "oneof":    field + " must be one of: " + param,
        "gte":      field + " must be >= " + param,
        "lte":      field + " must be <= " + param,
    }
    if msg, ok := msgs[tag]; ok {
        return msg
    }
    return field + " failed " + tag + " validation"
}

// Custom validator: strong password
var strongPassword validator.Func = func(fl validator.FieldLevel) bool {
    pwd, _ := fl.Field().Interface().(string)
    if len(pwd) < 8 {
        return false
    }
    hasUpper := regexp.MustCompile(`[A-Z]`).MatchString(pwd)
    hasLower := regexp.MustCompile(`[a-z]`).MatchString(pwd)
    hasDigit := regexp.MustCompile(`[0-9]`).MatchString(pwd)
    return hasUpper && hasLower && hasDigit
}

type CreateUserReq struct {
    Name     string `json:"name" binding:"required,min=2,max=50"`
    Email    string `json:"email" binding:"required,email"`
    Password string `json:"password" binding:"required,strongpwd"`
    Role     string `json:"role" binding:"required,oneof=admin user guest"`
}

func main() {
    r := gin.Default()

    if v, ok := binding.Validator.Engine().(*validator.Validate); ok {
        v.RegisterValidation("strongpwd", strongPassword)
    }

    r.POST("/users", func(c *gin.Context) {
        var req CreateUserReq
        if err := c.ShouldBindJSON(&req); err != nil {
            c.JSON(http.StatusBadRequest, APIResponse{
                Code:    400,
                Message: "Validation failed",
                Errors:  formatValidationErrors(err),
            })
            return
        }
        c.JSON(http.StatusCreated, APIResponse{
            Code:    201,
            Message: "User created",
            Data:    req,
        })
    })

    r.Run(":8080")
}
```
