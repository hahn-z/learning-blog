## 一、概念讲解

Web安全是上线前必须重视的环节。OWASP Top 10是业界公认的最常见Web安全风险清单。本节覆盖Gin中的核心安全防护。

### OWASP Top 10（2021版核心）

1. **A01 - 权限控制失效**：越权访问
2. **A02 - 加密失败**：敏感信息明文传输/存储
3. **A03 - 注入**：SQL注入、XSS、命令注入
4. **A04 - 不安全设计**：架构设计缺陷
5. **A05 - 安全配置错误**：默认配置、多余功能
6. **A07 - 认证失败**：弱密码、会话管理不当

### v1：基础防护（输入验证+安全头部）

```go
// middleware/security.go - Security middleware
package middleware

import (
    "net/http"
    "strings"
    "github.com/gin-gonic/gin"
)

// SecurityHeaders adds security-related HTTP headers
func SecurityHeaders() gin.HandlerFunc {
    return func(c *gin.Context) {
        c.Header("X-Content-Type-Options", "nosniff")
        c.Header("X-Frame-Options", "DENY")
        c.Header("X-XSS-Protection", "1; mode=block")
        c.Header("Strict-Transport-Security", "max-age=31536000; includeSubDomains")
        c.Header("Content-Security-Policy", "default-src 'self'")
        c.Header("Referrer-Policy", "strict-origin-when-cross-origin")
        c.Next()
    }
}

// InputSanitizer middleware sanitizes common injection patterns
type SanitizeConfig struct {
    MaxContentLength int
    StripTags        bool
}

func InputSanitizer(cfg SanitizeConfig) gin.HandlerFunc {
    return func(c *gin.Context) {
        if cfg.MaxContentLength > 0 {
            if c.Request.ContentLength > int64(cfg.MaxContentLength) {
                c.AbortWithStatusJSON(http.StatusRequestEntityTooLarge, gin.H{
                    "error": "request body too large",
                })
                return
            }
        }
        c.Next()
    }
}
```

### v2：认证安全（JWT+bcrypt+CSRF）

```go
// internal/auth/jwt.go - JWT authentication
package auth

import (
    "errors"
    "time"
    "github.com/golang-jwt/jwt/v5"
)

var jwtSecret = []byte("your-256-bit-secret-key-change-in-production")

type Claims struct {
    UserID int64  `json:"user_id"`
    Role   string `json:"role"`
    jwt.RegisteredClaims
}

// GenerateToken creates a JWT access token
func GenerateToken(userID int64, role string) (string, error) {
    claims := Claims{
        UserID: userID,
        Role:   role,
        RegisteredClaims: jwt.RegisteredClaims{
            ExpiresAt: jwt.NewNumericDate(time.Now().Add(2 * time.Hour)),
            IssuedAt:  jwt.NewNumericDate(time.Now()),
            Issuer:    "myapp",
        },
    }
    token := jwt.NewWithClaims(jwt.SigningMethodHS256, claims)
    return token.SignedString(jwtSecret)
}

// ParseToken validates and parses a JWT token
func ParseToken(tokenStr string) (*Claims, error) {
    token, err := jwt.ParseWithClaims(tokenStr, &Claims{},
        func(token *jwt.Token) (interface{}, error) {
            if _, ok := token.Method.(*jwt.SigningMethodHMAC); !ok {
                return nil, errors.New("unexpected signing method")
            }
            return jwtSecret, nil
        },
    )
    if err != nil {
        return nil, err
    }
    claims, ok := token.Claims.(*Claims)
    if !ok || !token.Valid {
        return nil, errors.New("invalid token")
    }
    return claims, nil
}
```

```go
// internal/auth/password.go - Password security
package auth

import (
    "golang.org/x/crypto/bcrypt"
)

// HashPassword hashes a password using bcrypt
func HashPassword(password string) (string, error) {
    bytes, err := bcrypt.GenerateFromPassword([]byte(password), bcrypt.DefaultCost)
    return string(bytes), err
}

// CheckPassword compares a password with its hash
func CheckPassword(password, hash string) bool {
    err := bcrypt.CompareHashAndPassword([]byte(hash), []byte(password))
    return err == nil
}
```

```go
// middleware/auth.go - Auth middleware
package middleware

import (
    "net/http"
    "strings"
    "myapp/internal/auth"
    "github.com/gin-gonic/gin"
)

// JWTAuth validates JWT tokens from Authorization header
func JWTAuth() gin.HandlerFunc {
    return func(c *gin.Context) {
        authHeader := c.GetHeader("Authorization")
        if authHeader == "" {
            c.AbortWithStatusJSON(http.StatusUnauthorized, gin.H{"error": "missing token"})
            return
        }
        parts := strings.SplitN(authHeader, " ", 2)
        if len(parts) != 2 || parts[0] != "Bearer" {
            c.AbortWithStatusJSON(http.StatusUnauthorized, gin.H{"error": "invalid token format"})
            return
        }
        claims, err := auth.ParseToken(parts[1])
        if err != nil {
            c.AbortWithStatusJSON(http.StatusUnauthorized, gin.H{"error": "invalid token"})
            return
        }
        c.Set("userID", claims.UserID)
        c.Set("userRole", claims.Role)
        c.Next()
    }
}

// RBAC checks user role against required roles
func RBAC(allowedRoles ...string) gin.HandlerFunc {
    roleSet := make(map[string]bool)
    for _, r := range allowedRoles {
        roleSet[r] = true
    }
    return func(c *gin.Context) {
        role, _ := c.Get("userRole")
        if !roleSet[role.(string)] {
            c.AbortWithStatusJSON(http.StatusForbidden, gin.H{"error": "insufficient permissions"})
            return
        }
        c.Next()
    }
}

// RateLimit simple rate limiter per IP
func RateLimit(maxRequests int, per time.Duration) gin.HandlerFunc {
    type visitor struct {
        count    int
        lastSeen time.Time
    }
    visitors := make(map[string]*visitor)
    mu := sync.Mutex{}

    return func(c *gin.Context) {
        ip := c.ClientIP()
        mu.Lock()
        defer mu.Unlock()

        v, exists := visitors[ip]
        if !exists || time.Since(v.lastSeen) > per {
            visitors[ip] = &visitor{count: 1, lastSeen: time.Now()}
            c.Next()
            return
        }
        v.count++
        v.lastSeen = time.Now()
        if v.count > maxRequests {
            c.AbortWithStatusJSON(http.StatusTooManyRequests, gin.H{"error": "rate limit exceeded"})
            return
        }
        c.Next()
    }
}
```

```go
// repository/user.go - Safe SQL with parameterized queries
package repository

import (
    "context"
    "database/sql"
    "myapp/internal/model"
)

type UserRepository struct {
    db *sql.DB
}

// GetByEmail uses parameterized query to prevent SQL injection
func (r *UserRepository) GetByEmail(ctx context.Context, email string) (*model.User, error) {
    // ✅ Safe: parameterized query
    query := "SELECT id, name, email, password_hash FROM users WHERE email = ?"
    row := r.db.QueryRowContext(ctx, query, email)
    
    var user model.User
    if err := row.Scan(&user.ID, &user.Name, &user.Email, &user.PasswordHash); err != nil {
        return nil, err
    }
    return &user, nil
}

// NEVER do this:
// query := fmt.Sprintf("SELECT * FROM users WHERE email = '%s'", email) // ❌ SQL INJECTION!
```

### v3：完整安全方案

```go
// cmd/server/main.go - Security-hardened server
package main

import (
    "crypto/tls"
    "log"
    "net/http"
    "time"
    "github.com/gin-gonic/gin"
    "myapp/middleware"
)

func main() {
    gin.SetMode(gin.ReleaseMode)
    r := gin.New()

    // Global security middleware
    r.Use(
        middleware.SecurityHeaders(),              // Security headers
        middleware.InputSanitizer(middleware.SanitizeConfig{
            MaxContentLength: 10 << 20, // 10MB max
        }),
        middleware.RateLimit(100, time.Minute),    // 100 req/min per IP
        gin.Logger(),                              // Access logging
        gin.Recovery(),                            // Panic recovery
    )

    // Public routes
    r.POST("/login", handleLogin)
    r.POST("/register", handleRegister)

    // Protected routes
    auth := r.Group("/api")
    auth.Use(middleware.JWTAuth())
    {
        auth.GET("/profile", handleProfile)
        
        // Admin-only routes
        admin := auth.Group("/admin")
        admin.Use(middleware.RBAC("admin"))
        {
            admin.GET("/users", handleListUsers)
        }
    }

    // TLS configuration
    tlsCfg := &tls.Config{
        MinVersion: tls.VersionTLS12,
        CurvePreferences: []tls.CurveID{
            tls.X25519,
            tls.CurveP256,
        },
    }

    srv := &http.Server{
        Addr:         ":8443",
        Handler:      r,
        TLSConfig:    tlsCfg,
        ReadTimeout:  10 * time.Second,
        WriteTimeout: 30 * time.Second,
        IdleTimeout:  60 * time.Second,
    }

    log.Println("Server on :8443 (HTTPS)")
    log.Fatal(srv.ListenAndServeTLS("cert.pem", "key.pem"))
}

func handleLogin(c *gin.Context) {
    var req struct {
        Email    string `json:"email" binding:"required,email"`
        Password string `json:"password" binding:"required,min=8"`
    }
    if err := c.ShouldBindJSON(&req); err != nil {
        c.JSON(http.StatusBadRequest, gin.H{"error": "invalid input"})
        return
    }
    // Verify credentials...
    token, _ := auth.GenerateToken(1, "user")
    c.JSON(http.StatusOK, gin.H{"token": token})
}
```

## 五、注意事项

| 事项 | 说明 |
|------|------|
| JWT密钥 | 生产环境必须从环境变量/Vault加载，不硬编码 |
| bcrypt cost | DefaultCost(10)够用，别设太高影响性能 |
| HTTPS | 生产环境必须开启，Let's Encrypt免费证书 |
| 错误信息 | 不要返回内部错误详情给客户端 |
| 日志审计 | 敏感操作必须记录，但不能记录密码 |

## 六、避坑指南

| ❌ 错误做法 | ✅ 正确做法 |
|------------|-----------|
| `fmt.Sprintf`拼接SQL | 参数化查询/ORM |
| `md5/sha1`存密码 | `bcrypt`/`argon2id` |
| JWT存在localStorage | HttpOnly Cookie + CSRF Token |
| 明文HTTP传输 | HTTPS + HSTS |
| 返回完整错误堆栈 | 通用错误信息 + 服务端日志 |
| 硬编码密钥 | 环境变量/Vault/配置中心 |
