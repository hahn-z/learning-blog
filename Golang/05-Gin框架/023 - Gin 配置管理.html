一、概念讲解

配置管理是应用的"大脑"。数据库连接、端口、密钥、超时时间等都通过配置管理。Viper 是 Go 生态最流行的配置库，支持多种格式和环境变量覆盖。

配置来源优先级（从高到低）：

优先级 来源 示例
1 命令行 Flag --port=8080
2 环境变量 APP_PORT=8080
3 配置文件 config.yaml
4 默认值 viper.SetDefault("port", 3000)

为什么用 Viper？

  • 🔄 支持热更新：运行时修改配置文件，自动生效
  • 🌍 多环境支持：dev/staging/prod 各一份配置
  • 🔐 敏感信息处理：环境变量覆盖文件中的密钥
  • 📦 多格式支持：JSON、YAML、TOML、ENV、INI

    // main.go - v1 basic config
package main

import (
	"fmt"
	"log"
	"net/http"

	"github.com/gin-gonic/gin"
	"github.com/spf13/viper"
)

func main() {
	// Load config from YAML file
	viper.SetConfigName("config")
	viper.SetConfigType("yaml")
	viper.AddConfigPath(".")
	viper.AddConfigPath("./configs")

	if err := viper.ReadInConfig(); err != nil {
		log.Fatalf("Failed to read config: %v", err)
	}

	// Set defaults
	viper.SetDefault("server.port", 8080)
	viper.SetDefault("server.mode", "debug")

	fmt.Printf("Config loaded: %s\n", viper.ConfigFileUsed())
	fmt.Printf("Server port: %d\n", viper.GetInt("server.port"))
	fmt.Printf("Database host: %s\n", viper.GetString("database.host"))

	// Set Gin mode from config
	gin.SetMode(viper.GetString("server.mode"))

	r := gin.Default()
	r.GET("/config", func(c *gin.Context) {
		c.JSON(http.StatusOK, gin.H{
			"port": viper.GetInt("server.port"),
			"mode": viper.GetString("server.mode"),
			"db":   viper.GetString("database.host"),
		})
	})

	addr := fmt.Sprintf(":%d", viper.GetInt("server.port"))
	r.Run(addr)
}

    v2：结构体映射 + 多环境 + 环境变量覆盖

    // config/config.go
package config

import (
	"fmt"
	"log"
	"strings"

	"github.com/spf13/viper"
)

// Config represents the entire application configuration
type Config struct {
	Server   ServerConfig   `mapstructure:"server"`
	Database DatabaseConfig `mapstructure:"database"`
	Redis    RedisConfig    `mapstructure:"redis"`
}

// ServerConfig holds server settings
type ServerConfig struct {
	Port int    `mapstructure:"port"`
	Mode string `mapstructure:"mode"`
}

// DatabaseConfig holds database connection settings
type DatabaseConfig struct {
	Host     string `mapstructure:"host"`
	Port     int    `mapstructure:"port"`
	Name     string `mapstructure:"name"`
	User     string `mapstructure:"user"`
	Password string `mapstructure:"password"`
}

// RedisConfig holds Redis connection settings
type RedisConfig struct {
	Host string `mapstructure:"host"`
	Port int    `mapstructure:"port"`
}

// Load reads config from file and environment variables
func Load() (*Config, error) {
	// Determine environment (default: dev)
	env := viper.GetString("APP_ENV")
	if env == "" {
		env = "dev"
	}

	viper.SetConfigName(fmt.Sprintf("config.%s", env))
	viper.SetConfigType("yaml")
	viper.AddConfigPath("./configs")
	viper.AddConfigPath(".")

	// Environment variable overrides
	viper.SetEnvPrefix("APP")          // APP_SERVER_PORT
	viper.SetEnvKeyReplacer(strings.NewReplacer(".", "_"))
	viper.AutomaticEnv()

	// Defaults
	viper.SetDefault("server.port", 8080)
	viper.SetDefault("server.mode", "debug")
	viper.SetDefault("database.host", "localhost")
	viper.SetDefault("database.port", 3306)

	if err := viper.ReadInConfig(); err != nil {
		return nil, fmt.Errorf("read config: %w", err)
	}

	var cfg Config
	if err := viper.Unmarshal(&cfg); err != nil {
		return nil, fmt.Errorf("unmarshal config: %w", err)
	}

	return &cfg, nil
}

    v3：热更新 + 敏感信息处理

    // config/config.go - v3 hot reload
package config

import (
	"fmt"
	"log"
	"strings"
	"sync"

	"github.com/fsnotify/fsnotify"
	"github.com/spf13/viper"
)

type Config struct {
	Server   ServerConfig   `mapstructure:"server"`
	Database DatabaseConfig `mapstructure:"database"`
	Redis    RedisConfig    `mapstructure:"redis"`
	JWT      JWTConfig      `mapstructure:"jwt"`
}

type ServerConfig struct {
	Port         int `mapstructure:"port"`
	ReadTimeout  int `mapstructure:"read_timeout"`  // seconds
	WriteTimeout int `mapstructure:"write_timeout"` // seconds
	Mode         string `mapstructure:"mode"`
}

type DatabaseConfig struct {
	Host     string `mapstructure:"host"`
	Port     int    `mapstructure:"port"`
	Name     string `mapstructure:"name"`
	User     string `mapstructure:"user"`
	Password string `mapstructure:"password"` // from env: APP_DATABASE_PASSWORD
	MaxOpen  int    `mapstructure:"max_open"`
	MaxIdle  int    `mapstructure:"max_idle"`
}

type RedisConfig struct {
	Host     string `mapstructure:"host"`
	Port     int    `mapstructure:"port"`
	Password string `mapstructure:"password"` // from env: APP_REDIS_PASSWORD
	DB       int    `mapstructure:"db"`
}

type JWTConfig struct {
	Secret string `mapstructure:"secret"` // from env: APP_JWT_SECRET
	TTL    int    `mapstructure:"ttl"`    // hours
}

var (
	globalCfg *Config
	mu        sync.RWMutex
)

// Get returns the current config (thread-safe)
func Get() *Config {
	mu.RLock()
	defer mu.RUnlock()
	return globalCfg
}

// Load initializes config with hot reload support
func Load() (*Config, error) {
	env := getEnvOrDefault("APP_ENV", "dev")

	viper.SetConfigName(fmt.Sprintf("config.%s", env))
	viper.SetConfigType("yaml")
	viper.AddConfigPath("./configs")
	viper.AddConfigPath(".")

	viper.SetEnvPrefix("APP")
	viper.SetEnvKeyReplacer(strings.NewReplacer(".", "_"))
	viper.AutomaticEnv()

	setDefaults()

	if err := viper.ReadInConfig(); err != nil {
		return nil, fmt.Errorf("read config: %w", err)
	}

	cfg, err := unmarshal()
	if err != nil {
		return nil, err
	}

	globalCfg = cfg

	// Watch for config file changes
	viper.WatchConfig()
	viper.OnConfigChange(func(e fsnotify.Event) {
		log.Printf("Config file changed: %s", e.Name)
		newCfg, err := unmarshal()
		if err != nil {
			log.Printf("Failed to reload config: %v", err)
			return
		}
		mu.Lock()
		globalCfg = newCfg
		mu.Unlock()
		log.Println("Config reloaded successfully")
	})

	return cfg, nil
}

func unmarshal() (*Config, error) {
	var cfg Config
	if err := viper.Unmarshal(&cfg); err != nil {
		return nil, fmt.Errorf("unmarshal: %w", err)
	}
	return &cfg, nil
}

func setDefaults() {
	viper.SetDefault("server.port", 8080)
	viper.SetDefault("server.mode", "debug")
	viper.SetDefault("server.read_timeout", 10)
	viper.SetDefault("server.write_timeout", 30)
	viper.SetDefault("database.host", "localhost")
	viper.SetDefault("database.port", 3306)
	viper.SetDefault("database.max_open", 100)
	viper.SetDefault("database.max_idle", 10)
	viper.SetDefault("redis.host", "localhost")
	viper.SetDefault("redis.port", 6379)
	viper.SetDefault("redis.db", 0)
	viper.SetDefault("jwt.ttl", 24)
}

func getEnvOrDefault(key, def string) string {
	if v := strings.TrimSpace(os.Getenv(key)); v != "" {
		return v
	}
	return def
}
    // main.go - v3 hot reload in action
package main

import (
	"fmt"
	"log"
	"net/http"

	"myapp/config"

	"github.com/gin-gonic/gin"
)

func main() {
	cfg, err := config.Load()
	if err != nil {
		log.Fatalf("Config error: %v", err)
	}

	gin.SetMode(cfg.Server.Mode)
	r := gin.Default()

	// This endpoint always reads the latest config
	r.GET("/config", func(c *gin.Context) {
		current := config.Get()
		c.JSON(http.StatusOK, gin.H{
			"port":      current.Server.Port,
			"db_host":   current.Database.Host,
			"jwt_ttl":   current.JWT.TTL,
			"redis_host": current.Redis.Host,
		})
	})

	r.GET("/health", func(c *gin.Context) {
		current := config.Get()
		c.JSON(http.StatusOK, gin.H{
			"status":   "ok",
			"database": current.Database.Host,
		})
	})

	log.Printf("Starting server on :%d", cfg.Server.Port)
	r.Run(fmt.Sprintf(":%d", cfg.Server.Port))
}

    五、注意事项

    事项 说明
    环境变量命名 APP_SERVER_PORT 对应 server.port，用 _ 替代 .
    密码不走配置文件 敏感信息通过环境变量注入，不提交到 Git
    热更新范围 不是所有配置都适合热更新（如端口变更需要重启）
    类型转换 viper.GetInt() 可能对 string 数字返回 0，注意原始类型
    配置校验 加载后应该校验必要字段，避免运行时空指针

    六、避坑指南

    ❌ 错误做法 ✅ 正确做法
    密码写死在 config.yaml 中 通过 APP_DB_PASSWORD 环境变量注入
    到处用 viper.GetString() Unmarshal 到结构体，类型安全
    只有一个 config.yaml 按环境分文件：config.dev.yaml / config.prod.yaml
    热更新不加锁 sync.RWMutex 保护全局配置对象
    配置文件提交到 Git .gitignore 排除 config.prod.yaml，提供 config.example.yaml
    不设默认值 SetDefault 确保缺失配置不会 panic
