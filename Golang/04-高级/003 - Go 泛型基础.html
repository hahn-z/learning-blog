概念讲解

Go 泛型是什么？

Go 1.18 引入了泛型（Generics），允许函数和类型使用类型参数，实现真正的类型安全复用。在此之前，Go 只能用 interface{} 实现通用逻辑，牺牲了类型安全和性能。

核心概念速览

概念 说明 示例
类型参数 函数/类型的占位类型 func PrintT any
类型约束 限制类型参数的范围 `[T int \ float64]`
类型实例化 用具体类型替换类型参数 Printint
类型推断 编译器自动推断类型参数 Print(42) → T=int

any vs interface{}

// 以下两种写法完全等价（any 是 interface{} 的别名）
func Old(v interface{}) {}
func New(v any)          {}

// 但泛型比 interface{} 强在：编译时类型安全 + 无装箱开销
func Generic[T any](v T) T { return v }

内置约束：comparable 和 any

  • any：无约束，等价于 interface{}
  • comparable：支持 == 和 != 的类型（数字、字符串、布尔、指针……）

    代码演进

    v1：泛型函数 — Filter / Map / Reduce

    package main

import "fmt"

// Filter returns elements matching the predicate
func Filter[T any](slice []T, predicate func(T) bool) []T {
    var result []T
    for _, v := range slice {
        if predicate(v) {
            result = append(result, v)
        }
    }
    return result
}

// Map transforms each element
func Map[T any, U any](slice []T, transform func(T) U) []U {
    result := make([]U, len(slice))
    for i, v := range slice {
        result[i] = transform(v)
    }
    return result
}

// Reduce folds the slice into a single value
func Reduce[T any, U any](slice []T, initial U, fn func(U, T) U) U {
    acc := initial
    for _, v := range slice {
        acc = fn(acc, v)
    }
    return acc
}

func main() {
    nums := []int{1, 2, 3, 4, 5, 6, 7, 8}

    // Filter even numbers
    evens := Filter(nums, func(n int) bool { return n%2 == 0 })
    fmt.Println("Evens:", evens)

    // Map to strings
    strs := Map(nums, func(n int) string {
        return fmt.Sprintf("<%d>", n)
    })
    fmt.Println("Mapped:", strs)

    // Reduce to sum
    sum := Reduce(nums, 0, func(acc, n int) int { return acc + n })
    fmt.Println("Sum:", sum)
}

    执行预览：

    Evens: [2 4 6 8]
Mapped: [<1> <2> <3> <4> <5> <6> <7> <8>]
Sum: 36

    v2：泛型类型 — Stack / Set / Pair

    package main

import "fmt"

// Stack is a generic LIFO data structure
type Stack[T any] struct {
    items []T
}

// Push adds an element
func (s *Stack[T]) Push(v T) {
    s.items = append(s.items, v)
}

// Pop removes and returns the top element
func (s *Stack[T]) Pop() (T, bool) {
    if len(s.items) == 0 {
        var zero T
        return zero, false
    }
    top := s.items[len(s.items)-1]
    s.items = s.items[:len(s.items)-1]
    return top, true
}

// Peek returns the top element without removing
func (s *Stack[T]) Peek() (T, bool) {
    if len(s.items) == 0 {
        var zero T
        return zero, false
    }
    return s.items[len(s.items)-1], true
}

// Len returns stack size
func (s *Stack[T]) Len() int {
    return len(s.items)
}

// Set is a generic unique collection
type Set[T comparable] struct {
    items map[T]struct{}
}

// NewSet creates a new Set
func NewSet[T comparable]() *Set[T] {
    return &Set[T]{items: make(map[T]struct{})}
}

// Add inserts a value
func (s *Set[T]) Add(v T) {
    s.items[v] = struct{}{}
}

// Contains checks membership
func (s *Set[T]) Contains(v T) bool {
    _, ok := s.items[v]
    return ok
}

// Elements returns all elements as a slice
func (s *Set[T]) Elements() []T {
    result := make([]T, 0, len(s.items))
    for k := range s.items {
        result = append(result, k)
    }
    return result
}

func main() {
    // Stack demo
    s := &Stack[string]{}
    s.Push("hello")
    s.Push("world")
    s.Push("!")

    for s.Len() > 0 {
        if v, ok := s.Pop(); ok {
            fmt.Println("Pop:", v)
        }
    }

    // Set demo
    set := NewSet[int]()
    set.Add(1)
    set.Add(2)
    set.Add(2) // duplicate, ignored
    set.Add(3)
    fmt.Println("Contains 2:", set.Contains(2))
    fmt.Println("Contains 4:", set.Contains(4))
    fmt.Println("Elements:", set.Elements())
}

    执行预览：

    Pop: !
Pop: world
Pop: hello
Contains 2: true
Contains 4: false
Elements: [1 2 3]

    v3：类型约束 + 泛型算法 — 排序、最大值、二分查找

    package main

import "fmt"

// Ordered is a constraint for ordered types
type Ordered interface {
    ~int | ~int8 | ~int16 | ~int32 | ~int64 |
    ~uint | ~uint8 | ~uint16 | ~uint32 | ~uint64 |
    ~float32 | ~float64 |
    ~string
}

// Max returns the larger of two values
func Max[T Ordered](a, b T) T {
    if a > b {
        return a
    }
    return b
}

// Min returns the smaller of two values
func Min[T Ordered](a, b T) T {
    if a < b {
        return a
    }
    return b
}

// Sort sorts a slice using insertion sort (simple, for demo)
func Sort[T Ordered](slice []T) []T {
    result := make([]T, len(slice))
    copy(result, slice)

    for i := 1; i < len(result); i++ {
        key := result[i]
        j := i - 1
        for j >= 0 && result[j] > key {
            result[j+1] = result[j]
            j--
        }
        result[j+1] = key
    }
    return result
}

// BinarySearch finds the index of target, or -1
func BinarySearch[T Ordered](slice []T, target T) int {
    lo, hi := 0, len(slice)-1
    for lo <= hi {
        mid := lo + (hi-lo)/2
        if slice[mid] == target {
            return mid
        }
        if slice[mid] < target {
            lo = mid + 1
        } else {
            hi = mid - 1
        }
    }
    return -1
}

// Clamp restricts value to [min, max]
func Clamp[T Ordered](val, minVal, maxVal T) T {
    return Max(minVal, Min(val, maxVal))
}

func main() {
    // Max / Min
    fmt.Println("Max(3, 7):", Max(3, 7))
    fmt.Println("Min(\"apple\", \"banana\"):", Min("apple", "banana"))

    // Sort
    nums := []int{5, 3, 8, 1, 9, 2, 7}
    sorted := Sort(nums)
    fmt.Println("Sorted:", sorted)

    // BinarySearch
    idx := BinarySearch(sorted, 7)
    fmt.Printf("BinarySearch(7) → index %d\n", idx)

    // Clamp
    fmt.Println("Clamp(15, 0, 10):", Clamp(15, 0, 10))
    fmt.Println("Clamp(-3, 0, 10):", Clamp(-3, 0, 10))

    // Works with string slices too
    words := []string{"cherry", "apple", "banana", "date"}
    fmt.Println("Sorted strings:", Sort(words))
}

    执行预览：

    Max(3, 7): 7
Min("apple", "banana"): apple
Sorted: [1 2 3 5 7 8 9]
BinarySearch(7) → index 4
Clamp(15, 0, 10): 10
Clamp(-3, 0, 10): 0
Sorted strings: [apple banana cherry date]

    注意事项

    要点 说明
    Go 版本 泛型需要 Go 1.18+，旧版本不可用
    类型推断限制 复杂场景编译器可能推断失败，需显式指定
    方法限制 泛型类型的方法不能引入新的类型参数
    运行时开销 泛型通过 GC shape stenciling 实现，大部分场景无额外开销
    不支持特化 Go 不支持 C++ 风格的模板特化
    调试体验 泛型错误信息可能很长，1.21 后有所改善

    避坑指南

    ❌ 错误做法 ✅ 正确做法
    泛型所有东西 只在真正需要类型参数时用泛型
    T any 然后类型断言 用类型约束限制范围
    泛型方法加新类型参数 用泛型函数代替，或在类型上定义足购的参数
    忽略 comparable 约束 做 map key 或比较时必须加 comparable
    在泛型代码里调 reflect 用接口约束代替运行时检查
    过度嵌套 []T 泛型 保持简单，不超过 2 个类型参数

            2. 知识点总结

            Go 泛型
├── 语法
│   ├── [T any]          → 类型参数声明
│   ├── [T comparable]   → 可比较约束
│   ├── [T Ordered]      → 自定义约束（联合类型）
│   └── [T any, U any]   → 多类型参数
├── 泛型函数
│   ├── Filter / Map / Reduce
│   ├── Max / Min / Clamp
│   ├── Sort / BinarySearch
│   └── 类型推断 → 大部分场景自动推断
├── 泛型类型
│   ├── Stack[T]
│   ├── Set[T comparable]
│   ├── Pair[T, U]
│   └── 方法不能引入新类型参数
├── 类型约束
│   ├── any          → 无约束
│   ├── comparable   → 可 == / !=
│   ├── 联合类型     → int | float64 | string
│   ├── ~底层类型    → ~int 包含 type MyInt int
│   └── 接口语法     → type Ordered interface { ... }
└── 局限
    ├── 不支持特化
    ├── 不支持运算符重载
    └── 方法不能加新类型参数
