093-工厂模式与抽象工厂

难度：🟡 中高级 | 预计阅读：25 分钟 | 2024 年更新

一、概念讲解

工厂模式是创建型设计模式中最常用的一族，分为三种：

  1. 简单工厂（Simple Factory）：一个函数根据参数返回不同类型
    1. 工厂方法（Factory Method）：定义接口，子类决定创建哪个类型
      1. 抽象工厂（Abstract Factory）：创建一组相关对象的接口

        2. Go 没有继承和虚函数，但通过 接口 + 闭包 + 函数类型 可以优雅地实现所有工厂变体。

        核心思想： 将对象创建逻辑从使用方解耦，调用方不需要知道具体类型。

        二、脑图

        工厂模式家族
├── 简单工厂
│   ├── switch-case 分发
│   ├── map[string]func() 注册
│   └── 适用：类型少且稳定
├── 工厂方法
│   ├── 接口定义 Creator
│   ├── 每个产品一个 Creator
│   └── 适用：需要扩展新产品
├── 抽象工厂
│   ├── 一组相关产品
│   ├── 工厂接口多方法
│   └── 适用：产品族概念
└── Go 特色
    ├── 函数类型即工厂
    ├── 闭包捕获配置
    ├── Option 模式
    └── 接口隐式实现

        三、完整 Go 代码

        package main

import "fmt"

// ============================================
// Domain: Message queue with different providers
// ============================================

// Message represents a queue message
type Message struct {
	ID      string
	Body    []byte
	Headers map[string]string
}

// Queue is the product interface
type Queue interface {
	Publish(topic string, msg Message) error
	Subscribe(topic string) (<-chan Message, error)
	Close() error
}

// ============================================
// V1: Simple Factory — switch-case
// ============================================

func NewQueue(provider string) (Queue, error) {
	switch provider {
	case "redis":
		return newRedisQueue(), nil
	case "kafka":
		return newKafkaQueue(), nil
	case "memory":
		return newMemoryQueue(), nil
	default:
		return nil, fmt.Errorf("unknown provider: %s", provider)
	}
}

// ============================================
// V2: Factory Method with registry
// ============================================

// QueueFactory is a function type that creates a Queue
type QueueFactory func(config map[string]string) (Queue, error)

var registry = map[string]QueueFactory{}

// Register adds a queue factory to the registry
func Register(name string, factory QueueFactory) {
	registry[name] = factory
}

// CreateQueue uses the registry to find and invoke the right factory
func CreateQueue(provider string, config map[string]string) (Queue, error) {
	factory, ok := registry[provider]
	if !ok {
		return nil, fmt.Errorf("unregistered provider: %s", provider)
	}
	return factory(config)
}

// ============================================
// V3: Abstract Factory — product families
// ============================================

// MessageInfrastructure creates related products
type MessageInfrastructure interface {
	NewQueue() Queue
	NewSerializer() Serializer
	NewMonitor() Monitor
}

// Serializer converts messages to/from bytes
type Serializer interface {
	Serialize(msg Message) ([]byte, error)
	Deserialize(data []byte) (Message, error)
}

// Monitor tracks queue metrics
type Monitor interface {
	Lag(topic string) int64
	Throughput(topic string) float64
}

// --- Redis family ---
type redisInfrastructure struct{ config map[string]string }

func (r *redisInfrastructure) NewQueue() Queue       { return newRedisQueue() }
func (r *redisInfrastructure) NewSerializer() Serializer { return &jsonSerializer{} }
func (r *redisInfrastructure) NewMonitor() Monitor    { return &redisMonitor{} }

// --- Kafka family ---
type kafkaInfrastructure struct{ config map[string]string }

func (k *kafkaInfrastructure) NewQueue() Queue       { return newKafkaQueue() }
func (k *kafkaInfrastructure) NewSerializer() Serializer { return &avroSerializer{} }
func (k *kafkaInfrastructure) NewMonitor() Monitor    { return &kafkaMonitor{} }

// NewInfrastructure is the abstract factory entry point
func NewInfrastructure(provider string, config map[string]string) MessageInfrastructure {
	switch provider {
	case "redis":
		return &redisInfrastructure{config: config}
	case "kafka":
		return &kafkaInfrastructure{config: config}
	default:
		return &redisInfrastructure{config: config}
	}
}

// ============================================
// Concrete implementations (stubs)
// ============================================

type redisQueue struct{ msgs chan Message }
func newRedisQueue() *redisQueue { return &redisQueue{msgs: make(chan Message, 100)} }
func (r *redisQueue) Publish(topic string, msg Message) error { r.msgs <- msg; return nil }
func (r *redisQueue) Subscribe(topic string) (<-chan Message, error) { return r.msgs, nil }
func (r *redisQueue) Close() error { close(r.msgs); return nil }

type kafkaQueue struct{ msgs chan Message }
func newKafkaQueue() *kafkaQueue { return &kafkaQueue{msgs: make(chan Message, 100)} }
func (k *kafkaQueue) Publish(topic string, msg Message) error { k.msgs <- msg; return nil }
func (k *kafkaQueue) Subscribe(topic string) (<-chan Message, error) { return k.msgs, nil }
func (k *kafkaQueue) Close() error { close(k.msgs); return nil }

type memoryQueue struct{ msgs chan Message }
func newMemoryQueue() *memoryQueue { return &memoryQueue{msgs: make(chan Message, 100)} }
func (m *memoryQueue) Publish(topic string, msg Message) error { m.msgs <- msg; return nil }
func (m *memoryQueue) Subscribe(topic string) (<-chan Message, error) { return m.msgs, nil }
func (m *memoryQueue) Close() error { close(m.msgs); return nil }

type jsonSerializer struct{}
func (j *jsonSerializer) Serialize(msg Message) ([]byte, error)   { return []byte(msg.ID), nil }
func (j *jsonSerializer) Deserialize(data []byte) (Message, error) { return Message{ID: string(data)}, nil }

type avroSerializer struct{}
func (a *avroSerializer) Serialize(msg Message) ([]byte, error)   { return []byte("avro:" + msg.ID), nil }
func (a *avroSerializer) Deserialize(data []byte) (Message, error) { return Message{ID: string(data[5:])}, nil }

type redisMonitor struct{}
func (r *redisMonitor) Lag(topic string) int64      { return 0 }
func (r *redisMonitor) Throughput(topic string) float64 { return 1000.0 }

type kafkaMonitor struct{}
func (k *kafkaMonitor) Lag(topic string) int64      { return 42 }
func (k *kafkaMonitor) Throughput(topic string) float64 { return 5000.0 }

// ============================================
// Demo
// ============================================

func main() {
	// V2: Registry pattern
	Register("redis", func(config map[string]string) (Queue, error) {
		return newRedisQueue(), nil
	})
	Register("kafka", func(config map[string]string) (Queue, error) {
		return newKafkaQueue(), nil
	})

	q, _ := CreateQueue("redis", map[string]string{"addr": "localhost:6379"})
	q.Publish("events", Message{ID: "1", Body: []byte("hello")})
	fmt.Println("V2: Published to redis")

	// V3: Abstract factory
	infra := NewInfrastructure("kafka", nil)
	queue := infra.NewQueue()
	ser := infra.NewSerializer()
	mon := infra.NewMonitor()

	data, _ := ser.Serialize(Message{ID: "msg-42"})
	fmt.Printf("V3: Serialized=%s, Lag=%d, Throughput=%.0f\n", data, mon.Lag("events"), mon.Throughput("events"))
	queue.Close()
}

        五、注意事项

        维度 注意点 说明
        接口设计 保持最小接口 Queue 只管收发，不暴露底层实现
        注册时机 init() 注册 vs 手动注册 init() 方便但隐式，手动更可控
        错误处理 工厂可能返回 error 未注册的 provider 要友好提示
        并发注册 map 非线程安全 init() 阶段注册无此问题
        配置传递 工厂函数接收 config 避免全局配置耦合

        六、避坑指南

        ❌ 错误做法 ✅ 正确做法
        switch-case 硬编码所有类型 用注册表模式，新类型无需改工厂代码
        工厂返回具体类型 返回接口类型，隐藏实现
        抽象工厂接口过于庞大 按产品族拆分，保持接口精简
        忽略 Close/cleanup 工厂创建的对象要提供资源释放方法
        用 reflect 做动态工厂 用函数注册表，类型安全且高效

        七、练习题

        🟢 初级： 用简单工厂创建 3 种 Logger（File、Console、Null），支持 Log(level, msg)。

        🟡 中级： 实现一个带注册表的工厂，支持 Register(name, factory) 和 Create(name, config)，处理未知 name 的错误。

        🔴 高级： 设计一个数据库抽象工厂，支持 MySQL/PostgreSQL 产品族，每个族包含 Connection、QueryBuilder、Migration 三种产品。

        八、知识点总结

        工厂模式知识树
├── 简单工厂
│   ├── 参数化创建
│   ├── 违反 OCP（开闭原则）
│   └── 适用：3-5 种固定类型
├── 工厂方法
│   ├── Creator 接口 + 具体 Creator
│   ├── 符合 OCP
│   └── Go 用函数类型替代继承
├── 抽象工厂
│   ├── 产品族概念
│   ├── 一致性约束
│   └── 扩展新产品需改接口
├── Go 惯用写法
│   ├── func() T 即工厂
│   ├── Option Functional 模式
│   ├── map 注册表
│   └── 接口隐式满足
└── 选型指南
    ├── 类型少 → 简单工厂
    ├── 需扩展 → 工厂方法 + 注册表
    └── 产品族 → 抽象工厂

        九、举一反三

        场景 模式选择 理由
        多种日志输出 简单工厂 类型固定，不需扩展
        插件系统 工厂方法 + 注册表 第三方插件可自注册
        多云 SDK 抽象工厂 AWS/GCP/Azure 各有存储、计算等产品
        数据库驱动 工厂方法 + 注册表 sql.Register 就是经典实现
        序列化格式 简单工厂 JSON/Protobuf/Avro 固定几种

          • 十一、代码演进

          V1 — 简单工厂（switch-case）：

          • 直观易懂，适合入门
          • 新增类型需修改工厂函数，违反开闭原则

            • V2 — 工厂方法 + 注册表：

            • 用 map[string]QueueFactory 注册表解耦
            • 新类型只需 Register() 即可，无需改工厂
            • 类似 database/sql 的 Register 模式

              • V3 — 抽象工厂（产品族）：

              • MessageInfrastructure 接口创建一组相关产品
              • 保证 Queue + Serializer + Monitor 来自同一个族
              • 适合需要一致性约束的复杂系统
