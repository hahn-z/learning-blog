一、概念讲解

策略模式（Strategy Pattern） 定义一系列算法，将每个算法封装起来，并使它们可以互相替换。策略模式让算法的变化独立于使用算法的客户端。

核心要素：

  • Context（上下文）：持有策略引用，将请求委托给策略对象
  • Strategy（策略接口）：定义算法的统一接口
  • ConcreteStrategy（具体策略）：实现策略接口的具体算法

    • Go 的优势： 函数是一等公民，简单策略直接用函数类型即可，无需定义 struct。复杂策略才需要接口+struct。

    典型场景：

    • 支付方式选择（微信/支付宝/银行卡）
    • 排序算法切换
    • 压缩格式选择（gzip/lz4/zstd）
    • 路由策略（轮询/随机/权重）

      三、完整 Go 代码

      package main

import (
	"fmt"
	"math/rand"
	"time"
)

// ============================================
// V1: Function-type strategy (lightweight)
// ============================================

// DiscountStrategy calculates the discounted price
type DiscountStrategy func(original float64) float64

// No discount
func NoDiscount() DiscountStrategy {
	return func(original float64) float64 { return original }
}

// Percentage discount
func PercentDiscount(pct float64) DiscountStrategy {
	return func(original float64) float64 { return original * (1 - pct) }
}

// Fixed amount discount
func FixedDiscount(amount float64) DiscountStrategy {
	return func(original float64) float64 {
		if original < amount {
			return 0
		}
		return original - amount
	}
}

// Tiered discount based on total amount
func TieredDiscount() DiscountStrategy {
	return func(original float64) float64 {
		switch {
		case original >= 1000:
			return original * 0.8 // 20% off
		case original >= 500:
			return original * 0.9 // 10% off
		case original >= 100:
			return original * 0.95 // 5% off
		default:
			return original
		}
	}
}

// Order uses a discount strategy
type Order struct {
	Product  string
	Price    float64
	strategy DiscountStrategy
}

func (o *Order) SetStrategy(s DiscountStrategy) { o.strategy = s }
func (o *Order) FinalPrice() float64            { return o.strategy(o.Price) }

// ============================================
// V2: Interface-based strategy (standard)
// ============================================

// LoadBalanceStrategy selects a backend server
type LoadBalanceStrategy interface {
	Pick(servers []Server) Server
}

// Server represents a backend server
type Server struct {
	ID     string
	Weight int
	Addr   string
}

// RoundRobin strategy
type RoundRobin struct{ current int }

func (rr *RoundRobin) Pick(servers []Server) Server {
	s := servers[rr.current%len(servers)]
	rr.current++
	return s
}

// WeightedRandom strategy
type WeightedRandom struct {
	rng   *rand.Rand
	total int
}

func NewWeightedRandom(servers []Server) *WeightedRandom {
	total := 0
	for _, s := range servers {
		total += s.Weight
	}
	return &WeightedRandom{rng: rand.New(rand.NewSource(time.Now().UnixNano())), total: total}
}

func (wr *WeightedRandom) Pick(servers []Server) Server {
	r := wr.rng.Intn(wr.total)
	cum := 0
	for _, s := range servers {
		cum += s.Weight
		if r < cum {
			return s
		}
	}
	return servers[0]
}

// LeastConnection strategy
type LeastConnection struct{ connections map[string]int }

func NewLeastConnection() *LeastConnection {
	return &LeastConnection{connections: make(map[string]int)}
}

func (lc *LeastConnection) Pick(servers []Server) Server {
	var best Server
	min := int(^uint(0) >> 1) // max int
	for _, s := range servers {
		c := lc.connections[s.ID]
		if c < min {
			min = c
			best = s
		}
	}
	lc.connections[best.ID]++
	return best
}

// LoadBalancer is the context
type LoadBalancer struct {
	strategy LoadBalanceStrategy
	servers  []Server
}

func (lb *LoadBalancer) SetStrategy(s LoadBalanceStrategy) { lb.strategy = s }
func (lb *LoadBalancer) Next() Server                      { return lb.strategy.Pick(lb.servers) }

// ============================================
// V3: Generic strategy with config-driven selection
// ============================================

// StrategyRegistry holds named strategies
type StrategyRegistry[T any] struct {
	strategies map[string]T
	default_   string
}

func NewStrategyRegistry[T any]() *StrategyRegistry[T] {
	return &StrategyRegistry[T]{strategies: make(map[string]T)}
}

func (sr *StrategyRegistry[T]) Register(name string, strategy T) {
	sr.strategies[name] = strategy
	if sr.default_ == "" {
		sr.default_ = name
	}
}

func (sr *StrategyRegistry[T]) Get(name string) (T, bool) {
	if name == "" {
		name = sr.default_
	}
	s, ok := sr.strategies[name]
	return s, ok
}

// ============================================
// Demo
// ============================================

func main() {
	// V1: Function strategy
	order := Order{Product: "MacBook", Price: 1299, strategy: TieredDiscount()}
	fmt.Printf("V1: %s $%.2f → $%.2f\n", order.Product, order.Price, order.FinalPrice())

	order.SetStrategy(PercentDiscount(0.15))
	fmt.Printf("V1: %s $%.2f → $%.2f (15%% off)\n", order.Product, order.Price, order.FinalPrice())

	// V2: Interface strategy
	servers := []Server{
		{ID: "s1", Weight: 3, Addr: "10.0.0.1:8080"},
		{ID: "s2", Weight: 1, Addr: "10.0.0.2:8080"},
		{ID: "s3", Weight: 1, Addr: "10.0.0.3:8080"},
	}

	lb := &LoadBalancer{servers: servers}

	lb.SetStrategy(&RoundRobin{})
	fmt.Printf("V2 RoundRobin: ")
	for i := 0; i < 5; i++ { fmt.Printf("%s ", lb.Next().ID) }
	fmt.Println()

	lb.SetStrategy(NewLeastConnection())
	fmt.Printf("V2 LeastConn: ")
	for i := 0; i < 5; i++ { fmt.Printf("%s ", lb.Next().ID) }
	fmt.Println()

	// V3: Registry
	reg := NewStrategyRegistry[DiscountStrategy]()
	reg.Register("tiered", TieredDiscount())
	reg.Register("vip", PercentDiscount(0.3))
	reg.Register("none", NoDiscount())

	if s, ok := reg.Get("vip"); ok {
		order.SetStrategy(s)
		fmt.Printf("V3 Registry(vip): $%.2f → $%.2f\n", order.Price, order.FinalPrice())
	}
}

      五、注意事项

      维度 注意点 说明
      函数 vs 接口 简单逻辑用函数，复杂状态用接口 函数类型更轻量
      策略无状态 尽量让策略无状态 有状态策略需注意并发安全
      默认策略 提供合理的默认行为 避免空策略导致 panic
      配置驱动 从配置文件/环境变量选择策略 运行时切换无需改代码

      六、避坑指南

      ❌ 错误做法 ✅ 正确做法
      if-else/switch 散落在业务代码中 封装为策略，业务代码只调用接口
      策略内部依赖全局变量 通过构造函数注入依赖
      每种策略一个独立包 简单策略放在同一文件，复杂才拆包
      策略创建时做重量级初始化 懒加载或对象池
      忘记设置策略（nil panic） 提供默认策略或零值策略

      七、练习题

      🟢 初级： 用函数类型实现 3 种排序策略（升序、降序、按长度），对字符串切片排序。

      🟡 中级： 实现负载均衡器的加权轮询策略（Weighted Round Robin），确保请求按权重比例分配。

      🔴 高级： 设计一个策略注册中心，支持从 YAML 配置文件动态加载和切换策略，运行时通过 HTTP API 热替换。

        V1 — 函数类型策略（轻量级）：

        • 用 func 类型定义策略，闭包捕获参数
        • 适合无状态的简单算法
        • 折扣计算示例：百分比/固定/阶梯

          • V2 — 接口策略（标准模式）：

          • 定义 LoadBalanceStrategy 接口
          • 实现轮询/加权随机/最少连接
          • Context（LoadBalancer）持有策略引用

            • V3 — 泛型注册表（配置驱动）：

            • StrategyRegistry[T] 类型安全注册
            • 支持命名注册和默认策略
            • 可从配置文件读取策略名，运行时选择
