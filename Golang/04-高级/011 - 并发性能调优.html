一、概念讲解

Go 语言的并发模型基于 CSP（Communicating Sequential Processes），goroutine + channel 是其核心原语。但"能跑"不等于"跑得快"——并发性能调优需要理解调度器原理、内存分配、锁竞争、GC 压力等多个维度。

核心问题：

  • Goroutine 泄漏导致内存持续增长
  • Channel 使用不当引发死锁或低吞吐
  • 锁竞争成为瓶颈
  • GC 压力过大影响延迟

    • 调优方法论：

    1. Measure First — 先用 pprof/benchmark 量化问题
  1. Identify Bottleneck — CPU? Memory? Lock contention?
    1. Optimize Incrementally — 一次改一个变量
      1. Validate — benchmark 对比，确认改善

        package main

import (
	"context"
	"fmt"
	"runtime"
	"sync"
	"sync/atomic"
	"time"
)

// ============================================
// V1: Naive goroutine spawning — resource leak
// ============================================

func processNaive(tasks []int) []int {
	results := make([]int, len(tasks))
	for i, t := range tasks {
		go func(idx, val int) {
			results[idx] = val * 2
		}(i, t)
	}
	return results // BUG: race condition, no synchronization
}

// ============================================
// V2: Worker pool with bounded concurrency
// ============================================

func processWithPool(ctx context.Context, tasks []int, workers int) ([]int, error) {
	results := make([]int, len(tasks))
	taskCh := make(chan int, len(tasks))
	resultCh := make(chan struct{}, len(tasks))

	// Start worker pool
	var wg sync.WaitGroup
	for w := 0; w < workers; w++ {
		wg.Add(1)
		go func() {
			defer wg.Done()
			for {
				select {
				case idx, ok := <-taskCh:
					if !ok {
						return
					}
					results[idx] = tasks[idx] * 2
					resultCh <- struct{}{}
				case <-ctx.Done():
					return
				}
			}
		}()
	}

	// Send tasks
	for i := range tasks {
		taskCh <- i
	}
	close(taskCh)

	// Wait for completion or timeout
	for i := 0; i < len(tasks); i++ {
		select {
		case <-resultCh:
		case <-ctx.Done():
			wg.Wait()
			return nil, ctx.Err()
		}
	}
	wg.Wait()
	return results, nil
}

// ============================================
// V3: Optimized — sync.Pool + atomic counter
// ============================================

var bufPool = sync.Pool{
	New: func() interface{} {
		buf := make([]byte, 1024)
		return &buf
	},
}

type Processor struct {
	workers  int
	processed atomic.Int64
}

func NewProcessor(workers int) *Processor {
	return &Processor{workers: workers}
}

func (p *Processor) Process(ctx context.Context, tasks []int) ([]int, error) {
	results := make([]int, len(tasks))
	taskCh := make(chan int, len(tasks))

	var wg sync.WaitGroup
	for w := 0; w < p.workers; w++ {
		wg.Add(1)
		go func() {
			defer wg.Done()
			// Reuse buffer from pool
			_ = bufPool.Get().(*[]byte)
			defer bufPool.Put(bufPool.Get())

			for {
				select {
				case idx, ok := <-taskCh:
					if !ok {
						return
					}
					// Simulate CPU work
					results[idx] = tasks[idx] * 2
					p.processed.Add(1)
				case <-ctx.Done():
					return
				}
			}
		}()
	}

	for i := range tasks {
		select {
		case taskCh <- i:
		case <-ctx.Done():
			close(taskCh)
			wg.Wait()
			return nil, ctx.Err()
		}
	}
	close(taskCh)
	wg.Wait()
	return results, nil
}

func (p *Processor) Stats() int64 {
	return p.processed.Load()
}

// ============================================
// Benchmark helper
// ============================================

func main() {
	tasks := make([]int, 10000)
	for i := range tasks {
		tasks[i] = i
	}

	// V2: Worker pool
	ctx, cancel := context.WithTimeout(context.Background(), 5*time.Second)
	defer cancel()

	fmt.Printf("GOMAXPROCS: %d\n", runtime.GOMAXPROCS(0))

	start := time.Now()
	res, err := processWithPool(ctx, tasks, runtime.NumCPU())
	if err != nil {
		fmt.Println("Error:", err)
		return
	}
	fmt.Printf("V2 Worker Pool: processed %d tasks in %v\n", len(res), time.Since(start))

	// V3: Optimized processor
	proc := NewProcessor(runtime.NumCPU() * 2)
	start = time.Now()
	res2, err := proc.Process(ctx, tasks)
	if err != nil {
		fmt.Println("Error:", err)
		return
	}
	fmt.Printf("V3 Optimized:   processed %d tasks in %v, total=%d\n",
		len(res2), time.Since(start), proc.Stats())
}

        五、注意事项

        维度 注意点 说明
        Goroutine 数量 不要无限制创建 10w+ goroutine 会吃掉大量内存
        Channel 方向 区分发送/接收 channel 编译期类型安全
        Context 超时 务必设置合理超时 防止 goroutine 泄漏
        sync.Pool 适合短生命周期对象 GC 前会被回收，不保证持久
        Race Detector 开发阶段始终开启 go run -race 或 go test -race
        GOMAXPROCS 默认等于 CPU 核心数 容器环境可能需要手动设置

        六、避坑指南

        ❌ 错误做法 ✅ 正确做法
        go func() 不加 WaitGroup 用 sync.WaitGroup 或 channel 等待完成
        无缓冲 channel 在同一 goroutine 收发 分离发送方和接收方
        runtime.GOMAXPROCS(1) 假设单线程 Go 调度器是并发的，需要正确的同步
        用 time.Sleep 等待 goroutine 完成 用 sync.WaitGroup 或 context
        每次请求 make 大 buffer 用 sync.Pool 复用
        Mutex 嵌套锁定顺序不一致 统一锁定顺序，避免死锁

        七、练习题

        🟢 初级： 写一个带缓冲 channel 的生产者-消费者模型，限制并发数为 10。

        🟡 中级： 使用 sync.Pool 优化一个 JSON 序列化函数的内存分配，对比 benchmark 结果。

        🔴 高级： 实现一个自适应 worker pool：当队列积压超过阈值时自动扩容，空闲超时后自动缩容。

          • 十一、代码演进

          V1 — 裸启动 goroutine（有竞态）：

          • 直接 go func() 启动，无同步机制
          • 存在 data race，结果不可预测

            • V2 — Worker Pool + Context：

            • 固定 worker 数量，通过 channel 分发任务
            • 支持 context 取消和超时
            • 用 WaitGroup 确保所有 worker 完成

              • V3 — sync.Pool + atomic + 自定义 Processor：

              • 对象复用减少 GC 压力
              • atomic 计数器替代 Mutex，减少锁竞争
              • 封装为可复用的 Processor 结构体
              • 支持统计和监控
