	"context"

	"sync/atomic"
	"time"

// ============================================
// V1: Naive goroutine spawning вҖ” resource leak
// ============================================

func processNaive(tasks []int) []int {
	results := make([]int, len(tasks))
	for i, t := range tasks {
		go func(idx, val int) {
			results[idx] = val * 2
		}(i, t)
	}
	return results // BUG: race condition, no synchronization
}

// ============================================
// V2: Worker pool with bounded concurrency
// ============================================

func processWithPool(ctx context.Context, tasks []int, workers int) ([]int, error) {
	results := make([]int, len(tasks))
	taskCh := make(chan int, len(tasks))
	resultCh := make(chan struct{}, len(tasks))

	// Start worker pool
	var wg sync.WaitGroup
	for w := 0; w < workers; w++ {
		wg.Add(1)
		go func() {
			defer wg.Done()
			for {
				select {
				case idx, ok := <-taskCh:
					if !ok {
						return
					}
					results[idx] = tasks[idx] * 2
					resultCh <- struct{}{}
				case <-ctx.Done():
					return
				}
			}
		}()
	}

	// Send tasks
	for i := range tasks {
		taskCh <- i
	}
	close(taskCh)

	// Wait for completion or timeout
	for i := 0; i < len(tasks); i++ {
		select {
		case <-resultCh:
		case <-ctx.Done():
			wg.Wait()
			return nil, ctx.Err()
		}
	}
	wg.Wait()
	return results, nil
}

// ============================================
// V3: Optimized вҖ” sync.Pool + atomic counter
// ============================================

var bufPool = sync.Pool{
	New: func() interface{} {
		buf := make([]byte, 1024)
		return &buf
	},
}

type Processor struct {
	workers  int
	processed atomic.Int64
}

func NewProcessor(workers int) *Processor {
	return &Processor{workers: workers}
}

func (p *Processor) Process(ctx context.Context, tasks []int) ([]int, error) {
	results := make([]int, len(tasks))
	taskCh := make(chan int, len(tasks))

	var wg sync.WaitGroup
	for w := 0; w < p.workers; w++ {
		wg.Add(1)
		go func() {
			defer wg.Done()
			// Reuse buffer from pool
			_ = bufPool.Get().(*[]byte)
			defer bufPool.Put(bufPool.Get())

			for {
				select {
				case idx, ok := <-taskCh:
					if !ok {
						return
					}
					// Simulate CPU work
					results[idx] = tasks[idx] * 2
					p.processed.Add(1)
				case <-ctx.Done():
					return
				}
			}
		}()
	}

	for i := range tasks {
		select {
		case taskCh <- i:
		case <-ctx.Done():
			close(taskCh)
			wg.Wait()
			return nil, ctx.Err()
		}
	}
	close(taskCh)
	wg.Wait()
	return results, nil
}

func (p *Processor) Stats() int64 {
	return p.processed.Load()
}

// ============================================
// Benchmark helper
// ============================================

func main() {
	tasks := make([]int, 10000)
	for i := range tasks {
		tasks[i] = i
	}

	// V2: Worker pool
	ctx, cancel := context.WithTimeout(context.Background(), 5*time.Second)
	defer cancel()

	fmt.Printf("GOMAXPROCS: %d\n", runtime.GOMAXPROCS(0))

	start := time.Now()
	res, err := processWithPool(ctx, tasks, runtime.NumCPU())
	if err != nil {
		fmt.Println("Error:", err)
		return
	}
	fmt.Printf("V2 Worker Pool: processed %d tasks in %v\n", len(res), time.Since(start))

	// V3: Optimized processor
	proc := NewProcessor(runtime.NumCPU() * 2)
	start = time.Now()
	res2, err := proc.Process(ctx, tasks)
	if err != nil {
		fmt.Println("Error:", err)
		return
	}
	fmt.Printf("V3 Optimized:   processed %d tasks in %v, total=%d\n",
		len(res2), time.Since(start), proc.Stats())
}
