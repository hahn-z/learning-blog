一、概念讲解

1.1 微服务协作的挑战

在真实电商场景中，一个下单操作涉及用户服务、商品服务、订单服务、库存服务、支付服务的协同。核心挑战：

挑战 说明
服务间通信 HTTP vs gRPC，同步 vs 异步
数据一致性 跨服务的数据如何保持一致
服务发现 服务实例动态变化，如何互相找到
链路追踪 一次请求跨多个服务，如何追踪
故障隔离 一个服务挂了，不影响整体

1.2 分布式事务方案

方案 适用场景 复杂度 一致性
2PC (两阶段提交) 数据库层面 中 强一致
TCC (Try-Confirm-Cancel) 资源预留型业务 高 最终一致
Saga (编排式) 长事务、异步流程 中 最终一致
本地消息表 简单可靠场景 低 最终一致

package order.v1;

option go_package = "ecommerce/api/order/v1";

import "google/api/annotations.proto";
import "google/protobuf/empty.proto";

service OrderService {
  rpc CreateOrder(CreateOrderRequest) returns (CreateOrderReply) {
    option (google.api.http) = {post: "/v1/orders"};
  }
  rpc GetOrder(GetOrderRequest) returns (OrderReply) {
    option (google.api.http) = {get: "/v1/orders/{id}"};
  }
  rpc ListOrders(ListOrdersRequest) returns (ListOrdersReply) {
    option (google.api.http) = {get: "/v1/orders"};
  }
  rpc CancelOrder(CancelOrderRequest) returns (google.protobuf.Empty) {
    option (google.api.http) = {put: "/v1/orders/{id}/cancel"};
  }
}

message CreateOrderRequest {
  string user_id = 1;
  repeated OrderItem items = 2;
}

message OrderItem {
  string product_id = 1;
  int32 quantity = 2;
}

message CreateOrderReply {
  string id = 1;
  string status = 2;
}

message GetOrderRequest { string id = 1; }

message OrderReply {
  string id = 1;
  string user_id = 2;
  string status = 3;
  repeated OrderItemReply items = 4;
  float total_amount = 5;
  string created_at = 6;
}

message OrderItemReply {
  string product_id = 1;
  string product_name = 2;
  int32 quantity = 3;
  float price = 4;
}

message ListOrdersRequest {
  string user_id = 1;
  int32 page = 2;
  int32 page_size = 3;
}

message ListOrdersReply {
  repeated OrderReply orders = 1;
  int32 total = 2;
}

message CancelOrderRequest { string id = 1; }

3.3 订单服务实现（含多服务协作）

// internal/order/service.go - Order service with multi-service collaboration
package order

import (
	"context"
	"fmt"
	"log"
	"time"

	pb "ecommerce/api/order/v1"
	pbProduct "ecommerce/api/product/v1"
	pbUser "ecommerce/api/user/v1"

	"github.com/go-kratos/kratos/v2/middleware/recovery"
	"google.golang.org/grpc"
)

// OrderServiceImpl implements OrderService
type OrderServiceImpl struct {
	pb.UnimplementedOrderServiceServer

	repo          OrderRepository
	productClient pbProduct.ProductServiceClient
	userClient    pbUser.UserServiceClient
	eventPublisher EventPublisher
}

// NewOrderService creates a new order service
func NewOrderService(
	repo OrderRepository,
	productConn *grpc.ClientConn,
	userConn *grpc.ClientConn,
	publisher EventPublisher,
) *OrderServiceImpl {
	return &OrderServiceImpl{
		repo:           repo,
		productClient:  pbProduct.NewProductServiceClient(productConn),
		userClient:     pbUser.NewUserServiceClient(userConn),
		eventPublisher: publisher,
	}
}

// CreateOrder handles order creation with multi-service collaboration
// Flow: validate user → check stock → create order → deduct stock → emit event
func (s *OrderServiceImpl) CreateOrder(ctx context.Context, req *pb.CreateOrderRequest) (*pb.CreateOrderReply, error) {
	// Step 1: Validate user exists
	user, err := s.userClient.GetUser(ctx, &pbUser.GetUserRequest{Id: req.UserId})
	if err != nil {
		return nil, fmt.Errorf("user validation failed: %w", err)
	}
	log.Printf("User validated: %s (%s)", user.Name, req.UserId)

	// Step 2: Check product availability and get prices
	productIDs := make([]string, 0, len(req.Items))
	for _, item := range req.Items {
		productIDs = append(productIDs, item.ProductId)
	}
	products, err := s.productClient.BatchGetProducts(ctx, &pbProduct.BatchGetRequest{Ids: productIDs})
	if err != nil {
		return nil, fmt.Errorf("product query failed: %w", err)
	}

	// Validate stock availability
	for _, item := range req.Items {
		product := findProduct(products.Products, item.ProductId)
		if product == nil {
			return nil, fmt.Errorf("product %s not found", item.ProductId)
		}
		if product.Stock < item.Quantity {
			return nil, fmt.Errorf("product %s insufficient stock: have %d, need %d",
				item.ProductId, product.Stock, item.Quantity)
		}
	}

	// Step 3: Calculate total and create order in DB
	order := &Order{
		UserID:    req.UserId,
		Status:    "pending",
		Items:     make([]OrderItem, 0, len(req.Items)),
		CreatedAt: time.Now(),
	}
	var total float64
	for _, item := range req.Items {
		product := findProduct(products.Products, item.ProductId)
		order.Items = append(order.Items, OrderItem{
			ProductID:   item.ProductId,
			ProductName: product.Name,
			Quantity:    item.Quantity,
			Price:       product.Price,
		})
		total += product.Price * float64(item.Quantity)
	}
	order.TotalAmount = total

	if err := s.repo.Create(ctx, order); err != nil {
		return nil, fmt.Errorf("create order failed: %w", err)
	}

	// Step 4: Deduct stock (synchronous call with compensation on failure)
	for _, item := range req.Items {
		_, err := s.productClient.DeductStock(ctx, &pbProduct.DeductStockRequest{
			ProductId: item.ProductId,
			Quantity:  item.Quantity,
		})
		if err != nil {
			// Compensation: cancel the order
			s.repo.UpdateStatus(ctx, order.ID, "cancelled")
			return nil, fmt.Errorf("stock deduction failed for %s: %w", item.ProductId, err)
		}
	}

	// Step 5: Update order status and emit event
	s.repo.UpdateStatus(ctx, order.ID, "confirmed")
	s.eventPublisher.Publish(ctx, "order.created", map[string]interface{}{
		"order_id":  order.ID,
		"user_id":   req.UserId,
		"total":     total,
		"item_count": len(req.Items),
	})

	log.Printf("Order created: id=%s user=%s total=%.2f", order.ID, user.Name, total)

	return &pb.CreateOrderReply{
		Id:     order.ID,
		Status: "confirmed",
	}, nil
}

// GetOrder returns order details
func (s *OrderServiceImpl) GetOrder(ctx context.Context, req *pb.GetOrderRequest) (*pb.OrderReply, error) {
	order, err := s.repo.GetByID(ctx, req.Id)
	if err != nil {
		return nil, fmt.Errorf("order not found: %w", err)
	}
	return orderToReply(order), nil
}

// CancelOrder cancels an order and restores stock
func (s *OrderServiceImpl) CancelOrder(ctx context.Context, req *pb.CancelOrderRequest) (*pb.Empty, error) {
	order, err := s.repo.GetByID(ctx, req.Id)
	if err != nil {
		return nil, fmt.Errorf("order not found: %w", err)
	}
	if order.Status == "cancelled" {
		return nil, fmt.Errorf("order already cancelled")
	}

	// Restore stock for each item
	for _, item := range order.Items {
		_, err := s.productClient.RestoreStock(ctx, &pbProduct.RestoreStockRequest{
			ProductId: item.ProductID,
			Quantity:  item.Quantity,
		})
		if err != nil {
			log.Printf("Warning: stock restore failed for %s: %v", item.ProductID, err)
			// Continue anyway - can be reconciled later
		}
	}

	// Update order status
	if err := s.repo.UpdateStatus(ctx, req.Id, "cancelled"); err != nil {
		return nil, err
	}

	// Emit cancellation event
	s.eventPublisher.Publish(ctx, "order.cancelled", map[string]interface{}{
		"order_id": req.Id,
	})

	return &pb.Empty{}, nil
}

func findProduct(products []*pbProduct.ProductReply, id string) *pbProduct.ProductReply {
	for _, p := range products {
		if p.Id == id {
			return p
		}
	}
	return nil
}

五、注意事项

注意点 说明 影响
超时设置 每个gRPC调用都要设超时 防止级联阻塞
重试策略 幂等操作可重试，非幂等要谨慎 重复扣款风险
数据库隔离 每个服务独立数据库 数据一致性边界
服务发现 动态地址需注册中心 服务上下线感知
日志关联 全链路TraceID透传 排障效率
库存扣减 先扣后创建 vs 先创建后扣 并发安全策略
优雅停机 服务关闭时完成进行中的请求 数据完整性

六、避坑指南

❌ 错误做法 ✅ 正确做法
所有服务共享一个数据库 每个服务独立数据库，通过API通信
同步调用链过长（A→B→C→D） 限制调用链深度，异步解耦长链路
不做超时控制 每层设置合理超时，使用context传递
忽略部分失败 库存扣减失败要回滚已成功的操作
服务间直接传DB对象 通过Protobuf定义清晰的API契约
不做服务发现 用Consul/etcd注册发现，不硬编码地址
单体思维设计API 按领域拆分，每个服务有独立生命周期

                              2. 十一、代码演进

                              v1：单体应用

                              // v1: Everything in one service, one database
func CreateOrder(db *sql.DB, req OrderReq) error {
    user := queryUser(db, req.UserID)
    product := queryProduct(db, req.ProductID)
    // Direct DB transaction for everything
    tx := db.Begin()
    tx.Exec("INSERT INTO orders ...")
    tx.Exec("UPDATE products SET stock = stock - ?", req.Quantity)
    tx.Commit()
}

                              v2：微服务拆分 + 同步调用

                              // v2: Separate services, synchronous gRPC calls
func (s *OrderService) CreateOrder(ctx context.Context, req *CreateReq) (*Order, error) {
    user := s.userClient.GetUser(ctx, &GetUserReq{Id: req.UserID})
    product := s.productClient.GetProduct(ctx, &GetProductReq{Id: req.ProductID})
    // Cross-service calls, but no transaction guarantees
    s.productClient.DeductStock(ctx, &DeductReq{...})
    s.repo.Create(ctx, order)
}

                              v3：微服务 + 事件驱动 + 补偿事务

                              // v3: Event-driven with saga compensation
func (s *OrderService) CreateOrder(ctx context.Context, req *CreateReq) (*Order, error) {
    // Create order in pending state
    order := s.repo.CreatePending(ctx, req)
    
    // Publish event for async processing
    s.eventBus.Publish(ctx, OrderCreatedEvent{OrderID: order.ID, Items: req.Items})
    
    // Saga coordinator handles: deduct stock → confirm → notify
    // With compensation: cancel → restore stock → notify cancellation
    return order, nil
}

                              演进路线：单体事务 → 微服务同步调用 → 事件驱动 + Saga补偿
