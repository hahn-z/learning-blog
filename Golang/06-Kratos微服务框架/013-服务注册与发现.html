013-服务注册与发现

难度： ⭐⭐⭐⭐ | 适合： 需要搭建多服务协作的微服务开发者

一、概念讲解

微服务架构中，服务实例动态变化（扩容、重启、迁移），调用方不能硬编码地址。服务注册与发现解决这个问题：

  1. 注册：服务启动时将自己的地址信息写入注册中心
    1. 发现：调用方从注册中心获取目标服务的可用实例列表
      1. 心跳/健康检查：注册中心定期检查实例是否存活，剔除不健康实例

        2. 核心角色

        ┌──────────┐    注册    ┌──────────────┐    查询    ┌──────────┐
│ Service A │ ────────→ │  Registry    │ ←──────── │ Service B │
│ (Provider)│           │ (Consul/Etcd)│           │(Consumer) │
└──────────┘           └──────────────┘           └──────────┘
     ↑                       │                         │
     └── 心跳保活 ───────────┘                         │
     └── 健康检查 ──────────┘                         │
                                         负载均衡选择实例 ↓

        Kratos Registry 接口

        Kratos抽象了统一的注册中心接口，适配多种实现：

        // registry/registry.go
type Registrar interface {
    Register(ctx context.Context, service *ServiceInstance) error
    Deregister(ctx context.Context, service *ServiceInstance) error
}

type Discovery interface {
    GetService(ctx context.Context, serviceName string) ([]*ServiceInstance, error)
    Watch(ctx context.Context, serviceName string) (Watcher, error)
}

        三、完整Go代码

        v1: Consul注册基础

        // go.mod dependencies
// require (
//     github.com/go-kratos/kratos/v2 latest
//     github.com/go-kratos/kratos/contrib/registry/consul/v2 latest
//     github.com/hashicorp/consul/api latest

        // cmd/server/main.go - Service with Consul registration
package main

import (
    "log"

    "github.com/go-kratos/kratos/contrib/registry/consul/v2"
    "github.com/go-kratos/kratos/v2"
    "github.com/go-kratos/kratos/v2/transport/grpc"
    "github.com/go-kratos/kratos/v2/transport/http"
    "github.com/hashicorp/consul/api"
)

func main() {
    // Create Consul client
    consulClient, err := api.NewClient(api.DefaultConfig())
    if err != nil {
        log.Fatal(err)
    }

    // Create Kratos Consul registrar
    r := consul.New(consulClient)

    // Create servers with registrar
    httpSrv := http.NewServer(
        http.Address(":8000"),
    )
    grpcSrv := grpc.NewServer(
        grpc.Address(":9000"),
    )

    app := kratos.New(
        kratos.Name("helloworld"),        // Service name for registration
        kratos.Server(httpSrv, grpcSrv),
        kratos.Registrar(r),              // Registrar for service registration
    )

    if err := app.Run(); err != nil {
        log.Fatal(err)
    }
}

package server

import (
    "log"

    "github.com/go-kratos/kratos/contrib/registry/consul/v2"
    "github.com/go-kratos/kratos/v2"
    "github.com/go-kratos/kratos/v2/transport/grpc"
    "github.com/go-kratos/kratos/v2/transport/http"
    "github.com/hashicorp/consul/api"
)

// NewApp create kratos application with registry
func NewApp(httpSrv *http.Server, grpcSrv *grpc.Server) *kratos.App {
    consulClient, err := api.NewClient(&api.Config{
        Address:    "127.0.0.1:8500",   // Consul address
        Scheme:     "http",
        Token:      "",                  // ACL token if needed
    })
    if err != nil {
        log.Fatalf("failed to create consul client: %v", err)
    }

    r := consul.New(consulClient)

    return kratos.New(
        kratos.Name("helloworld"),
        kratos.Version("v1.0.0"),
        kratos.Metadata(map[string]string{
            "env":     "production",
            "version": "v1.0.0",
        }),
        kratos.Server(httpSrv, grpcSrv),
        kratos.Registrar(r),
    )
}
        // internal/data/data.go - Service discovery for consumers
package data

import (
    "context"

    "github.com/go-kratos/kratos/contrib/registry/consul/v2"
    "github.com/go-kratos/kratos/v2/registry"
    "github.com/go-kratos/kratos/v2/transport/grpc"
    "github.com/hashicorp/consul/api"
    "google.golang.org/grpc/credentials/insecure"
)

// NewDiscovery create service discovery from Consul
func NewDiscovery() registry.Discovery {
    consulClient, _ := api.NewClient(api.DefaultConfig())
    return consul.New(consulClient)
}

// NewGreeterClient create gRPC client with service discovery
func NewGreeterClient(r registry.Discovery) pb.GreeterClient {
    conn, err := grpc.DialInsecure(
        context.Background(),
        grpc.WithEndpoint("discovery:///helloworld"),   // Use discovery scheme
        grpc.WithDiscovery(r),                           // Inject discovery
    )
    if err != nil {
        panic(err)
    }
    return pb.NewGreeterClient(conn)
}

        v3: Etcd注册 + 健康检查 + 负载均衡

        // internal/server/server.go - Etcd registry variant
package server

import (
    "log"
    "time"

    etcdreg "github.com/go-kratos/kratos/contrib/registry/etcd/v2"
    "github.com/go-kratos/kratos/v2"
    "github.com/go-kratos/kratos/v2/transport/grpc"
    "github.com/go-kratos/kratos/v2/transport/http"
    clientv3 "go.etcd.io/etcd/client/v3"
)

// NewAppWithEtcd use Etcd as registry
func NewAppWithEtcd(httpSrv *http.Server, grpcSrv *grpc.Server) *kratos.App {
    etcdClient, err := clientv3.New(clientv3.Config{
        Endpoints:   []string{"127.0.0.1:2379"},
        DialTimeout: 5 * time.Second,
    })
    if err != nil {
        log.Fatal(err)
    }

    r := etcdreg.New(etcdClient)

    return kratos.New(
        kratos.Name("order-service"),
        kratos.Version("v1.0.0"),
        kratos.Server(httpSrv, grpcSrv),
        kratos.Registrar(r),
    )
}
        // internal/data/client.go - Consumer with load balancing
package data

import (
    "context"
    "time"

    "github.com/go-kratos/kratos/v2/selector"
    "github.com/go-kratos/kratos/v2/selector/p2c"
    "github.com/go-kratos/kratos/v2/transport/grpc"
)

// NewUserClient create client with P2C load balancing
func NewUserClient(r registry.Discovery) pb.UserClient {
    conn, err := grpc.DialInsecure(
        context.Background(),
        grpc.WithEndpoint("discovery:///user-service"),
        grpc.WithDiscovery(r),
        grpc.WithSelector(
            selector.SetGlobalSelector(p2c.New()),  // P2C load balancing
        ),
        grpc.WithTimeout(5*time.Second),
    )
    if err != nil {
        panic(err)
    }
    return pb.NewUserClient(conn)
}
        // internal/server/health.go - Health check implementation
package server

import (
    "context"

    "google.golang.org/grpc/health/grpc_health_v1"
)

// HealthServer implements gRPC health checking
type HealthServer struct {
    grpc_health_v1.UnimplementedHealthServer
}

// Check implements health check
func (h *HealthServer) Check(ctx context.Context, req *grpc_health_v1.HealthCheckRequest) (*grpc_health_v1.HealthCheckResponse, error) {
    return &grpc_health_v1.HealthCheckResponse{
        Status: grpc_health_v1.HealthCheckResponse_SERVING,
    }, nil
}

// Watch implements health watch
func (h *HealthServer) Watch(req *grpc_health_v1.HealthCheckRequest, stream grpc_health_v1.Health_WatchServer) error {
    return stream.Send(&grpc_health_v1.HealthCheckResponse{
        Status: grpc_health_v1.HealthCheckResponse_SERVING,
    })
}

        五、注意事项

        注意点 说明 建议
        注册中心可用性 注册中心宕机导致全链路故障 集群部署 + 本地缓存
        心跳间隔 太短浪费资源，太长切换慢 建议10-30秒
        优雅下线 kill -9会导致残留节点 使用kratos.New的Stop钩子
        服务名规范 名称冲突导致路由错误 使用 {team}.{service}.{version} 格式
        多注册中心 不同环境用不同注册中心 通过配置文件切换
        实例元数据 版本/权重/环境信息 用于灰度发布和流量控制

        六、避坑指南

        ❌ 错误做法 ✅ 正确做法
        硬编码服务地址 使用 discovery:///service-name 方案
        注册中心单节点 至少3节点集群部署
        忽略服务deregister 实现优雅关闭，调用Deregister
        心跳超时设太长 根据业务容忍度设置（通常15-30秒）
        不做健康检查 实现gRPC health check协议
        所有环境共用注册中心 dev/staging/prod 隔离
        忽略DNS缓存 客户端设置合理的刷新间隔
