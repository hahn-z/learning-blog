// api/order/v1/order.proto
syntax = "proto3";
package order.v1;

option go_package = "order-service/api/order/v1";

import "google/protobuf/empty.proto";

service OrderService {
    rpc CreateOrder(CreateOrderRequest) returns (CreateOrderReply) {}
    rpc GetOrder(GetOrderRequest) returns (GetOrderReply) {}
    rpc UpdateOrder(UpdateOrderRequest) returns (UpdateOrderReply) {}
    rpc DeleteOrder(DeleteOrderRequest) returns (google.protobuf.Empty) {}
    rpc ListOrders(ListOrdersRequest) returns (ListOrdersReply) {}
}

    int64 user_id = 1;

    double price = 3;

    int64 id = 1;

    int64 id = 1;

// internal/biz/order.go - Domain entity
package biz

import "time"

// Order is the domain model
type Order struct {
    ID          int64
    OrderNo     string
    UserID      int64
    Status      string // PENDING, PAID, SHIPPED, COMPLETED, CANCELLED
    TotalAmount float64
    Items       []OrderItem
    CreatedAt   time.Time
    UpdatedAt   time.Time
}

type OrderItem struct {
    ProductID int64
    Quantity  int32
    Price     float64
}

// OrderRepo defines repository interface
type OrderRepo interface {
    Create(order *Order) (*Order, error)
    GetByID(id int64) (*Order, error)
    Update(id int64, status string) (*Order, error)
    Delete(id int64) error
    ListByUserID(userID int64, page, pageSize int32) ([]*Order, int32, error)
}

v2: 数据层 + 业务层实现

// internal/data/order.go - Data access with GORM
package data

import (
    "context"
    "fmt"
    "time"

    "gorm.io/gorm"
    "order-service/internal/biz"
)

type OrderPO struct {
    ID          int64          `gorm:"primaryKey"`
    OrderNo     string         `gorm:"uniqueIndex;size:32"`
    UserID      int64          `gorm:"index"`
    Status      string         `gorm:"size:20;default:PENDING"`
    TotalAmount float64
    CreatedAt   time.Time
    UpdatedAt   time.Time
    DeletedAt   gorm.DeletedAt `gorm:"index"`
}

func (OrderPO) TableName() string { return "orders" }

type orderRepo struct {
    db *gorm.DB
}

// NewOrderRepo creates order repository
func NewOrderRepo(db *gorm.DB) biz.OrderRepo {
    return &orderRepo{db: db}
}

// Create inserts a new order
func (r *orderRepo) Create(order *biz.Order) (*biz.Order, error) {
    po := &OrderPO{
        OrderNo:     fmt.Sprintf("ORD%s%d", time.Now().Format("20060102"), order.UserID),
        UserID:      order.UserID,
        Status:      "PENDING",
        TotalAmount: r.calcTotal(order.Items),
    }
    if err := r.db.Create(po).Error; err != nil {
        return nil, fmt.Errorf("create order failed: %w", err)
    }
    order.ID = po.ID
    order.OrderNo = po.OrderNo
    order.Status = po.Status
    return order, nil
}

// GetByID fetches order by primary key
func (r *orderRepo) GetByID(id int64) (*biz.Order, error) {
    var po OrderPO
    if err := r.db.First(&po, id).Error; err != nil {
        if err == gorm.ErrRecordNotFound {
            return nil, biz.ErrOrderNotFound
        }
        return nil, fmt.Errorf("get order failed: %w", err)
    }
    return r.toDomain(&po), nil
}

// Update modifies order status
func (r *orderRepo) Update(id int64, status string) (*biz.Order, error) {
    result := r.db.Model(&OrderPO{}).Where("id = ?", id).Update("status", status)
    if result.Error != nil {
        return nil, fmt.Errorf("update order failed: %w", result.Error)
    }
    if result.RowsAffected == 0 {
        return nil, biz.ErrOrderNotFound
    }
    return r.GetByID(id)
}

// Delete soft-deletes an order
func (r *orderRepo) Delete(id int64) error {
    result := r.db.Delete(&OrderPO{}, id)
    if result.RowsAffected == 0 {
        return biz.ErrOrderNotFound
    }
    return result.Error
}

// ListByUserID returns paginated orders for a user
func (r *orderRepo) ListByUserID(userID int64, page, pageSize int32) ([]*biz.Order, int32, error) {
    var total int64
    var pos []OrderPO
    query := r.db.Where("user_id = ?", userID)
    query.Model(&OrderPO{}).Count(&total)

    offset := int((page - 1) * pageSize)
    if err := query.Offset(offset).Limit(int(pageSize)).Order("created_at DESC").Find(&pos).Error; err != nil {
        return nil, 0, err
    }

    orders := make([]*biz.Order, 0, len(pos))
    for _, po := range pos {
        orders = append(orders, r.toDomain(&po))
    }
    return orders, int32(total), nil
}

func (r *orderRepo) calcTotal(items []biz.OrderItem) float64 {
    var total float64
    for _, item := range items {
        total += float64(item.Quantity) * item.Price
    }
    return total
}

func (r *orderRepo) toDomain(po *OrderPO) *biz.Order {
    return &biz.Order{
        ID:          po.ID,
        OrderNo:     po.OrderNo,
        UserID:      po.UserID,
        Status:      po.Status,
        TotalAmount: po.TotalAmount,
        CreatedAt:   po.CreatedAt,
        UpdatedAt:   po.UpdatedAt,
    }
}

    db, err := gorm.Open(postgres.Open(dsn), &gorm.Config{})

// internal/biz/order.go - Business errors
package biz

import "github.com/go-kratos/kratos/v2/errors"

var (
    ErrOrderNotFound = errors.NotFound("ORDER_NOT_FOUND", "order not found")
    ErrInvalidStatus = errors.BadRequest("INVALID_STATUS", "invalid order status")
    ErrEmptyItems    = errors.BadRequest("EMPTY_ITEMS", "order must have at least one item")
)

// Validate validates order creation request
func (o *Order) Validate() error {
    if len(o.Items) == 0 {
        return ErrEmptyItems
    }
    return nil
}
// internal/service/order.go - Service layer (implements both HTTP & gRPC)
package service

import (
    "context"
    "fmt"

    pb "order-service/api/order/v1"
    "order-service/internal/biz"

    "github.com/go-kratos/kratos/v2/log"
)

type OrderService struct {
    pb.UnimplementedOrderServiceServer
    repo biz.OrderRepo
    log  *log.Helper
}

// NewOrderService creates order service
func NewOrderService(repo biz.OrderRepo, logger log.Logger) *OrderService {
    return &OrderService{repo: repo, log: log.NewHelper(logger)}
}

// CreateOrder handles order creation
func (s *OrderService) CreateOrder(ctx context.Context, req *pb.CreateOrderRequest) (*pb.CreateOrderReply, error) {
    s.log.WithContext(ctx).Infof("CreateOrder user_id=%d items=%d", req.UserId, len(req.Items))

    // Build domain object
    items := make([]biz.OrderItem, 0, len(req.Items))
    for _, item := range req.Items {
        items = append(items, biz.OrderItem{
            ProductID: item.ProductId,
            Quantity:  item.Quantity,
            Price:     item.Price,
        })
    }
    order := &biz.Order{UserID: req.UserId, Items: items}

    // Validate
    if err := order.Validate(); err != nil {
        return nil, err
    }

    // Persist
    created, err := s.repo.Create(order)
    if err != nil {
        return nil, fmt.Errorf("create order failed: %w", err)
    }

    return &pb.CreateOrderReply{
        Id:      created.ID,
        OrderNo: created.OrderNo,
        Status:  created.Status,
    }, nil
}

// GetOrder retrieves order by ID
func (s *OrderService) GetOrder(ctx context.Context, req *pb.GetOrderRequest) (*pb.GetOrderReply, error) {
    order, err := s.repo.GetByID(req.Id)
    if err != nil {
        return nil, err
    }
    return &pb.GetOrderReply{
        Id:          order.ID,
        OrderNo:     order.OrderNo,
        UserId:      order.UserID,
        Status:      order.Status,
        TotalAmount: order.TotalAmount,
        CreatedAt:   order.CreatedAt.Format("2006-01-02 15:04:05"),
    }, nil
}

// UpdateOrder updates order status
func (s *OrderService) UpdateOrder(ctx context.Context, req *pb.UpdateOrderRequest) (*pb.UpdateOrderReply, error) {
    validStatuses := map[string]bool{"PAID": true, "SHIPPED": true, "COMPLETED": true, "CANCELLED": true}
    if !validStatuses[req.Status] {
        return nil, biz.ErrInvalidStatus
    }

    order, err := s.repo.Update(req.Id, req.Status)
    if err != nil {
        return nil, err
    }
    return &pb.UpdateOrderReply{Id: order.ID, Status: order.Status}, nil
}

// DeleteOrder soft-deletes an order
func (s *OrderService) DeleteOrder(ctx context.Context, req *pb.DeleteOrderRequest) (*pb.DeleteOrderReply, error) {
    // Return Empty, but proto uses google.protobuf.Empty workaround
    return nil, s.repo.Delete(req.Id)
}

// ListOrders returns paginated order list
func (s *OrderService) ListOrders(ctx context.Context, req *pb.ListOrdersRequest) (*pb.ListOrdersReply, error) {
    if req.Page <= 0 {
        req.Page = 1
    }
    if req.PageSize <= 0 {
        req.PageSize = 10
    }

    orders, total, err := s.repo.ListByUserID(req.UserId, req.Page, req.PageSize)
    if err != nil {
        return nil, err
    }

    reply := &pb.ListOrdersReply{Total: total}
    for _, o := range orders {
        reply.Orders = append(reply.Orders, &pb.GetOrderReply{
            Id:          o.ID,
            OrderNo:     o.OrderNo,
            UserId:      o.UserID,
            Status:      o.Status,
            TotalAmount: o.TotalAmount,
            CreatedAt:   o.CreatedAt.Format("2006-01-02 15:04:05"),
        })
    }
    return reply, nil
}

// internal/server/http.go - HTTP server with full middleware
package server

import (
    "github.com/go-kratos/kratos/v2/log"
    "github.com/go-kratos/kratos/v2/middleware/logging"
    "github.com/go-kratos/kratos/v2/middleware/recovery"
    "github.com/go-kratos/kratos/v2/middleware/tracing"
    "github.com/go-kratos/kratos/v2/middleware/validate"
    khttp "github.com/go-kratos/kratos/v2/transport/http"
    pb "order-service/api/order/v1"
    "order-service/internal/service"
)

func NewHTTPServer(c *conf.Server, svc *service.OrderService) *khttp.Server {
    var opts = []khttp.ServerOption{
        khttp.Address(c.Http.Addr),
        khttp.Middleware(
            recovery.Recovery(),
            tracing.Server(),
            logging.Server(log.DefaultLogger),
            validate.Validator(),
        ),
    }
    srv := khttp.NewServer(opts...)
    pb.RegisterOrderServiceHTTPServer(srv, svc)
    return srv
}

// cmd/server/main.go - Application entry point
package main

import (
    "log"

    "github.com/go-kratos/kratos/contrib/registry/consul/v2"
    "github.com/go-kratos/kratos/v2"
    "github.com/go-kratos/kratos/v2/config"
    "github.com/go-kratos/kratos/v2/config/file"
    "github.com/hashicorp/consul/api"
    "order-service/internal/data"
    "order-service/internal/server"
    "order-service/internal/service"
)

func main() {
    // Load config
    c := config.New(
        config.WithSource(file.NewSource("configs/config.yaml")),
    )
    if err := c.Load(); err != nil {
        log.Fatal(err)
    }
    var conf conf.Server
    if err := c.Scan(&conf); err != nil {
        log.Fatal(err)
    }

    // Init data layer
    db := data.NewDB(conf.Database.Dsn)
    orderRepo := data.NewOrderRepo(db)

    // Init service
    orderSvc := service.NewOrderService(orderRepo, log.DefaultLogger)

    // Init servers
    httpSrv := server.NewHTTPServer(&conf, orderSvc)
    grpcSrv := server.NewGRPCServer(&conf, orderSvc)

    // Registry
    consulClient, _ := api.NewClient(api.DefaultConfig())
    r := consul.New(consulClient)

    // Create and run app
    app := kratos.New(
        kratos.Name("order-service"),
        kratos.Version("v1.0.0"),
        kratos.Server(httpSrv, grpcSrv),
        kratos.Registrar(r),
    )

    if err := app.Run(); err != nil {
        log.Fatal(err)
    }
}

五、注意事项

注意点 说明 建议
事务边界 创建订单需同时写入订单和订单项 使用GORM事务包裹
并发安全 同一订单并发更新可能冲突 乐观锁(version字段)或分布式锁
分页参数 page=0或pageSize=0会查全表 设置默认值和上限
软删除 GORM软删除后查询自动过滤 注意唯一索引需配合DeletedAt
状态机 PENDING→PAID→SHIPPED→COMPLETED 校验状态转换合法性
错误传播 gRPC内部错误需映射为HTTP错误 使用Kratos errors统一处理
ID生成 自增ID暴露业务量 考虑雪花算法等分布式ID

六、避坑指南

❌ 错误做法 ✅ 正确做法
Service直接操作数据库 Service→Biz接口→Data层，依赖倒置
硬编码数据库连接 配置文件管理，支持环境变量覆盖
不校验请求参数 proto validate注解 + validate中间件
返回裸error 用 errors.New/BadRequest/NotFound 等
不处理数据库NotFound 区分"不存在"和"数据库故障"
忽略RowsAffected 更新/删除时检查影响行数
用int当ID 大系统用int64，分布式用string雪花ID
不写日志 关键操作加结构化日志，方便排查
