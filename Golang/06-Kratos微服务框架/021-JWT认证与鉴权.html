021-JWT认证与鉴权

难度：⭐⭐⭐⭐（高级）

本文深入讲解在 Kratos 微服务中实现 JWT 认证与鉴权的完整方案，涵盖 gRPC metadata 传递、中间件设计、角色鉴权、Token 刷新及多服务间 Token 透传。

一、概念讲解

1.1 JWT 基础回顾

JWT（JSON Web Token）由三部分组成：Header.Payload.Signature。在微服务架构中，JWT 是无状态认证的首选方案——服务端不需要存储 Session，只需验证签名即可信任 Token 中的 Claims。

1.2 Kratos 中的认证架构

Kratos 的认证体系基于 Middleware 机制。请求经过 Transport 层时，中间件从 gRPC metadata 或 HTTP Header 中提取 Token，解析后注入 Context，供下游 biz 层使用。

Client → Transport(HTTP/gRPC) → Auth Middleware → Service → Biz → Data
                                      ↓
                              提取并验证JWT
                              注入Claims到Context

1.3 gRPC 中的 Token 传递

gRPC 不使用 HTTP Header，而是使用 Metadata（类似 HTTP Header 的键值对）。客户端通过 metadata.AppendToOutgoingContext 注入 Token，服务端通过 metadata.FromIncomingContext 提取。

1.4 与 Gin JWT 的异同

对比项 Gin (HTTP) Kratos (gRPC + HTTP)
Token 提取 c.GetHeader("Authorization") gRPC: metadata; HTTP: Header
中间件形式 gin.HandlerFunc transport.Middleware
Context 注入 c.Set("claims", claims) context.WithValue
路由级控制 r.Use(jwtMiddleware) Kratos Route/Selector
Token 刷新 手动实现 中间件统一处理

三、完整代码

v1: 基础 JWT 认证

// pkg/jwt/jwt.go - JWT utility package
package jwt

import (
	"errors"
	"time"

	"github.com/golang-jwt/jwt/v5"
)

var (
	ErrTokenExpired = errors.New("token has expired")
	ErrTokenInvalid = errors.New("token is invalid")
)

// Claims represents the JWT payload
type Claims struct {
	UserID   int64  `json:"user_id"`
	Username string `json:"username"`
	Role     string `json:"role"`
	jwt.RegisteredClaims
}

// JWTManager handles token operations
type JWTManager struct {
	secretKey     []byte
	accessExpiry  time.Duration
	refreshExpiry time.Duration
}

// NewJWTManager creates a new JWT manager
func NewJWTManager(secretKey string, accessExp, refreshExp time.Duration) *JWTManager {
	return &JWTManager{
		secretKey:     []byte(secretKey),
		accessExpiry:  accessExp,
		refreshExpiry: refreshExp,
	}
}

// GenerateAccessToken generates a short-lived access token
func (m *JWTManager) GenerateAccessToken(userID int64, username, role string) (string, error) {
	claims := &Claims{
		UserID:   userID,
		Username: username,
		Role:     role,
		RegisteredClaims: jwt.RegisteredClaims{
			ExpiresAt: jwt.NewNumericDate(time.Now().Add(m.accessExpiry)),
			IssuedAt:  jwt.NewNumericDate(time.Now()),
			Issuer:    "kratos-microservice",
		},
	}
	token := jwt.NewWithClaims(jwt.SigningMethodHS256, claims)
	return token.SignedString(m.secretKey)
}

// GenerateRefreshToken generates a long-lived refresh token
func (m *JWTManager) GenerateRefreshToken(userID int64) (string, error) {
	claims := &jwt.RegisteredClaims{
		Subject:   fmt.Sprintf("%d", userID),
		ExpiresAt: jwt.NewNumericDate(time.Now().Add(m.refreshExpiry)),
		IssuedAt:  jwt.NewNumericDate(time.Now()),
		Issuer:    "kratos-microservice",
	}
	token := jwt.NewWithClaims(jwt.SigningMethodHS256, claims)
	return token.SignedString(m.secretKey)
}

// ParseToken parses and validates a JWT token
func (m *JWTManager) ParseToken(tokenString string) (*Claims, error) {
	token, err := jwt.ParseWithClaims(tokenString, &Claims{}, func(token *jwt.Token) (interface{}, error) {
		// Validate signing method
		if _, ok := token.Method.(*jwt.SigningMethodHMAC); !ok {
			return nil, ErrTokenInvalid
		}
		return m.secretKey, nil
	})
	if err != nil {
		if errors.Is(err, jwt.ErrTokenExpired) {
			return nil, ErrTokenExpired
		}
		return nil, ErrTokenInvalid
	}
	claims, ok := token.Claims.(*Claims)
	if !ok || !token.Valid {
		return nil, ErrTokenInvalid
	}
	return claims, nil
}

v2: 认证中间件 + gRPC Metadata

// internal/middleware/auth.go - Authentication middleware
package middleware

import (
	"context"
	"fmt"
	"strings"

	"google.golang.org/grpc/metadata"

	"github.com/go-kratos/kratos/v2/middleware"
	"github.com/go-kratos/kratos/v2/transport"

	jwtpkg "your-project/pkg/jwt"
)

type authKey struct{}

// ClaimsFromContext extracts claims from context
func ClaimsFromContext(ctx context.Context) *jwtpkg.Claims {
	claims, _ := ctx.Value(authKey{}).(*jwtpkg.Claims)
	return claims
}

// JWTAuth returns a Kratos middleware for JWT authentication
func JWTAuth(jwtMgr *jwtpkg.JWTManager) middleware.Middleware {
	return func(handler middleware.Handler) middleware.Handler {
		return func(ctx context.Context, req interface{}) (interface{}, error) {
			// Extract token from transport (works for both HTTP and gRPC)
			tokenString, err := extractToken(ctx)
			if err != nil {
				return nil, err
			}

			// Parse and validate token
			claims, err := jwtMgr.ParseToken(tokenString)
			if err != nil {
				return nil, fmt.Errorf("authentication failed: %w", err)
			}

			// Inject claims into context
			ctx = context.WithValue(ctx, authKey{}, claims)

			return handler(ctx, req)
		}
	}
}

// extractToken extracts JWT from HTTP header or gRPC metadata
func extractToken(ctx context.Context) (string, error) {
	// Try HTTP first
	if tr, ok := transport.FromServerContext(ctx); ok {
		header := tr.RequestHeader()
		auth := header.Get("Authorization")
		if auth != "" {
			return strings.TrimPrefix(auth, "Bearer "), nil
		}
	}

	// Try gRPC metadata
	md, ok := metadata.FromIncomingContext(ctx)
	if !ok {
		return "", fmt.Errorf("no metadata found")
	}
	values := md.Get("authorization")
	if len(values) == 0 {
		return "", fmt.Errorf("no authorization token")
	}
	return strings.TrimPrefix(values[0], "Bearer "), nil
}

v2: 角色鉴权中间件

// internal/middleware/rbac.go - Role-based access control
package middleware

// RequireRole creates a middleware that checks user role
func RequireRole(roles ...string) middleware.Middleware {
	roleSet := make(map[string]struct{}, len(roles))
	for _, r := range roles {
		roleSet[r] = struct{}{}
	}
	return func(handler middleware.Handler) middleware.Handler {
		return func(ctx context.Context, req interface{}) (interface{}, error) {
			claims := ClaimsFromContext(ctx)
			if claims == nil {
				return nil, fmt.Errorf("no auth claims in context")
			}
			if _, ok := roleSet[claims.Role]; !ok {
				return nil, fmt.Errorf("permission denied: role %s not allowed", claims.Role)
			}
			return handler(ctx, req)
		}
	}
}

v3: 完整集成（Token 刷新 + 多服务透传）

// internal/service/auth.go - Auth service with refresh
package service

import (
	"context"
	"crypto/rand"
	"encoding/hex"
	"time"

	pb "your-project/api/user/v1"
	jwtpkg "your-project/pkg/jwt"
)

type AuthService struct {
	pb.UnimplementedAuthServer
	jwtMgr    *jwtpkg.JWTManager
	refreshStore map[string]refreshInfo // In production, use Redis
}

type refreshInfo struct {
	UserID    int64
	ExpiresAt time.Time
}

// Login authenticates user and returns token pair
func (s *AuthService) Login(ctx context.Context, req *pb.LoginRequest) (*pb.LoginReply, error) {
	// Verify credentials (simplified)
	user, err := s.verifyCredentials(req.Username, req.Password)
	if err != nil {
		return nil, err
	}

	// Generate token pair
	accessToken, err := s.jwtMgr.GenerateAccessToken(user.ID, user.Username, user.Role)
	if err != nil {
		return nil, err
	}
	refreshToken, err := s.jwtMgr.GenerateRefreshToken(user.ID)
	if err != nil {
		return nil, err
	}

	// Store refresh token
	s.refreshStore[refreshToken] = refreshInfo{
		UserID:    user.ID,
		ExpiresAt: time.Now().Add(7 * 24 * time.Hour),
	}

	return &pb.LoginReply{
		AccessToken:  accessToken,
		RefreshToken: refreshToken,
		ExpiresIn:    int64(15 * time.Minute / time.Second),
	}, nil
}

// RefreshToken renews access token using refresh token
func (s *AuthService) RefreshToken(ctx context.Context, req *pb.RefreshTokenRequest) (*pb.RefreshTokenReply, error) {
	info, ok := s.refreshStore[req.RefreshToken]
	if !ok || time.Now().After(info.ExpiresAt) {
		return nil, fmt.Errorf("invalid or expired refresh token")
	}
	// Delete old refresh token (rotation)
	delete(s.refreshStore, req.RefreshToken)

	// Generate new token pair
	accessToken, _ := s.jwtMgr.GenerateAccessToken(info.UserID, "", "")
	newRefresh, _ := s.jwtMgr.GenerateRefreshToken(info.UserID)
	s.refreshStore[newRefresh] = refreshInfo{
		UserID:    info.UserID,
		ExpiresAt: time.Now().Add(7 * 24 * time.Hour),
	}
	return &pb.RefreshTokenReply{
		AccessToken:  accessToken,
		RefreshToken: newRefresh,
	}, nil
}
// internal/middleware/forward.go - Token forwarding for inter-service calls
package middleware

import (
	"context"
	"strings"

	"google.golang.org/grpc/metadata"
)

// ForwardToken injects JWT into outgoing gRPC metadata
func ForwardToken() middleware.Middleware {
	return func(handler middleware.Handler) middleware.Handler {
		return func(ctx context.Context, req interface{}) (interface{}, error) {
			token, err := extractToken(ctx)
			if err == nil {
				ctx = metadata.AppendToOutgoingContext(ctx, "authorization", "Bearer "+token)
			}
			return handler(ctx, req)
		}
	}
}
// cmd/server/main.go - Wire up middleware
func NewHTTPServer(c *conf.Server, jwtMgr *jwtpkg.JWTManager) *http.Server {
	var opts = []http.ServerOption{
		http.Middleware(
			middleware.JWTAuth(jwtMgr),
		),
	}
	// Public routes use Selector to skip auth
	srv := http.NewServer(opts...)
	// ... register routes
	return srv
}

五、注意事项

事项 说明
密钥管理 生产环境使用 RSA/ECDSA 非对称密钥，方便服务间只验证不签发
Token 有效期 Access Token 15分钟，Refresh Token 7天
HTTPS 生产环境必须 HTTPS，防止 Token 被嗅探
黑名单 如需即时吊销，使用 Redis 维护黑名单（牺牲无状态性）
Context Key 使用 unexported struct 作为 context key，避免冲突

六、避坑指南

❌ 错误做法 ✅ 正确做法
在 Service 层提取 Token 在 Middleware 层统一处理
使用 context.Value(string) 作为 key 使用 unexported struct 作为 key
gRPC 客户端不传 Authorization 使用 ForwardToken 中间件自动透传
JWT Secret 硬编码 从环境变量/Vault 读取
只用 Access Token 不做 Refresh Access + Refresh 双 Token 机制
Metadata 中 Token 不加 Bearer 前缀 保持与 HTTP 一致的 Bearer xxx 格式
