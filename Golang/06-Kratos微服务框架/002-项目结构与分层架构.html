002-项目结构与分层架构

难度：⭐⭐⭐（进阶级）

需要理解DDD基本概念。读完本文，你将掌握Kratos项目的完整结构，理解各层之间的依赖关系。

一、概念讲解：DDD分层架构

1.1 什么是DDD分层？

DDD（Domain-Driven Design，领域驱动设计）将系统分为四个核心层：

┌───────────────────────────┐
│     Interface Layer       │  ← 接口层：处理请求/响应
│   (API / Transport)       │
├───────────────────────────┤
│     Application Layer     │  ← 应用层：编排业务流程
│     (Service)             │
├───────────────────────────┤
│     Domain Layer          │  ← 领域层：核心业务逻辑
│     (Biz)                 │
├───────────────────────────┤
│     Infrastructure Layer  │  ← 基础设施层：技术实现
│     (Data)                │
└───────────────────────────┘

关键原则：依赖倒置（Dependency Inversion）

  • 上层依赖下层定义的接口，不依赖具体实现
  • 下层实现上层定义的接口
  • 通过依赖注入（Wire）在运行时组装

    • 1.2 Kratos中的分层映射

    DDD层 Kratos目录 职责
    接口层 api/ Proto定义 + 生成代码
    应用层 internal/service/ 接收请求，调用Biz层
    领域层 internal/biz/ 核心业务逻辑
    基础设施层 internal/data/ 数据库、缓存、外部API
    入口层 cmd/ 程序入口，依赖注入组装
    服务层 internal/server/ HTTP/gRPC服务器配置

    main.go — 程序启动入口：

    package main

import (
    "flag"
    "os"

    "helloworld/internal/conf"

    // Import wire generated injector
    _ "helloworld/cmd/helloworld/wire"

    "github.com/go-kratos/kratos/v2"
    "github.com/go-kratos/kratos/v2/config"
    "github.com/go-kratos/kratos/v2/config/file"
    "github.com/go-kratos/kratos/v2/log"
    "github.com/go-kratos/kratos/v2/middleware/tracing"
    "github.com/go-kratos/kratos/v2/transport/grpc"
    "github.com/go-kratos/kratos/v2/transport/http"
)

// go build -ldflags "-X main.Name=x"
var (
    Name     = "helloworld"
    // Version holds the service version, injected at build time
    Version  = "1.0.0"
)

func newApp(logger log.Logger, gs *grpc.Server, hs *http.Server) *kratos.App {
    return kratos.New(
        kratos.ID(Name),
        kratos.Version(Version),
        kratos.Metadata(map[string]string{}),
        kratos.Logger(logger),
        kratos.Server(gs, hs),
    )
}

func main() {
    flag.Parse()
    logger := log.With(log.NewStdLogger(os.Stdout),
        "ts", log.DefaultTimestamp,
        "caller", log.DefaultCaller,
        "service_name", Name,
        "service_version", Version,
        "trace_id", tracing.TraceID(),
        "span_id", tracing.SpanID(),
    )

    c := config.New(
        config.WithSource(
            file.NewSource(flag.String("conf", "configs", "config path")),
        ),
    )
    if err := c.Load(); err != nil {
        panic(err)
    }

    var bc conf.Bootstrap
    if err := c.Scan(&bc); err != nil {
        panic(err)
    }

    // Wire injector builds all dependencies
    app, cleanup, err := wireApp(bc.Server, bc.Data, logger)
    if err != nil {
        panic(err)
    }
    defer cleanup()

    // Start and block
    if err := app.Run(); err != nil {
        panic(err)
    }
}

    3.2 internal/service/ — 应用层（Service）

    Service层是最薄的一层，只做请求转换和调用Biz层：

    package service

import (
    "context"

    pb "helloworld/api/helloworld/v1"
    "helloworld/internal/biz"

    "github.com/go-kratos/kratos/v2/log"
)

// GreeterService implements the greeter service defined in proto
type GreeterService struct {
    pb.UnimplementedGreeterServer

    // uc is the use case (biz layer) for greeter
    uc  *biz.GreeterUsecase
    log *log.Helper
}

// NewGreeterService creates a new greeter service
func NewGreeterService(uc *biz.GreeterUsecase, logger log.Logger) *GreeterService {
    return &GreeterService{
        uc:  uc,
        log: log.NewHelper(logger),
    }
}

// SayHello implements helloworld.v1.GreeterService
func (s *GreeterService) SayHello(ctx context.Context, req *pb.HelloRequest) (*pb.HelloReply, error) {
    s.log.WithContext(ctx).Infof("SayHello received: %v", req.GetName())

    // Delegate to biz layer (core logic)
    greeting, err := s.uc.CreateGreeting(ctx, req.GetName())
    if err != nil {
        return nil, err
    }

    return &pb.HelloReply{Message: greeting}, nil
}

    3.3 internal/biz/ — 领域层（核心）

    Biz层包含核心业务逻辑，定义Repo接口但不实现：

    package biz

import (
    "context"
    "fmt"

    "github.com/go-kratos/kratos/v2/log"
)

// Greeter is the domain entity
type Greeter struct {
    ID      int64
    Name    string
    Greeting string
}

// GreeterRepo defines the repository interface (implemented by data layer)
type GreeterRepo interface {
    Save(ctx context.Context, g *Greeter) (*Greeter, error)
    GetByName(ctx context.Context, name string) (*Greeter, error)
}

// GreeterUsecase holds business logic for greeter
type GreeterUsecase struct {
    repo GreeterRepo
    log  *log.Helper
}

// NewGreeterUsecase creates a new use case
func NewGreeterUsecase(repo GreeterRepo, logger log.Logger) *GreeterUsecase {
    return &GreeterUsecase{
        repo: repo,
        log:  log.NewHelper(logger),
    }
}

// CreateGreeting contains core business logic
func (uc *GreeterUsecase) CreateGreeting(ctx context.Context, name string) (string, error) {
    // Business rule: name cannot be empty
    if name == "" {
        name = "World"
    }

    greeting := fmt.Sprintf("Hello, %s!", name)

    // Save to repository (data layer handles actual storage)
    _, err := uc.repo.Save(ctx, &Greeter{
        Name:     name,
        Greeting: greeting,
    })
    if err != nil {
        uc.log.Warnf("Failed to save greeting: %v", err)
        // Don't fail the request, just log
    }

    return greeting, nil
}

    3.5 internal/server/ — 服务器配置

    // internal/server/http.go
package server

import (
    pb "helloworld/api/helloworld/v1"
    "helloworld/internal/conf"
    "helloworld/internal/service"

    "github.com/go-kratos/kratos/v2/log"
    "github.com/go-kratos/kratos/v2/middleware/recovery"
    "github.com/go-kratos/kratos/v2/transport/http"
)

// NewHTTPServer creates an HTTP server
func NewHTTPServer(c *conf.Server, greeter *service.GreeterService, logger log.Logger) *http.Server {
    opts := []http.ServerOption{
        http.Middleware(
            recovery.Recovery(),
        ),
    }
    if c.Http.Network != "" {
        opts = append(opts, http.Network(c.Http.Network))
    }
    if c.Http.Addr != "" {
        opts = append(opts, http.Address(c.Http.Addr))
    }

    srv := http.NewServer(opts...)
    pb.RegisterGreeterHTTPServer(srv, greeter)
    return srv
}

    五、注意事项

    注意点 说明 严重度
    Service层不要写业务逻辑 Service只做编排，逻辑放Biz 🔴 高
    Biz层定义接口，Data层实现 依赖倒置的核心 🔴 高
    ProviderSet必须正确声明 Wire依赖注入需要ProviderSet 🔴 高
    wire.go的build tag 注意//go:build wireinject 🟡 中
    不要在Biz层直接import Data 通过接口解耦 🔴 高

    六、避坑指南

    ❌ 常见错误 ✅ 正确做法
    在Service层直接操作数据库 Service调Biz，Biz定义Repo接口，Data实现
    Biz层import Data层的包 Biz定义接口，通过Wire注入Data实现
    忘记添加Wire ProviderSet 每个包都声明var ProviderSet = wire.NewSet(...)
    修改接口后不重新生成Wire 执行make generate更新wire_gen.go
    把所有逻辑塞在一个文件 按职责分文件：biz/greeter.go、data/greeter.go

          2. 八、知识点总结

          项目结构与分层
├── cmd/
│   ├── main.go (入口、配置加载)
│   └── wire/ (依赖注入)
├── internal/
│   ├── service/ (应用层 - 请求编排)
│   ├── biz/ (领域层 - 业务逻辑)
│   │   └── 定义Repo接口
│   ├── data/ (基础设施层 - 数据访问)
│   │   └── 实现Repo接口
│   ├── server/ (服务器配置)
│   │   ├── http.go
│   │   └── grpc.go
│   └── conf/ (配置结构)
├── api/ (Proto定义+生成代码)
├── configs/ (YAML配置文件)
└── 依赖关系:
    Service → Biz(接口) ← Data(实现)
    Wire负责运行时组装

          九、举一反三

          场景 思路 涉及的层
          新增一个业务实体 Biz定义实体+Repo接口→Data实现→Service暴露→Proto定义 全层
          接入MySQL Data层初始化GORM，实现Repo接口 Data层
          添加缓存 Data层增加Redis依赖，实现缓存逻辑 Data层
          添加中间件 Server层注册中间件 Server层
          多服务共享逻辑 Biz层抽取公共包，独立module Biz层
