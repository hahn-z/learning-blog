1. 概念讲解

什么是消息队列？

消息队列（Message Queue，简称MQ）是一种应用程序间的通信方式，它将消息发送到队列中，由接收者从队列中获取并处理。消息队列实现了异步通信和解耦，是分布式系统中重要的基础设施。

为什么使用消息队列？

在高并发和分布式系统中，消息队列解决了以下问题：

1. 异步处理

  • 传统同步调用：客户端 → 服务端处理（耗时） → 返回响应（阻塞）
  • 异步处理：客户端 → 消息队列 → 服务端异步处理（快速返回）

    • 2. 系统解耦

    • 生产者和消费者互不依赖，可以独立部署和扩展
    • 新增消费者无需修改生产者代码

      • 3. 流量削峰

      • 高并发时，消息队列作为缓冲，保护后端服务
      • 请求先进入队列，消费者按自己的节奏处理

        • 4. 可靠性

        • 消息持久化，防止丢失
        • 支持重试机制，保证消息可靠投递

          • Go语言与消息队列

          Go语言提供了丰富的MQ客户端库：

          MQ类型 推荐库 特点
          RabbitMQ streadway/amqp 功能强大，路由灵活
          Kafka IBM/sarama 高吞吐量，适合大数据
          Redis Queue go-redis/redis 轻量级，适合简单场景
          NSQ nsqio/go-nsq 分布式，支持实时和离线处理

          本文使用RabbitMQ进行实战，因为它功能完整、社区活跃、文档丰富。

          package main

import (
	"context"
	"encoding/json"
	"fmt"
	"log"
	"time"

	amqp "github.com/rabbitmq/amqp091-go"
)

// RabbitMQConfig RabbitMQ配置
type RabbitMQConfig struct {
	Host     string
	Port     int
	Username string
	Password string
	VHost    string
}

// RabbitMQClient RabbitMQ客户端
type RabbitMQClient struct {
	conn    *amqp.Connection
	channel *amqp.Channel
	ctx     context.Context
}

// NewRabbitMQClient 创建RabbitMQ客户端
func NewRabbitMQClient(config RabbitMQConfig) (*RabbitMQClient, error) {
	url := fmt.Sprintf("amqp://%s:%s@%s:%d%s",
		config.Username, config.Password, config.Host, config.Port, config.VHost)

	conn, err := amqp.Dial(url)
	if err != nil {
		return nil, fmt.Errorf("failed to connect to RabbitMQ: %w", err)
	}

	channel, err := conn.Channel()
	if err != nil {
		conn.Close()
		return nil, fmt.Errorf("failed to open channel: %w", err)
	}

	return &RabbitMQClient{
		conn:    conn,
		channel: channel,
		ctx:     context.Background(),
	}, nil
}

// Close 关闭连接
func (c *RabbitMQClient) Close() error {
	if c.channel != nil {
		c.channel.Close()
	}
	if c.conn != nil {
		return c.conn.Close()
	}
	return nil
}

          3.3 生产者实现

          3.3.1 简单生产者

          // SimpleProducer 简单生产者
type SimpleProducer struct {
	client *RabbitMQClient
	queue  string
}

// NewSimpleProducer 创建简单生产者
func NewSimpleProducer(client *RabbitMQClient, queue string) (*SimpleProducer, error) {
	// 声明队列
	_, err := client.channel.QueueDeclare(
		queue, // 队列名称
		true,  // durable（持久化）
		false, // delete when unused
		false, // exclusive
		false, // no-wait
		nil,   // arguments
	)
	if err != nil {
		return nil, fmt.Errorf("failed to declare queue: %w", err)
	}

	return &SimpleProducer{
		client: client,
		queue:  queue,
	}, nil
}

// Publish 发布消息
func (p *SimpleProducer) Publish(message EmailMessage) error {
	body, err := json.Marshal(message)
	if err != nil {
		return fmt.Errorf("marshal error: %w", err)
	}

	ctx, cancel := context.WithTimeout(p.client.ctx, 5*time.Second)
	defer cancel()

	err = p.client.channel.PublishWithContext(
		ctx,
		"",     // exchange
		p.queue, // routing key
		false,  // mandatory
		false,  // immediate
		amqp.Publishing{
			ContentType:  "application/json",
			Body:         body,
			DeliveryMode: amqp.Persistent, // 持久化消息
			Timestamp:    time.Now(),
		},
	)

	if err != nil {
		return fmt.Errorf("failed to publish message: %w", err)
	}

	log.Printf("✅ Published message: %s", message.Subject)
	return nil
}

		return fmt.Errorf("failed to publish message: %w", err)
	}

          3.4 消费者实现

          3.4.1 简单消费者

          // SimpleConsumer 简单消费者
type SimpleConsumer struct {
	client *RabbitMQClient
	queue  string
}

// NewSimpleConsumer 创建简单消费者
func NewSimpleConsumer(client *RabbitMQClient, queue string) (*SimpleConsumer, error) {
	// 声明队列
	_, err := client.channel.QueueDeclare(
		queue,
		true,
		false,
		false,
		false,
		nil,
	)
	if err != nil {
		return nil, fmt.Errorf("failed to declare queue: %w", err)
	}

	return &SimpleConsumer{
		client: client,
		queue:  queue,
	}, nil
}

// Consume 消费消息
func (c *SimpleConsumer) Consume(handler func(EmailMessage) error) error {
	msgs, err := c.client.channel.Consume(
		c.queue, // queue
		"",      // consumer tag
		false,   // auto-ack（手动确认）
		false,   // exclusive
		false,   // no-local
		false,   // no-wait
		nil,     // args
	)
	if err != nil {
		return fmt.Errorf("failed to register consumer: %w", err)
	}

	log.Printf("📥 Waiting for messages from queue: %s", c.queue)

	for msg := range msgs {
		var email EmailMessage
		if err := json.Unmarshal(msg.Body, &email); err != nil {
			log.Printf("❌ Failed to unmarshal message: %v", err)
			msg.Nack(false, false) // 拒绝且不重新入队
			continue
		}

		log.Printf("📨 Received email: %s -> %s", email.From, email.To)

		// 处理消息
		if err := handler(email); err != nil {
			log.Printf("❌ Failed to process message: %v", err)
			msg.Nack(false, true) // 拒绝并重新入队（重试）
			continue
		}

		// 确认消息
		msg.Ack(false)
		log.Printf("✅ Message processed successfully")
	}

	return nil
}

          3.4.2 带重试的消费者

          // RetryConsumer 带重试的消费者
type RetryConsumer struct {
	client        *RabbitMQClient
	queue         string
	retryQueue    string
	dlxQueue      string
	maxRetries    int
}

// NewRetryConsumer 创建带重试的消费者
func NewRetryConsumer(client *RabbitMQClient, queue string, maxRetries int) (*RetryConsumer, error) {
	// 声明死信交换机和队列
	dlxExchange := fmt.Sprintf("%s.dlx", queue)
	retryQueue := fmt.Sprintf("%s.retry", queue)
	dlxQueue := fmt.Sprintf("%s.dlx", queue)

	// 声明死信交换机
	err := client.channel.ExchangeDeclare(
		dlxExchange,
		"direct",
		true,
		false,
		false,
		false,
		nil,
	)
	if err != nil {
		return nil, fmt.Errorf("failed to declare dlx exchange: %w", err)
	}

	// 声明主队列，绑定死信交换机
	args := amqp.Table{
		"x-dead-letter-exchange":    dlxExchange,
		"x-dead-letter-routing-key": queue,
	}

	_, err = client.channel.QueueDeclare(
		queue,
		true,
		false,
		false,
		false,
		args,
	)
	if err != nil {
		return nil, fmt.Errorf("failed to declare queue: %w", err)
	}

	// 声明重试队列
	err = client.channel.QueueDeclare(
		retryQueue,
		true,
		false,
		false,
		false,
		amqp.Table{
			"x-dead-letter-exchange":    dlxExchange,
			"x-dead-letter-routing-key": queue,
		},
	)
	if err != nil {
		return nil, fmt.Errorf("failed to declare retry queue: %w", err)
	}

	// 声明死信队列
	_, err = client.channel.QueueDeclare(
		dlxQueue,
		true,
		false,
		false,
		false,
		nil,
	)
	if err != nil {
		return nil, fmt.Errorf("failed to declare dlx queue: %w", err)
	}

	// 绑定路由
	err = client.channel.QueueBind(
		queue,
		queue,
		dlxExchange,
		false,
		nil,
	)
	if err != nil {
		return nil, fmt.Errorf("failed to bind queue: %w", err)
	}

	return &RetryConsumer{
		client:     client,
		queue:      queue,
		retryQueue: retryQueue,
		dlxQueue:   dlxQueue,
		maxRetries: maxRetries,
	}, nil
}

// ConsumeWithRetry 带重试机制的消费
func (c *RetryConsumer) ConsumeWithRetry(handler func(TaskMessage) error) error {
	msgs, err := c.client.channel.Consume(
		c.queue,
		"",
		false,
		false,
		false,
		false,
		nil,
	)
	if err != nil {
		return fmt.Errorf("failed to register consumer: %w", err)
	}

	log.Printf("📥 Waiting for messages from queue: %s (max retries: %d)", c.queue, c.maxRetries)

	for msg := range msgs {
		var task TaskMessage
		if err := json.Unmarshal(msg.Body, &task); err != nil {
			log.Printf("❌ Failed to unmarshal message: %v", err)
			msg.Nack(false, false)
			continue
		}

		log.Printf("📨 Received task: %s (retry: %d/%d)", task.TaskID, task.Retry, task.MaxRetry)

		// 处理消息
		if err := handler(task); err != nil {
			task.Retry++

			// 超过最大重试次数，移入死信队列
			if task.Retry > c.maxRetries {
				log.Printf("💀 Max retries exceeded for task: %s", task.TaskID)

				// 发送到死信队列
				body, _ := json.Marshal(task)
				c.client.channel.Publish(
					"",            // exchange
					c.dlxQueue,    // routing key
					false,
					false,
					amqp.Publishing{
						ContentType:  "application/json",
						Body:         body,
						DeliveryMode: amqp.Persistent,
					},
				)

				msg.Ack(false)
				continue
			}

			// 更新重试次数并重新入队
			body, _ := json.Marshal(task)
			delay := time.Duration(task.Retry) * time.Second

			c.client.channel.Publish(
				"",            // exchange
				c.retryQueue,  // routing key
				false,
				false,
				amqp.Publishing{
					ContentType:  "application/json",
					Body:         body,
					DeliveryMode: amqp.Persistent,
					Expiration:   fmt.Sprintf("%d", int64(delay.Seconds()*1000)),
				},
			)

			log.Printf("🔄 Retrying task: %s in %v (attempt %d/%d)",
				task.TaskID, delay, task.Retry, c.maxRetries)

			msg.Ack(false)
			continue
		}

		// 处理成功
		msg.Ack(false)
		log.Printf("✅ Task completed: %s", task.TaskID)
	}

	return nil
}

type TopicConsumer struct {

// NewTopicConsumer 创建主题交换机消费者
func NewTopicConsumer(client *RabbitMQClient, exchange, queue, pattern string) (*TopicConsumer, error) {
	// 声明交换机
	err := client.channel.ExchangeDeclare(
		exchange,
		"topic",
		true,
		false,
		false,
		false,
		nil,
	)
	if err != nil {
		return nil, fmt.Errorf("failed to declare exchange: %w", err)
	}

	// 声明队列
	_, err = client.channel.QueueDeclare(
		queue,
		true,
		false,
		false,
		false,
		nil,
	)
	if err != nil {
		return nil, fmt.Errorf("failed to declare queue: %w", err)
	}

	// 绑定队列到交换机（使用路由模式）
	err = client.channel.QueueBind(
		queue,
		pattern,
		exchange,
		false,
		nil,
	)
	if err != nil {
		return nil, fmt.Errorf("failed to bind queue: %w", err)
	}

	return &TopicConsumer{
		client:   client,
		exchange: exchange,
		queue:    queue,
		pattern:  pattern,
	}, nil
}

// ConsumeByPattern 按模式消费消息
func (c *TopicConsumer) ConsumeByPattern(handler func(OrderMessage) error) error {

		"",
		false,
		false,
		false,
		false,
		nil,
	)
	if err != nil {
		return fmt.Errorf("failed to register consumer: %w", err)
	}

			continue
		}

		log.Printf("📦 Received order: %s (routing key: %s)", order.OrderID, msg.RoutingKey)

			log.Printf("❌ Failed to process order: %v", err)
			msg.Nack(false, true)
			continue
		}

		msg.Ack(false)
		log.Printf("✅ Order processed: %s", order.OrderID)
	}

}

          3.6 完整示例

          func main() {
	config := RabbitMQConfig{
		Host:     "localhost",
		Port:     5672,
		Username: "guest",
		Password: "guest",
		VHost:    "/",
	}

	client, err := NewRabbitMQClient(config)
	if err != nil {
		log.Fatalf("❌ Failed to create RabbitMQ client: %v", err)
	}
	defer client.Close()

	fmt.Println("✅ Connected to RabbitMQ")

	// 示例1：简单生产者-消费者模式
	fmt.Println("\n========== 示例1: 简单生产者-消费者 ==========")

	producer, err := NewSimpleProducer(client, "email_queue")
	if err != nil {
		log.Fatal(err)
	}

	consumer, err := NewSimpleConsumer(client, "email_queue")
	if err != nil {
		log.Fatal(err)
	}

	// 启动消费者（在goroutine中）
	go func() {
		err := consumer.Consume(func(email EmailMessage) error {
			log.Printf("📧 Sending email: %s", email.Subject)
			time.Sleep(500 * time.Millisecond) // 模拟处理时间
			return nil
		})
		if err != nil {
			log.Printf("❌ Consumer error: %v", err)
		}
	}()

	// 发送消息
	emails := []EmailMessage{
		{From: "noreply@example.com", To: "user1@example.com", Subject: "Welcome!", Body: "Welcome to our service!", Priority: 1},
		{From: "noreply@example.com", To: "user2@example.com", Subject: "Order Confirm", Body: "Your order #12345 is confirmed.", Priority: 2},
		{From: "noreply@example.com", To: "user3@example.com", Subject: "Password Reset", Body: "Click here to reset password.", Priority: 3},
	}

	for _, email := range emails {
		producer.Publish(email)
		time.Sleep(100 * time.Millisecond)
	}

	time.Sleep(2 * time.Second)

	// 示例2：带路由的生产者-消费者
	fmt.Println("\n========== 示例2: 路由键模式 ==========")

	routingProducer, err := NewRoutingProducer(client, "tasks_exchange")
	if err != nil {
		log.Fatal(err)
	}

	// 为不同类型的任务创建队列
	taskTypes := []string{"task.high", "task.normal", "task.low"}
	for _, taskType := range taskTypes {
		queue := fmt.Sprintf("%s_queue", taskType)
		_, err = client.channel.QueueDeclare(queue, true, false, false, false, nil)
		if err != nil {
			log.Fatal(err)
		}

		// 绑定队列到交换机
		err = client.channel.QueueBind(queue, taskType, "tasks_exchange", false, nil)
		if err != nil {
			log.Fatal(err)
		}
	}

	// 发送不同优先级的任务
	tasks := []TaskMessage{
		{TaskID: "001", Type: "urgent", Payload: map[string]interface{}{"action": "process_payment"}, Retry: 0, MaxRetry: 3},
		{TaskID: "002", Type: "normal", Payload: map[string]interface{}{"action": "send_report"}, Retry: 0, MaxRetry: 3},
		{TaskID: "003", Type: "low", Payload: map[string]interface{}{"action": "cleanup_logs"}, Retry: 0, MaxRetry: 3},
	}

	routingKeys := []string{"task.high", "task.normal", "task.low"}
	for i, task := range tasks {
		routingProducer.PublishWithRouting(routingKeys[i], task)
	}

	time.Sleep(1 * time.Second)

	// 示例3：主题交换机
	fmt.Println("\n========== 示例3: 主题交换机 ==========")

	// 订单相关队列
	orderQueues := []struct {
		name    string
		pattern string
	}{
		{"order.created.queue", "order.created.*"},
		{"order.paid.queue", "order.paid.*"},
		{"order.shipped.queue", "order.shipped.*"},
	}

	for _, q := range orderQueues {
		topicConsumer, err := NewTopicConsumer(client, "orders_exchange", q.name, q.pattern)
		if err != nil {
			log.Fatal(err)
		}

		// 启动消费者
		go func(pattern string) {
			err := topicConsumer.ConsumeByPattern(func(order OrderMessage) error {
				log.Printf("📦 Processing order: %s (pattern: %s)", order.OrderID, pattern)
				time.Sleep(300 * time.Millisecond)
				return nil
			})
			if err != nil {
				log.Printf("❌ Consumer error: %v", err)
			}
		}(q.pattern)
	}

	for _, o := range orders {

		client.channel.Publish(
			"orders_exchange",

			},

	}

	retryConsumer, err := NewRetryConsumer(client, "task_retry_queue", 3)

	go func() {

			}

			log.Printf("✅ Task succeeded: %s", task.TaskID)
			return nil
		})
		if err != nil {
			log.Printf("❌ Consumer error: %v", err)
		}
	}()

	failedTask := TaskMessage{

          5. 注意事项

          注意事项 说明 解决方案
          消息丢失 网络故障或服务器宕机导致消息丢失 启用消息持久化和队列持久化
          重复消费 消费者处理失败重试导致重复消费 使用消息ID做去重（幂等性设计）
          消息积压 生产速度 > 消费速度，消息堆积 增加消费者数量，优化处理逻辑
          连接泄漏 goroutine未正确关闭连接 使用defer关闭，实现优雅关闭
          死信队列 处理失败多次后如何处理 设置死信队列，人工介入处理
          顺序性 消息消费顺序可能与发送顺序不同 使用单消费者队列或分区队列

          6. 避坑指南

          ❌ 错误做法

          // ❌ 自动确认，处理失败导致消息丢失
func BadConsume() {
    msgs, _ := channel.Consume(queue, "", true, ...) // auto-ack = true
    for msg := range msgs {
        // 如果这里panic，消息已确认，丢失
        process(msg)
    }
}

// ❌ 无限重试，导致消息积压
func BadRetry(msg Message) {
    for {
        err := process(msg)
        if err != nil {
            time.Sleep(time.Second) // 无限重试
            continue
        }
        break
    }
}

// ❌ 忽略错误，导致资源泄漏
func BadConnection() {
    conn, _ := amqp.Dial(url)
    channel, _ := conn.Channel()
    // 忘记关闭连接和channel
}

func BadBlockingPublish() {
    for i := 0; i < 100000; i++ {
        // 同步发布，阻塞主线程
        channel.Publish(exchange, key, false, false, msg)
    }
}

func BadNonDurable() {
    channel.QueueDeclare(queue, false, ...) // durable = false

        DeliveryMode: amqp.Transient, // 不持久化
    })

          ✅ 正确做法

          // ✅ 手动确认，确保消息可靠
func GoodConsume() {
    msgs, _ := channel.Consume(queue, "", false, ...) // auto-ack = false
    for msg := range msgs {
        if err := process(msg); err != nil {
            msg.Nack(false, true) // 重新入队
            continue
        }
        msg.Ack(false) // 确认成功
    }
}

// ✅ 有限重试 + 死信队列
func GoodRetry(msg Message, maxRetries int) {
    retryCount := 0
    for retryCount < maxRetries {
        if err := process(msg); err != nil {
            retryCount++
            time.Sleep(time.Duration(retryCount) * time.Second)
            continue
        }
        msg.Ack(false)
        return
    }
    // 超过重试次数，移入死信队列
    moveToDLQ(msg)
}

// ✅ 优雅关闭
func GoodConnection() {
    conn, _ := amqp.Dial(url)
    channel, _ := conn.Channel()
    defer channel.Close()
    defer conn.Close()

    // 监听关闭信号
    sigChan := make(chan os.Signal, 1)
    signal.Notify(sigChan, syscall.SIGINT, syscall.SIGTERM)
    <-sigChan
    log.Println("Shutting down gracefully...")
}
