📖 概念讲解

结构型设计模式关注"如何组合对象"，通过接口组合、结构体嵌入、代理包装等方式，在不改变原有代码的基础上扩展功能。Go语言天然适合结构型模式，因为Go的嵌入(embedding)机制就是天然的装饰器和代理。

本章覆盖4种结构型模式：适配器(Adapter)、装饰器(Decorator)、代理(Proxy)、外观(Facade)。

结构型设计模式脑图
├── Adapter 适配器
│   ├── 接口转换
│   ├── 第三方库适配
│   └── 不兼容接口桥接
├── Decorator 装饰器
│   ├── 中间件模式
│   ├── 功能叠加
│   └── Go嵌入天然支持
├── Proxy 代理
│   ├── 远程代理
│   ├── 虚代理(延迟加载)
│   └── 保护代理(权限控制)
└── Facade 外观
    ├── 简化复杂子系统
    ├── 统一入口
    └── 微服务网关

v1 基础版本：适配器 + 简单装饰器

// v1_basic.go - Adapter and simple decorator
package main

import "fmt"

// ===== Adapter Pattern =====

// LegacyPrinter is an old interface we need to adapt
type LegacyPrinter interface {
    PrintOld(text string) string
}

type OldPrinter struct{}

func (o *OldPrinter) PrintOld(text string) string {
    return "[OLD] " + text
}

// ModernPrinter is the new interface our system uses
type ModernPrinter interface {
    Print(text string) string
}

// PrinterAdapter adapts LegacyPrinter to ModernPrinter
type PrinterAdapter struct {
    legacy LegacyPrinter
}

func NewPrinterAdapter(legacy LegacyPrinter) ModernPrinter {
    return &PrinterAdapter{legacy: legacy}
}

func (a *PrinterAdapter) Print(text string) string {
    return a.legacy.PrintOld(text)
}

// ===== Simple Decorator =====

// Notifier interface
type Notifier interface {
    Send(message string)
}

type EmailNotifier struct{}

func (e *EmailNotifier) Send(message string) {
    fmt.Println("Email:", message)
}

// SMSDecorator adds SMS capability
type SMSDecorator struct {
    wrapped Notifier
}

func (s *SMSDecorator) Send(message string) {
    s.wrapped.Send(message) // Call original
    fmt.Println("SMS:", message) // Add new behavior
}

func main() {
    // Adapter
    old := &OldPrinter{}
    adapted := NewPrinterAdapter(old)
    fmt.Println(adapted.Print("Hello World"))

    // Decorator
    notifier := &SMSDecorator{wrapped: &EmailNotifier{}}
    notifier.Send("Server is down!")
}

v2 进阶版本：完整4种结构型模式

// v2_patterns.go - Full structural patterns with Go idioms
package main

import (
    "fmt"
    "sync"
    "time"
)

// ===== 1. Adapter - Multi-logger unification =====

// Logger is our unified interface
type Logger interface {
    Log(level, msg string)
}

// JSONLogger outputs JSON format
type JSONLogger struct{}

func (j *JSONLogger) Log(level, msg string) {
    fmt.Printf(`{"level":"%s","msg":"%s"}` + "\n", level, msg)
}

// TextLogger outputs plain text (legacy)
type TextLogger struct{}

func (t *TextLogger) Write(level, msg string) {
    fmt.Printf("[%s] %s\n", level, msg)
}

// TextLoggerAdapter adapts TextLogger to Logger interface
type TextLoggerAdapter struct {
    textLogger *TextLogger
}

func NewTextLoggerAdapter(t *TextLogger) Logger {
    return &TextLoggerAdapter{textLogger: t}
}

func (a *TextLoggerAdapter) Log(level, msg string) {
    a.textLogger.Write(level, msg) // Delegate to legacy method
}

// ===== 2. Decorator - Middleware chain =====

// Handler represents an HTTP handler
type Handler interface {
    Handle(data string) string
}

// CoreHandler does the actual work
type CoreHandler struct{}

func (c *CoreHandler) Handle(data string) string {
    return "processed: " + data
}

// LoggingDecorator adds logging
type LoggingDecorator struct {
    handler Handler
    prefix  string
}

func (d *LoggingDecorator) Handle(data string) string {
    fmt.Printf("%s> Input: %s\n", d.prefix, data)
    result := d.handler.Handle(data)
    fmt.Printf("%s> Output: %s\n", d.prefix, result)
    return result
}

// CacheDecorator adds caching
type CacheDecorator struct {
    handler Handler
    cache   map[string]string
    mu      sync.RWMutex
}

func (d *CacheDecorator) Handle(data string) string {
    d.mu.RLock()
    if cached, ok := d.cache[data]; ok {
        d.mu.RUnlock()
        fmt.Println("[Cache] Hit!")
        return cached
    }
    d.mu.RUnlock()

    result := d.handler.Handle(data)
    d.mu.Lock()
    d.cache[data] = result
    d.mu.Unlock()
    fmt.Println("[Cache] Miss, cached")
    return result
}

// MetricsDecorator adds timing metrics
type MetricsDecorator struct {
    handler Handler
}

func (d *MetricsDecorator) Handle(data string) string {
    start := time.Now()
    result := d.handler.Handle(data)
    fmt.Printf("[Metrics] Took: %v\n", time.Since(start))
    return result
}

// ===== 3. Proxy - Protection & Virtual proxy =====

// Database represents an expensive resource
type Database interface {
    Query(sql string) string
}

type RealDatabase struct {
    connected bool
}

func (db *RealDatabase) connect() {
    if !db.connected {
        fmt.Println("[DB] Connecting to database...")
        time.Sleep(100 * time.Millisecond) // Simulate expensive connection
        db.connected = true
    }
}

func (db *RealDatabase) Query(sql string) string {
    db.connect()
    return fmt.Sprintf("Result of: %s", sql)
}

// ConnectionPoolProxy manages connections
type ConnectionPoolProxy struct {
    real    *RealDatabase
    maxConn int
    active  int
    mu      sync.Mutex
}

func NewConnectionPoolProxy(maxConn int) Database {
    return &ConnectionPoolProxy{
        real:    &RealDatabase{},
        maxConn: maxConn,
    }
}

func (p *ConnectionPoolProxy) Query(sql string) string {
    p.mu.Lock()
    if p.active >= p.maxConn {
        p.mu.Unlock()
        return "[Error] Connection pool exhausted"
    }
    p.active++
    p.mu.Unlock()

    result := p.real.Query(sql)

    p.mu.Lock()
    p.active--
    p.mu.Unlock()
    return result
}

// ProtectionProxy controls access
type ProtectionProxy struct {
    real   Database
    admins map[string]bool
}

func NewProtectionProxy(real Database, admins []string) Database {
    a := make(map[string]bool)
    for _, name := range admins {
        a[name] = true
    }
    return &ProtectionProxy{real: real, admins: a}
}

func (p *ProtectionProxy) Query(sql string) string {
    // In real code, check user from context
    if len(p.admins) == 0 {
        return "[Error] Unauthorized"
    }
    return p.real.Query(sql)
}

// ===== 4. Facade - Simplified subsystem =====

type InventorySystem struct{}

func (i *InventorySystem) CheckStock(item string) bool {
    fmt.Printf("[Inventory] Checking stock for %s\n", item)
    return true
}
func (i *InventorySystem) Reserve(item string) { fmt.Printf("[Inventory] Reserved %s\n", item) }

type PaymentSystem struct{}

func (p *PaymentSystem) Charge(amount float64) bool {
    fmt.Printf("[Payment] Charged $%.2f\n", amount)
    return true
}

type ShippingSystem struct{}

func (s *ShippingSystem) Schedule(item, address string) {
    fmt.Printf("[Shipping] Scheduled %s to %s\n", item, address)
}

// OrderFacade simplifies the ordering process
type OrderFacade struct {
    inventory *InventorySystem
    payment   *PaymentSystem
    shipping  *ShippingSystem
}

func NewOrderFacade() *OrderFacade {
    return &OrderFacade{
        inventory: &InventorySystem{},
        payment:   &PaymentSystem{},
        shipping:  &ShippingSystem{},
    }
}

func (f *OrderFacade) PlaceOrder(item string, amount float64, address string) error {
    if !f.inventory.CheckStock(item) {
        return fmt.Errorf("out of stock: %s", item)
    }
    f.inventory.Reserve(item)
    if !f.payment.Charge(amount) {
        return fmt.Errorf("payment failed")
    }
    f.shipping.Schedule(item, address)
    fmt.Println("[OrderFacade] Order placed successfully!")
    return nil
}

// ===== Demo =====
func main() {
    fmt.Println("=== 1. Adapter ===")
    var logger Logger = NewTextLoggerAdapter(&TextLogger{})
    logger.Log("INFO", "System started")

    fmt.Println("\n=== 2. Decorator Chain ===")
    var handler Handler = &CoreHandler{}
    handler = &LoggingDecorator{handler: handler, prefix: "APP"}
    handler = &CacheDecorator{handler: handler, cache: make(map[string]string)}
    handler = &MetricsDecorator{handler: handler}
    result := handler.Handle("user_data")
    handler.Handle("user_data") // Cache hit

    fmt.Println("\n=== 3. Proxy ===")
    pool := NewConnectionPoolProxy(2)
    fmt.Println(pool.Query("SELECT * FROM users"))
    fmt.Println(pool.Query("SELECT * FROM orders"))

    fmt.Println("\n=== 4. Facade ===")
    facade := NewOrderFacade()
    facade.PlaceOrder("Go Book", 49.99, "123 Main St")
}

v3 生产级版本：HTTP中间件链 + 反向代理

// v3_production.go - Real-world middleware chain and reverse proxy
package main

import (
    "fmt"
    "log"
    "net/http"
    "net/http/httputil"
    "net/url"
    "sync"
    "time"
)

// ===== Middleware Decorator (production pattern) =====

// Middleware is a decorator for http.Handler
type Middleware func(http.Handler) http.Handler

// Chain creates a middleware chain
func Chain(middlewares ...Middleware) Middleware {
    return func(final http.Handler) http.Handler {
        for i := len(middlewares) - 1; i >= 0; i-- {
            final = middlewares[i](final)
        }
        return final
    }
}

// LoggingMiddleware logs requests
func LoggingMiddleware(next http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        start := time.Now()
        next.ServeHTTP(w, r)
        log.Printf("%s %s %v", r.Method, r.URL.Path, time.Since(start))
    })
}

// RateLimitMiddleware limits requests per IP
func RateLimitMiddleware(maxRequests int, window time.Duration) Middleware {
    var mu sync.Mutex
    counters := make(map[string]int)

    return func(next http.Handler) http.Handler {
        return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
            ip := r.RemoteAddr
            mu.Lock()
            counters[ip]++
            count := counters[ip]
            mu.Unlock()

            if count > maxRequests {
                http.Error(w, "Rate limited", http.StatusTooManyRequests)
                return
            }
            next.ServeHTTP(w, r)
        })
    }
}

// CORSMiddleware adds CORS headers
func CORSMiddleware(next http.Handler) http.Handler {
    return http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        w.Header().Set("Access-Control-Allow-Origin", "*")
        w.Header().Set("Access-Control-Allow-Methods", "GET, POST, PUT, DELETE")
        if r.Method == "OPTIONS" {
            w.WriteHeader(http.StatusOK)
            return
        }
        next.ServeHTTP(w, r)
    })
}

// ===== Reverse Proxy (Proxy pattern) =====

func NewReverseProxy(target string) (http.Handler, error) {
    url, err := url.Parse(target)
    if err != nil {
        return nil, err
    }
    proxy := httputil.NewSingleHostReverseProxy(url)
    return proxy, nil
}

func main() {
    // Build handler with middleware chain
    finalHandler := http.HandlerFunc(func(w http.ResponseWriter, r *http.Request) {
        fmt.Fprintf(w, "Hello from %s", r.URL.Path)
    })

    chain := Chain(
        CORSMiddleware,
        LoggingMiddleware,
        RateLimitMiddleware(100, time.Minute),
    )

    http.Handle("/", chain(finalHandler))

    log.Println("Server on :8080 with middleware chain")
    log.Fatal(http.ListenAndServe(":8080", nil))
}

⚠️ 注意事项

事项 说明 严重度
装饰器顺序 中间件链的执行顺序取决于Chain的组装顺序 🔴 高
代理连接泄漏 使用完代理连接必须释放，否则池耗尽 🔴 高
适配器性能 适配器增加一层调用，热路径注意开销 ⚠️ 中
外观层膨胀 外观层不要堆积业务逻辑，只做协调 ⚠️ 中
装饰器与接口 装饰器必须和被装饰对象实现同一接口 ⚠️ 中

🚫→✅ 避坑指南

❌ 错误做法 ✅ 正确做法
装饰器链用继承实现 Go用接口组合+结构体嵌入，天然支持
适配器修改已有接口 适配器包一层，不动原有代码
代理不加超时控制 代理层必须设置超时，防止级联阻塞
外观层暴露子系统细节 外观只暴露简化后的接口，隐藏内部复杂性
中间件不传递ResponseWriter 始终调用 next.ServeHTTP(w, r) 传递下去
所有功能塞进一个装饰器 每个装饰器只加一个关注点（日志/缓存/限流）
