# 设计模式（行为型） ⭐⭐⭐⭐高级

> **难度标注：** ⭐⭐⭐⭐ 高级 | 预计阅读时间：25分钟 | 前置知识：Go接口、channel、函数类型

## 📖 概念讲解

行为型设计模式关注**"对象之间的通信和职责分配"**，解决"谁做什么、怎么协作"的问题。Go语言的函数是一等公民，channel天然支持发布订阅，很多行为型模式在Go中有更简洁的实现方式。

本章覆盖5种行为型模式：**策略(Strategy)、观察者(Observer)、责任链(Chain of Responsibility)、状态(State)、模板方法(Template Method)**。

```
行为型设计模式脑图
├── Strategy 策略
│   ├── 算法族封装
│   ├── 运行时切换
│   └── Go函数类型天然支持
├── Observer 观察者
│   ├── 一对多依赖
│   ├── channel实现
│   └── 事件总线
├── Chain of Responsibility 责任链
│   ├── 请求沿链传递
│   ├── 处理或转发
│   └── HTTP中间件就是这个模式
├── State 状态
│   ├── 状态机
│   ├── 订单/工作流
│   └── 状态隔离
└── Template Method 模板方法
    ├── 固定骨架，可变步骤
    ├── Go用嵌入实现
    └── sort.Interface就是模板
```

## 💻 完整Go代码

### v1 基础版本：策略 + 简单观察者

```go
// v1_basic.go - Strategy pattern and simple observer
package main

import "fmt"

// ===== Strategy Pattern =====

// SortStrategy defines how to sort
type SortStrategy interface {
    Sort([]int) []int
}

type BubbleSort struct{}

func (s *BubbleSort) Sort(data []int) []int {
    result := make([]int, len(data))
    copy(result, data)
    for i := 0; i < len(result)-1; i++ {
        for j := 0; j < len(result)-i-1; j++ {
            if result[j] > result[j+1] {
                result[j], result[j+1] = result[j+1], result[j]
            }
        }
    }
    return result
}

type QuickSort struct{}

func (s *QuickSort) Sort(data []int) []int {
    if len(data) <= 1 {
        return data
    }
    pivot := data[0]
    var left, right []int
    for _, v := range data[1:] {
        if v <= pivot {
            left = append(left, v)
        } else {
            right = append(right, v)
        }
    }
    result := append(s.Sort(left), pivot)
    result = append(result, s.Sort(right)...)
    return result
}

// Sorter uses a strategy
type Sorter struct {
    strategy SortStrategy
}

func (s *Sorter) SetStrategy(strategy SortStrategy) {
    s.strategy = strategy
}

func (s *Sorter) Sort(data []int) []int {
    return s.strategy.Sort(data)
}

// ===== Simple Observer =====

type EventListener interface {
    OnEvent(event string)
}

type EventEmitter struct {
    listeners []EventListener
}

func (e *EventEmitter) Subscribe(listener EventListener) {
    e.listeners = append(e.listeners, listener)
}

func (e *EventEmitter) Emit(event string) {
    for _, l := range e.listeners {
        l.OnEvent(event)
    }
}

type Logger struct{ Name string }
func (l *Logger) OnEvent(event string) {
    fmt.Printf("[%s] Received: %s\n", l.Name, event)
}

func main() {
    // Strategy
    sorter := &Sorter{}
    data := []int{5, 3, 8, 1, 9, 2}

    sorter.SetStrategy(&BubbleSort{})
    fmt.Println("Bubble:", sorter.Sort(data))

    sorter.SetStrategy(&QuickSort{})
    fmt.Println("Quick:", sorter.Sort(data))

    // Observer
    emitter := &EventEmitter{}
    emitter.Subscribe(&Logger{Name: "FileLogger"})
    emitter.Subscribe(&Logger{Name: "Console"})
    emitter.Emit("user_login")
}
```

### v2 进阶版本：完整5种行为型模式

```go
// v2_patterns.go - All 5 behavioral patterns
package main

import (
    "fmt"
    "strings"
    "sync"
)

// ===== 1. Strategy - Payment processing =====

type PaymentStrategy interface {
    Pay(amount float64) string
}

type CreditCard struct{ Owner string }
func (c *CreditCard) Pay(amount float64) string {
    return fmt.Sprintf("Paid $%.2f via CreditCard (%s)", amount, c.Owner)
}

type Alipay struct{ Account string }
func (a *Alipay) Pay(amount float64) string {
    return fmt.Sprintf("Paid $%.2f via Alipay (%s)", amount, a.Account)
}

type WechatPay struct{ OpenID string }
func (w *WechatPay) Pay(amount float64) string {
    return fmt.Sprintf("Paid $%.2f via WechatPay (%s)", amount, w.OpenID)
}

// Go idiom: function type as strategy
type PayFunc func(amount float64) string

func (f PayFunc) Pay(amount float64) string { return f(amount) }

// ===== 2. Observer - Event Bus with channels =====

type EventBus struct {
    subscribers map[string][]chan string
    mu          sync.RWMutex
}

func NewEventBus() *EventBus {
    return &EventBus{
        subscribers: make(map[string][]chan string),
    }
}

func (bus *EventBus) Subscribe(topic string) chan string {
    bus.mu.Lock()
    defer bus.mu.Unlock()
    ch := make(chan string, 10) // Buffered channel
    bus.subscribers[topic] = append(bus.subscribers[topic], ch)
    return ch
}

func (bus *EventBus) Publish(topic, message string) {
    bus.mu.RLock()
    defer bus.mu.RUnlock()
    for _, ch := range bus.subscribers[topic] {
        select {
        case ch <- message:
        default:
            fmt.Printf("[WARN] Channel full, dropped: %s\n", message)
        }
    }
}

// ===== 3. Chain of Responsibility - Request validation =====

type Request struct {
    Data   map[string]string
    Errors []string
}

func (r *Request) AddError(err string) {
    r.Errors = append(r.Errors, err)
}

type Handler interface {
    SetNext(handler Handler) Handler
    Handle(request *Request) bool
}

type BaseHandler struct {
    next Handler
}

func (h *BaseHandler) SetNext(handler Handler) Handler {
    h.next = handler
    return handler
}

func (h *BaseHandler) HandleNext(request *Request) bool {
    if h.next != nil {
        return h.next.Handle(request)
    }
    return true
}

// AuthHandler checks authentication
type AuthHandler struct{ BaseHandler }

func (h *AuthHandler) Handle(request *Request) bool {
    if token, ok := request.Data["token"]; !ok || token == "" {
        request.AddError("Missing auth token")
        return false
    }
    fmt.Println("[Auth] Token validated")
    return h.HandleNext(request)
}

// ValidationHandler validates input fields
type ValidationHandler struct{ BaseHandler }

func (h *ValidationHandler) Handle(request *Request) bool {
    if email, ok := request.Data["email"]; !ok || !strings.Contains(email, "@") {
        request.AddError("Invalid email")
        return false
    }
    fmt.Println("[Validation] Fields valid")
    return h.HandleNext(request)
}

// RateLimitHandler checks rate limit
type RateLimitHandler struct {
    BaseHandler
    maxRequests int
    counter     int
}

func (h *RateLimitHandler) Handle(request *Request) bool {
    h.counter++
    if h.counter > h.maxRequests {
        request.AddError("Rate limit exceeded")
        return false
    }
    fmt.Println("[RateLimit] OK")
    return h.HandleNext(request)
}

// ===== 4. State - Order state machine =====

type OrderState interface {
    Next() OrderState
    Cancel() OrderState
    String() string
}

type Order struct {
    state OrderState
}

func NewOrder() *Order {
    return &Order{state: &CreatedState{}}
}

func (o *Order) Next() {
    o.state = o.state.Next()
    fmt.Printf("State → %s\n", o.state.String())
}

func (o *Order) Cancel() {
    o.state = o.state.Cancel()
    fmt.Printf("State → %s\n", o.state.String())
}

type CreatedState struct{}

func (s *CreatedState) Next() OrderState { return &PaidState{} }
func (s *CreatedState) Cancel() OrderState { return &CancelledState{} }
func (s *CreatedState) String() string { return "Created" }

type PaidState struct{}

func (s *PaidState) Next() OrderState { return &ShippedState{} }
func (s *PaidState) Cancel() OrderState { return &CancelledState{} }
func (s *PaidState) String() string { return "Paid" }

type ShippedState struct{}

func (s *ShippedState) Next() OrderState { return &DeliveredState{} }
func (s *ShippedState) Cancel() OrderState { return &CancelledState{} }
func (s *ShippedState) String() string { return "Shipped" }

type DeliveredState struct{}

func (s *DeliveredState) Next() OrderState { return s } // Terminal
func (s *DeliveredState) Cancel() OrderState { return s }
func (s *DeliveredState) String() string { return "Delivered" }

type CancelledState struct{}

func (s *CancelledState) Next() OrderState { return s }
func (s *CancelledState) Cancel() OrderState { return s }
func (s *CancelledState) String() string { return "Cancelled" }

// ===== 5. Template Method - Data processing pipeline =====

// DataProcessor defines the skeleton algorithm
type DataProcessor interface {
    Read() []string
    Process(data []string) []string
    Write(data []string)
}

// RunPipeline is the template method (free function in Go)
func RunPipeline(p DataProcessor) {
    data := p.Read()
    processed := p.Process(data)
    p.Write(processed)
}

// CSVProcessor processes CSV files
type CSVProcessor struct {
    Source string
}

func (c *CSVProcessor) Read() []string {
    fmt.Printf("[CSV] Reading from %s\n", c.Source)
    return []string{"name,age", "Alice,30", "Bob,25"}
}

func (c *CSVProcessor) Process(data []string) []string {
    fmt.Println("[CSV] Parsing and transforming")
    var result []string
    for _, line := range data[1:] { // Skip header
        result = append(result, "processed: "+line)
    }
    return result
}

func (c *CSVProcessor) Write(data []string) {
    fmt.Printf("[CSV] Writing %d records\n", len(data))
}

// JSONProcessor processes JSON files
type JSONProcessor struct {
    Source string
}

func (j *JSONProcessor) Read() []string {
    fmt.Printf("[JSON] Reading from %s\n", j.Source)
    return []string{`{"name":"Alice","age":30}`, `{"name":"Bob","age":25}`}
}

func (j *JSONProcessor) Process(data []string) []string {
    fmt.Println("[JSON] Parsing and transforming")
    var result []string
    for _, line := range data {
        result = append(result, "transformed: "+line)
    }
    return result
}

func (j *JSONProcessor) Write(data []string) {
    fmt.Printf("[JSON] Writing %d records\n", len(data))
}

// ===== Demo =====
func main() {
    fmt.Println("=== 1. Strategy (Function Type) ===")
    strategies := []PaymentStrategy{
        &CreditCard{Owner: "Alice"},
        &Alipay{Account: "alice@example.com"},
        PayFunc(func(amount float64) string {
            return fmt.Sprintf("Crypto payment: $%.2f", amount)
        }),
    }
    for _, s := range strategies {
        fmt.Println(s.Pay(99.99))
    }

    fmt.Println("\n=== 2. Observer (Event Bus) ===")
    bus := NewEventBus()
    ch1 := bus.Subscribe("order.created")
    ch2 := bus.Subscribe("order.created")
    go func() {
        for msg := range ch1 {
            fmt.Println("[EmailService]", msg)
        }
    }()
    go func() {
        for msg := range ch2 {
            fmt.Println("[InventoryService]", msg)
        }
    }()
    bus.Publish("order.created", "Order #123")
    bus.Publish("order.created", "Order #124")

    fmt.Println("\n=== 3. Chain of Responsibility ===")
    auth := &AuthHandler{}
    validation := &ValidationHandler{}
    rateLimit := &RateLimitHandler{maxRequests: 5}
    auth.SetNext(validation).SetNext(rateLimit)

    req := &Request{Data: map[string]string{"token": "abc123", "email": "test@example.com"}}
    if auth.Handle(req) {
        fmt.Println("Request approved!")
    }

    fmt.Println("\n=== 4. State Machine ===")
    order := NewOrder()
    order.Next()   // Created → Paid
    order.Next()   // Paid → Shipped
    order.Next()   // Shipped → Delivered

    fmt.Println("\n=== 5. Template Method ===")
    RunPipeline(&CSVProcessor{Source: "data.csv"})
    RunPipeline(&JSONProcessor{Source: "data.json"})
}
```

### v3 生产级版本：泛型事件总线 + 状态机框架

```go
// v3_production.go - Generic event bus and state machine framework
package main

import (
    "fmt"
    "reflect"
    "sync"
)

// ===== Generic Event Bus =====

type GenericEventBus struct {
    handlers map[reflect.Type][]interface{}
    mu       sync.RWMutex
}

func NewGenericEventBus() *GenericEventBus {
    return &GenericEventBus{
        handlers: make(map[reflect.Type][]interface{}),
    }
}

// Subscribe registers a handler function for event type T
func Subscribe[T any](bus *GenericEventBus, handler func(T)) {
    bus.mu.Lock()
    defer bus.mu.Unlock()
    t := reflect.TypeOf((*T)(nil)).Elem()
    bus.handlers[t] = append(bus.handlers[t], handler)
}

// Publish sends event to all subscribers of type T
func Publish[T any](bus *GenericEventBus, event T) {
    bus.mu.RLock()
    defer bus.mu.RUnlock()
    t := reflect.TypeOf((*T)(nil)).Elem()
    for _, h := range bus.handlers[t] {
        if fn, ok := h.(func(T)); ok {
            fn(event)
        }
    }
}

// ===== State Machine Framework =====

type Transition struct {
    From string
    To   string
    On   string // Event that triggers transition
}

type StateMachine struct {
    current     string
    transitions []Transition
    onEnter     map[string]func()
    mu          sync.Mutex
}

func NewStateMachine(initial string) *StateMachine {
    return &StateMachine{
        current: initial,
        onEnter: make(map[string]func()),
    }
}

func (sm *StateMachine) AddTransition(from, to, event string) *StateMachine {
    sm.transitions = append(sm.transitions, Transition{from, to, event})
    return sm
}

func (sm *StateMachine) OnEnter(state string, fn func()) *StateMachine {
    sm.onEnter[state] = fn
    return sm
}

func (sm *StateMachine) Fire(event string) error {
    sm.mu.Lock()
    defer sm.mu.Unlock()
    for _, t := range sm.transitions {
        if t.From == sm.current && t.On == event {
            sm.current = t.To
            if fn, ok := sm.onEnter[t.To]; ok {
                fn()
            }
            return nil
        }
    }
    return fmt.Errorf("no transition from %s on event %s", sm.current, event)
}

func (sm *StateMachine) State() string { return sm.current }

// ===== Demo =====
func main() {
    // Generic Event Bus
    bus := NewGenericEventBus()

    type UserCreated struct {
        ID   int
        Name string
    }
    type OrderPlaced struct {
        OrderID string
        Amount  float64
    }

    Subscribe(bus, func(e UserCreated) {
        fmt.Printf("[Handler1] New user: %s (%d)\n", e.Name, e.ID)
    })
    Subscribe(bus, func(e OrderPlaced) {
        fmt.Printf("[Handler2] Order: %s $%.2f\n", e.OrderID, e.Amount)
    })

    Publish(bus, UserCreated{ID: 1, Name: "Alice"})
    Publish(bus, OrderPlaced{OrderID: "ORD-001", Amount: 99.99})

    // State Machine
    sm := NewStateMachine("draft").
        AddTransition("draft", "review", "submit").
        AddTransition("review", "approved", "approve").
        AddTransition("review", "draft", "reject").
        AddTransition("approved", "published", "publish").
        OnEnter("approved", func() { fmt.Println("→ Notified: article approved!") }).
        OnEnter("published", func() { fmt.Println("→ Article is now live!") })

    fmt.Printf("State: %s\n", sm.State())
    sm.Fire("submit")
    fmt.Printf("State: %s\n", sm.State())
    sm.Fire("approve")
    fmt.Printf("State: %s\n", sm.State())
    sm.Fire("publish")
    fmt.Printf("State: %s\n", sm.State())
}
```

## ⚠️ 注意事项

| 事项 | 说明 | 严重度 |
|------|------|--------|
| 观察者内存泄漏 | subscriber channel必须关闭或消费，否则goroutine泄漏 | 🔴 高 |
| 责任链顺序 | 链的顺序影响结果（如先鉴权再限流） | 🔴 高 |
| 状态机终态 | 终态不能再转换，需明确处理 | ⚠️ 中 |
| 反射性能 | 泛型事件总线用reflect，热路径有性能开销 | ⚠️ 中 |
| 策略切换线程安全 | 运行时切换策略需加锁保护 | ⚠️ 中 |

## 🚫→✅ 避坑指南

| ❌ 错误做法 | ✅ 正确做法 |
|------------|------------|
| 策略用switch硬编码 | 用接口/函数类型，运行时注入 |
| 观察者同步通知阻塞发布者 | 用channel或goroutine异步通知 |
| 责任链每个handler都处理 | 明确返回true/false决定是否继续传递 |
| 用大if-else实现状态机 | 独立State接口，每个状态一个实现 |
| 模板方法用继承 | Go用嵌入或组合，Template用接口 |
| 事件总线无错误处理 | handler panic要recover，防止整个系统崩溃 |
