概念讲解

一句话定义： 运算符是执行特定运算（算术、比较、逻辑等）的符号，Go 提供了丰富的运算符集并规定了明确的优先级和结合性。

现实类比： 运算符就像数学课堂上的运算符号——+ - × ÷ 大家都会用，但 Go 还提供了位运算（像开关灯的控制面板）和指针运算（像地址导航）。

技术场景： 日常编码中处处用到运算符——计算价格、判断条件、位掩码权限、指针操作、数值溢出检测等。

完整代码

package main

import (
	"fmt"
	"math"
)

func main() {
	// ========================================
	// 1. Arithmetic Operators
	// ========================================
	a, b := 17, 5

	fmt.Println("=== Arithmetic ===")
	fmt.Printf("%d + %d = %d\n", a, b, a+b)   // 22
	fmt.Printf("%d - %d = %d\n", a, b, a-b)   // 12
	fmt.Printf("%d * %d = %d\n", a, b, a*b)   // 85
	fmt.Printf("%d / %d = %d\n", a, b, a/b)   // 3 (integer division)
	fmt.Printf("%d %% %d = %d\n", a, b, a%b)  // 2

	// Float division
	fa, fb := 17.0, 5.0
	fmt.Printf("%.1f / %.1f = %.2f\n", fa, fb, fa/fb) // 3.40

	// Increment / Decrement (statement, not expression)
	count := 10
	count++ // count = 11
	count-- // count = 10
	fmt.Println("count:", count)

	// ========================================
	// 2. Comparison / Relational Operators
	// ========================================
	fmt.Println("\n=== Comparison ===")
	x, y := 10, 20
	fmt.Printf("%d == %d: %t\n", x, y, x == y) // false
	fmt.Printf("%d != %d: %t\n", x, y, x != y) // true
	fmt.Printf("%d <  %d: %t\n", x, y, x < y)  // true
	fmt.Printf("%d >  %d: %t\n", x, y, x > y)  // false
	fmt.Printf("%d <= %d: %t\n", x, y, x <= y) // true
	fmt.Printf("%d >= %d: %t\n", x, y, x >= y) // false

	// Comparables: strings compared lexicographically
	fmt.Printf("\"abc\" < \"abd\": %t\n", "abc" < "abd") // true

	// ========================================
	// 3. Logical Operators
	// ========================================
	fmt.Println("\n=== Logical ===")
	t, f := true, false
	fmt.Printf("true && false = %t\n", t && f)  // false
	fmt.Printf("true || false = %t\n", t || f)  // true
	fmt.Printf("!true = %t\n", !t)              // false

	// Short-circuit evaluation
	// In &&: if left is false, right is not evaluated
	// In ||: if left is true, right is not evaluated
	slice := []int{1, 2, 3}
	idx := 5
	if idx < len(slice) && slice[idx] > 0 { // short-circuit prevents panic
		fmt.Println("safe access")
	} else {
		fmt.Println("index out of range, but no panic!")
	}

	// ========================================
	// 4. Bitwise Operators
	// ========================================
	fmt.Println("\n=== Bitwise ===")
	m, n := 0b1100, 0b1010 // 12 and 10
	fmt.Printf("m     = %04b (%d)\n", m, m)
	fmt.Printf("n     = %04b (%d)\n", n, n)
	fmt.Printf("m & n = %04b (%d)\n", m&n, m&n)   // AND: 1000 (8)
	fmt.Printf("m | n = %04b (%d)\n", m|n, m|n)   // OR:  1110 (14)
	fmt.Printf("m ^ n = %04b (%d)\n", m^n, m^n)   // XOR: 0110 (6)
	fmt.Printf("m &^ n= %04b (%d)\n", m&^n, m&^n) // AND NOT: 0100 (4)

	// Shift
	val := 1
	fmt.Printf("1 << 3 = %d\n", val<<3) // 8
	fmt.Printf("16 >> 2 = %d\n", 16>>2)  // 4

	// ========================================
	// 5. Assignment Operators
	// ========================================
	fmt.Println("\n=== Assignment ===")
	num := 100
	num += 10  // 110
	num -= 5   // 105
	num *= 2   // 210
	num /= 3   // 70
	num %= 3   // 1
	num <<= 4  // 16
	num >>= 1  // 8
	num |= 0b1 // 9
	num &= ^0  // 9 (no change)
	fmt.Println("num after ops:", num)

	// ========================================
	// 6. Pointer Operators
	// ========================================
	fmt.Println("\n=== Pointer ===")
	val2 := 42
	ptr := &val2           // & takes address
	fmt.Printf("val2=%d ptr=%v *ptr=%d\n", val2, ptr, *ptr) // * dereferences
	*ptr = 99              // modify through pointer
	fmt.Println("after *ptr=99, val2 =", val2) // 99

	// ========================================
	// 7. Operator Precedence (high to low)
	// ========================================
	fmt.Println("\n=== Precedence ===")
	// 1. * / % << >> & &^
	// 2. + - | ^
	// 3. == != < <= > >=
	// 4. &&
	// 5. ||
	result := 2 + 3*4  // 14, not 20
	fmt.Println("2 + 3*4 =", result)
	logic := true || false && false // true (&& before ||)
	fmt.Println("true || false && false =", logic)

	// ========================================
	// 8. Overflow Handling
	// ========================================
	fmt.Println("\n=== Overflow ===")
	var maxUint8 uint8 = math.MaxUint8 // 255
	fmt.Println("max uint8:", maxUint8)
	overflow := maxUint8 + 1 // wraps to 0
	fmt.Println("maxUint8 + 1 =", overflow) // 0 (wraps!)

	// Detect overflow before it happens
	safeAdd := func(a, b uint8) (uint8, bool) {
		if a > math.MaxUint8-b {
			return 0, false // would overflow
		}
		return a + b, true
	}
	if r, ok := safeAdd(200, 100); ok {
		fmt.Printf("200 + 100 = %d\n", r)
	} else {
		fmt.Println("200 + 100 would overflow!")
	}

	// Using int overflow detection
	safeIntAdd := func(a, b int) (int, bool) {
		if b > 0 && a > math.MaxInt-b {
			return 0, false
		}
		if b < 0 && a < math.MinInt-b {
			return 0, false
		}
		return a + b, true
	}
	if r, ok := safeIntAdd(math.MaxInt, 1); ok {
		fmt.Printf("MaxInt + 1 = %d\n", r)
	} else {
		fmt.Println("MaxInt + 1 would overflow!")
	}
}

注意事项

事项 说明
++/-- 是语句不是表达式 不能写 x := y++，只能单独一行
整数除法截断 17/5 = 3，不是 3.4；需要浮点结果必须用浮点数
位运算只支持整数 float64 不能用 & ` ^` 等位运算符
^ 既取反又异或 一元 ^x 是取反，二元 x^y 是异或
溢出不报错 Go 整数溢出是静默的（wrap around），不会 panic
&^ 是 Go 特有 "位清除"运算符，x &^ y 清除 x 中 y 为 1 的位
字符串不可比较大小 只能用 == !=，但可以用 < > 按字典序比较

避坑指南

❌ 整数除法期望浮点结果

result := 1 / 3  // 0, not 0.333...

✅ 用浮点数

result := 1.0 / 3.0  // 0.333...

❌ ++ 当表达式用

x := 5
y := x++  // compile error

✅ 单独作为语句

x := 5
x++       // OK
y := x    // then assign

❌ 忽略溢出

var a uint8 = 250
a += 10  // wraps to 4, no error!

✅ 检查溢出

var a uint8 = 250
if a + 10 < a {
    // overflow detected
}
// or use explicit check as shown in the code

❌ ^ 混淆

x := ^2   // -3 (bitwise NOT of 2)
y := 6 ^ 3 // 5 (XOR)

✅ 记住一元和二元的区别

// Unary ^: bitwise complement (flip all bits)
// Binary ^: XOR (bits different = 1)

├── 优先级

    ├── 整数静默wrap
    └── 需手动检测

              举一反三

              运算符 Go C/C++ Python 差异
              位清除 x &^ y x & ~y x & ~y Go 有专用符号
              自增 x++（语句） x++（表达式） 无 Go 不能嵌入表达式
              乘方 无 无 x ** y Go 用 math.Pow()
              整除 / / // Go 和 C 自动截断
              取反 ^x ~x ~x Go 用 ^ 代替 ~
              指针 & * & * 无 Go 无指针算术
