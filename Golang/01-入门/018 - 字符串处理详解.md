# 018-字符串处理详解

> **难度：⭐⭐** | 适合掌握基础语法，想系统学习字符串操作的开发者

## 一、概念讲解

Go 的字符串是**不可变的字节序列**，底层是一个只读的 `byte` 切片。关键概念：

- **string = 只读 []byte**：字符串的内容在编译时就确定了，运行时不可修改
- **UTF-8 编码**：Go 源码默认 UTF-8，字符串字面量也是 UTF-8
- **rune = int32**：代表一个 Unicode 码点，一个中文字符 = 一个 rune
- **byte 和 rune 的区别**：`byte` 是 `uint8`（ASCII），`rune` 是 `int32`（Unicode）

```
字符串 "Go语言"
├── bytes: [G] [o] [语] [语] [言] [言]  →  6 bytes（UTF-8）
│         47  6F  E8 AF AD  E8 A8 80
├── runes: ['G'] ['o'] ['语'] ['言']     →  4 runes
└── len() = 6 (字节数，不是字符数！)
```

## 三、完整 Go 代码

```go
package main

import (
	"fmt"
	"strconv"
	"strings"
)

// DemoStringBasics shows string internal representation
func DemoStringBasics() {
	s := "Go语言"

	// len returns byte count, not character count
	fmt.Printf("string: %q\n", s)
	fmt.Printf("bytes:  %d (len)\n", len(s))
	fmt.Printf("runes:  %d (utf8.RuneCountInString)\n", len([]rune(s)))

	// Access by index gives bytes, not characters
	fmt.Printf("s[0]=%c (byte)\n", s[0]) // 'G'

	// Range iterates over runes
	fmt.Print("Runes: ")
	for i, r := range s {
		fmt.Printf("[%d:%c] ", i, r)
	}
	fmt.Println()
}

// DemoRuneVsByte shows the difference between byte and rune
func DemoRuneVsByte() {
	s := "Hello,世界"

	// Byte slice — each position is one byte
	fmt.Printf("Bytes: %v\n", []byte(s))

	// Rune slice — each position is one character
	fmt.Printf("Runes: %v\n", []rune(s))

	// Convert between string and rune slice
	runes := []rune(s)
	runes[0] = 'h' // Change 'H' to 'h'
	fmt.Printf("Modified: %s\n", string(runes))
}

// DemoStringsPkg demonstrates common strings package functions
func DemoStringsPkg() {
	s := "Hello, Go Programming!"

	// Search functions
	fmt.Printf("Contains: %v\n", strings.Contains(s, "Go"))
	fmt.Printf("HasPrefix: %v\n", strings.HasPrefix(s, "Hello"))
	fmt.Printf("HasSuffix: %v\n", strings.HasSuffix(s, "!"))
	fmt.Printf("Index: %d\n", strings.Index(s, "Go"))
	fmt.Printf("Count: %d\n", strings.Count(s, "o"))

	// Transform functions
	fmt.Printf("ToUpper: %s\n", strings.ToUpper(s))
	fmt.Printf("ToLower: %s\n", strings.ToLower(s))
	fmt.Printf("TrimSpace: [%s]\n", strings.TrimSpace("  hi  "))
	fmt.Printf("Trim: [%s]\n", strings.Trim("xxxhiyyy", "xy"))
	fmt.Printf("Replace: %s\n", strings.Replace(s, "Go", "Golang", 1))
	fmt.Printf("ReplaceAll: %s\n", strings.ReplaceAll("aaa", "a", "b"))

	// Split and Join
	parts := strings.Split("a,b,c,d", ",")
	fmt.Printf("Split: %v (len=%d)\n", parts, len(parts))

	joined := strings.Join(parts, "-")
	fmt.Printf("Join: %s\n", joined)

	// Repeat
	fmt.Printf("Repeat: %s\n", strings.Repeat("Go", 3))

	// Fields (split by whitespace)
	words := strings.Fields("  hello   world  go  ")
	fmt.Printf("Fields: %v\n", words)
}

// DemoStrconvPkg shows string conversions using strconv
func DemoStrconvPkg() {
	// int to string
	s1 := strconv.Itoa(42)
	fmt.Printf("Itoa: %q\n", s1)

	// string to int
	n, err := strconv.Atoi("42")
	if err != nil {
		fmt.Println("Parse error:", err)
	} else {
		fmt.Printf("Atoi: %d (type: %T)\n", n, n)
	}

	// FormatFloat
	pi := strconv.FormatFloat(3.14159, 'f', 2, 64)
	fmt.Printf("FormatFloat: %s\n", pi)

	// ParseFloat
	f, _ := strconv.ParseFloat("3.14", 64)
	fmt.Printf("ParseFloat: %v\n", f)

	// ParseBool
	b, _ := strconv.ParseBool("true")
	fmt.Printf("ParseBool: %v\n", b)

	// Quote (for debugging)
	fmt.Printf("Quote: %s\n", strconv.Quote(`hello "world"`))
}

// DemoStringBuilder shows efficient string concatenation
func DemoStringBuilder() {
	// Inefficient: + operator in loop
	result := ""
	for i := 0; i < 5; i++ {
		result += "Go"
	}
	fmt.Printf("+ operator: %s\n", result)

	// Efficient: strings.Builder
	var sb strings.Builder
	for i := 0; i < 5; i++ {
		sb.WriteString("Go")
	}
	fmt.Printf("Builder: %s\n", sb.String())

	// Builder also supports WriteRune, WriteByte, Grow
	var sb2 strings.Builder
	sb2.Grow(100) // preallocate
	sb2.WriteString("Hello")
	sb2.WriteRune(' ')
	sb2.WriteString("世界")
	fmt.Printf("Builder2: %s (len=%d, cap=%d)\n",
		sb2.String(), sb2.Len(), sb2.Cap())
}

// DemoMultilineStrings shows raw string literals
func DemoMultilineStrings() {
	// Raw string literal (backtick) — no escape processing
	multiline := `Line 1
Line 2
Line 3
	Tab indented`
	fmt.Println("Raw string:")
	fmt.Println(multiline)

	// Interpret string literal (double quote) — needs \n
	regular := "Line 1\nLine 2\nLine 3"
	fmt.Println("\nRegular string:")
	fmt.Println(regular)

	// Useful for regex, JSON templates, SQL
	sql := `SELECT * FROM users WHERE name = ?`
	_ = sql
}

// DemoStringSlicing shows safe string slicing with UTF-8 awareness
func DemoStringSlicing() {
	s := "Go语言"

	// Byte slicing — can break UTF-8 characters!
	fmt.Printf("s[0:4] = %q (might be broken)\n", s[0:4])

	// Safe: convert to rune slice first
	runes := []rune(s)
	fmt.Printf("runes[0:2] = %q\n", string(runes[0:2]))

	// Using range to safely iterate
	for i, r := range s {
		fmt.Printf("byte_idx=%d rune=%c\n", i, r)
	}
}

func main() {
	fmt.Println("=== String Basics ===")
	DemoStringBasics()

	fmt.Println("\n=== Rune vs Byte ===")
	DemoRuneVsByte()

	fmt.Println("\n=== strings Package ===")
	DemoStringsPkg()

	fmt.Println("\n=== strconv Package ===")
	DemoStrconvPkg()

	fmt.Println("\n=== strings.Builder ===")
	DemoStringBuilder()

	fmt.Println("\n=== Multiline Strings ===")
	DemoMultilineStrings()

	fmt.Println("\n=== String Slicing ===")
	DemoStringSlicing()
}
```

## 五、注意事项（表格）

| 注意点 | 说明 | 影响 |
|--------|------|------|
| `len(string)` 返回字节数 | 不是字符数 | 中文字符占 3 字节 |
| 字符串不可变 | 不能修改 `s[0]` | 需要转 `[]rune` 或 `[]byte` |
| `range` 按 rune 遍历 | 索引是字节位置 | 索引可能跳跃 |
| 字符串拼接性能 | `+` 在循环中低效 | 每次都分配新内存 |
| `strings.Builder` 非 goroutine 安全 | 不要并发调用 Write | 需要外部加锁 |

## 六、避坑指南

| ❌ 错误写法 | ✅ 正确写法 | 说明 |
|------------|-----------|------|
| `len(s)` 获取字符数 | `utf8.RuneCountInString(s)` | len 返回字节数 |
| `s[i]` 取中文字符 | `[]rune(s)[i]` | 按字节索引会乱码 |
| 循环中 `s += "x"` | `strings.Builder` | 拼接性能差 |
| `s[0] = 'h'` | 转 `[]byte` 或 `[]rune` 修改 | 字符串不可变 |
| `==` 比较 byte slice 和 string | `string(bs) == s` | 类型不同 |

## 九、举一反三（表格）

| 场景 | 方案 | 关键点 |
|------|------|--------|
| 统计字符数 | `utf8.RuneCountInString` | 不要用 `len` |
| 按字符遍历 | `for i, r := range s` | `i` 是字节偏移 |
| 大量拼接 | `strings.Builder` + `Grow` | 预分配更优 |

| JSON/SQL 模板 | 反引号多行字符串 | 无需转义 |
| 字符串转数字 | `strconv.Atoi` | 处理 error |

## 十一、代码演进

### v1 — 基础：循环拼接字符串

```go
func buildQuery(items []string) string {
    q := "SELECT * FROM t WHERE id IN ("
    for i, item := range items {
        if i > 0 {
            q += ","
        }
        q += item
    }
    q += ")"
    return q
}
// 问题：每次 + 都分配新内存，O(n²) 复杂度
```

### v2 — 使用 strings.Builder

```go
func buildQuery(items []string) string {
    var sb strings.Builder
    sb.WriteString("SELECT * FROM t WHERE id IN (")
    for i, item := range items {
        if i > 0 {
            sb.WriteByte(',')
        }
        sb.WriteString(item)
    }
    sb.WriteString(")")
    return sb.String()
}
// 改进：O(n) 复杂度，内存分配少
```

### v3 — 预分配 + 泛型

```go
func join[T ~string](sep string, items ...T) string {
    if len(items) == 0 {
        return ""
    }
    // Estimate size for preallocation
    total := len(sep) * (len(items) - 1)
    for _, item := range items {
        total += len(string(item))
    }
    var sb strings.Builder
    sb.Grow(total)
    for i, item := range items {
        if i > 0 {
            sb.WriteString(sep)
        }
        sb.WriteString(string(item))
    }
    return sb.String()
}
// 改进：泛型支持 + 预分配 + 零额外分配
```
