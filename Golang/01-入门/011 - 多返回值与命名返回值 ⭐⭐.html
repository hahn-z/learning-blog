难度标注

⭐⭐ 中等基础 | 适合：已掌握函数定义，需要深入理解 Go 返回值机制的初学者

概念讲解

Go 的多返回值是其最独特的设计之一，也是 Go 错误处理模式的基石。理解多返回值和命名返回值，是写好 Go 代码的关键一步。

核心要点：

  • 多返回值：一个函数可以返回多个值，func f() (int, error)
  • 命名返回值：给返回值取名字，自动初始化为零值
  • 裸 return：命名返回值函数中可直接 return
  • error 模式：(result, error) 是 Go 的标准错误处理范式
  • 最佳实践：何时用命名返回值，何时避免

    完整 Go 代码

    v1：多返回值基础

    package main

import (
	"fmt"
	"math"
)

// Basic multiple return values
func divide(a, b float64) (float64, error) {
	if b == 0 {
		return 0, fmt.Errorf("cannot divide by zero")
	}
	return a / b, nil
}

// Return min and max of a slice
func minMax(nums []int) (int, int) {
	if len(nums) == 0 {
		return 0, 0
	}
	min, max := nums[0], nums[0]
	for _, n := range nums[1:] {
		if n < min {
			min = n
		}
		if n > max {
			max = n
		}
	}
	return min, max
}

// Swap using multiple return values
func swapVals(a, b int) (int, int) {
	return b, a
}

func main() {
	// --- Divide with error ---
	result, err := divide(10, 3)
	if err != nil {
		fmt.Println("Error:", err)
	} else {
		fmt.Printf("10 / 3 = %.2f\n", result)
	}

	_, err2 := divide(10, 0)
	fmt.Println("divide by zero:", err2)

	// --- MinMax ---
	nums := []int{5, 2, 8, 1, 9, 3}
	min, max := minMax(nums)
	fmt.Printf("min=%d, max=%d\n", min, max)

	// --- Swap ---
	a, b := swapVals(3, 7)
	fmt.Printf("swapped: a=%d, b=%d\n", a, b)

	// --- Ignoring return value ---
	_, max2 := minMax([]int{10, 20})
	fmt.Println("max only:", max2)

	// --- math.Round example ---
	fmt.Println("rounded:", math.Round(3.7))
}

    执行预览：

    10 / 3 = 3.33
divide by zero: cannot divide by zero
min=1, max=9
swapped: a=7, b=3
max only: 20
rounded: 4

    v2：命名返回值与裸 return

    package main

import "fmt"

// Named return values: auto-initialized to zero values
func counter(limit int) (count int, message string) {
	for i := 0; i < limit; i++ {
		count++
	}
	message = fmt.Sprintf("counted to %d", count)
	return // bare return
}

// Named returns make the purpose of each return value clear
func parseRange(s string) (start int, end int, err error) {
	if s == "" {
		err = fmt.Errorf("empty string")
		return // bare return: start=0, end=0, err is set
	}
	// Simplified parsing
	start = 0
	end = len(s)
	return
}

// Good use of named returns: documentation
func findUser(id int) (name string, age int, found bool) {
	users := map[int][2]interface{}{
		1: {"Alice", 30},
		2: {"Bob", 25},
	}
	if data, ok := users[id]; ok {
		name = data[0].(string)
		age = data[1].(int)
		found = true
	}
	return // bare return
}

func main() {
	count, msg := counter(5)
	fmt.Println(count, msg)

	start, end, err := parseRange("hello")
	fmt.Printf("start=%d, end=%d, err=%v\n", start, end, err)

	_, _, err2 := parseRange("")
	fmt.Println("empty parse:", err2)

	name, age, found := findUser(1)
	fmt.Printf("user: name=%s, age=%d, found=%v\n", name, age, found)

	_, _, found2 := findUser(99)
	fmt.Println("user 99 found:", found2)
}

    执行预览：

    5 counted to 5
start=0, end=5, err=<nil>
empty parse: empty string
user: name=Alice, age=30, found=true
user 99 found: false

    v3：error 返回模式与最佳实践

    package main

import (
	"fmt"
	"strconv"
	"strings"
)

// Standard error return pattern: (result, error)
func parseInt(s string) (int, error) {
	n, err := strconv.Atoi(s)
	if err != nil {
		return 0, fmt.Errorf("parseInt(%q): %w", s, err)
	}
	return n, nil
}

// Multiple useful returns + error
func parseFullName(fullName string) (firstName, lastName string, err error) {
	parts := strings.SplitN(fullName, " ", 2)
	if len(parts) != 2 {
		err = fmt.Errorf("expected 'first last', got %q", fullName)
		return
	}
	firstName = parts[0]
	lastName = parts[1]
	return
}

// Functional pattern: collect results with error handling
func sumStrings(nums []string) (total int, errors []error) {
	for i, s := range nums {
		n, err := parseInt(s)
		if err != nil {
			errors = append(errors, fmt.Errorf("index %d: %w", i, err))
			continue
		}
		total += n
	}
	return
}

func main() {
	// --- Standard error pattern ---
	n, err := parseInt("42")
	if err != nil {
		fmt.Println("Error:", err)
	} else {
		fmt.Println("Parsed:", n)
	}

	_, err2 := parseInt("abc")
	fmt.Println("Parse error:", err2)

	// --- Parse full name ---
	first, last, err := parseFullName("John Doe")
	if err != nil {
		fmt.Println("Error:", err)
	} else {
		fmt.Printf("First: %s, Last: %s\n", first, last)
	}

	// --- Batch with partial errors ---
	total, errs := sumStrings([]string{"1", "two", "3", "four", "5"})
	fmt.Printf("Total: %d, Errors: %d\n", total, len(errs))
	for _, e := range errs {
		fmt.Println(" ", e)
	}
}

    执行预览：

    Parsed: 42
Parse error: parseInt("abc"): strconv.Atoi: parsing "abc": invalid syntax
First: John, Last: Doe
Total: 9, Errors: 2
  index 1: parseInt("two"): strconv.Atoi: parsing "two": invalid syntax
  index 4: parseInt("four"): strconv.Atoi: parsing "four": invalid syntax

    注意事项

    项目 说明
    返回值数量 调用时必须用相同数量的变量接收，或用 _ 忽略
    命名返回值初始化 自动初始化为零值（int→0, string→"", error→nil）
    裸 return 仅在有命名返回值的函数中使用
    error 位置 error 总是放在最后一个返回值位置（惯例）
    %w 包装 用 fmt.Errorf("...: %w", err) 包装错误，保留错误链

    避坑指南

    ❌ 错误做法 ✅ 正确做法 说明
    长函数用裸 return 长函数显式 return x, y, z 裸 return 在长函数中难以追踪
    忽略 error 不检查 result, err := f(); if err != nil { handle } 不检查 error 是最常见的 Go Bug
    返回 3 个以上值 返回结构体 func f() Result 超过 3 个返回值用 struct 更清晰
    命名返回值与 shadow 混淆 注意 defer 中对命名返回值的修改 defer 可以修改命名返回值
    不用 %w 包装错误 fmt.Errorf("context: %w", err) %w 支持错误链和 errors.Is/As

    练习题

    🟢 入门：字符串截取

    编写 splitAt(s string, sep string) (string, string, error)，在分隔符处分割字符串。

    func splitAt(s, sep string) (string, string, error) {
    i := strings.Index(s, sep)
    if i == -1 {
        return "", "", fmt.Errorf("separator %q not found", sep)
    }
    return s[:i], s[i+len(sep):], nil
}

    🟡 进阶：安全除法链

    编写 safeDivide(a, b, c int) (float64, error)，计算 a / b / c，任何除以零都返回 error。

    func safeDivide(a, b, c int) (float64, error) {
    if b == 0 { return 0, fmt.Errorf("division by zero: b") }
    if c == 0 { return 0, fmt.Errorf("division by zero: c") }
    return float64(a) / float64(b) / float64(c), nil
}

    🔴 挑战：批量解析器

    编写 parseNumbers(inputs []string) (results []int, errors []error)，解析所有字符串，收集成功和失败。

    func parseNumbers(inputs []string) (results []int, errors []error) {
    for _, s := range inputs {
        n, err := strconv.Atoi(s)
        if err != nil {
            errors = append(errors, err)
            continue
        }
        results = append(results, n)
    }
    return
}

    ├── error 放最后
    ├── >3个返回值用 struct
    └── 命名返回值用于短函数

    举一反三

    Go 模式 其他语言等价 说明
    (val, error) Python try/except, Java throws Go 用返回值而非异常
    _, err := f() Python except 只关注错误 忽略正常返回值
    命名返回值 Python -> Tuple[int, str] 类型注解 文档化返回值含义
    %w 错误包装 Java exception chaining 保留错误上下文
    裸 return — Go 特有，需谨慎使用
