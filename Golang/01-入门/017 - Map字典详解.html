017-Map字典详解

难度：⭐⭐⭐ | 适合掌握基础语法，想全面理解 map 机制的开发者

一、概念讲解

Map 是 Go 内置的哈希表实现，提供键值对（key-value）存储。它的核心特点：

  • 引用类型：map 变量本身是一个指针，指向底层的哈希表数据结构
  • 无序遍历：每次 range 遍历的顺序是随机的（Go 故意设计）
  • 零值为 nil：未初始化的 map 是 nil，写入会 panic
  • 并发不安全：多个 goroutine 同时读写会触发 fatal error

    • Go 的 map 底层使用哈希桶（buckets）实现，每个桶存 8 个键值对。当负载因子超过 6.5 时触发扩容（渐进式，不是一次完成）。

    三、完整 Go 代码

    package main

import (
	"fmt"
	"strconv"
	"sync"
)

// DemoCreateMap shows different ways to create a map
func DemoCreateMap() {
	// Method 1: make
	m1 := make(map[string]int)
	m1["go"] = 1
	fmt.Printf("make: %v\n", m1)

	// Method 2: literal
	m2 := map[string]int{
		"python": 2,
		"java":   3,
	}
	fmt.Printf("literal: %v\n", m2)

	// Method 3: make with capacity hint (Go 1.11+)
	m3 := make(map[string]int, 100) // hint for 100 entries
	_ = m3
	fmt.Println("capacity hint: helpful for large maps")
}

// DemoCRUD demonstrates Create, Read, Update, Delete operations
func DemoCRUD() {
	m := make(map[string]int)

	// Create / Update
	m["alice"] = 90
	m["bob"] = 85
	fmt.Printf("After set: %v\n", m)

	// Read
	score := m["alice"]
	fmt.Printf("alice's score: %d\n", score)

	// Read non-existent key (returns zero value)
	unknown := m["charlie"]
	fmt.Printf("charlie's score (not exist): %d\n", unknown)

	// Delete
	delete(m, "bob")
	fmt.Printf("After delete bob: %v\n", m)
}

// DemoCheckKeyExist shows how to check if a key exists
func DemoCheckKeyExist() {
	m := map[string]int{"go": 2009, "rust": 2010}

	// Two-value assignment to check existence
	if v, ok := m["go"]; ok {
		fmt.Printf("go exists: %d\n", v)
	}

	if _, ok := m["python"]; !ok {
		fmt.Println("python does not exist")
	}
}

// DemoIteration shows that map iteration order is random
func DemoIteration() {
	m := map[string]int{
		"a": 1, "b": 2, "c": 3, "d": 4, "e": 5,
	}
	fmt.Print("Iteration order: ")
	for k, v := range m {
		fmt.Printf("%s=%d ", k, v)
	}
	fmt.Println()
	// Run multiple times — order will differ!
}

// DemoMapAsParameter shows map is passed by reference
func DemoMapAsParameter() {
	m := map[string]int{"x": 1}
	modifyMap(m)
	fmt.Printf("After function call: %v\n", m) // x is modified!
}

func modifyMap(m map[string]int) {
	m["x"] = 999
}

// DemoNilMapPanic shows nil map behavior
func DemoNilMapPanic() {
	var m map[string]int // nil map

	// Reading from nil map is safe — returns zero value
	fmt.Printf("Read nil map: %d\n", m["key"])

	// Writing to nil map panics!
	// m["key"] = 1 // ← would panic: assignment to entry in nil map

	// Fix: always initialize before writing
	m = make(map[string]int)
	m["key"] = 1
	fmt.Printf("After init: %v\n", m)
}

// DemoConcurrentUnsafe shows that map is NOT concurrent-safe
// This would panic if run concurrently without sync
func DemoConcurrentUnsafe() {
	m := make(map[int]string)
	var wg sync.WaitGroup

	// Use mutex to protect map access
	var mu sync.Mutex

	for i := 0; i < 10; i++ {
		wg.Add(1)
		go func(n int) {
			defer wg.Done()
			mu.Lock()
			m[n] = "value" + strconv.Itoa(n)
			mu.Unlock()
		}(i)
	}
	wg.Wait()
	fmt.Printf("Concurrent safe with mutex: %d entries\n", len(m))
}

// DemoSyncMap shows sync.Map for concurrent access
func DemoSyncMap() {
	var sm sync.Map

	// Store
	sm.Store("go", 2009)
	sm.Store("rust", 2010)

	// Load
	if v, ok := sm.Load("go"); ok {
		fmt.Printf("sync.Map load: go=%v\n", v)
	}

	// LoadOrStore
	actual, loaded := sm.LoadOrStore("go", 9999)
	fmt.Printf("LoadOrStore: actual=%v, loaded=%v\n", actual, loaded)

	// Range
	fmt.Print("sync.Map range: ")
	sm.Range(func(k, v any) bool {
		fmt.Printf("%v=%v ", k, v)
		return true
	})
	fmt.Println()

	// Delete
	sm.Delete("rust")
}

func main() {
	fmt.Println("=== Create Map ===")
	DemoCreateMap()

	fmt.Println("\n=== CRUD ===")
	DemoCRUD()

	fmt.Println("\n=== Check Key Exist ===")
	DemoCheckKeyExist()

	fmt.Println("\n=== Iteration (Random Order) ===")
	DemoIteration()

	fmt.Println("\n=== Map As Parameter ===")
	DemoMapAsParameter()

	fmt.Println("\n=== Nil Map ===")
	DemoNilMapPanic()

	fmt.Println("\n=== Concurrent Safe ===")
	DemoConcurrentUnsafe()

	fmt.Println("\n=== sync.Map ===")
	DemoSyncMap()
}

    五、注意事项（表格）

    注意点 说明 后果
    nil map 读取安全 返回零值 不会 panic
    nil map 写入 panic 未初始化直接赋值 运行时 panic
    遍历顺序随机 Go 故意随机化 不能依赖遍历顺序
    并发读写 fatal 无锁保护 concurrent map read and map write
    key 必须可比较 支持 == 运算符的类型 slice/map/func 不能做 key
    不可取地址 &m["key"] 编译不过 map 扩容会迁移数据

    六、避坑指南

    ❌ 错误写法 ✅ 正确写法 说明
    var m map[string]int; m["a"]=1 m := make(map[string]int) nil map 写入 panic
    多协程直接读写 map sync.Mutex 或 sync.Map 并发 fatal error
    for k := range m { ... } 假设有序 先收集 key 排序再遍历 遍历顺序不确定
    if m["key"] != 0 判断存在 if v, ok := m["key"]; ok 零值也可能是合法值
    用 float 做 key 用 string 或 int 浮点精度问题

    七、练习题

    🟢 基础题： 编写函数 WordCount(s string) map[string]int，统计字符串中每个单词出现的次数。

    🟡 进阶题： 实现一个 Set 结构体（基于 map），支持 Add、Remove、Contains、Size 方法，要求并发安全。

    🔴 挑战题： 实现 LRU 缓存（基于 map + 双向链表），支持 Get、Put 操作，容量超出时淘汰最久未使用的条目。

    ├── 标准map + Mutex

      • 十一、代码演进

      v1 — 基础频率统计

      func WordCount(words []string) map[string]int {
    m := make(map[string]int)
    for _, w := range words {
        m[w]++
    }
    return m
}
// 简单直接，但不是并发安全的

      v2 — 加锁的并发安全版本

      type SafeCounter struct {
    mu sync.Mutex
    m  map[string]int
}

func (c *SafeCounter) Inc(key string) {
    c.mu.Lock()
    c.m[key]++
    c.mu.Unlock()
}
// 改进：并发安全，适合读写字典都频繁的场景

      v3 — 使用 sync.Map 的泛型包装

      type ConcurrentMap[K comparable, V any] struct {
    data sync.Map
}

func (m *ConcurrentMap[K, V]) Store(key K, value V) {
    m.data.Store(key, value)
}

func (m *ConcurrentMap[K, V]) Load(key K) (V, bool) {
    v, ok := m.data.Load(key)
    if !ok {
        var zero V
        return zero, false
    }
    return v.(V), true
}
// 改进：类型安全 + 泛型 + 并发安全
