一、概念讲解

上一篇我们掌握了接口基础。本篇深入三个核心主题：

  1. 类型断言与 type switch：从接口值中提取具体类型
    1. 接口嵌入组合：用小接口组合出复杂行为
      1. 实战模式：sort.Interface、error 接口、自定义错误类型、接口设计原则

        2. Go 的接口哲学："The bigger the interface, the weaker the abstraction."（接口越大，抽象越弱）—— Rob Pike。Go 鼓励小接口，1-3 个方法最佳。

        三、完整 Go 代码

        package main

import (
	"errors"
	"fmt"
	"sort"
	"strconv"
	"strings"
)

// ========== 1. Type Assertion & Type Switch ==========

type Animal interface {
	Sound() string
}

type Dog struct{ Name string }
type Cat struct{ Name string }
type Bird struct{ Name string }

func (d Dog) Sound() string { return d.Name + ": Woof!" }
func (c Cat) Sound() string { return c.Name + ": Meow!" }
func (b Bird) Sound() string { return b.Name + ": Tweet!" }

// Type switch: clean pattern for handling multiple types
func identify(a Animal) {
	switch v := a.(type) {
	case Dog:
		fmt.Printf("  Dog named %s says %s\n", v.Name, v.Sound())
	case Cat:
		fmt.Printf("  Cat named %s says %s\n", v.Name, v.Sound())
	case Bird:
		fmt.Printf("  Bird named %s says %s\n", v.Name, v.Sound())
	default:
		fmt.Printf("  Unknown animal type: %T\n", v)
	}
}

// Safe type assertion with comma-ok
func tryAsDog(a Animal) {
	if dog, ok := a.(Dog); ok {
		fmt.Printf("  Found dog: %s\n", dog.Name)
	} else {
		fmt.Printf("  Not a dog, it's %T\n", a)
	}
}

// ========== 2. sort.Interface in Practice ==========

// Custom type that sorts by multiple criteria
type Employee struct {
	Name   string
	Age    int
	Salary float64
}

type ByAge []Employee

func (a ByAge) Len() int           { return len(a) }
func (a ByAge) Swap(i, j int)      { a[i], a[j] = a[j], a[i] }
func (a ByAge) Less(i, j int) bool { return a[i].Age < a[j].Age }

type BySalary []Employee

func (a BySalary) Len() int           { return len(a) }
func (a BySalary) Swap(i, j int)      { a[i], a[j] = a[j], a[i] }
func (a BySalary) Less(i, j int) bool { return a[i].Salary > a[j].Salary } // descending

// Multi-sort using sort.Slice (simpler alternative)
func demoSliceSort() {
	emps := []Employee{
		{"Alice", 30, 80000},
		{"Bob", 25, 90000},
		{"Charlie", 35, 70000},
	}

	sort.Slice(emps, func(i, j int) bool {
		return emps[i].Name < emps[j].Name
	})
	fmt.Println("  Sorted by name:", namesOf(emps))
}

func namesOf(emps []Employee) string {
	ns := make([]string, len(emps))
	for i, e := range emps {
		ns[i] = e.Name
	}
	return strings.Join(ns, ", ")
}

// ========== 3. Custom Error Types ==========

// Basic custom error
type NotFoundError struct {
	Resource string
	ID       int
}

func (e *NotFoundError) Error() string {
	return fmt.Sprintf("%s with id %d not found", e.Resource, e.ID)
}

// Wrapped error with context
type ValidationError struct {
	Field   string
	Message string
	Cause   error
}

func (e *ValidationError) Error() string {
	return fmt.Sprintf("validation failed on %s: %s", e.Field, e.Message)
}

func (e *ValidationError) Unwrap() error {
	return e.Cause
}

// Business logic that returns custom errors
func findUser(id int) (string, error) {
	if id <= 0 {
		return "", &ValidationError{
			Field:   "id",
			Message: "must be positive",
		}
	}
	if id > 100 {
		return "", &NotFoundError{Resource: "User", ID: id}
	}
	return "user_" + strconv.Itoa(id), nil
}

// ========== 4. Stringer Interface ==========

type Money struct {
	Amount   float64
	Currency string
}

func (m Money) String() string {
	return fmt.Sprintf("%.2f%s", m.Amount, m.Currency)
}

type Percentage float64

func (p Percentage) String() string {
	return fmt.Sprintf("%.1f%%", float64(p))
}

// ========== 5. Interface Composition ==========

type Serializer interface {
	Serialize() string
}

type Validator interface {
	Validate() error
}

// Composed interface: must implement both
type ValidatedSerializer interface {
	Serializer
	Validator
}

type Config struct {
	Key   string
	Value string
}

func (c Config) Serialize() string {
	return c.Key + "=" + c.Value
}

func (c Config) Validate() error {
	if c.Key == "" {
		return errors.New("key cannot be empty")
	}
	return nil
}

func processConfig(vs ValidatedSerializer) error {
	if err := vs.Validate(); err != nil {
		return fmt.Errorf("config validation: %w", err)
	}
	fmt.Printf("  Serialized: %s\n", vs.Serialize())
	return nil
}

// ========== main ==========

func main() {
	fmt.Println("=== 1. Type Switch ===")
	animals := []Animal{
		Dog{"Buddy"},
		Cat{"Whiskers"},
		Bird{"Tweety"},
	}
	for _, a := range animals {
		identify(a)
	}

	fmt.Println("\n=== Type Assertion (comma-ok) ===")
	tryAsDog(Dog{"Rex"})
	tryAsDog(Cat{"Mittens"})

	fmt.Println("\n=== 2. sort.Interface ===")
	emps := []Employee{
		{"Alice", 30, 80000},
		{"Bob", 25, 90000},
		{"Charlie", 35, 70000},
		{"Diana", 28, 85000},
	}

	sort.Sort(ByAge(emps))
	fmt.Println("  By Age:", namesOf(emps))

	sort.Sort(BySalary(emps))
	fmt.Println("  By Salary (desc):", namesOf(emps))

	fmt.Println("\n  Using sort.Slice:")
	demoSliceSort()

	fmt.Println("\n=== 3. Custom Error Types ===")
	testCases := []int{-1, 42, 200}
	for _, id := range testCases {
		user, err := findUser(id)
		if err != nil {
			// Use errors.As to check specific error type
			var nfe *NotFoundError
			var ve *ValidationError
			if errors.As(err, &nfe) {
				fmt.Printf("  ID %d: NotFound → %v\n", id, nfe)
			} else if errors.As(err, &ve) {
				fmt.Printf("  ID %d: Validation → %v\n", id, ve)
			}
			continue
		}
		fmt.Printf("  ID %d: Found %s\n", id, user)
	}

	fmt.Println("\n=== 4. Stringer ===")
	price := Money{99.5, "USD"}
	tax := Percentage(8.25)
	fmt.Printf("  Price: %s, Tax: %s\n", price, tax)

	fmt.Println("\n=== 5. Interface Composition ===")
	cfg := Config{Key: "host", Value: "localhost"}
	processConfig(cfg)

	badCfg := Config{Key: "", Value: "x"}
	if err := processConfig(badCfg); err != nil {
		fmt.Printf("  Error: %v\n", err)
	}

	fmt.Println("\n=== Bonus: errors.Is for error chain ===")
	baseErr := errors.New("connection refused")
	wrappedErr := fmt.Errorf("db query failed: %w", baseErr)
	fmt.Printf("  errors.Is(wrapped, base)? %v\n", errors.Is(wrappedErr, baseErr))
}

        五、注意事项

        事项 说明
        类型断言安全 除非100%确定类型，否则用 v, ok := i.(T)
        type switch 只能在 switch 语句中使用 i.(type)，不能赋值
        errors.As vs 类型断言 errors.As 会沿 Unwrap 链查找，类型断言只看当前层
        sort.Interface 需同时实现 Len/Less/Swap 三个方法
        Unwrap 链 errors.Is 和 errors.As 依赖 Unwrap 遍历链
        接口大小 1-3 个方法最佳，标准库 io.Reader 只有 1 个方法
        组合接口 嵌入多个接口时，方法不能重名（除非签名一致）

        六、避坑指南

        ❌ 类型断言不检查直接用

        func process(i interface{}) {
    s := i.(string) // panics if i is int!
    fmt.Println(s)
}

        ✅ 安全模式

        func process(i interface{}) {
    if s, ok := i.(string); ok {
        fmt.Println(s)
    }
}

        ❌ 在 switch case 外使用 i.(type)

        t := i.(type) // compile error!

        ✅ 只在 type switch 内使用

        switch v := i.(type) {
case string:
    fmt.Println(v)
}

        ❌ 用 == 比较自定义错误

        if err == io.EOF { ... } // fragile, misses wrapped errors

        ✅ 用 errors.Is

        if errors.Is(err, io.EOF) { ... } // handles wrapped errors

        七、练习题

        🟢 基础题：type switch 实战

        写一个函数 func classify(v any) string，用 type switch 识别 int、string、bool、[]int、map[string]int，返回描述字符串如 "int: 42"。

        🟡 进阶题：自定义排序

        定义 type Student struct { Name string; Score int }，实现按 Score 降序排序，Score 相同时按 Name 升序。同时用 sort.Slice 写一个简化版本对比。

        🔴 挑战题：错误链与 errors.Is/As

        定义三层错误类型：DBError → ServiceError → HandlerError，每层用 Unwrap() 链接。用 errors.Is 判断底层错误类型，用 errors.As 提取中间层错误信息。
