# 042 - 泛型基础

> **难度：** ⭐⭐⭐⭐
> **关键词：** Go 1.18、泛型、类型参数、类型约束、comparable、类型集、泛型函数

## 一、概念讲解

Go 1.18（2022年3月）正式引入泛型（Generics），这是 Go 语言最大的语法变革之一。泛型允许函数和类型**参数化类型**，编写更通用、类型安全且无需反射的代码。

### 核心概念

- **类型参数（Type Parameter）**：`[T any]` 中的 `T` 是类型参数，在使用时由调用方指定具体类型
- **类型约束（Constraint）**：限制类型参数必须满足的条件，如 `comparable`、`any`、自定义接口
- **类型集（Type Set）**：Go 1.18+ 的接口可以包含类型列表，定义允许的类型集合

### 泛型 vs interface{}

| 维度 | 泛型 | interface{} |
|------|------|-------------|
| 类型安全 | 编译期检查 | 运行时断言 |
| 性能 | 无装箱开销 | 有装箱/拆箱 |
| 代码复用 | 类型参数化 | 类型断言 |
| 可读性 | 签名明确 | 运行时才知道 |

## 三、完整代码

```go
package main

import (
	"fmt"
	"sort"
	"strings"
)

// === 1. Generic Functions ===

// Contains checks if a slice contains a value
func Contains[T comparable](slice []T, target T) bool {
	for _, v := range slice {
		if v == target {
			return true
		}
	}
	return false
}

// Filter returns elements that satisfy the predicate
func Filter[T any](slice []T, pred func(T) bool) []T {
	var result []T
	for _, v := range slice {
		if pred(v) {
			result = append(result, v)
		}
	}
	return result
}

// Map transforms each element using the function
func Map[T any, R any](slice []T, fn func(T) R) []R {
	result := make([]R, len(slice))
	for i, v := range slice {
		result[i] = fn(v)
	}
	return result
}

// Reduce reduces a slice to a single value
func Reduce[T any, R any](slice []T, init R, fn func(R, T) R) R {
	acc := init
	for _, v := range slice {
		acc = fn(acc, v)
	}
	return acc
}

// Keys returns all keys of a map
func Keys[K comparable, V any](m map[K]V) []K {
	keys := make([]K, 0, len(m))
	for k := range m {
		keys = append(keys, k)
	}
	return keys
}

// Values returns all values of a map
func Values[K comparable, V any](m map[K]V) []V {
	vals := make([]V, 0, len(m))
	for _, v := range m {
		vals = append(vals, v)
	}
	return vals
}

// === 2. Custom Type Constraints ===

// Number constraint: allows all numeric types
type Number interface {
	~int | ~int8 | ~int16 | ~int32 | ~int64 |
		~float32 | ~float64 |
		~uint | ~uint8 | ~uint16 | ~uint32 | ~uint64
}

// Sum returns the sum of all elements in a numeric slice
func Sum[T Number](slice []T) T {
	var total T
	for _, v := range slice {
		total += v
	}
	return total
}

// Max returns the maximum value (requires ordering)
type Ordered interface {
	~int | ~int8 | ~int16 | ~int32 | ~int64 |
		~float32 | ~float64 |
		~string
}

func Max[T Ordered](slice []T) T {
	if len(slice) == 0 {
		var zero T
		return zero
	}
	max := slice[0]
	for _, v := range slice[1:] {
		if v > max {
			max = v
		}
	}
	return max
}

// Min returns the minimum value
func Min[T Ordered](slice []T) T {
	if len(slice) == 0 {
		var zero T
		return zero
	}
	min := slice[0]
	for _, v := range slice[1:] {
		if v < min {
			min = v
		}
	}
	return min
}

// === 3. Generic Data Structure: Stack ===

type Stack[T any] struct {
	items []T
}

func NewStack[T any]() *Stack[T] {
	return &Stack[T]{items: make([]T, 0)}
}

func (s *Stack[T]) Push(v T) {
	s.items = append(s.items, v)
}

func (s *Stack[T]) Pop() (T, bool) {
	if len(s.items) == 0 {
		var zero T
		return zero, false
	}
	top := s.items[len(s.items)-1]
	s.items = s.items[:len(s.items)-1]
	return top, true
}

func (s *Stack[T]) Peek() (T, bool) {
	if len(s.items) == 0 {
		var zero T
		return zero, false
	}
	return s.items[len(s.items)-1], true
}

func (s *Stack[T]) Len() int {
	return len(s.items)
}

// === 4. Generic Pair ===

type Pair[K comparable, V any] struct {
	Key   K
	Value V
}

func NewPair[K comparable, V any](k K, v V) Pair[K, V] {
	return Pair[K, V]{Key: k, Value: v}
}

// SortBy sorts a slice of pairs by key (requires Ordered key)
func SortBy[K Ordered, V any](pairs []Pair[K, V]) {
	sort.Slice(pairs, func(i, j int) bool {
		return pairs[i].Key < pairs[j].Key
	})
}

// === 5. Type alias for clarity ===

type MyInt int // ~int allows this with Number constraint

func main() {
	// === 1. Generic Functions ===
	fmt.Println("=== Generic Functions ===")
	nums := []int{1, 2, 3, 4, 5}
	fmt.Printf("Contains(3): %v\n", Contains(nums, 3))
	fmt.Printf("Contains(6): %v\n", Contains(nums, 6))

	evens := Filter(nums, func(n int) bool { return n%2 == 0 })
	fmt.Printf("Evens: %v\n", evens)

	squares := Map(nums, func(n int) int { return n * n })
	fmt.Printf("Squares: %v\n", squares)

	sum := Reduce(nums, 0, func(acc, n int) int { return acc + n })
	fmt.Printf("Sum via Reduce: %d\n", sum)

	// === 2. String operations ===
	words := []string{"hello", "world", "go"}
	upper := Map(words, strings.ToUpper)
	fmt.Printf("Upper: %v\n", upper)

	longWords := Filter(words, func(s string) bool { return len(s) > 2 })
	fmt.Printf("Long words: %v\n", longWords)

	// === 3. Map operations ===
	fmt.Println("\n=== Map Utils ===")
	m := map[string]int{"apple": 3, "banana": 5, "cherry": 2}
	fmt.Printf("Keys: %v\n", Keys(m))
	fmt.Printf("Values: %v\n", Values(m))

	// === 4. Number constraint with type alias ===
	fmt.Println("\n=== Number Constraint ===")
	myNums := []MyInt{10, 20, 30}
	fmt.Printf("Sum(MyInt): %v\n", Sum(myNums)) // ~int allows MyInt

	floatNums := []float64{1.5, 2.5, 3.0}
	fmt.Printf("Sum(float64): %v\n", Sum(floatNums))

	// === 5. Max/Min ===
	fmt.Println("\n=== Max/Min ===")
	fmt.Printf("Max(nums): %d\n", Max(nums))
	fmt.Printf("Min(nums): %d\n", Min(nums))
	fmt.Printf("Max(words): %s\n", Max(words))

	// === 6. Generic Stack ===
	fmt.Println("\n=== Stack[int] ===")
	stack := NewStack[int]()
	stack.Push(10)
	stack.Push(20)
	stack.Push(30)
	fmt.Printf("Len: %d\n", stack.Len())
	if v, ok := stack.Peek(); ok {
		fmt.Printf("Peek: %d\n", v)
	}
	if v, ok := stack.Pop(); ok {
		fmt.Printf("Pop: %d\n", v)
	}
	if v, ok := stack.Pop(); ok {
		fmt.Printf("Pop: %d\n", v)
	}

	// Stack with strings
	fmt.Println("\n=== Stack[string] ===")
	strStack := NewStack[string]()
	strStack.Push("hello")
	strStack.Push("world")
	if v, ok := strStack.Pop(); ok {
		fmt.Printf("Pop: %s\n", v)
	}

	// === 7. Pair ===
	fmt.Println("\n=== Pair ===")
	pairs := []Pair[string, int]{
		NewPair("banana", 5),
		NewPair("apple", 3),
		NewPair("cherry", 2),
	}
	SortBy(pairs)
	for _, p := range pairs {
		fmt.Printf("  %s: %d\n", p.Key, p.Value)
	}

	// === 8. Type inference ===
	fmt.Println("\n=== Type Inference ===")
	// Go infers T from arguments, no need to specify
	result := Map([]int{1, 2, 3}, func(n int) string {
		return fmt.Sprintf("0x%x", n)
	})
	fmt.Printf("Int to Hex: %v\n", result)
}
```

## 五、注意事项

| 项目 | 说明 |
|------|------|
| 类型推断 | 大多数场景无需显式指定类型参数，Go 会自动推断 |
| `~` 前缀 | 约束中 `~int` 表示底层类型为 int 的所有类型（含 type MyInt int） |
| 不支持泛型方法 | Go 不支持在泛型参数上定义额外类型参数的方法 |
| comparable | 只保证 == 和 !=，不保证 < > |
| 编译速度 | 泛型可能略微增加编译时间 |
| 运行时无泛型 | Go 采用单态化（monomorphization），编译期展开 |

## 六、避坑指南

❌ **方法不能有额外的类型参数**
```go
type Container[T any] struct{ data T }
func (c Container[T]) Map[R any](fn func(T) R) R { // compile error!
    return fn(c.data)
}
```
✅ **用包级泛型函数代替**
```go
func MapValue[T any, R any](c Container[T], fn func(T) R) R {
    return fn(c.data)
}
```

❌ **忘记 ~ 导致自定义类型不匹配**
```go
type Number interface { int | float64 } // MyInt won't match!
```
✅ **使用 ~ 允许底层类型**
```go
type Number interface { ~int | ~float64 } // MyInt matches
```

❌ **泛型约束过宽导致运行时 panic**
```go
func First[T any](slice []T) T { return slice[0] } // panics on empty slice!
```
✅ **处理边界情况**
```go
func First[T any](slice []T) (T, bool) {
    if len(slice) == 0 {
        var zero T
        return zero, false
    }
    return slice[0], true
}
```

🟢 **基础：** 实现 `Unique[T comparable](slice []T) []T`，去除切片中的重复元素。

## 九、举一反三

| 场景 | 泛型用法 | 替代方案 |
|------|----------|----------|
| 通用集合类型 | Stack[T]、Queue[T]、Set[T] | interface{} + 断言 |
| 数据转换管道 | Map/Filter/Reduce | for 循环 |
| 类型安全的配置 | Pair[string, V] | map[string]interface{} |
| 排序/比较 | Ordered 约束 | sort.Slice + 反射 |
| 错误处理 | Result[T] 类型 | value, error 模式 |
| 并发模式 | FanIn[T]、FanOut[T] | interface{} channel |
| 缓存 | Cache[K comparable, V any] | sync.Map |

## 十一、代码演进

### v1 — interface{} 实现通用函数

```go
func Contains(slice []interface{}, target interface{}) bool {
    for _, v := range slice {
        if v == target {
            return true
        }
    }
    return false
}
// No type safety, requires boxing, ugly to call
```

### v2 — 泛型函数（基础版）

```go
func Contains[T comparable](slice []T, target T) bool {
    for _, v := range slice {
        if v == target {
            return true
        }
    }
    return false
}
// Type safe, no boxing, clean API
```

### v3 — 泛型工具库（含自定义约束和泛型类型）

```go
type Number interface {
    ~int | ~int8 | ~int16 | ~int32 | ~int64 | ~float32 | ~float64
}

type Stack[T any] struct { items []T }

func Sum[T Number](s []T) T { ... }
func Map[T, R any](s []T, f func(T) R) []R { ... }
// Full generic toolkit: constraints + functions + types
```

> **小结：** Go 泛型是类型安全的代码复用利器。掌握类型参数、约束和类型集语法后，可以编写通用函数和数据结构，告别 interface{} 的类型断言和反射的性能开销。
