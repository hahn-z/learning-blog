一、概念讲解

上篇我们学习了反射的基础操作。本篇深入修改值、动态 select、struct tag 应用 和 反射实现 ORM 映射等进阶主题，最后讨论反射 vs 代码生成的选型。

反射修改值三步曲

  1. 传指针：reflect.ValueOf(&x)
    1. 解引用：.Elem()
      1. 设置值：.Setxxx() 或 .Set(reflect.Value)

        2. 前提条件：CanSet() 返回 true。

        struct tag

        Go 结构体字段的 tag 是编译期绑定的元数据字符串，通过反射在运行时读取。这是 JSON、数据库映射、验证等库的基础。

        reflect.Select

        reflect.Select 允许在运行时动态构建 select case 列表，实现不确定数量的 channel 同时监听。

        二、脑图

        反射进阶
├── 修改值
│   ├── 传指针 → Elem() → CanSet() → Set()
│   ├── SetInt / SetString / SetFloat ...
│   └── Set(reflect.Value)
├── struct tag
│   ├── Tag.Get("key")
│   ├── Tag.Lookup("key") → (value, ok)
│   └── 应用: json / db / validate / form
├── reflect.Select
│   ├── SelectDir (SelectSend/SelectRecv/SelectDefault)
│   ├── 动态构建 case 列表
│   └── 返回 (chosen, recv, ok)
├── 简易 ORM
│   ├── 结构体 → 表名/字段映射
│   ├── SELECT 构建器
│   └── Scan 行 → 结构体
└── 反射 vs 代码生成
    ├── 反射: 灵活、运行时、慢
    ├── 代码生成: 编译期、快、需生成步骤
    └── 选型: 热路径用生成, 工具用反射

        三、完整代码

        package main

import (
	"fmt"
	"reflect"
	"strings"
)

// === 1. Struct with tags for ORM mapping ===

type User struct {
	ID    int    `db:"id" json:"id"`
	Name  string `db:"name" json:"name"`
	Email string `db:"email" json:"email"`
}

// TableName returns the table name for ORM
func (User) TableName() string {
	return "users"
}

// === 2. Simple ORM using reflection ===

// Schema holds field-to-column mapping
type Schema struct {
	TableName   string
	Fields      []reflect.StructField
	FieldValues []reflect.Value
	ColumnMap   map[string]int // column name -> field index
}

// NewSchema creates a schema from a struct pointer
func NewSchema(model interface{}) *Schema {
	t := reflect.TypeOf(model)
	if t.Kind() == reflect.Ptr {
		t = t.Elem()
	}
	v := reflect.ValueOf(model)
	if v.Kind() == reflect.Ptr {
		v = v.Elem()
	}

	s := &Schema{
		TableName:   t.Name() + "s", // pluralize naively
		ColumnMap:   make(map[string]int),
	}

	// Check for TableName method
	if tn := reflect.ValueOf(model).MethodByName("TableName"); tn.IsValid() {
		result := tn.Call(nil)
		s.TableName = result[0].String()
	}

	for i := 0; i < t.NumField(); i++ {
		sf := t.Field(i)
		col := sf.Tag.Get("db")
		if col == "" {
			col = strings.ToLower(sf.Name)
		}
		s.Fields = append(s.Fields, sf)
		s.FieldValues = append(s.FieldValues, v.Field(i))
		s.ColumnMap[col] = i
	}
	return s
}

// BuildSelect generates a SELECT query
func (s *Schema) BuildSelect(conditions ...string) string {
	cols := make([]string, len(s.Fields))
	for i, f := range s.Fields {
		col := f.Tag.Get("db")
		if col == "" {
			col = strings.ToLower(f.Name)
		}
		cols[i] = col
	}
	query := fmt.Sprintf("SELECT %s FROM %s", strings.Join(cols, ", "), s.TableName)
	if len(conditions) > 0 {
		query += " WHERE " + strings.Join(conditions, " AND ")
	}
	return query
}

// Scan maps a row (map[string]interface{}) into a struct
func (s *Schema) Scan(row map[string]interface{}, dest interface{}) {
	v := reflect.ValueOf(dest).Elem()
	for col, val := range row {
		if idx, ok := s.ColumnMap[col]; ok {
			field := v.Field(idx)
			if field.CanSet() {
				field.Set(reflect.ValueOf(val))
			}
		}
	}
}

// === 3. Validator using struct tags ===

func Validate(x interface{}) []string {
	var errors []string
	v := reflect.ValueOf(x)
	if v.Kind() == reflect.Ptr {
		v = v.Elem()
	}
	t := v.Type()

	for i := 0; i < t.NumField(); i++ {
		sf := t.Field(i)
		fv := v.Field(i)

		// Check "required" tag
		if sf.Tag.Get("validate") == "required" {
			switch fv.Kind() {
			case reflect.String:
				if fv.String() == "" {
					errors = append(errors, fmt.Sprintf("%s is required", sf.Name))
				}
			case reflect.Int:
				if fv.Int() == 0 {
					errors = append(errors, fmt.Sprintf("%s is required", sf.Name))
				}
			}
		}
	}
	return errors
}

// === 4. Reflect modify value ===

func setField(ptr interface{}, fieldName string, value interface{}) error {
	v := reflect.ValueOf(ptr).Elem()
	f := v.FieldByName(fieldName)
	if !f.IsValid() {
		return fmt.Errorf("field %s not found", fieldName)
	}
	if !f.CanSet() {
		return fmt.Errorf("field %s cannot be set", fieldName)
	}
	val := reflect.ValueOf(value)
	if val.Type() != f.Type() {
		// Try to convert
		if val.Type().ConvertibleTo(f.Type()) {
			val = val.Convert(f.Type())
		} else {
			return fmt.Errorf("type mismatch: field=%s, value=%s", f.Type(), val.Type())
		}
	}
	f.Set(val)
	return nil
}

func main() {
	// === 1. Modify value via reflection ===
	fmt.Println("=== Modify Value ===")
	u := User{ID: 1, Name: "Alice", Email: "alice@example.com"}
	fmt.Printf("Before: %+v\n", u)

	// Modify fields via pointer
	setField(&u, "Name", "Bob")
	setField(&u, "Email", "bob@example.com")
	fmt.Printf("After:  %+v\n", u)

	// Direct approach
	v := reflect.ValueOf(&u).Elem()
	nameField := v.FieldByName("Name")
	fmt.Printf("CanSet Name: %v\n", nameField.CanSet())
	nameField.SetString("Charlie")
	fmt.Printf("Direct set:  %+v\n", u)

	// === 2. ORM Schema ===
	fmt.Println("\n=== ORM Schema ===")
	schema := NewSchema(&User{})
	fmt.Printf("Table: %s\n", schema.TableName)
	fmt.Printf("Select: %s\n", schema.BuildSelect("id = ?", "active = true"))

	// === 3. ORM Scan ===
	fmt.Println("\n=== ORM Scan ===")
	row := map[string]interface{}{
		"id":    42,
		"name":  "Dave",
		"email": "dave@example.com",
	}
	var u2 User
	schema.Scan(row, &u2)
	fmt.Printf("Scanned: %+v\n", u2)

	// === 4. Read all tags ===
	fmt.Println("\n=== Struct Tags ===")
	t := reflect.TypeOf(User{})
	for i := 0; i < t.NumField(); i++ {
		f := t.Field(i)
		fmt.Printf("  %s: db=%q, json=%q\n", f.Name, f.Tag.Get("db"), f.Tag.Get("json"))
	}

	// === 5. reflect.Select (dynamic select) ===
	fmt.Println("\n=== reflect.Select ===")
	demoReflectSelect()
}

func demoReflectSelect() {
	ch1 := make(chan string, 1)
	ch2 := make(chan string, 1)

	ch1 <- "from channel 1"
	// ch2 has no data

	// Build select cases dynamically
	cases := []reflect.SelectCase{
		{Dir: reflect.SelectRecv, Chan: reflect.ValueOf(ch1)},
		{Dir: reflect.SelectRecv, Chan: reflect.ValueOf(ch2)},
		{Dir: reflect.SelectDefault},
	}

	chosen, value, ok := reflect.Select(cases)
	fmt.Printf("  Chosen case %d, value=%v, ok=%v\n", chosen, value, ok)
}

        五、注意事项

        项目 说明
        CanSet 条件 必须传指针 + Elem()，且字段是导出的
        unexported 字段 私有字段即使通过指针也无法 Set
        Tag.Get vs Tag.Lookup Get 返回空串表示不存在；Lookup 返回 (val, ok) 区分空 tag
        reflect.Select 性能 比原生 select 慢，只在动态 channel 场景使用
        ORM Scan 类型匹配 map 的 value 类型必须与结构体字段类型兼容

        六、避坑指南

        ❌ 忘记 Elem() 导致 CanSet 为 false

        v := reflect.ValueOf(&x)
v.SetInt(10) // panic: SetInt on ptr, not the value

        ✅ 先 Elem() 解引用

        v := reflect.ValueOf(&x).Elem()
v.SetInt(10) // OK

        ❌ Tag 写法错误

        Name string `json:"name"`db:"user_name`` // syntax error

        ✅ 用空格分隔多个 tag

        Name string `json:"name" db:"user_name"`

        ❌ 反射修改 slice/map 的容量

        s := []int{1, 2, 3}
reflect.ValueOf(&s).Elem().SetLen(10) // panic: len > cap

        ✅ 先确保 cap 足够，或用 MakeSlice

        newSlice := reflect.MakeSlice(reflect.TypeOf(s), 10, 10)
reflect.ValueOf(&s).Elem().Set(newSlice)

        八、知识点总结

        反射进阶
├── 值修改
│   ├── ValueOf(&x).Elem() → 可寻址 Value
│   ├── CanSet() / CanAddr()
│   ├── SetInt / SetString / Set / SetLen
│   └── unexported 字段不可 Set
├── struct tag
│   ├── Tag.Get("key") → string
│   ├── Tag.Lookup("key") → (string, bool)
│   ├── json / db / validate / form
│   └── struct tag 在编译期绑定，运行时只读
├── reflect.Select
│   ├── SelectCase{Dir, Chan, Send}
│   ├── SelectSend / SelectRecv / SelectDefault
│   └── 返回 (chosen, value, ok)
├── ORM 映射
│   ├── Type → 表结构
│   ├── Tag → 列名
│   ├── Value → 数据
│   └── Scan: row → struct
└── 反射 vs 代码生成
    ├── 反射: 运行时、灵活、较慢
    ├── 代码生成: 编译期、快速、需工具链
    └── 推荐: 热路径用生成，框架用反射

          • 十一、代码演进

          v1 — 硬编码字段映射

          func scanUser(row map[string]interface{}) User {
    return User{
        ID:    row["id"].(int),
        Name:  row["name"].(string),
        Email: row["email"].(string),
    }
}
// Works but not reusable

          v2 — 通用反射 Scan

          func scan(row map[string]interface{}, dest interface{}) {
    v := reflect.ValueOf(dest).Elem()
    t := v.Type()
    for i := 0; i < t.NumField(); i++ {
        col := t.Field(i).Tag.Get("db")
        if val, ok := row[col]; ok {
            v.Field(i).Set(reflect.ValueOf(val))
        }
    }
}
// Reusable but slower than hand-written code

          v3 — Schema 缓存 + ORM 构建器

          // Schema parsed once, cached per type
var schemaCache sync.Map

func getSchema(t reflect.Type) *Schema {
    if s, ok := schemaCache.Load(t); ok {
        return s.(*Schema)
    }
    s := parseSchema(t)
    schemaCache.Store(t, s)
    return s
}
// Parse once, use many times — best of both worlds

          小结： 反射进阶的核心是理解可寻址性、struct tag 和动态 select。实际项目中，ORM 和验证库大量使用反射；但在性能敏感场景，代码生成是更好的选择。
